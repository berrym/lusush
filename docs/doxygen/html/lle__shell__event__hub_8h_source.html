1/**
2 * @file lle_shell_event_hub.h
3 * @brief LLE Shell Event Hub - Central routing for shell lifecycle events
4 *
5 * This module provides the shell event hub that coordinates shell-level events
6 * (directory changes, pre/post command) and routes them to registered handlers.
7 * It acts as a thin coordinator layer on top of the existing LLE event system.
8 *
9 * Shell lifecycle events:
10 * - Directory changed: Fired when cd changes the working directory
11 * - Pre-command: Fired just before command execution
12 * - Post-command: Fired after command completes with exit code and duration
13 *
14 * Architecture:
15 * - Handlers registered via lle_shell_event_hub_register()
16 * - Events fired via lle_fire_*() convenience functions
17 * - Also propagates to LLE event system (LLE_EVENT_DIRECTORY_CHANGED, etc.)
18 *
19 * Specification: docs/lle_specification/26_initialization_system_complete.md
20 * Date: 2025-01-16
21 */
22

38/**
39 * @brief Shell event type enumeration
40 *
41 * These map to the existing LLE event system types:
42 * - LLE_SHELL_EVENT_DIRECTORY_CHANGED -> LLE_EVENT_DIRECTORY_CHANGED (0xD000)
43 * - LLE_SHELL_EVENT_PRE_COMMAND -> LLE_EVENT_PRE_COMMAND (0xD001)
44 * - LLE_SHELL_EVENT_POST_COMMAND -> LLE_EVENT_POST_COMMAND (0xD002)
45 */
46typedef enum lle_shell_event_type {
47 LLE_SHELL_EVENT_DIRECTORY_CHANGED = 0, /**< Working directory changed */
48 LLE_SHELL_EVENT_PRE_COMMAND = 1, /**< Before command execution */
49 LLE_SHELL_EVENT_POST_COMMAND = 2, /**< After command execution */
50 LLE_SHELL_EVENT_TYPE_COUNT = 3 /**< Number of event types */
51} lle_shell_event_type_t;

53/* ============================================================================
54 * EVENT DATA STRUCTURES

58/**
59 * @brief Directory changed event data
60 *
61 * Fired when the shell's working directory changes (e.g., via cd builtin).
62 * Used to trigger cache invalidation for directory-dependent data like
63 * git status, autosuggestions, and prompt segments.
64 */
65typedef struct lle_directory_changed_event {
66 const char *old_dir; /**< Previous working directory (may be NULL) */
67 const char *new_dir; /**< New working directory (must not be NULL) */
68} lle_directory_changed_event_t;
69
70/**
71 * @brief Pre-command event data
72 *
73 * Fired just before a command is executed. Allows subsystems to record
74 * state for transient prompt display and command timing.
75 */
76typedef struct lle_pre_command_event {
77 const char *command; /**< Command about to be executed */
78 bool is_background; /**< True if command will run in background (&) */
79} lle_pre_command_event_t;
80
81/**
82 * @brief Post-command event data
83 *
84 * Fired after a command completes execution. Provides exit code and duration
85 * for prompt status display, history annotation, and timing segments.
86 */
87typedef struct lle_post_command_event {
88 const char *command; /**< Command that was executed */
89 int exit_code; /**< Command exit code (0 = success) */
90 uint64_t duration_us; /**< Execution duration in microseconds */
91} lle_post_command_event_t;

98/**
99 * @brief Shell event handler callback signature
100 *
101 * @param event_data Pointer to event-specific data structure:
102 * - lle_directory_changed_event_t* for DIRECTORY_CHANGED
103 * - lle_pre_command_event_t* for PRE_COMMAND
104 * - lle_post_command_event_t* for POST_COMMAND
105 * @param user_data User-provided context pointer from registration

107typedef void (*lle_shell_event_handler_t)(void *event_data, void *user_data);

114/** Maximum handlers per event type */
115#define LLE_SHELL_EVENT_MAX_HANDLERS 8
116
117/**
118 * @brief Handler registration entry
119 */
120typedef struct lle_shell_handler_entry {
121 lle_shell_event_handler_t handler; /**< Handler callback function */
122 void *user_data; /**< User context for callback */
123 const char *name; /**< Handler name for debugging */
124} lle_shell_handler_entry_t;
125
126/**
127 * @brief Shell event hub structure
128 *
129 * Central coordinator for shell lifecycle events. Maintains registered
130 * handlers for each event type and tracks shell state for event generation.
131 */
132typedef struct lle_shell_event_hub {
133 /** Handler arrays indexed by event type */
134 lle_shell_handler_entry_t handlers[LLE_SHELL_EVENT_TYPE_COUNT]
135 [LLE_SHELL_EVENT_MAX_HANDLERS];
136
137 /** Number of registered handlers per event type */
138 size_t handler_counts[LLE_SHELL_EVENT_TYPE_COUNT];
139
140 /** Current working directory (for change detection) */
141 char current_dir[PATH_MAX];
142
143 /** Command start time for duration calculation */
144 uint64_t command_start_time_us;
145
146 /** Current command being executed (for post-command event) */
147 char current_command[4096];
148
149 /** Whether hub is initialized */
150 bool initialized;
151
152 /* Statistics */
153 uint64_t events_fired; /**< Total events fired */
154 uint64_t directory_changes; /**< Directory change events */
155 uint64_t commands_executed; /**< Commands executed (post-command count) */
156} lle_shell_event_hub_t;
