1/**
2 * @file lle_watchdog.h
3 * @brief LLE Watchdog Timer for Deadlock Detection
4 *
5 * Provides a SIGALRM-based watchdog mechanism that detects when the
6 * readline input loop is stuck and forces recovery. This catches
7 * scenarios where:
8 * - Readline is stuck in infinite loop while processing
9 * - Event handler never returns
10 * - Deadlock in completion/search subsystem
11 *
12 * Design Principles:
13 * - Non-invasive: Signal handler sets flag, main loop checks flag
14 * - Fail-safe: If alarm fires, force hard reset
15 * - Self-resetting: Each successful input resets watchdog timer
16 * - Configurable: Timeout adjustable (default 10 seconds)
17 *
18 * Usage:
19 * lle_watchdog_init(); // At shell startup
20 *
21 * while (!done) {
22 * lle_watchdog_pet(0); // Reset timer before blocking
23 * event = read_input(); // May block
24 * if (lle_watchdog_check_and_clear()) {
25 * // Watchdog fired - force recovery
26 * lle_hard_reset();
27 * }
28 * process(event);
29 * }
30 *
31 * lle_watchdog_stop(); // On normal exit
32 * lle_watchdog_cleanup(); // At shell shutdown
33 *
34 * @see src/lle/lle_watchdog.c for implementation
35 * @see src/lle/lle_readline.c for integration
36 */

45/**
46 * @brief Default watchdog timeout in seconds

54/**
55 * @brief Initialize watchdog system
56 *
57 * Installs SIGALRM handler. Safe to call multiple times - subsequent
58 * calls are no-ops.
59 *
60 * @return LLE_SUCCESS on success
61 * @return LLE_ERROR_SYSTEM if sigaction fails

63lle_result_t lle_watchdog_init(void);

65/**
66 * @brief Cleanup watchdog system
67 *
68 * Cancels any pending alarm and restores previous SIGALRM handler.
69 * Safe to call even if init was never called.
70 */
71void lle_watchdog_cleanup(void);
72
73/**
74 * @brief Start/reset watchdog timer ("pet the dog")
75 *
76 * Call this before any potentially-blocking operation. If the operation
77 * takes longer than timeout_seconds, the watchdog will fire.
78 *
79 * The watchdog is automatically disarmed if it fires, so you must
80 * call pet() again after handling a timeout.
81 *
82 * @param timeout_seconds Timeout value in seconds (0 = use default)
83 */
84void lle_watchdog_pet(unsigned int timeout_seconds);
85
86/**
87 * @brief Stop watchdog timer
88 *
89 * Cancels any pending alarm without cleaning up the handler.
90 * Call this when readline exits normally to prevent spurious timeouts.
91 */
92void lle_watchdog_stop(void);
93
94/**
95 * @brief Check if watchdog has fired and clear the flag
96 *
97 * This is the main check point in the input loop. Call this after
98 * any potentially-blocking operation.
99 *
100 * The flag is automatically cleared by this call, so you only get
101 * one chance to handle each timeout.
102 *
103 * @return true if watchdog timeout occurred, false otherwise
104 */
105bool lle_watchdog_check_and_clear(void);
