45bool lle_shell_is_builtin(const char *text) {
46 if (!text) {
47 return false;
48 }
49
50 for (size_t i = 0; i < builtins_count; i++) {
51 if (strcmp(text, builtins[i].name) == 0) {
52 return true;
53 }
54 }
55
56 return false;

61 return false;
62 }
63
64 char *alias_value = lookup_alias(text);

66 free(alias_value);

103lle_result_t lle_completion_source_aliases(lle_memory_pool_t *memory_pool,
104 const char *prefix,
105 lle_completion_result_t *result) {
106 if (!memory_pool || !prefix || !result) {
107 return LLE_ERROR_INVALID_PARAMETER;
108 }
109
110 if (!aliases) {

112 }

115
116 ht_enum_t *aliases_e = ht_strstr_enum_create(aliases);
117 if (!aliases_e) {

120
121 const char *alias_name = NULL;
122 const char *alias_value = NULL;
123
124 while (ht_strstr_enum_next(aliases_e, &alias_name, &alias_value)) {
125 if (alias_name && strncmp(alias_name, prefix, prefix_len) == 0) {
126 lle_result_t res = lle_completion_result_add(

128
129 if (res != LLE_SUCCESS) {
130 ht_strstr_enum_destroy(aliases_e);
131 return res;
132 }
133 }
134 }
135
136 ht_strstr_enum_destroy(aliases_e);
137 return LLE_SUCCESS;

140// ============================================================================
141// PATH COMMANDS SOURCE
142// ============================================================================
143
144lle_result_t lle_completion_source_commands(lle_memory_pool_t *memory_pool,
145 const char *prefix,
146 lle_completion_result_t *result) {
147 if (!memory_pool || !prefix || !result) {
148 return LLE_ERROR_INVALID_PARAMETER;
149 }
150
151 char *path_env = getenv("PATH");
152 if (!path_env) {
153 return LLE_SUCCESS; // No PATH, not an error
154 }
155
156 char *path = strdup(path_env);
157 if (!path) {
158 return LLE_ERROR_OUT_OF_MEMORY;
159 }
160
161 size_t prefix_len = strlen(prefix);
162 char *dir = strtok(path, ":");
163 lle_result_t final_result = LLE_SUCCESS;
164
165 while (dir) {
166 DIR *d = opendir(dir);
167 if (d) {
168 struct dirent *entry;
169 while ((entry = readdir(d)) != NULL) {
170 // Skip hidden files
171 if (entry->d_name[0] == '.') {
172 continue;
173 }
174
175 // Check if name matches prefix
176 if (strncmp(entry->d_name, prefix, prefix_len) != 0) {
177 continue;
178 }
179
180 // Build full path
181 size_t path_size = strlen(dir) + strlen(entry->d_name) + 2;
182 char *full_path = malloc(path_size);
183 if (!full_path) {
184 continue; // Skip this entry on allocation failure
185 }
186
187 snprintf(full_path, path_size, "%s/%s", dir, entry->d_name);
188
189 // Check if executable
190 struct stat st;
191 if (stat(full_path, &st) == 0 && (st.st_mode & S_IXUSR)) {
192 lle_result_t res = lle_completion_result_add(
193 result, entry->d_name, " ", LLE_COMPLETION_TYPE_COMMAND,
194 800);
195
196 if (res != LLE_SUCCESS && final_result == LLE_SUCCESS) {
197 final_result = res;
198 }
199 }
200
201 free(full_path);
202 }
203 closedir(d);
204 }
205
206 dir = strtok(NULL, ":");
207 }
208
209 free(path);
210 return final_result;
211}

217/**
218 * Expand tilde (~) to home directory for completion
219 *
220 * Handles:
221 * ~/path -> /home/user/path
222 * ~user/path -> /home/user/path (if user exists)
223 *
224 * @param path The path to expand
225 * @return Newly allocated expanded path, or NULL on error
226 */
227static char *lle_completion_expand_tilde(const char *path) {
228 if (!path || path[0] != '~') {
229 return path ? strdup(path) : NULL;
230 }
231
232 // Find the end of tilde expression (until '/' or end)
233 const char *slash = strchr(path, '/');
234 const char *rest = slash ? slash : "";
235 size_t tilde_len = slash ? (size_t)(slash - path) : strlen(path);
236
237 if (tilde_len == 1) {
238 // Simple ~ expansion to $HOME
239 const char *home = getenv("HOME");
240 if (!home) {
241 struct passwd *pw = getpwuid(getuid());
242 home = pw ? pw->pw_dir : NULL;
243 }
244 if (!home) {
245 return strdup(path); // Can't expand, return original
246 }
247
248 size_t result_len = strlen(home) + strlen(rest) + 1;
249 char *result = malloc(result_len);
250 if (result) {
251 snprintf(result, result_len, "%s%s", home, rest);
252 }
253 return result;
254 } else {
255 // ~user expansion
256 char *username = strndup(path + 1, tilde_len - 1);
257 if (!username) {
258 return strdup(path);
259 }
260
261 struct passwd *pw = getpwnam(username);
262 free(username);
263
264 if (!pw) {
265 return strdup(path); // User not found, return original
266 }
267
268 size_t result_len = strlen(pw->pw_dir) + strlen(rest) + 1;
269 char *result = malloc(result_len);
270 if (result) {
271 snprintf(result, result_len, "%s%s", pw->pw_dir, rest);
272 }
273 return result;
274 }
275}

277/**
278 * Expand a single environment variable for completion
279 *
280 * Handles:
281 * $VAR/path -> /value/path
282 * ${VAR}/path -> /value/path
283 *
284 * @param path The path to expand
285 * @return Newly allocated expanded path, or NULL on error
286 */
287static char *lle_completion_expand_variable(const char *path) {
288 if (!path || path[0] != '$') {
289 return path ? strdup(path) : NULL;
290 }
291
292 const char *var_start = path + 1;
293 const char *var_end = NULL;
294 const char *rest = NULL;
295
296 if (var_start[0] == '{') {
297 // ${VAR} format
298 var_start++;
299 var_end = strchr(var_start, '}');
300 if (!var_end) {
301 return strdup(path); // Unclosed brace, return original
302 }
303 rest = var_end + 1;
304 } else {
305 // $VAR format - variable name is alphanumeric + underscore
306 var_end = var_start;
307 while (*var_end &&
308 (isalnum((unsigned char)*var_end) || *var_end == '_')) {
309 var_end++;
310 }
311 rest = var_end;
312 }
313
314 if (var_end == var_start) {
315 return strdup(path); // Empty variable name
316 }
317
318 // Extract variable name
319 size_t var_len = var_end - var_start;
320 char *var_name = strndup(var_start, var_len);
321 if (!var_name) {
322 return strdup(path);
323 }
324
325 // Look up value
326 const char *value = getenv(var_name);
327 free(var_name);
328
329 if (!value) {
330 return strdup(path); // Variable not set, return original
331 }
332
333 // Build result
334 size_t result_len = strlen(value) + strlen(rest) + 1;
335 char *result = malloc(result_len);
336 if (result) {
337 snprintf(result, result_len, "%s%s", value, rest);
338 }
339 return result;
340}

342/**
343 * Expand path for completion (tilde and variables)
344 *
345 * @param path The path to expand
346 * @param original_prefix_len Output: length of the original unexpanded prefix
347 * @return Newly allocated expanded path, or NULL on error
348 */
349static char *lle_completion_expand_path(const char *path,
350 size_t *original_prefix_len) {
351 if (!path) {
352 if (original_prefix_len)
353 *original_prefix_len = 0;
354 return NULL;
355 }
356
357 if (original_prefix_len) {
358 *original_prefix_len = strlen(path);
359 }
360
361 // Try tilde expansion first
362 if (path[0] == '~') {
363 return lle_completion_expand_tilde(path);
364 }
365
366 // Try variable expansion
367 if (path[0] == '$') {
368 return lle_completion_expand_variable(path);
369 }
370
371 // No expansion needed
372 return strdup(path);
373}

379/**
380 * Internal implementation that supports filtering
381 *
382 * @param directories_only If true, only return directories (for cd, rmdir)
383 */
384static lle_result_t lle_completion_source_files_internal(
385 lle_memory_pool_t *memory_pool, const char *prefix,
386 lle_completion_result_t *result, bool directories_only) {
387 if (!memory_pool || !prefix || !result) {
388 return LLE_ERROR_INVALID_PARAMETER;
389 }
390
391 // Expand tilde and variables in the prefix for directory access
392 size_t original_prefix_len = 0;
393 char *expanded_prefix =
394 lle_completion_expand_path(prefix, &original_prefix_len);
395 if (!expanded_prefix) {
396 return LLE_ERROR_OUT_OF_MEMORY;
397 }
398
399 // Determine the original unexpanded directory prefix to preserve in results
400 // e.g., "~/" or "$HOME/" should be preserved in completion output
401 const char *original_dir_prefix = NULL;
402 size_t original_dir_prefix_len = 0;
403 const char *orig_last_slash = strrchr(prefix, '/');
404 if (orig_last_slash) {
405 original_dir_prefix_len =
406 orig_last_slash - prefix + 1; // Include the slash
407 original_dir_prefix = prefix;
408 }
409
410 // Extract directory and filename parts from expanded path
411 const char *last_slash = strrchr(expanded_prefix, '/');
412 const char *dir_path = ".";
413 const char *file_prefix = expanded_prefix;
414
415 char *dir_copy = NULL;
416 if (last_slash) {
417 size_t dir_len = last_slash - expanded_prefix;
418 if (dir_len == 0) {
419 dir_path = "/";
420 } else {
421 dir_copy = strndup(expanded_prefix, dir_len);
422 if (dir_copy) {
423 dir_path = dir_copy;
424 }
425 }
426 file_prefix = last_slash + 1;
427 }
428
429 size_t file_prefix_len = strlen(file_prefix);
430 bool show_hidden = (file_prefix[0] == '.');
431
432 DIR *d = opendir(dir_path);
433 if (!d) {
434 free(dir_copy);
435 free(expanded_prefix);
436 return LLE_SUCCESS; // Directory doesn't exist, not an error
437 }
438
439 lle_result_t final_result = LLE_SUCCESS;
440 struct dirent *entry;
441
442 while ((entry = readdir(d)) != NULL) {
443 // Skip . and ..
444 if (strcmp(entry->d_name, ".") == 0 ||
445 strcmp(entry->d_name, "..") == 0) {
446 continue;
447 }
448
449 // Skip hidden files unless explicitly requested
450 if (!show_hidden && entry->d_name[0] == '.') {
451 continue;
452 }
453
454 // Check if name matches prefix
455 if (strncmp(entry->d_name, file_prefix, file_prefix_len) != 0) {
456 continue;
457 }
458
459 // Build full path for stat
460 size_t path_size = strlen(dir_path) + strlen(entry->d_name) + 2;
461 char *full_path = malloc(path_size);
462 if (!full_path) {
463 continue;
464 }
465
466 snprintf(full_path, path_size, "%s/%s", dir_path, entry->d_name);
467
468 // Determine if directory or file
469 struct stat st;
470 bool is_dir = false;
471 lle_completion_type_t type = LLE_COMPLETION_TYPE_FILE;
472 const char *suffix = "";
473 int32_t score = 600;
474
475 if (stat(full_path, &st) == 0 && S_ISDIR(st.st_mode)) {
476 is_dir = true;
477 type = LLE_COMPLETION_TYPE_DIRECTORY;
478 suffix = "/";
479 score = 700;
480 }
481
482 free(full_path);
483
484 // Skip files if directories_only is set
485 if (directories_only && !is_dir) {
486 continue;
487 }
488
489 // Build completion text preserving the original unexpanded prefix
490 // e.g., ~/Documents instead of /home/user/Documents
491 char *completion_text = NULL;
492 if (original_dir_prefix && original_dir_prefix_len > 0) {
493 // Use original unexpanded prefix (e.g., ~/, $HOME/)
494 size_t text_size =
495 original_dir_prefix_len + strlen(entry->d_name) + 1;
496 completion_text = malloc(text_size);
497 if (completion_text) {
498 memcpy(completion_text, original_dir_prefix,
499 original_dir_prefix_len);
500 strcpy(completion_text + original_dir_prefix_len,
501 entry->d_name);
502 }
503 } else if (last_slash && dir_copy) {
504 // No expansion happened, use expanded dir_copy
505 size_t text_size = strlen(dir_copy) + strlen(entry->d_name) + 2;
506 completion_text = malloc(text_size);
507 if (completion_text) {
508 snprintf(completion_text, text_size, "%s/%s", dir_copy,
509 entry->d_name);
510 }
511 } else if (last_slash) { // Root directory
512 size_t text_size = strlen(entry->d_name) + 2;
513 completion_text = malloc(text_size);
514 if (completion_text) {
515 snprintf(completion_text, text_size, "/%s", entry->d_name);
516 }
517 }
518
519 lle_result_t res = lle_completion_result_add(
520 result, completion_text ? completion_text : entry->d_name, suffix,
521 type, score);
522
523 free(completion_text);
524
525 if (res != LLE_SUCCESS && final_result == LLE_SUCCESS) {
526 final_result = res;
527 }
528 }
529
530 closedir(d);
531 free(dir_copy);
532 free(expanded_prefix);
533
534 return final_result;
535}

553// VARIABLES SOURCE

555
556lle_result_t lle_completion_source_variables(lle_memory_pool_t *memory_pool,
557 const char *prefix,
558 lle_completion_result_t *result) {
559 if (!memory_pool || !prefix || !result) {
560 return LLE_ERROR_INVALID_PARAMETER;
561 }
562
563 size_t prefix_len = strlen(prefix);
564 lle_result_t final_result = LLE_SUCCESS;
565
566 // Complete environment variables
567 for (char **env = environ; *env != NULL; env++) {
568 const char *equals = strchr(*env, '=');
569 if (!equals) {
570 continue;
571 }
572
573 size_t var_len = equals - *env;
574 if (var_len < prefix_len) {
575 continue;
576 }
577
578 if (strncmp(*env, prefix, prefix_len) == 0) {
579 char *var_name = strndup(*env, var_len);
580 if (var_name) {
581 lle_result_t res = lle_completion_result_add(
582 result, var_name, "", LLE_COMPLETION_TYPE_VARIABLE, 500);
583
584 free(var_name);
585
586 if (res != LLE_SUCCESS && final_result == LLE_SUCCESS) {
587 final_result = res;
588 }
589 }
590 }
591 }
592
593 // Add common shell special variables
594 const char *special_vars[] = {"?", "$", "!", "0", "#", "*", "@", "-", "_"};
595
596 for (size_t i = 0; i < sizeof(special_vars) / sizeof(special_vars[0]);
597 i++) {
598 if (strncmp(special_vars[i], prefix, prefix_len) == 0) {
599 lle_result_t res = lle_completion_result_add(
600 result, special_vars[i], "", LLE_COMPLETION_TYPE_VARIABLE, 600);
601
602 if (res != LLE_SUCCESS && final_result == LLE_SUCCESS) {
603 final_result = res;
604 }
605 }
606 }
607
608 return final_result;
609}

635lle_result_t lle_completion_source_ssh_hosts(lle_memory_pool_t *memory_pool,
636 const char *prefix,
637 lle_completion_result_t *result) {
638 if (!memory_pool || !prefix || !result) {
639 return LLE_ERROR_INVALID_PARAMETER;
640 }
641
642 ssh_host_cache_t *cache = get_ssh_host_cache();
643 if (!cache || cache->count == 0) {
644 return LLE_SUCCESS;
645 }
646
647 size_t prefix_len = strlen(prefix);
648 lle_result_t final_result = LLE_SUCCESS;
649
650 for (size_t i = 0; i < cache->count; i++) {
651 ssh_host_t *host = &cache->hosts[i];
652
653 /* Try hostname match */
654 if (strncmp(host->hostname, prefix, prefix_len) == 0) {
655 /* Format: hostname or user@hostname */
656 char completion[320];
657 if (host->user[0]) {
658 snprintf(completion, sizeof(completion), "%s@%s", host->user,
659 host->hostname);
660 } else {
661 snprintf(completion, sizeof(completion), "%s", host->hostname);
662 }
663
664 lle_result_t res = lle_completion_result_add(
665 result, completion, " ", LLE_COMPLETION_TYPE_CUSTOM,
666 800 + host->priority);
667
668 if (res != LLE_SUCCESS && final_result == LLE_SUCCESS) {
669 final_result = res;
670 }
671 continue;
672 }
673
674 /* Try alias match */
675 if (host->alias[0] && strncmp(host->alias, prefix, prefix_len) == 0) {
676 lle_result_t res = lle_completion_result_add(
677 result, host->alias, " ", LLE_COMPLETION_TYPE_CUSTOM,
678 850 + host->priority);
679
680 if (res != LLE_SUCCESS && final_result == LLE_SUCCESS) {
681 final_result = res;
682 }
683 }
684 }
685
686 return final_result;
