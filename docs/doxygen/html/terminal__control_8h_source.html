103 TERMINAL_CAP_COLOR_TRUECOLOR = (1 << 3), // 24-bit RGB color support
104 TERMINAL_CAP_CURSOR_POSITIONING = (1 << 4), // Cursor positioning support
105 TERMINAL_CAP_CURSOR_VISIBILITY = (1 << 5), // Cursor visibility control

745/*
746 * Basic usage example:
747 *
748 * ```c
749 * #include "terminal_control.h"
750 *
751 * int main() {
752 * // Create base terminal
753 * base_terminal_t *base = base_terminal_create();
754 * base_terminal_init(base);
755 *
756 * // Create terminal control layer
757 * terminal_control_t *control = terminal_control_create(base);
758 * if (!control) return 1;
759 *
760 * if (terminal_control_init(control) != TERMINAL_CONTROL_SUCCESS) {
761 * terminal_control_destroy(control);
762 * return 1;
763 * }
764 *
765 * // Use terminal control features
766 * terminal_control_clear_screen(control);
767 * terminal_control_move_cursor(control, 1, 1);
768 *
769 * // Set colors
770 * terminal_color_t red =
771 * terminal_control_color_from_basic(TERMINAL_COLOR_RED);
772 * terminal_control_set_foreground_color(control, red);
