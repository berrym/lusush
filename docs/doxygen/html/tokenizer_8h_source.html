18// Simple, clean token classification for parser
19typedef enum {
20 // Basic token types
21 TOK_EOF, // End of input
22 TOK_WORD, // Regular word (command, argument, variable name)
23 TOK_STRING, // Quoted string ('...' - literal)
24 TOK_EXPANDABLE_STRING, // Double-quoted string ("...") - needs variable
25 // expansion
26 TOK_NUMBER, // Numeric literal
27 TOK_VARIABLE, // Variable reference ($var, ${var}, etc.)
28
29 // Operators and separators
30 TOK_SEMICOLON, // ;
31 TOK_PIPE, // |
32 TOK_AND, // &
33 TOK_LOGICAL_AND, // &&
34 TOK_LOGICAL_OR, // ||
35 TOK_REDIRECT_IN, // <
36 TOK_REDIRECT_OUT, // >
37 TOK_APPEND, // >>
38 TOK_HEREDOC, // <<
39 TOK_HEREDOC_STRIP, // <<-
40 TOK_HERESTRING, // <<<
41 TOK_REDIRECT_ERR, // 2>
42 TOK_REDIRECT_BOTH, // &>
43 TOK_APPEND_ERR, // 2>>
44 TOK_REDIRECT_FD, // &1, &2, etc.
45 TOK_REDIRECT_CLOBBER, // >|
46 TOK_ASSIGN, // =
47 TOK_NOT_EQUAL, // !=
48 TOK_PLUS, // +
49 TOK_MINUS, // -
50 TOK_MULTIPLY, // *
51 TOK_DIVIDE, // /
52 TOK_MODULO, // %
53 TOK_GLOB, // * (when used for globbing)
54 TOK_QUESTION, // ?
55 TOK_COMMAND_SUB, // $(...)
56 TOK_ARITH_EXP, // $((...))
57 TOK_BACKQUOTE, // `
58
59 // Delimiters
60 TOK_LPAREN, // (
61 TOK_RPAREN, // )
62 TOK_LBRACE, // {
63 TOK_RBRACE, // }
64 TOK_LBRACKET, // [
65 TOK_RBRACKET, // ]
66
67 // Keywords (recognized contextually)
68 TOK_IF,
69 TOK_THEN,
70 TOK_ELSE,
71 TOK_ELIF,
72 TOK_FI,
73 TOK_WHILE,
74 TOK_DO,
75 TOK_DONE,
76 TOK_FOR,
77 TOK_IN,
78 TOK_CASE,
79 TOK_ESAC,
80 TOK_UNTIL,
81 TOK_FUNCTION,
82
83 // Special
84 TOK_NEWLINE, // \n (significant in shell)
85 TOK_WHITESPACE, // Spaces, tabs (usually ignored)
86 TOK_COMMENT, // # comment
87 TOK_ERROR // Invalid token
88} token_type_t;

91typedef struct token {

97 size_t position; // Absolute position in input
