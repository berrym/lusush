19/**
20 * __doubledup:
21 * Create a pointer duplicating val.
22 */
23static double *__doubledup(const double *val) {
24 double *d = calloc(1, sizeof(double));
25 if (!d) {
26 perror("__floatdup");
27 return NULL;
28 }
29
30 return memcpy(d, val, sizeof(double));
31}

37ht_strdouble_t *ht_strdouble_create(unsigned int flags) {

43
44 if (flags & HT_STR_CASECMP) {
45 hash = fnv1a_hash_str_casecmp;
46 keyeq = str_caseeq;
47 }
48
49 return (ht_strdouble_t *)ht_create(hash, keyeq, &callbacks, flags);
50}
