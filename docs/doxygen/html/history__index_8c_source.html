24/**
25 * Hash function for uint64_t keys
26 * Uses a simple but effective integer hash
27 */
28static uint64_t hash_uint64(const void *key, uint64_t seed) {
29 (void)seed; /* Unused for deterministic integer hashing */
30
31 uint64_t k = *(const uint64_t *)key;
32
33 /* Thomas Wang's 64-bit integer hash */
34 k = (~k) + (k << 21);
35 k = k ^ (k >> 24);
36 k = (k + (k << 3)) + (k << 8);
37 k = k ^ (k >> 14);
38 k = (k + (k << 2)) + (k << 4);
39 k = k ^ (k >> 28);
40 k = k + (k << 31);
41
42 return k;
43}

45/**
46 * Equality function for uint64_t keys
47 */
48static bool eq_uint64(const void *key1, const void *key2) {
49 return *(const uint64_t *)key1 == *(const uint64_t *)key2;
50}

91/**
92 * Create hashtable index for fast ID lookup
93 *
94 * @param index Output pointer for created hashtable
95 * @param initial_capacity Initial capacity hint
96 * @return LLE_SUCCESS or error code
97 */
98lle_result_t lle_history_index_create(lle_hashtable_t **index,
99 size_t initial_capacity) {
100 if (!index) {
101 return LLE_ERROR_INVALID_PARAMETER;
102 }
103
104 /* Set up callbacks for uint64_t keys and entry pointer values */
105 ht_callbacks_t callbacks = {.key_copy = copy_uint64_key,
106 .key_free = free_uint64_key,
107 .val_copy = copy_entry_ptr,
108 .val_free = free_entry_ptr};
109
110 /* Create hashtable with custom hash and equality functions */
111 ht_t *ht = ht_create(hash_uint64, eq_uint64, &callbacks,
112 (unsigned int)initial_capacity);
113 if (!ht) {
114 return LLE_ERROR_OUT_OF_MEMORY;
115 }
116
117 *index = ht;
118 return LLE_SUCCESS;
119}

242/**
243 * Rebuild index from history core entries
244 *
245 * This function rebuilds the entire index from the history core's
246 * entry array. Useful after bulk operations or corruption recovery.
247 *
248 * @param core History core
249 * @return LLE_SUCCESS or error code
250 */
251lle_result_t lle_history_rebuild_index(lle_history_core_t *core) {
252 if (!core) {
253 return LLE_ERROR_INVALID_PARAMETER;
254 }
255
256 /* If no index exists, create one */
257 if (!core->entry_lookup) {
258 lle_result_t result = lle_history_index_create(
259 &core->entry_lookup, core->config->initial_capacity);
260 if (result != LLE_SUCCESS) {
261 return result;
262 }
263 } else {
264 /* Clear existing index */
265 lle_history_index_clear(core->entry_lookup);
266 }
267
268 /* Rebuild from entries array */
269 for (size_t i = 0; i < core->entry_count; i++) {
270 lle_history_entry_t *entry = core->entries[i];
271 if (entry) {
272 lle_result_t result = lle_history_index_insert(
273 core->entry_lookup, entry->entry_id, entry);
274 if (result != LLE_SUCCESS) {
275 return result;
276 }
277 }
278 }
279
280 return LLE_SUCCESS;
281}

296/**
297 * Get last N entries
298 *
299 * Returns the most recent N entries from history.
300 *
301 * @param core History core
302 * @param n Number of entries to retrieve
303 * @param entries Output array (caller must allocate at least n pointers)
304 * @param count Output pointer for actual number retrieved
305 * @return LLE_SUCCESS or error code
306 */
307lle_result_t lle_history_get_last_n_entries(lle_history_core_t *core, size_t n,
308 lle_history_entry_t **entries,
309 size_t *count) {
310 if (!core || !entries || !count) {
311 return LLE_ERROR_INVALID_PARAMETER;
312 }
313
314 pthread_rwlock_rdlock(&core->lock);
315
316 /* Calculate actual count to return */
317 size_t actual_n = (n < core->entry_count) ? n : core->entry_count;
318
319 /* Copy last N entries from the array */
320 size_t start_index = core->entry_count - actual_n;
321 for (size_t i = 0; i < actual_n; i++) {
322 entries[i] = core->entries[start_index + i];
323 }
324
325 *count = actual_n;
326
327 pthread_rwlock_unlock(&core->lock);
328 return LLE_SUCCESS;
329}

331/**
332 * Get entry by reverse index (for Up arrow - most recent first)
333 *
334 * Index 0 = most recent entry
335 * Index 1 = second most recent
336 * etc.
337 *
338 * @param core History core
339 * @param reverse_index Reverse index (0 = newest)
340 * @param entry Output pointer for entry
341 * @return LLE_SUCCESS or error code
342 */
343lle_result_t
344lle_history_get_entry_by_reverse_index(lle_history_core_t *core,
345 size_t reverse_index,
346 lle_history_entry_t **entry) {
347 if (!core || !entry) {
348 return LLE_ERROR_INVALID_PARAMETER;
349 }
350
351 pthread_rwlock_rdlock(&core->lock);
352
353 /* Check bounds */
354 if (reverse_index >= core->entry_count) {
355 pthread_rwlock_unlock(&core->lock);
356 return LLE_ERROR_INVALID_RANGE; /* Index out of bounds */
357 }
358
359 /* Calculate forward index */
360 size_t forward_index = core->entry_count - 1 - reverse_index;
361 *entry = core->entries[forward_index];
362
363 pthread_rwlock_unlock(&core->lock);
364 return LLE_SUCCESS;
365}
