29/**
30 * @brief Preprocessing stage execution
31 *
32 * First stage of the rendering pipeline. Prepares buffer content for
33 * subsequent stages by normalizing line endings and handling special
34 * characters.
35 *
36 * @param context Render context with buffer and configuration
37 * @param output Output pointer to receive preprocessed content
38 * @return LLE_SUCCESS on success, error code on failure
39 */
40static lle_result_t
41lle_pipeline_stage_preprocess(lle_render_context_t *context,
42 lle_render_output_t **output) {
43 if (!context || !context->buffer || !output) {
44 return LLE_ERROR_INVALID_PARAMETER;
45 }
46
47 /* Allocate output structure */
48 lle_render_output_t *out = lle_pool_alloc(sizeof(lle_render_output_t));
49 if (!out) {
50 return LLE_ERROR_OUT_OF_MEMORY;
51 }
52 memset(out, 0, sizeof(lle_render_output_t));
53
54 /* Estimate size (buffer length + overhead for processing) */
55 size_t estimated_size = context->buffer->length + 256;
56
57 /* Allocate content buffer */
58 out->content = lle_pool_alloc(estimated_size);
59 if (!out->content) {
60 lle_pool_free(out);
61 return LLE_ERROR_OUT_OF_MEMORY;
62 }
63 out->content_capacity = estimated_size;
64
65 /* Copy buffer content (preprocessing is simple copy for now) */
66 if (context->buffer->length > 0) {
67 memcpy(out->content, context->buffer->data, context->buffer->length);
68 out->content_length = context->buffer->length;
69 } else {
70 out->content[0] = '\0';
71 out->content_length = 0;
72 }
73
74 *output = out;
75 return LLE_SUCCESS;
76}

230 *
231 * Creates and initializes a multi-stage rendering pipeline with 4 stages:

242 /* Step 1: Validate parameters */

244 return LLE_ERROR_INVALID_PARAMETER;
245 }
246
247 /* Step 2: Allocate pipeline structure */
248 lle_render_pipeline_t *pipe = lle_pool_alloc(sizeof(lle_render_pipeline_t));
249 if (!pipe) {
250 return LLE_ERROR_OUT_OF_MEMORY;

254 /* Step 3: Store memory pool reference */

257 /* Step 4: Initialize pipeline configuration */
258 pipe->stage_capacity =

260 pipe->stage_count = 4;

264 pipe->stages =

348/**
349 * @brief Execute rendering pipeline
350 *
351 * Executes all enabled stages of the rendering pipeline in sequence,
352 * passing the output of each stage to the next.
353 *
354 * @param pipeline Pipeline to execute
355 * @param context Render context with input data
356 * @param output Output pointer to receive final rendered content
357 * @return LLE_SUCCESS on success, error code on failure
358 *
359 * SPEC COMPLIANCE: Section 3.4.3 "Pipeline Execution"
360 * PERFORMANCE: Tracks execution time for each stage
361 */
362lle_result_t lle_render_pipeline_execute(lle_render_pipeline_t *pipeline,
363 lle_render_context_t *context,
364 lle_render_output_t **output) {
365 struct timespec stage_start, stage_end;
366 lle_result_t result = LLE_SUCCESS;
367
368 /* Step 1: Validate parameters */
369 if (!pipeline || !context || !output) {
370 return LLE_ERROR_INVALID_PARAMETER;
371 }
372
373 /* Step 2: Lock pipeline for thread safety */
374 pthread_mutex_lock(&pipeline->pipeline_lock);
375
376 /* Step 3: Execute each stage in sequence */
377 lle_render_output_t *stage_output = NULL;
378
379 for (size_t i = 0; i < pipeline->stage_count; i++) {
380 /* Skip disabled stages */
381 if (!pipeline->stages[i].enabled) {
382 continue;
383 }
384
385 /* Record start time */
386 clock_gettime(CLOCK_MONOTONIC, &stage_start);
387
388 /* Execute stage */
389 result = pipeline->stages[i].execute(context, &stage_output);
390
391 /* Record end time */
392 clock_gettime(CLOCK_MONOTONIC, &stage_end);
393
394 /* Check for errors */
395 if (result != LLE_SUCCESS) {
396 pthread_mutex_unlock(&pipeline->pipeline_lock);
397 return result;
398 }
399
400 /* Update stage metrics */
401 uint64_t stage_time_ns =
402 (stage_end.tv_sec - stage_start.tv_sec) * 1000000000ULL +
403 (stage_end.tv_nsec - stage_start.tv_nsec);
404 pipeline->stages[i].execution_count++;
405 pipeline->stages[i].total_execution_time_ns += stage_time_ns;
406
407 /* For now, each stage creates its own output */
408 /* In a more advanced implementation, stages would pass output between
409 * them */
410 /* For this basic implementation, we'll use the last stage's output */
411 if (i < pipeline->stage_count - 1 && stage_output) {
412 /* Free intermediate outputs (except the last one) */
413 lle_render_output_free(stage_output);
414 stage_output = NULL;
415 }
416 }
417
418 /* Step 4: Unlock pipeline */
419 pthread_mutex_unlock(&pipeline->pipeline_lock);
420
421 /* Step 5: Return final output */
422 *output = stage_output;
423 return LLE_SUCCESS;
424}
