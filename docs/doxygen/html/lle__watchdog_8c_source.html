55/**
56 * SIGALRM handler - must be async-signal-safe.
57 *
58 * Only uses atomic operations and avoids any non-reentrant functions.
59 * This is called from signal context, so must be minimal.
60 */
61static void watchdog_signal_handler(int sig) {
62 (void)sig; /* Unused, but required by signal handler signature */
63
64 /* Set the fired flag atomically */
65 atomic_store(&g_watchdog_fired, true);
66
67 /* Disarm the watchdog to prevent repeated firing */
68 atomic_store(&g_watchdog_armed, false);
69
70 /* Increment fire counter (atomic) */
71 atomic_fetch_add(&g_stats_fires, 1);
72}

79lle_result_t lle_watchdog_init(void) {
80 if (g_initialized) {
81 return LLE_SUCCESS; /* Already initialized */
82 }
83
84 struct sigaction sa;
85 sa.sa_handler = watchdog_signal_handler;
86 sa.sa_flags = 0; /* No SA_RESTART - we want to interrupt syscalls */
87 sigemptyset(&sa.sa_mask);
88
89 if (sigaction(SIGALRM, &sa, &g_old_sigalrm_action) < 0) {
90 return LLE_ERROR_SYSTEM_CALL;
91 }
92
93 g_initialized = true;
94 atomic_store(&g_watchdog_fired, false);
95 atomic_store(&g_watchdog_armed, false);
96 g_current_timeout = 0;
97
98 return LLE_SUCCESS;
99}

101void lle_watchdog_cleanup(void) {
102 if (!g_initialized) {
103 return;
104 }
105
106 /* Cancel any pending alarm */
107 alarm(0);
108
109 /* Restore previous handler */
110 sigaction(SIGALRM, &g_old_sigalrm_action, NULL);
111
112 g_initialized = false;
113 atomic_store(&g_watchdog_armed, false);
114 atomic_store(&g_watchdog_fired, false);
115 g_current_timeout = 0;
116}

118void lle_watchdog_pet(unsigned int timeout_seconds) {
119 if (!g_initialized) {
120 return;
121 }
122
123 if (timeout_seconds == 0) {
124 timeout_seconds = LLE_WATCHDOG_TIMEOUT_DEFAULT;
125 }
126
127 /* Clear fired flag and arm the watchdog */
128 atomic_store(&g_watchdog_fired, false);
129 atomic_store(&g_watchdog_armed, true);
130 g_current_timeout = timeout_seconds;

133 alarm(timeout_seconds);

153bool lle_watchdog_check_and_clear(void) {
154 bool was_fired = atomic_exchange(&g_watchdog_fired, false);
155
156 if (was_fired) {
157 /* Count successful recovery (we caught the timeout) */
158 atomic_fetch_add(&g_stats_recoveries, 1);
159 }
160
161 return was_fired;
162}
