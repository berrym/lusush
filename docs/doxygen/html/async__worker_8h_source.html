1/**
2 * @file async_worker.h
3 * @brief LLE Async Worker Thread Pool
4 *
5 * Provides a pthread-based worker thread for async operations like git status.
6 * This enables non-blocking prompt generation where expensive operations run
7 * in the background.
8 *
9 * Specification: docs/lle_specification/25_prompt_theme_system_complete.md
10 * Section: 7 - Async Operations
11 *
12 * Design Principles:
13 * - Single worker thread (sufficient for git operations)
14 * - Request queue with mutex/condition variable synchronization
15 * - Completion callbacks for async responses
16 * - Graceful shutdown with pending request draining
17 *
18 * Example Usage:
19 *
20 * // Callback for completed requests
21 * void on_git_status(const lle_async_response_t *resp, void *user_data) {
22 * if (resp->result == LLE_SUCCESS) {
23 * printf("Branch: %s\n", resp->data.git_status.branch);
24 * }
25 * }
26 *
27 * // Initialize and start worker
28 * lle_async_worker_t *worker;
29 * lle_async_worker_init(&worker, on_git_status, NULL);
30 * lle_async_worker_start(worker);
31 *
32 * // Submit async request
33 * lle_async_request_t *req =
34 * lle_async_request_create(LLE_ASYNC_GIT_STATUS); strncpy(req->cwd,
35 * "/path/to/repo", sizeof(req->cwd) - 1); lle_async_worker_submit(worker, req);
36 *
37 * // Later: shutdown
38 * lle_async_worker_shutdown(worker);
39 * lle_async_worker_destroy(worker);
40 */

182/**
183 * Initialize async worker
184 *
185 * Creates the worker structure but does not start the thread. Call
186 * lle_async_worker_start() to begin processing requests.
187 *
188 * @param worker Output pointer for created worker (must not be NULL)
189 * @param on_complete Callback for completed requests (may be NULL)
190 * @param user_data User data passed to callback
191 * @return LLE_SUCCESS on success
192 * @return LLE_ERROR_INVALID_PARAMETER if worker is NULL
193 * @return LLE_ERROR_OUT_OF_MEMORY if allocation fails
194 */
195lle_result_t lle_async_worker_init(lle_async_worker_t **worker,
196 lle_async_completion_fn on_complete,
197 void *user_data);

199/**
200 * Start async worker thread
201 *
202 * Starts the worker thread. The worker will begin processing queued requests.
203 *
204 * @param worker Worker to start (must not be NULL)
205 * @return LLE_SUCCESS on success
206 * @return LLE_ERROR_INVALID_PARAMETER if worker is NULL or already running
207 * @return LLE_ERROR_INTERNAL if thread creation fails
208 */
209lle_result_t lle_async_worker_start(lle_async_worker_t *worker);
210
211/**
212 * Request worker shutdown
213 *
214 * Signals the worker to stop accepting new requests and finish processing
215 * pending requests. This is non-blocking; use lle_async_worker_wait() to
216 * wait for completion.
217 *
218 * @param worker Worker to shutdown (must not be NULL)
219 * @return LLE_SUCCESS on success
220 * @return LLE_ERROR_INVALID_PARAMETER if worker is NULL
221 */
222lle_result_t lle_async_worker_shutdown(lle_async_worker_t *worker);
223
224/**
225 * Wait for worker to complete
226 *
227 * Blocks until the worker thread exits. Should be called after shutdown.
228 *
229 * @param worker Worker to wait for (must not be NULL)
230 * @return LLE_SUCCESS on success
231 * @return LLE_ERROR_INVALID_PARAMETER if worker is NULL
232 */
233lle_result_t lle_async_worker_wait(lle_async_worker_t *worker);
234
235/**
236 * Destroy async worker
237 *
238 * Frees all resources. Must call lle_async_worker_shutdown() and
239 * lle_async_worker_wait() first if the worker is running.
240 *
241 * @param worker Worker to destroy (may be NULL)
