64/**
65 * Check if a Unicode codepoint is a word character (alphanumeric or underscore)
66 * Uses wctype.h for Unicode-aware character classification.
67 */
68LLE_MAYBE_UNUSED
69static bool is_word_codepoint(uint32_t cp) {
70 /* Underscore is always a word character */
71 if (cp == '_')
72 return true;
73
74 /* Use Unicode-aware classification for alphanumeric */
75 return iswalnum((wint_t)cp);
76}

81static bool is_shell_metachar(uint32_t cp) {
82 return cp == '|' || cp == '&' || cp == ';' || cp == '(' || cp == ')' ||
83 cp == '<' || cp == '>' || cp == '\'' || cp == '"' || cp == '`' ||
84 cp == '$' || cp == '\\';

101/**
102 * Find start of previous grapheme cluster from byte position
103 * Returns byte offset of the start of the previous grapheme cluster.
104 */
105static size_t find_prev_grapheme_start(const char *text, size_t len,
106 size_t pos) {
107 if (pos == 0)
108 return 0;
109
110 const char *start = text;
111 const char *end = text + len;
112 const char *ptr = text + pos;
113
114 /* Move back one byte at minimum */
115 ptr--;
116
117 /* Scan backwards until we find a grapheme boundary */
118 while (ptr > start && !lle_is_grapheme_boundary(ptr, start, end)) {
119 ptr--;
120 }
121
122 return (size_t)(ptr - start);
123}

125/**
126 * Find end of current grapheme cluster from byte position
127 * Returns byte offset just past the end of the current grapheme cluster.
128 */
129static size_t find_next_grapheme_end(const char *text, size_t len, size_t pos) {
130 if (pos >= len)
131 return len;
132
133 const char *start = text;
134 const char *end = text + len;
135 const char *ptr = text + pos;
136
137 /* Move to next UTF-8 character */
138 int seq_len = lle_utf8_sequence_length((unsigned char)*ptr);
139 if (seq_len <= 0)
140 seq_len = 1;
141 ptr += seq_len;
142
143 /* Continue until we find a grapheme boundary */
144 while (ptr < end && !lle_is_grapheme_boundary(ptr, start, end)) {
145 seq_len = lle_utf8_sequence_length((unsigned char)*ptr);
146 if (seq_len <= 0)
147 seq_len = 1;
148 ptr += seq_len;
149 }
150
151 return (size_t)(ptr - start);
152}

154/**
155 * Decode codepoint at grapheme cluster start position
156 */
157static uint32_t decode_codepoint_at(const char *text, size_t len, size_t pos) {
158 if (pos >= len)
159 return 0;
160
161 uint32_t cp = 0;
162 int decoded = lle_utf8_decode_codepoint(text + pos, len - pos, &cp);
163 if (decoded <= 0) {
164 /* Fallback to byte value for invalid UTF-8 */
165 return (unsigned char)text[pos];
166 }
167 return cp;
168}

170/**
171 * Find start of current word from position (grapheme-aware)
172 *
173 * Works by scanning grapheme clusters backwards, checking if each
174 * grapheme's first codepoint is a word character.
175 */
176static size_t find_word_start(const char *text, size_t pos) {
177 if (pos == 0)
178 return 0;
179
180 size_t len = strlen(text);
181 size_t current = pos;
182
183 /* Skip whitespace backward (by grapheme clusters) */
184 while (current > 0) {
185 size_t prev = find_prev_grapheme_start(text, len, current);
186 uint32_t cp = decode_codepoint_at(text, len, prev);
187
188 if (!is_whitespace_codepoint(cp)) {
189 break;
190 }
191 current = prev;
192 }
193
194 /* Find beginning of word (by grapheme clusters) */
195 while (current > 0) {
196 size_t prev = find_prev_grapheme_start(text, len, current);
197 uint32_t cp = decode_codepoint_at(text, len, prev);
198
199 /* Stop at whitespace or shell metacharacters */
200 if (is_whitespace_codepoint(cp) || is_shell_metachar(cp)) {
201 break;
202 }
203 current = prev;
204 }
205
206 return current;
207}

209/**
210 * Find end of current word from position (grapheme-aware)
211 *
212 * Works by scanning grapheme clusters forward, checking if each
213 * grapheme's first codepoint is a word character.
214 */
215static size_t find_word_end(const char *text, size_t len, size_t pos) {
216 size_t current = pos;
217
218 /* Skip whitespace forward (by grapheme clusters) */
219 while (current < len) {
220 uint32_t cp = decode_codepoint_at(text, len, current);
221
222 if (!is_whitespace_codepoint(cp)) {
223 break;
224 }
225 current = find_next_grapheme_end(text, len, current);
226 }
227
228 /* Find end of word (by grapheme clusters) */
229 while (current < len) {
230 uint32_t cp = decode_codepoint_at(text, len, current);
231
232 /* Stop at whitespace or shell metacharacters */
233 if (is_whitespace_codepoint(cp) || is_shell_metachar(cp)) {
234 break;
235 }
236 current = find_next_grapheme_end(text, len, current);
237 }
238
239 return current;
240}

276/**
277 * Replace word at cursor with completion text
278 * Deletes the word being completed and inserts the selected completion
279 */
280static lle_result_t replace_word_at_cursor(lle_editor_t *editor,
281 size_t word_start,
282 size_t word_length,
283 const char *replacement) {
284 if (!editor || !editor->buffer || !replacement) {
285 return LLE_ERROR_INVALID_PARAMETER;
286 }
287
288 /* Delete the word being completed */
289 if (word_length > 0) {
290 lle_result_t result =
291 lle_buffer_delete_text(editor->buffer, word_start, word_length);
292 if (result != LLE_SUCCESS) {
293 return result;
294 }
295 }
296
297 /* Insert the replacement text at word_start */
298 lle_result_t result = lle_buffer_insert_text(
299 editor->buffer, word_start, replacement, strlen(replacement));
300 if (result != LLE_SUCCESS) {
301 return result;
302 }
303
304 /* Move cursor to end of inserted text */
305 size_t new_pos = word_start + strlen(replacement);
306 lle_result_t move_result =
307 lle_cursor_manager_move_to_byte_offset(editor->cursor_manager, new_pos);
308
309 /* CRITICAL: Sync buffer cursor back from cursor manager after movement */
310 if (move_result == LLE_SUCCESS) {
311 lle_cursor_manager_get_position(editor->cursor_manager,
312 &editor->buffer->cursor);
313 }
314
315 return move_result;
316}

375/**
376 * Get current line boundaries for multiline buffer
377 */
378static void get_current_line_bounds(lle_buffer_t *buffer, size_t *start,
379 size_t *end) {
380 const char *data = buffer->data;
381 size_t cursor = buffer->cursor.byte_offset;
382 size_t len = buffer->length;
383
384 /* Find start of current line */
385 *start = cursor;
386 while (*start > 0 && data[*start - 1] != '\n') {
387 (*start)--;
388 }
389
390 /* Find end of current line */
391 *end = cursor;
392 while (*end < len && data[*end] != '\n') {
393 (*end)++;
394 }
395}

667/**
668 * Move cursor to previous line (up arrow in multiline mode)
669 * Preserves horizontal column position using sticky_column
670 */
671lle_result_t lle_previous_line(lle_editor_t *editor) {
672 if (!editor || !editor->buffer || !editor->cursor_manager) {
673 return LLE_ERROR_INVALID_PARAMETER;
674 }
675
676 const char *data = editor->buffer->data;
677 size_t cursor = editor->buffer->cursor.byte_offset;
678
679 debug_log("previous_line: cursor=%zu, buffer_len=%zu", cursor,
680 editor->buffer->length);
681
682 /* Get current line boundaries */
683 size_t curr_line_start, curr_line_end;
684 get_current_line_bounds(editor->buffer, &curr_line_start, &curr_line_end);
685
686 debug_log("previous_line: curr_line_start=%zu, curr_line_end=%zu",
687 curr_line_start, curr_line_end);
688
689 /* If we're on the first line, can't move up */
690 if (curr_line_start == 0) {
691 debug_log("previous_line: already on first line, returning no-op");
692 return LLE_SUCCESS; /* No-op, stay on first line */
693 }
694
695 /* Find previous line boundaries */
696 /* curr_line_start points to first char of current line (after the '\n') */
697 /* So curr_line_start - 1 is the '\n' that ends the previous line */
698
699 /* The newline at curr_line_start - 1 terminates the previous line.
700 * We need to find where that previous line starts.
701 *
702 * Example: "line1\n\nline3" with cursor on line3
703 * 012345 6 789...
704 * line1 \n \n line3
705 * ^ ^
706 * | curr_line_start = 7
707 * prev line's terminating newline = 6
708 *
709 * The previous line (empty) spans from index 6 to 6 (just the newline).
710 * For an empty line, prev_line_start = prev_line_end = position after prior
711 * newline.
712 */
713
714 size_t prev_line_terminator =
715 curr_line_start - 1; /* The '\n' that ends prev line */
716
717 /* Find start of previous line by searching backwards for newline (or buffer
718 * start) */
719 size_t prev_line_start = prev_line_terminator;
720 while (prev_line_start > 0 && data[prev_line_start - 1] != '\n') {
721 prev_line_start--;
722 }
723
724 /* Calculate or retrieve preferred column */
725 size_t target_column;
726 if (editor->cursor_manager->sticky_column) {
727 /* Use saved preferred column */
728 target_column = editor->cursor_manager->preferred_visual_column;
729 } else {
730 /* First vertical movement - save current column */
731 target_column = cursor - curr_line_start;
732 editor->cursor_manager->preferred_visual_column = target_column;
733 editor->cursor_manager->sticky_column = true;
734 }

736 /* Calculate new cursor position on previous line */
737 /* prev_line_start points to first character of line content
738 * prev_line_terminator points to the '\n' that ends the line
739 * Line length = prev_line_terminator - prev_line_start (0 for empty lines)
740 */
741 size_t prev_line_length = prev_line_terminator - prev_line_start;
742 size_t new_cursor = prev_line_start + target_column;
743
744 /* Clamp to end of previous line if column is too far right */
745 if (target_column > prev_line_length) {
746 new_cursor = prev_line_terminator; /* Position cursor at end of line
747 (before newline) */
748 }
749
750 /* Temporarily disable sticky_column to prevent move_to_byte_offset from
751 * overwriting preferred_visual_column */
752 bool was_sticky = editor->cursor_manager->sticky_column;
753 size_t saved_preferred = editor->cursor_manager->preferred_visual_column;
754 editor->cursor_manager->sticky_column = false;
755
756 /* Use cursor_manager to move (updates all fields properly) */
757 lle_result_t result = lle_cursor_manager_move_to_byte_offset(
758 editor->cursor_manager, new_cursor);
759
760 /* CRITICAL: Sync buffer cursor back from cursor manager after movement */
761 if (result == LLE_SUCCESS) {
762 lle_cursor_manager_get_position(editor->cursor_manager,
763 &editor->buffer->cursor);
764 }
765
766 /* Restore sticky_column state */
767 editor->cursor_manager->sticky_column = was_sticky;
768 editor->cursor_manager->preferred_visual_column = saved_preferred;
769
770 return result;
771}

773/**
774 * Move cursor to next line (down arrow in multiline mode)
775 * Preserves horizontal column position using sticky_column
776 */
777lle_result_t lle_next_line(lle_editor_t *editor) {
778 if (!editor || !editor->buffer || !editor->cursor_manager) {
779 return LLE_ERROR_INVALID_PARAMETER;
780 }
781
782 const char *data = editor->buffer->data;
783 size_t cursor = editor->buffer->cursor.byte_offset;
784 size_t len = editor->buffer->length;
785
786 /* Get current line boundaries */
787 size_t curr_line_start, curr_line_end;
788 get_current_line_bounds(editor->buffer, &curr_line_start, &curr_line_end);
789
790 /* If we're on the last line, can't move down */
791 if (curr_line_end >= len || data[curr_line_end] != '\n') {
792 return LLE_SUCCESS; /* No-op, stay on last line */
793 }
794
795 /* Find next line boundaries */
796 size_t next_line_start = curr_line_end + 1; /* Skip the '\n' */
797 size_t next_line_end = next_line_start;
798
799 while (next_line_end < len && data[next_line_end] != '\n') {
800 next_line_end++;
801 }
802
803 /* Calculate or retrieve preferred column */
804 size_t target_column;
805 if (editor->cursor_manager->sticky_column) {
806 /* Use saved preferred column */
807 target_column = editor->cursor_manager->preferred_visual_column;
808 } else {
809 /* First vertical movement - save current column */
810 target_column = cursor - curr_line_start;
811 editor->cursor_manager->preferred_visual_column = target_column;
812 editor->cursor_manager->sticky_column = true;
813 }
814
815 /* Calculate new cursor position on next line */
816 /* next_line_end points to newline at end of line (or buffer end) */
817 /* next_line_start points to first character */
818 /* Line length is distance between them, cursor at next_line_end positions
819 * after last char */
820 size_t next_line_length = next_line_end - next_line_start;
821 size_t new_cursor = next_line_start + target_column;
822
823 /* Clamp to end of next line if column is too far right */
824 if (target_column > next_line_length) {
825 new_cursor =
826 next_line_end; /* Position cursor at/after last character */
827 }
828
829 /* Temporarily disable sticky_column to prevent move_to_byte_offset from
830 * overwriting preferred_visual_column */
831 bool was_sticky = editor->cursor_manager->sticky_column;
832 size_t saved_preferred = editor->cursor_manager->preferred_visual_column;
833 editor->cursor_manager->sticky_column = false;
834
835 /* Use cursor_manager to move (updates all fields properly) */
836 lle_result_t result = lle_cursor_manager_move_to_byte_offset(
837 editor->cursor_manager, new_cursor);
838
839 /* CRITICAL: Sync buffer cursor back from cursor manager after movement */
840 if (result == LLE_SUCCESS) {
841 lle_cursor_manager_get_position(editor->cursor_manager,
842 &editor->buffer->cursor);
843 }
844
845 /* Restore sticky_column state */
846 editor->cursor_manager->sticky_column = was_sticky;
847 editor->cursor_manager->preferred_visual_column = saved_preferred;
848
849 return result;
850}

1298lle_result_t lle_transpose_chars(lle_editor_t *editor) {
1299 if (!editor || !editor->buffer || !editor->cursor_manager) {
1300 return LLE_ERROR_INVALID_PARAMETER;
1301 }
1302
1303 const char *data = editor->buffer->data;
1304 size_t len = editor->buffer->length;
1305 size_t cursor = editor->buffer->cursor.byte_offset;
1306
1307 /* Need at least 2 graphemes to transpose */
1308 if (editor->buffer->grapheme_count < 2) {
1309 return LLE_SUCCESS;
1310 }
1311
1312 /* Find grapheme boundaries around cursor.
1313 * Transpose swaps the grapheme before cursor with the one at/after cursor.
1314 * If at end of buffer, swap the last two graphemes.
1315 */
1316 size_t g1_start, g1_end, g2_start, g2_end;
1317
1318 if (cursor >= len) {
1319 /* At end of buffer: swap last two graphemes */
1320 g2_end = len;
1321 g2_start = find_prev_grapheme_start(data, len, g2_end);
1322 g1_end = g2_start;
1323 g1_start = find_prev_grapheme_start(data, len, g1_end);
1324 } else if (cursor == 0) {
1325 /* At beginning: nothing before to transpose */
1326 return LLE_SUCCESS;
1327 } else {
1328 /* Normal case: swap grapheme before cursor with grapheme at cursor */
1329 g1_end = cursor;
1330 g1_start = find_prev_grapheme_start(data, len, g1_end);
1331 g2_start = cursor;
1332 g2_end = find_next_grapheme_end(data, len, g2_start);
1333 }
1334
1335 /* Validate boundaries */
1336 if (g1_start >= g1_end || g2_start >= g2_end || g1_end != g2_start) {
1337 return LLE_SUCCESS; /* Invalid state, no-op */
1338 }
1339
1340 size_t g1_len = g1_end - g1_start;
1341 size_t g2_len = g2_end - g2_start;
1342
1343 /* Allocate temp buffers for the two graphemes */
1344 char *g1_copy = malloc(g1_len);
1345 char *g2_copy = malloc(g2_len);
1346 if (!g1_copy || !g2_copy) {
1347 free(g1_copy);
1348 free(g2_copy);
1349 return LLE_ERROR_OUT_OF_MEMORY;
1350 }
1351
1352 /* Copy graphemes */
1353 memcpy(g1_copy, data + g1_start, g1_len);
1354 memcpy(g2_copy, data + g2_start, g2_len);
1355
1356 /* Delete both graphemes (from end to preserve positions) */
1357 lle_buffer_delete_text(editor->buffer, g2_start, g2_len);
1358 lle_buffer_delete_text(editor->buffer, g1_start, g1_len);
1359
1360 /* Insert in swapped order: g2 then g1 */
1361 lle_buffer_insert_text(editor->buffer, g1_start, g2_copy, g2_len);
1362 lle_buffer_insert_text(editor->buffer, g1_start + g2_len, g1_copy, g1_len);
1363
1364 free(g1_copy);
1365 free(g2_copy);
1366
1367 /* Move cursor to end of the swapped region */
1368 size_t new_cursor = g1_start + g1_len + g2_len;
1369 lle_cursor_manager_move_to_byte_offset(editor->cursor_manager, new_cursor);
1370 lle_cursor_manager_get_position(editor->cursor_manager,
1371 &editor->buffer->cursor);
1372
1373 return LLE_SUCCESS;
1374}

1382 const char *data = editor->buffer->data;
1383 size_t len = editor->buffer->length;
1384
1385 if (len == 0) {
1386 return LLE_SUCCESS;
1387 }
1388
1389 /* Find word2: the word at or after cursor using grapheme-aware boundaries
1390 */
1391 size_t word2_start = cursor;
1392 size_t word2_end;
1393
1394 /* If cursor is in the middle of a word, find its boundaries */
1395 /* First, check if we're inside a word by looking at current position */
1396 uint32_t cp_at_cursor = decode_codepoint_at(data, len, cursor);
1397
1398 if (cursor < len && !is_whitespace_codepoint(cp_at_cursor) &&
1399 !is_shell_metachar(cp_at_cursor)) {
1400 /* We're inside a word - find its start */
1401 word2_start = find_word_start(data, cursor);
1402 /* find_word_start returns position at start of word, but we need to
1403 * check if we need to look for next word instead */
1404 }
1405
1406 /* Find end of word2 */
1407 word2_end = find_word_end(data, len, word2_start);
1408
1409 /* If word2 is empty (e.g., cursor was in whitespace), find next word */
1410 if (word2_start >= word2_end) {
1411 word2_start = find_word_end(data, len, cursor); /* Skip whitespace */
1412 if (word2_start >= len) {
1413 return LLE_SUCCESS; /* No word at or after cursor */
1414 }
1415 word2_end = find_word_end(data, len, word2_start);
1416 }
1417
1418 /* Find word1: the word before word2 */
1419 size_t word1_end = word2_start;
1420
1421 /* Skip whitespace/punctuation backward to find end of word1 */
1422 while (word1_end > 0) {
1423 size_t prev = find_prev_grapheme_start(data, len, word1_end);
1424 uint32_t cp = decode_codepoint_at(data, len, prev);
1425 if (!is_whitespace_codepoint(cp) && !is_shell_metachar(cp)) {
1426 /* Found end of word1 */
1427 break;
1428 }
1429 word1_end = prev;
1430 }
1431
1432 if (word1_end == 0) {
1433 return LLE_SUCCESS; /* No previous word */
1434 }
1435
1436 /* Find start of word1 */
1437 size_t word1_start = find_word_start(data, word1_end);
1438
1439 /* Validate we have two distinct words */
1440 if (word1_start >= word1_end || word2_start >= word2_end ||
1441 word1_end > word2_start) {
1442 return LLE_SUCCESS; /* Invalid word boundaries */
1443 }
1444
1445 /* Extract words */
1446 size_t word1_len = word1_end - word1_start;
1447 size_t word2_len = word2_end - word2_start;
1448
1449 char *word1 = strndup(data + word1_start, word1_len);
1450 char *word2 = strndup(data + word2_start, word2_len);
1451
1452 if (!word1 || !word2) {
1453 free(word1);
1454 free(word2);
1455 return LLE_ERROR_OUT_OF_MEMORY;
1456 }
1457
1458 /* Delete word2 first (higher position), then word1 */
1459 lle_buffer_delete_text(editor->buffer, word2_start, word2_len);
1460 lle_buffer_delete_text(editor->buffer, word1_start, word1_len);
1461
1462 /* Calculate separator between the original words */
1463 size_t sep_len = word2_start - word1_end;
1464 char *separator = NULL;
1465 if (sep_len > 0) {
1466 separator = strndup(data + word1_end, sep_len);
1467 }
1468
1469 /* Insert in swapped order: word2, separator, word1 */
1470 lle_buffer_insert_text(editor->buffer, word1_start, word2, word2_len);
1471 if (separator) {
1472 lle_buffer_insert_text(editor->buffer, word1_start + word2_len,
1473 separator, sep_len);
1474 free(separator);
1475 }
1476 lle_buffer_insert_text(editor->buffer, word1_start + word2_len + sep_len,
1477 word1, word1_len);
1478
1479 free(word1);
1480 free(word2);
1481
1482 /* Move cursor to end of swapped region */
1483 size_t new_cursor = word1_start + word1_len + word2_len + sep_len;
1484 lle_cursor_manager_move_to_byte_offset(editor->cursor_manager, new_cursor);
1485 lle_cursor_manager_get_position(editor->cursor_manager,
1486 &editor->buffer->cursor);
1487
1488 return LLE_SUCCESS;
1489}

1516/**
1517 * Helper: Apply case transformation to a word region
1518 * mode: 0 = uppercase all, 1 = lowercase all, 2 = capitalize (first upper, rest
1519 * lower)
1520 */
1521static lle_result_t transform_word_case(lle_editor_t *editor, size_t word_start,
1522 size_t word_end, int mode) {
1523 if (!editor || !editor->buffer || word_start >= word_end) {
1524 return LLE_SUCCESS;
1525 }
1526
1527 const char *data = editor->buffer->data;
1528 size_t len = editor->buffer->length;
1529
1530 /* Build new word with transformed case */
1531 size_t word_len = word_end - word_start;
1532 /* Allocate generous buffer - case conversion can change byte length */
1533 size_t max_new_len = word_len * 4; /* UTF-8 max 4 bytes per codepoint */
1534 char *new_word = malloc(max_new_len + 1);
1535 if (!new_word) {
1536 return LLE_ERROR_OUT_OF_MEMORY;
1537 }
1538
1539 size_t new_pos = 0;
1540 size_t pos = word_start;
1541 bool first_alpha = true; /* For capitalize mode */
1542
1543 while (pos < word_end && pos < len) {
1544 /* Decode codepoint */
1545 uint32_t cp;
1546 int decoded = lle_utf8_decode_codepoint(data + pos, len - pos, &cp);
1547 if (decoded <= 0) {
1548 /* Invalid UTF-8, copy byte as-is */
1549 if (new_pos < max_new_len) {
1550 new_word[new_pos++] = data[pos];
1551 }
1552 pos++;
1553 continue;
1554 }
1555
1556 /* Transform case based on mode */
1557 char transformed[4];
1558 size_t transformed_len = 0;
1559
1560 if (mode == 0) {
1561 /* Uppercase all */
1562 transformed_len =
1563 codepoint_to_upper_utf8(cp, transformed, sizeof(transformed));
1564 } else if (mode == 1) {
1565 /* Lowercase all */
1566 transformed_len =
1567 codepoint_to_lower_utf8(cp, transformed, sizeof(transformed));
1568 } else if (mode == 2) {
1569 /* Capitalize: first alphabetic char uppercase, rest lowercase */
1570 if (first_alpha && iswalpha((wint_t)cp)) {
1571 transformed_len = codepoint_to_upper_utf8(cp, transformed,
1572 sizeof(transformed));
1573 first_alpha = false;
1574 } else {
1575 transformed_len = codepoint_to_lower_utf8(cp, transformed,
1576 sizeof(transformed));
1577 }
1578 }
1579
1580 if (transformed_len == 0) {
1581 /* Fallback: copy original bytes */
1582 if (new_pos + (size_t)decoded <= max_new_len) {
1583 memcpy(new_word + new_pos, data + pos, (size_t)decoded);
1584 new_pos += (size_t)decoded;
1585 }
1586 } else if (new_pos + transformed_len <= max_new_len) {
1587 memcpy(new_word + new_pos, transformed, transformed_len);
1588 new_pos += transformed_len;
1589 }
1590
1591 pos += (size_t)decoded;
1592 }
1593
1594 new_word[new_pos] = '\0';
1595
1596 /* Replace word in buffer */
1597 lle_buffer_delete_text(editor->buffer, word_start, word_len);
1598 lle_buffer_insert_text(editor->buffer, word_start, new_word, new_pos);
1599
1600 free(new_word);
1601
1602 /* Move cursor past transformed word */
1603 size_t new_cursor = word_start + new_pos;
1604 if (editor->cursor_manager) {
1605 lle_cursor_manager_move_to_byte_offset(editor->cursor_manager,
1606 new_cursor);
1607 lle_cursor_manager_get_position(editor->cursor_manager,
1608 &editor->buffer->cursor);
1609 }
1610
1611 return LLE_SUCCESS;
1612}

1820lle_result_t lle_history_previous(lle_editor_t *editor) {
1821 if (!editor || !editor->buffer || !editor->history_system) {
1822 return LLE_ERROR_INVALID_PARAMETER;
1823 }
1824
1825 size_t entry_count = 0;
1826 lle_result_t result =
1827 lle_history_get_entry_count(editor->history_system, &entry_count);
1828 if (result != LLE_SUCCESS || entry_count == 0) {
1829 return LLE_SUCCESS; /* No history */
1830 }
1831
1832 /* Check if navigation-time deduplication is enabled (default: true) */
1833 bool dedup_enabled = config.lle_dedup_navigation;
1834
1835 /* Check if unique-only navigation is enabled (default: true)
1836 * When enabled, each command is shown at most once per navigation session
1837 */
1838 bool unique_only = config.lle_dedup_navigation_unique;
1839
1840 /* Get current buffer content for deduplication comparison */
1841 const char *current_content =
1842 dedup_enabled ? get_current_buffer_content(editor) : NULL;
1843
1844 /* Move backward in history (toward older entries), skipping duplicates if
1845 * enabled */
1846 while (editor->history_navigation_pos < entry_count) {
1847 size_t idx = entry_count - 1 - editor->history_navigation_pos;
1848 lle_history_entry_t *entry = NULL;
1849 result =
1850 lle_history_get_entry_by_index(editor->history_system, idx, &entry);
1851
1852 editor->history_navigation_pos++; /* Always advance position */
1853
1854 if (result == LLE_SUCCESS && entry && entry->command) {
1855 /* Skip if dedup enabled and this entry matches current buffer
1856 * content */
1857 if (dedup_enabled && current_content &&
1858 history_nav_strings_equal(entry->command, current_content)) {
1859 continue; /* Skip duplicate, try next older entry */
1860 }
1861
1862 /* Skip if unique-only mode and we've already seen this command */
1863 if (unique_only) {
1864 uint32_t cmd_hash = hash_command_string(entry->command);
1865 if (history_nav_is_seen(editor, cmd_hash)) {
1866 continue; /* Already shown this command, skip it */
1867 }
1868 /* Mark as seen for future navigation */
1869 history_nav_mark_seen(editor, cmd_hash);
1870 }
1871
1872 /* Found entry to display */
1873 lle_buffer_clear(editor->buffer);
1874 lle_buffer_insert_text(editor->buffer, 0, entry->command,
1875 strlen(entry->command));
1876
1877 /* CRITICAL: Sync cursor_manager after insertion moves cursor to end
1878 */
1879 if (editor->cursor_manager) {
1880 lle_cursor_manager_move_to_byte_offset(
1881 editor->cursor_manager, editor->buffer->cursor.byte_offset);
1882 }
1883 break; /* Found and displayed entry, done */
1884 }
1885 }
1886
1887 return LLE_SUCCESS;
1888}

2046 }

2048 if (!editor->completion_system) {
2049 return LLE_SUCCESS; /* No completion system available */
2050 }

2052 /* If completion is already active, cycle to next item
2053 * This is standard shell behavior: TAB cycles through completions

2055 bool is_active = lle_completion_system_is_active(editor->completion_system);
2056 bool is_menu_visible =
2057 lle_completion_system_is_menu_visible(editor->completion_system);
2058
2059 if (is_active && is_menu_visible) {
2060 /* Completion is active WITH a visible menu - cycle through items */
2061 lle_completion_menu_state_t *menu =
2062 lle_completion_system_get_menu(editor->completion_system);
2063 if (menu) {

2065 lle_completion_menu_move_down(menu);

2068 * update) */
2069 lle_completion_state_t *state =
2070 lle_completion_system_get_state(editor->completion_system);
2071
2072 if (state) {

2077 display_controller_t *dc = display_integration_get_controller();
2078 if (dc) {

2080 }
2081 return LLE_SUCCESS;
2082 }
2083 /* Menu is NULL despite is_menu_visible - fall through to regenerate */
2084 }
2085
2086 /* If is_active but no menu, clear stale state before regenerating */
2087 if (is_active && !is_menu_visible) {
2088 lle_completion_system_clear(editor->completion_system);
2089 }
2090
2091 /* Generate completions for current cursor position */
2092 lle_cursor_position_t cursor_info;
2093 lle_cursor_manager_get_position(editor->cursor_manager, &cursor_info);
2094 size_t cursor_pos = cursor_info.byte_offset;
2095 const char *buffer = editor->buffer->data;
2096
2097 lle_completion_result_t *result = NULL;
2098
2099 /* Use Spec 12 generation (PROPER - with deduplication) */
2100 lle_result_t gen_result = lle_completion_system_generate(
2101 editor->completion_system, buffer, cursor_pos, &result);
2102
2103 if (gen_result != LLE_SUCCESS || !result) {
2104 return LLE_SUCCESS; /* No completions - not an error */
2105 }

2107 /* If no items, clean up and return */
2108 if (result->count == 0) {
2109 lle_completion_result_free(result);
2110 return LLE_SUCCESS;
2111 }
2112
2113 /* Extract word being completed */
2114 lle_completion_context_info_t context;
2115 lle_result_t ctx_result =
2116 lle_completion_analyze_context(buffer, cursor_pos, &context);
2117 if (ctx_result != LLE_SUCCESS) {
2118 lle_completion_result_free(result);
2119 return LLE_SUCCESS;
2120 }
2121
2122 /* If only one completion, insert it directly */
2123 if (result->count == 1) {
2124 const char *completion_text = result->items[0].text;
2125 lle_result_t replace_result = replace_word_at_cursor(
2126 editor, context.word_start, context.word_length, completion_text);
2127 lle_completion_result_free(result);
2128
2129 /* Clear completion system state since we auto-inserted the single
2130 * completion. Without this, the state remains active (is_active=true)
2131 * but with no menu, causing subsequent TAB presses to not regenerate
2132 * completions. */
2133 lle_completion_system_clear(editor->completion_system);
2134
2135 /* Trigger display refresh */
2136 display_controller_t *dc = display_integration_get_controller();
2137 if (dc) {
2138 refresh_after_completion(dc);
2139 }
2140
2141 return replace_result;
2142 }
2143
2144 /* Multiple completions - activate completion system with menu */
2145 /* Completion system stores state internally during generate, just need to
2146 * show menu
2147 */
2148 lle_completion_menu_state_t *menu =
2149 lle_completion_system_get_menu(editor->completion_system);
2150 if (!menu) {
2151 /* No menu despite multiple completions - clear state to avoid stuck
2152 * active flag */
2153 lle_completion_system_clear(editor->completion_system);
2154 lle_completion_result_free(result);
2155 return LLE_SUCCESS;
2156 }
2157
2158 /* Display menu via display_controller (proper architecture) */
2159 display_controller_t *dc = display_integration_get_controller();
2160 if (dc) {
2161 if (menu) {
2162 /* CRITICAL: Update text to show first selected item */
2163 lle_completion_state_t *state =
2164 lle_completion_system_get_state(editor->completion_system);
2165 if (state) {
2166 update_inline_completion(editor, menu, state);
2167 }
2168
2169 display_controller_set_completion_menu(dc, menu);
2170 /* NOTE: Don't call refresh_after_completion() here!
2171 * The caller (execute_keybinding_action) will call
2172 * refresh_display(ctx) which goes through lle_render
2173 * lle_display_bridge command_layer_set_command REDRAW_NEEDED
2174 * event dc_handle_redraw_needed menu rendered
2175 */
2176 }
2177 }
2178
2179 return LLE_SUCCESS;
2180}

2411lle_result_t lle_unix_word_rubout(lle_editor_t *editor) {
2412 if (!editor || !editor->buffer || !editor->cursor_manager) {
2413 return LLE_ERROR_INVALID_PARAMETER;
2414 }
2415
2416 if (editor->buffer->cursor.byte_offset == 0) {
2417 return LLE_SUCCESS; /* Already at beginning */
2418 }
2419
2420 /* Sync cursor_manager with buffer cursor */
2421 lle_cursor_manager_move_to_byte_offset(editor->cursor_manager,
2422 editor->buffer->cursor.byte_offset);
2423
2424 const char *data = editor->buffer->data;
2425 size_t cursor_pos = editor->buffer->cursor.byte_offset;
2426 size_t word_start = cursor_pos;
2427
2428 /* Move backward by graphemes, skipping trailing whitespace */
2429 while (word_start > 0) {
2430 /* Move back one grapheme */
2431 lle_cursor_manager_move_to_byte_offset(editor->cursor_manager,
2432 word_start);
2433 lle_result_t result =
2434 lle_cursor_manager_move_by_graphemes(editor->cursor_manager, -1);
2435 if (result != LLE_SUCCESS) {
2436 break;
2437 }
2438
2439 lle_cursor_position_t pos;
2440 lle_cursor_manager_get_position(editor->cursor_manager, &pos);
2441 size_t prev_pos = pos.byte_offset;
2442
2443 /* Check if this grapheme is whitespace (check first byte) */
2444 if (!is_unix_word_boundary(data[prev_pos])) {
2445 word_start = prev_pos;
2446 break;
2447 }
2448
2449 word_start = prev_pos;
2450 }
2451
2452 /* Now find beginning of word */
2453 while (word_start > 0) {
2454 /* Move back one grapheme */
2455 lle_cursor_manager_move_to_byte_offset(editor->cursor_manager,
2456 word_start);
2457 lle_result_t result =
2458 lle_cursor_manager_move_by_graphemes(editor->cursor_manager, -1);
2459 if (result != LLE_SUCCESS) {
2460 break;
2461 }
2462
2463 lle_cursor_position_t pos;
2464 lle_cursor_manager_get_position(editor->cursor_manager, &pos);
2465 size_t prev_pos = pos.byte_offset;
2466
2467 /* If we hit whitespace, stop */
2468 if (is_unix_word_boundary(data[prev_pos])) {
2469 break;
2470 }
2471
2472 word_start = prev_pos;
2473 }
2474
2475 if (cursor_pos > word_start) {
2476 size_t kill_len = cursor_pos - word_start;
2477 char *killed_text = strndup(data + word_start, kill_len);
2478
2479 if (killed_text) {
2480 /* Add to kill ring */
2481 if (editor->kill_ring) {
2482 lle_kill_ring_add(editor->kill_ring, killed_text, false);
2483 }
2484 free(killed_text);
2485 }
2486
2487 /* Delete the text */
2488 lle_result_t result =
2489 lle_buffer_delete_text(editor->buffer, word_start, kill_len);
2490 if (result == LLE_SUCCESS) {
2491 /* CRITICAL: Sync cursor_manager after deletion */
2492 lle_cursor_manager_move_to_byte_offset(editor->cursor_manager,
2493 word_start);
2494 }
2495 return result;
2496 }
2497
2498 return LLE_SUCCESS;
2499}
