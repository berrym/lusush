28/*
29 * Initialize terminal abstraction system
30 *
31 * This is the main entry point for LLE terminal abstraction. It initializes
32 * all 8 subsystems in the correct order with proper error handling.
33 */
34lle_result_t
35lle_terminal_abstraction_init(lle_terminal_abstraction_t **abstraction,
36 lusush_display_context_t *lusush_display) {
37 if (!abstraction) {
38 return LLE_ERROR_INVALID_PARAMETER;
39 }
40
41 lle_terminal_abstraction_t *abs = NULL;
42 lle_result_t result = LLE_SUCCESS;
43
44 /* Step 1: Allocate main structure */
45 abs = calloc(1, sizeof(lle_terminal_abstraction_t));
46 if (!abs) {
47 return LLE_ERROR_OUT_OF_MEMORY;
48 }
49
50 /* Step 2: Initialize Unix interface for basic terminal access */
51 result = lle_unix_interface_init(&abs->unix_interface);
52 if (result != LLE_SUCCESS) {
53 free(abs);
54 return result;
55 }
56
57 /* Step 3: One-time capability detection (NO terminal queries) */
58 result = lle_capabilities_detect_environment(&abs->capabilities,
59 abs->unix_interface);
60 if (result != LLE_SUCCESS) {
61 lle_unix_interface_destroy(abs->unix_interface);
62 free(abs);
63 return result;
64 }
65
66 /* Step 3.5: Initialize sequence parser now that capabilities are available
67 */
68 result = lle_unix_interface_init_sequence_parser(
69 abs->unix_interface, abs->capabilities,
70 (lle_memory_pool_t *)global_memory_pool);
71 if (result != LLE_SUCCESS) {
72 lle_capabilities_destroy(abs->capabilities);
73 lle_unix_interface_destroy(abs->unix_interface);
74 free(abs);
75 return result;
76 }
77
78 /* Step 4: Initialize internal state authority model */
79 result = lle_internal_state_init(&abs->internal_state, abs->capabilities);
80 if (result != LLE_SUCCESS) {
81 lle_capabilities_destroy(abs->capabilities);
82 lle_unix_interface_destroy(abs->unix_interface);
83 free(abs);
84 return result;
85 }
86
87 /* Step 5: Initialize display content generation system */
88 result = lle_display_generator_init(&abs->display_generator,
89 abs->capabilities, abs->internal_state);
90 if (result != LLE_SUCCESS) {
91 lle_internal_state_destroy(abs->internal_state);
92 lle_capabilities_destroy(abs->capabilities);
93 lle_unix_interface_destroy(abs->unix_interface);
94 free(abs);
95 return result;
96 }
97
98 /* Step 6: Initialize as Lusush display layer client */
99 result = lle_lusush_display_client_init(&abs->display_client,
100 lusush_display, abs->capabilities);
101 if (result != LLE_SUCCESS) {
102 lle_display_generator_destroy(abs->display_generator);
103 lle_internal_state_destroy(abs->internal_state);
104 lle_capabilities_destroy(abs->capabilities);
105 lle_unix_interface_destroy(abs->unix_interface);
106 free(abs);
107 return result;
108 }
109
110 /* Step 7: Initialize input processing system */
111 result = lle_input_processor_init(&abs->input_processor, abs->capabilities,
112 abs->unix_interface);
113 if (result != LLE_SUCCESS) {
114 lle_lusush_display_client_destroy(abs->display_client);
115 lle_display_generator_destroy(abs->display_generator);
116 lle_internal_state_destroy(abs->internal_state);
117 lle_capabilities_destroy(abs->capabilities);
118 lle_unix_interface_destroy(abs->unix_interface);
119 free(abs);
120 return result;
121 }
122
123 /* Step 8: Initialize error handling and performance monitoring */
124 /* NOTE: When Spec 16 Phase 2 and Spec 14 Phase 1 are implemented,
125 * we will call initialization functions here:
126 * - lle_error_context_init(&abs->error_ctx)
127 * - lle_performance_monitor_init(&abs->perf_monitor)
128 *
129 * For now, set to NULL to indicate not yet initialized.
130 */
131 abs->error_ctx = NULL;
132 abs->perf_monitor = NULL;
133
134 *abstraction = abs;
135 return LLE_SUCCESS;
136}

138/*
139 * Destroy terminal abstraction system
140 *
141 * Cleans up all subsystems in reverse order of initialization.
142 */
143void lle_terminal_abstraction_destroy(lle_terminal_abstraction_t *abstraction) {
144 if (!abstraction) {
145 return;
146 }
147
148 /* Destroy in reverse order of initialization */
149
150 /* Step 8: Destroy performance monitoring and error handling */
151 /* NOTE: When Spec 16 Phase 2 and Spec 14 Phase 1 are implemented,
152 * we will call destroy functions here. For now, they are NULL.
153 */
154 abstraction->perf_monitor = NULL;
155 abstraction->error_ctx = NULL;
156
157 /* Step 7: Destroy input processing system */
158 if (abstraction->input_processor) {
159 lle_input_processor_destroy(abstraction->input_processor);
160 }
161
162 /* Step 6: Destroy Lusush display client */
163 if (abstraction->display_client) {
164 lle_lusush_display_client_destroy(abstraction->display_client);
165 }
166
167 /* Step 5: Destroy display generator */
168 if (abstraction->display_generator) {
169 lle_display_generator_destroy(abstraction->display_generator);
170 }
171
172 /* Step 4: Destroy internal state */
173 if (abstraction->internal_state) {
174 lle_internal_state_destroy(abstraction->internal_state);
175 }
176
177 /* Step 3: Destroy capabilities */
178 if (abstraction->capabilities) {
179 lle_capabilities_destroy(abstraction->capabilities);
180 }
181
182 /* Step 2: Destroy Unix interface (ensures terminal restored) */
183 if (abstraction->unix_interface) {
184 lle_unix_interface_destroy(abstraction->unix_interface);
185 }
186
187 /* Step 1: Free main structure */
188 free(abstraction);
189}
