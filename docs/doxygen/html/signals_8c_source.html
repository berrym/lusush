38// SIGINT handler that properly manages shell vs child process behavior
39static void sigint_handler(int signo) {
40 (void)signo; // Suppress unused parameter warning
41
42 if (current_child_pid > 0) {
43 // We have an active child process - send SIGINT to it
44 kill(current_child_pid, SIGINT);
45 } else if (lle_readline_active) {
46 // LLE readline is active - set flag for LLE to handle
47 // LLE will check this flag in its input loop and abort the current line
48 sigint_received_during_readline = 1;
49 // Don't print newline here - LLE will handle display cleanup
50 } else {
51 // No active child process and not in LLE readline (GNU readline mode)
52 // Set the flag so the main loop knows this was SIGINT, not EOF
53 sigint_received_during_readline = 1;
54 // Print newline to move past current input
55 // NOTE: Using write() instead of printf/fflush for async-signal-safety
56 write(STDOUT_FILENO, "\n", 1);
57 }
58}

106// Set a trap for a signal
107int set_trap(int signal, const char *command) {
108 // Remove existing trap for this signal
109 remove_trap(signal);
110
111 if (!command || strlen(command) == 0) {
112 // Empty command means remove trap (already done above)
113 return 0;
114 }
115
116 // Create new trap entry
117 trap_entry_t *new_trap = malloc(sizeof(trap_entry_t));
118 if (!new_trap) {
119 return -1;
120 }
121
122 new_trap->signal = signal;
123 new_trap->command = strdup(command);
124 if (!new_trap->command) {
125 free(new_trap);
126 return -1;
127 }
128
129 // Add to list
130 new_trap->next = trap_list;
131 trap_list = new_trap;
132
133 // Set the signal handler
134 if (signal == SIGINT || signal == SIGTERM || signal == SIGQUIT ||
135 signal == SIGHUP || signal == SIGUSR1 || signal == SIGUSR2) {
136 set_signal_handler(signal, trap_signal_handler);
137 }
138
139 return 0;
140}

142// Remove a trap for a signal
143int remove_trap(int signal) {
144 trap_entry_t *current = trap_list;
145 trap_entry_t *prev = NULL;
146
147 while (current) {
148 if (current->signal == signal) {
149 // Remove from list
150 if (prev) {
151 prev->next = current->next;
152 } else {
153 trap_list = current->next;
154 }
155
156 // Free memory
157 free(current->command);
158 free(current);
159
160 // Reset signal handler to default (but not for EXIT trap which is
161 // special)
162 if (signal != 0) {
163 set_signal_handler(signal, SIG_DFL);
164 }
165
166 return 0;
167 }
168 prev = current;
169 current = current->next;
170 }
171
172 return -1; // Trap not found
173}
