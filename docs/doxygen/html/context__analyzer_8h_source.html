31/**
32 * Context types for completion
33 */
34typedef enum {
35 LLE_CONTEXT_COMMAND, /* Start of command (complete command names) */
36 LLE_CONTEXT_ARGUMENT, /* Command argument (complete files/dirs) */
37 LLE_CONTEXT_VARIABLE, /* Variable expansion ($VAR) */
38 LLE_CONTEXT_REDIRECT, /* After redirect operator (>, <, |) */
39 LLE_CONTEXT_ASSIGNMENT, /* Variable assignment (VAR=) */
40 LLE_CONTEXT_UNKNOWN /* Can't determine context */
41} lle_completion_context_type_t;
42
43/**
44 * Analysis result for current completion context
45 */
46typedef struct lle_context_analyzer {
47 /* Context type */
48 lle_completion_context_type_t type;
49
50 /* Position information */
51 size_t word_start; /* Start of word being completed */
52 size_t word_end; /* End of word being completed */
53 char *partial_word; /* The partial word to complete */
54
55 /* Command context (if applicable) */
56 char *command_name; /* Current command (if in argument position) */
57 int argument_index; /* Which argument (0-based) */
58
59 /* State flags */
60 bool in_quotes; /* Inside quotes? */
61 bool after_redirect; /* After >, <, |, etc. */
62 bool in_assignment; /* In VAR=value ? */
63
64 /* Memory pool reference */
65 lle_memory_pool_t *pool; /* For allocations */
66} lle_context_analyzer_t;
67
68/**
69 * Analyze buffer to determine completion context
70 *
71 * @param buffer Input buffer
72 * @param cursor_pos Cursor position in buffer
73 * @param pool Memory pool for allocations
74 * @param out_context Output context (allocated from pool)
75 * @return LLE_SUCCESS or error code
76 */
77lle_result_t lle_context_analyze(const char *buffer, size_t cursor_pos,
78 lle_memory_pool_t *pool,
79 lle_context_analyzer_t **out_context);
