185/**
186 * Find closing brace, handling nested braces
187 */
188static const char *template_find_closing_brace(const char *p) {
189 int depth = 1;
190 while (*p && depth > 0) {
191 if (*p == '{') {
192 depth++;
193 } else if (*p == '}') {
194 depth--;
195 }
196 if (depth > 0) {
197 p++;
198 }
199 }
200 return (depth == 0) ? p : NULL;
201}

203/**
204 * Parse conditional content: segment.prop:true:false or segment:true:false
205 */
206static lle_template_token_t *parse_conditional(const char *content) {
207 /* Format: ?segment:true:false or ?segment.prop:true:false */
208 char segment[LLE_TEMPLATE_SEGMENT_MAX] = {0};
209 char property[LLE_TEMPLATE_PROPERTY_MAX] = {0};
210 char true_val[LLE_TEMPLATE_LITERAL_MAX] = {0};
211 char false_val[LLE_TEMPLATE_LITERAL_MAX] = {0};
212
213 const char *p = content;
214
215 /* Parse segment name (and optional property) */
216 size_t seg_len = 0;
217 while (*p && *p != ':' && *p != '.' && seg_len < sizeof(segment) - 1) {
218 segment[seg_len++] = *p++;
219 }
220 segment[seg_len] = '\0';
221
222 /* Check for property */
223 if (*p == '.') {
224 p++; /* Skip . */
225 size_t prop_len = 0;
226 while (*p && *p != ':' && prop_len < sizeof(property) - 1) {
227 property[prop_len++] = *p++;
228 }
229 property[prop_len] = '\0';
230 }
231
232 /* Skip to true value */
233 if (*p == ':') {
234 p++;
235 size_t true_len = 0;
236 while (*p && *p != ':' && true_len < sizeof(true_val) - 1) {
237 true_val[true_len++] = *p++;
238 }
239 true_val[true_len] = '\0';
240 }
241
242 /* Skip to false value */
243 if (*p == ':') {
244 p++;
245 size_t false_len = 0;
246 while (*p && false_len < sizeof(false_val) - 1) {
247 false_val[false_len++] = *p++;
248 }
249 false_val[false_len] = '\0';
250 }
251
252 return lle_template_token_conditional(
253 segment, strlen(property) > 0 ? property : NULL, true_val, false_val);
254}

256/**
257 * Parse color application: color:text
258 */
259static lle_template_token_t *parse_color(const char *content) {
260 char color[LLE_TEMPLATE_SEGMENT_MAX] = {0};
261 char text[LLE_TEMPLATE_LITERAL_MAX] = {0};
262
263 const char *colon = strchr(content, ':');
264 if (!colon) {
265 return NULL;
266 }
267
268 size_t color_len = colon - content;
269 if (color_len >= sizeof(color)) {
270 color_len = sizeof(color) - 1;
271 }
272 memcpy(color, content, color_len);
273 color[color_len] = '\0';
274
275 snprintf(text, sizeof(text), "%s", colon + 1);
276
277 return lle_template_token_color(color, text);
278}

280/**
281 * Parse segment or property reference: segment or segment.property
282 */
283static lle_template_token_t *parse_segment_or_property(const char *content) {
284 const char *dot = strchr(content, '.');
285 if (dot) {
286 /* Property access */
287 char segment[LLE_TEMPLATE_SEGMENT_MAX] = {0};
288 char property[LLE_TEMPLATE_PROPERTY_MAX] = {0};
289
290 size_t seg_len = dot - content;
291 if (seg_len >= sizeof(segment)) {
292 seg_len = sizeof(segment) - 1;
293 }
294 memcpy(segment, content, seg_len);
295 segment[seg_len] = '\0';
296
297 snprintf(property, sizeof(property), "%s", dot + 1);
298
299 return lle_template_token_property(segment, property);
300 } else {
301 /* Simple segment */
302 return lle_template_token_segment(content);
303 }
304}

310lle_result_t lle_template_parse(const char *template_str,
311 lle_parsed_template_t **parsed) {
312 if (!template_str || !parsed) {
313 return LLE_ERROR_INVALID_PARAMETER;
314 }
315
316 lle_parsed_template_t *tmpl = calloc(1, sizeof(*tmpl));
317 if (!tmpl) {
318 return LLE_ERROR_OUT_OF_MEMORY;
319 }
320
321 snprintf(tmpl->original, sizeof(tmpl->original), "%s", template_str);
322
323 const char *p = template_str;
324 char literal_buf[LLE_TEMPLATE_LITERAL_MAX] = {0};
325 size_t literal_pos = 0;
326
327 while (*p) {
328 /* Check for escape sequences */
329 if (*p == '\\' && *(p + 1)) {
330 if (*(p + 1) == 'n') {
331 /* Flush literal buffer first */
332 if (literal_pos > 0) {
333 lle_template_token_t *tok =
334 lle_template_token_literal(literal_buf, literal_pos);
335 if (tok) {
336 template_add_token(tmpl, tok);
337 }
338 literal_pos = 0;
339 }
340 template_add_token(tmpl, lle_template_token_newline());
341 p += 2;
342 continue;
343 } else if (*(p + 1) == '$' || *(p + 1) == '\\') {
344 /* Escaped character - add to literal */
345 if (literal_pos < sizeof(literal_buf) - 1) {
346 literal_buf[literal_pos++] = *(p + 1);
347 }
348 p += 2;
349 continue;
350 }
351 }
352
353 /* Check for variable/segment reference ${...} */
354 if (*p == '$' && *(p + 1) == '{') {
355 /* Flush literal buffer */
356 if (literal_pos > 0) {
357 lle_template_token_t *tok =
358 lle_template_token_literal(literal_buf, literal_pos);
359 if (tok) {
360 template_add_token(tmpl, tok);
361 }
362 literal_pos = 0;
363 }
364
365 p += 2; /* Skip ${ */
366
367 /* Find closing brace */
368 const char *end = template_find_closing_brace(p);
369 if (!end) {
370 lle_template_free(tmpl);
371 return LLE_ERROR_INPUT_PARSING;
372 }
373
374 /* Extract content between braces */
375 size_t content_len = end - p;
376 char content[LLE_TEMPLATE_LITERAL_MAX] = {0};
377 if (content_len >= sizeof(content)) {
378 content_len = sizeof(content) - 1;
379 }
380 memcpy(content, p, content_len);
381 content[content_len] = '\0';
382
383 /* Parse the content based on type */
384 lle_template_token_t *token = NULL;
385
386 if (content[0] == '?') {
387 /* Conditional: ${?segment:true:false} */
388 token = parse_conditional(content + 1);
389 } else if (strchr(content, ':') && !strchr(content, '.')) {
390 /* Color: ${color:text} - colon present, no dot */
391 token = parse_color(content);
392 } else {
393 /* Segment or property: ${segment} or ${segment.property} */
394 token = parse_segment_or_property(content);
395 }
396
397 if (token) {
398 template_add_token(tmpl, token);
399 }
400
401 p = end + 1; /* Skip } */
402 continue;
403 }
404
405 /* Regular character */
406 if (literal_pos < sizeof(literal_buf) - 1) {
407 literal_buf[literal_pos++] = *p;
408 }
409 p++;
410 }
411
412 /* Flush remaining literal */
413 if (literal_pos > 0) {
414 lle_template_token_t *tok =
415 lle_template_token_literal(literal_buf, literal_pos);
416 if (tok) {
417 template_add_token(tmpl, tok);
418 }
419 }
420
421 /* Add end token */
422 template_add_token(tmpl, lle_template_token_end());
423
424 tmpl->valid = true;
425 *parsed = tmpl;
426
427 return LLE_SUCCESS;
428}

445bool lle_template_validate(const char *template_str) {
446 if (!template_str) {
447 return false;
448 }
449
450 const char *p = template_str;
451 int brace_depth = 0;
452
453 while (*p) {
454 if (*p == '\\' && *(p + 1)) {
455 /* Skip escaped character */
456 p += 2;
457 continue;
458 }
459
460 if (*p == '$' && *(p + 1) == '{') {
461 brace_depth++;
462 p += 2;
463 continue;
464 }
465
466 if (*p == '{' && brace_depth > 0) {
467 brace_depth++;
468 } else if (*p == '}' && brace_depth > 0) {
469 brace_depth--;
470 }
471
472 p++;
473 }
474
475 return (brace_depth == 0);
476}

482lle_result_t lle_template_render(const lle_parsed_template_t *tmpl,
483 const lle_template_render_ctx_t *render_ctx,
484 char *output, size_t output_size) {
485 if (!tmpl || !render_ctx || !output || output_size == 0) {
486 return LLE_ERROR_INVALID_PARAMETER;
487 }
488
489 if (!tmpl->valid) {
490 return LLE_ERROR_INPUT_PARSING;
491 }
492
493 size_t pos = 0;
494 lle_template_token_t *token = tmpl->head;
495
496 while (token && token->type != LLE_TOKEN_END && pos < output_size - 1) {
497 switch (token->type) {
498 case LLE_TOKEN_LITERAL:
499 /* Copy literal text */
500 {
501 size_t avail = output_size - pos - 1;
502 size_t copy_len = token->data.literal.length;
503 if (copy_len > avail) {
504 copy_len = avail;
505 }
506 memcpy(output + pos, token->data.literal.text, copy_len);
507 pos += copy_len;
508 }
509 break;
510
511 case LLE_TOKEN_NEWLINE:
512 if (pos < output_size - 1) {
513 output[pos++] = '\n';
514 }
515 break;
516
517 case LLE_TOKEN_SEGMENT:
518 /* Render segment */
519 if (render_ctx->get_segment) {
520 char *content =
521 render_ctx->get_segment(token->data.segment.segment_name,
522 NULL, render_ctx->user_data);
523 if (content) {
524 size_t len = strlen(content);
525 size_t avail = output_size - pos - 1;
526 if (len > avail) {
527 len = avail;
528 }
529 memcpy(output + pos, content, len);
530 pos += len;
531 free(content);
532 }
533 }
534 break;
535
536 case LLE_TOKEN_PROPERTY:
537 /* Render segment property */
538 if (render_ctx->get_segment) {
539 char *content = render_ctx->get_segment(
540 token->data.segment.segment_name,
541 token->data.segment.property_name, render_ctx->user_data);
542 if (content) {
543 size_t len = strlen(content);
544 size_t avail = output_size - pos - 1;
545 if (len > avail) {
546 len = avail;
547 }
548 memcpy(output + pos, content, len);
549 pos += len;
550 free(content);
551 }
552 }
553 break;
554
555 case LLE_TOKEN_CONDITIONAL:
556 /* Evaluate conditional */
557 {
558 bool condition_met = false;
559
560 if (render_ctx->is_visible) {
561 const char *prop =
562 token->data.conditional.check_property
563 ? token->data.conditional.condition_property
564 : NULL;
565 condition_met = render_ctx->is_visible(
566 token->data.conditional.condition_segment, prop,
567 render_ctx->user_data);
568 }
569
570 const char *branch = condition_met
571 ? token->data.conditional.true_value
572 : token->data.conditional.false_value;
573
574 if (branch && strlen(branch) > 0) {
575 /* Recursively evaluate the branch content to expand
576 * nested segment references like ${git} inside
577 * conditionals like ${?git: (${git})} */
578 size_t avail = output_size - pos - 1;
579 char *temp = malloc(avail + 1);
580 if (temp) {
581 lle_result_t eval_result = lle_template_evaluate(
582 branch, render_ctx, temp, avail + 1);
583 if (eval_result == LLE_SUCCESS) {
584 size_t len = strlen(temp);
585 memcpy(output + pos, temp, len);
586 pos += len;
587 } else {
588 /* Fallback: copy branch literally */
589 size_t len = strlen(branch);
590 if (len > avail) {
591 len = avail;
592 }
593 memcpy(output + pos, branch, len);
594 pos += len;
595 }
596 free(temp);
597 }
598 }
599 }
600 break;
601
602 case LLE_TOKEN_COLOR:
603 /* Apply color to text */
604 {
605 const char *color_code = "";
606 if (render_ctx->get_color) {
607 color_code = render_ctx->get_color(
608 token->data.color.color_name, render_ctx->user_data);
609 if (!color_code) {
610 color_code = "";
611 }
612 }
613
614 /* Add color code */
615 size_t color_len = strlen(color_code);
616 if (color_len > 0 && pos + color_len < output_size - 1) {
617 memcpy(output + pos, color_code, color_len);
618 pos += color_len;
619 }
620
621 /* Add text */
622 size_t text_len = strlen(token->data.color.text);
623 size_t avail = output_size - pos - 1;
624 if (text_len > avail) {
625 text_len = avail;
626 }
627 memcpy(output + pos, token->data.color.text, text_len);
628 pos += text_len;
629
630 /* Add reset if color was applied */
631 if (color_len > 0) {
632 const char *reset = "\033[0m";
633 size_t reset_len = strlen(reset);
634 if (pos + reset_len < output_size - 1) {
635 memcpy(output + pos, reset, reset_len);
636 pos += reset_len;
637 }
638 }
639 }
640 break;
641
642 default:
643 break;
644 }
645
646 token = token->next;
647 }
648
649 output[pos] = '\0';
650 return LLE_SUCCESS;
651}

653lle_result_t lle_template_evaluate(const char *template_str,
654 const lle_template_render_ctx_t *render_ctx,
655 char *output, size_t output_size) {
656 if (!template_str || !render_ctx || !output || output_size == 0) {
657 return LLE_ERROR_INVALID_PARAMETER;
658 }
659
660 lle_parsed_template_t *parsed = NULL;
661 lle_result_t result = lle_template_parse(template_str, &parsed);
662 if (result != LLE_SUCCESS) {
663 return result;
664 }
665
666 result = lle_template_render(parsed, render_ctx, output, output_size);
667
668 lle_template_free(parsed);
669 return result;
670}
