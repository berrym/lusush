22/**
23 * Get current time in microseconds for performance tracking
24 */
25static uint64_t get_time_microseconds(void) {
26 struct timespec ts;
27 if (clock_gettime(CLOCK_MONOTONIC, &ts) != 0) {
28 return 0;
29 }
30 return (uint64_t)ts.tv_sec * 1000000 + (uint64_t)ts.tv_nsec / 1000;
31}

36static char *pool_strdup(lle_memory_pool_t *pool, const char *str) {
37 (void)pool; /* Reserved for pool-based allocation */
38 if (!str) {
39 return NULL;
40 }
41
42 size_t len = strlen(str) + 1;
43 char *dup = lle_pool_alloc(len);
44 if (!dup) {
45 return NULL;
46 }
47
48 memcpy(dup, str, len);
49 return dup;
50}

57lle_result_t lle_widget_registry_init(lle_widget_registry_t **registry,
58 lle_memory_pool_t *memory_pool) {
59 if (!registry || !memory_pool) {
60 return LLE_ERROR_INVALID_PARAMETER;
61 }
62
63 /* Allocate registry structure */
64 lle_widget_registry_t *reg = lle_pool_alloc(sizeof(lle_widget_registry_t));
65 if (!reg) {
66 return LLE_ERROR_OUT_OF_MEMORY;
67 }
68
69 memset(reg, 0, sizeof(lle_widget_registry_t));
70
71 /* Initialize hash table (16 buckets initially, will grow as needed) */
72 reg->widgets = ht_create(fnv1a_hash_str, str_eq, NULL, 16);
73 if (!reg->widgets) {
74 lle_pool_free(reg);
75 return LLE_ERROR_OUT_OF_MEMORY;
76 }
77
78 /* Initialize state */
79 reg->widget_list = NULL;
80 reg->widget_count = 0;
81 reg->memory_pool = memory_pool;
82 reg->registry_active = true;
83
84 *registry = reg;
85 return LLE_SUCCESS;
86}

88lle_result_t lle_widget_registry_destroy(lle_widget_registry_t *registry) {
89 if (!registry) {
90 return LLE_ERROR_INVALID_PARAMETER;
91 }
92
93 /* Mark as inactive to prevent concurrent access */
94 registry->registry_active = false;
95
96 /* Free all widgets in linked list */
97 lle_widget_t *widget = registry->widget_list;
98 while (widget) {
99 lle_widget_t *next = widget->next;
100
101 /* Free widget name */
102 if (widget->name) {
103 lle_pool_free(widget->name);
104 }
105
106 /* Free widget structure */
107 lle_pool_free(widget);
108
109 widget = next;
110 }
111
112 /* Destroy hash table */
113 if (registry->widgets) {
114 ht_destroy(registry->widgets);
115 }
116
117 /* Free registry structure */
118 lle_pool_free(registry);
119
120 return LLE_SUCCESS;
121}

128lle_result_t lle_widget_register(lle_widget_registry_t *registry,
129 const char *name,
130 lle_widget_callback_t callback,
131 lle_widget_type_t type, void *user_data) {
132 if (!registry || !name || !callback) {
133 return LLE_ERROR_INVALID_PARAMETER;
134 }
135
136 if (!registry->registry_active) {
137 return LLE_ERROR_INVALID_STATE;
138 }
139
140 /* Check for duplicate name */
141 void *existing = ht_get(registry->widgets, name);
142 if (existing != NULL) {
143 return LLE_ERROR_ALREADY_EXISTS;
144 }
145
146 /* Allocate widget structure */
147 lle_widget_t *widget = lle_pool_alloc(sizeof(lle_widget_t));
148 if (!widget) {
149 return LLE_ERROR_OUT_OF_MEMORY;
150 }
151
152 memset(widget, 0, sizeof(lle_widget_t));
153
154 /* Copy widget name */
155 widget->name = pool_strdup(registry->memory_pool, name);
156 if (!widget->name) {
157 lle_pool_free(widget);
158 return LLE_ERROR_OUT_OF_MEMORY;
159 }
160
161 /* Initialize widget fields */
162 widget->callback = callback;
163 widget->type = type;
164 widget->user_data = user_data;
165 widget->execution_count = 0;
166 widget->total_execution_time_us = 0;
167 widget->enabled = true;
168 widget->next = NULL;
169
170 /* Insert into hash table */
171 ht_insert(registry->widgets, widget->name, widget);
172
173 /* Add to linked list at head */
174 widget->next = registry->widget_list;
175 registry->widget_list = widget;
176 registry->widget_count++;
177
178 return LLE_SUCCESS;
179}

181lle_result_t lle_widget_unregister(lle_widget_registry_t *registry,
182 const char *name) {
183 if (!registry || !name) {
184 return LLE_ERROR_INVALID_PARAMETER;
185 }
186
187 if (!registry->registry_active) {
188 return LLE_ERROR_INVALID_STATE;
189 }
190
191 /* Lookup widget in hash table */
192 lle_widget_t *widget = (lle_widget_t *)ht_get(registry->widgets, name);
193 if (!widget) {
194 return LLE_ERROR_NOT_FOUND;
195 }
196
197 /* Remove from hash table */
198 ht_remove(registry->widgets, name);
199
200 /* Remove from linked list */
201 if (registry->widget_list == widget) {
202 /* Widget is at head of list */
203 registry->widget_list = widget->next;
204 } else {
205 /* Find previous widget in list */
206 lle_widget_t *prev = registry->widget_list;
207 while (prev && prev->next != widget) {
208 prev = prev->next;
209 }
210
211 if (prev) {
212 prev->next = widget->next;
213 }
214 }
215
216 registry->widget_count--;
217
218 /* Free widget resources */
219 if (widget->name) {
220 lle_pool_free(widget->name);
221 }
222 lle_pool_free(widget);
223
224 return LLE_SUCCESS;
225}

256lle_result_t lle_widget_execute_direct(lle_widget_t *widget,
257 lle_editor_t *editor) {
258 if (!widget || !editor) {
259 return LLE_ERROR_INVALID_PARAMETER;
260 }
261
262 /* Check if widget is enabled */
263 if (!widget->enabled) {
264 return LLE_ERROR_DISABLED;
265 }
266
267 /* Verify callback exists */
268 if (!widget->callback) {
269 return LLE_ERROR_INVALID_STATE;
270 }
271
272 /* Track execution time */
273 uint64_t start_time = get_time_microseconds();
274
275 /* Execute widget callback */
276 lle_result_t result = widget->callback(editor, widget->user_data);
277
278 /* Update statistics */
279 uint64_t end_time = get_time_microseconds();
280 uint64_t execution_time = end_time - start_time;
281
282 widget->execution_count++;
283 widget->total_execution_time_us += execution_time;
284
285 return result;
286}
