118static char *lle_strdup_pool(const char *str) {
119 if (!str) {
120 return NULL;
121 }
122
123 size_t len = strlen(str) + 1;
124 char *copy = (char *)lle_pool_alloc(len);
125 if (!copy) {
126 return NULL;
127 }
128
129 memcpy(copy, str, len);
130 return copy;
131}

137lle_result_t lle_completion_item_create(lle_memory_pool_t *memory_pool,
138 const char *text, const char *suffix,

146lle_result_t lle_completion_item_create_with_description(
147 lle_memory_pool_t *memory_pool, const char *text, const char *suffix,
148 lle_completion_type_t type, int32_t relevance_score,
149 const char *description, lle_completion_item_t **item) {
150 if (!memory_pool || !text || !item) {
151 return LLE_ERROR_INVALID_PARAMETER;
152 }
153
154 // Allocate item structure
155 lle_completion_item_t *new_item =
156 (lle_completion_item_t *)lle_pool_alloc(sizeof(lle_completion_item_t));
157 if (!new_item) {
158 return LLE_ERROR_OUT_OF_MEMORY;
159 }
160
161 // Duplicate text
162 new_item->text = lle_strdup_pool(text);
163 if (!new_item->text) {
164 lle_pool_free(new_item);
165 return LLE_ERROR_OUT_OF_MEMORY;
166 }
167 new_item->owns_text = true;
168
169 // Duplicate suffix if provided
170 if (suffix) {
171 new_item->suffix = lle_strdup_pool(suffix);
172 if (!new_item->suffix) {
173 lle_pool_free(new_item->text);
174 lle_pool_free(new_item);
175 return LLE_ERROR_OUT_OF_MEMORY;
176 }
177 new_item->owns_suffix = true;
178 } else {
179 new_item->suffix = NULL;
180 new_item->owns_suffix = false;
181 }
182
183 // Set type and get indicator
184 new_item->type = type;
185 new_item->type_indicator = lle_completion_type_get_indicator(type);
186
187 // Set relevance score (clamp to valid range)
188 if (relevance_score < 0)
189 relevance_score = 0;
190 if (relevance_score > 1000)
191 relevance_score = 1000;
192 new_item->relevance_score = relevance_score;
193
194 // Duplicate description if provided
195 if (description) {
196 new_item->description = lle_strdup_pool(description);
197 new_item->owns_description = (new_item->description != NULL);
198 } else {
199 new_item->description = NULL;
200 new_item->owns_description = false;
201 }
202
203 *item = new_item;
204 return LLE_SUCCESS;
205}

207lle_result_t lle_completion_item_free(lle_memory_pool_t *memory_pool,
208 lle_completion_item_t *item) {
209 if (!memory_pool || !item) {
210 return LLE_ERROR_INVALID_PARAMETER;
211 }
212
213 if (item->owns_text && item->text) {
214 lle_pool_free(item->text);
215 }
216
217 if (item->owns_suffix && item->suffix) {
218 lle_pool_free(item->suffix);
219 }
220
221 if (item->owns_description && item->description) {
222 lle_pool_free(item->description);
223 }
224
225 lle_pool_free(item);
226 return LLE_SUCCESS;
227}

233lle_result_t lle_completion_result_create(lle_memory_pool_t *memory_pool,
234 size_t initial_capacity,
235 lle_completion_result_t **result) {
236 if (!memory_pool || !result) {
237 return LLE_ERROR_INVALID_PARAMETER;
238 }
239
240 if (initial_capacity == 0) {
241 initial_capacity = 16; // Default capacity
242 }
243
244 // Allocate result structure
245 lle_completion_result_t *new_result =
246 (lle_completion_result_t *)lle_pool_alloc(
247 sizeof(lle_completion_result_t));
248 if (!new_result) {
249 return LLE_ERROR_OUT_OF_MEMORY;
250 }
251
252 // Allocate items array
253 new_result->items = (lle_completion_item_t *)lle_pool_alloc(
254 sizeof(lle_completion_item_t) * initial_capacity);
255 if (!new_result->items) {
256 lle_pool_free(new_result);
257 return LLE_ERROR_OUT_OF_MEMORY;
258 }
259
260 new_result->count = 0;
261 new_result->capacity = initial_capacity;
262 new_result->memory_pool = memory_pool;
263
264 // Initialize category counts
265 new_result->builtin_count = 0;
266 new_result->command_count = 0;
267 new_result->file_count = 0;
268 new_result->directory_count = 0;
269 new_result->variable_count = 0;
270 new_result->alias_count = 0;
271 new_result->history_count = 0;
272
273 *result = new_result;
274 return LLE_SUCCESS;
275}

277lle_result_t lle_completion_result_add_item(lle_completion_result_t *result,
278 lle_completion_item_t *item) {
279 if (!result || !item) {
280 return LLE_ERROR_INVALID_PARAMETER;
281 }
282
283 // Grow array if needed
284 if (result->count >= result->capacity) {
285 size_t new_capacity = result->capacity * 2;
286 lle_completion_item_t *new_items =
287 (lle_completion_item_t *)lle_pool_alloc(
288 sizeof(lle_completion_item_t) * new_capacity);
289 if (!new_items) {
290 return LLE_ERROR_OUT_OF_MEMORY;
291 }
292
293 // Copy existing items
294 memcpy(new_items, result->items,
295 sizeof(lle_completion_item_t) * result->count);
296
297 // Free old array
298 lle_pool_free(result->items);
299
300 result->items = new_items;
301 result->capacity = new_capacity;
302 }
303
304 // Add item (transfer ownership)
305 result->items[result->count] = *item;
306 result->count++;
307
308 // Update category count
309 switch (item->type) {
310 case LLE_COMPLETION_TYPE_BUILTIN:
311 result->builtin_count++;
312 break;
313 case LLE_COMPLETION_TYPE_COMMAND:
314 result->command_count++;
315 break;
316 case LLE_COMPLETION_TYPE_FILE:
317 result->file_count++;
318 break;
319 case LLE_COMPLETION_TYPE_DIRECTORY:
320 result->directory_count++;
321 break;
322 case LLE_COMPLETION_TYPE_VARIABLE:
323 result->variable_count++;
324 break;
325 case LLE_COMPLETION_TYPE_ALIAS:
326 result->alias_count++;
327 break;
328 case LLE_COMPLETION_TYPE_HISTORY:
329 result->history_count++;
330 break;
331 case LLE_COMPLETION_TYPE_CUSTOM:
332 result->custom_count++;
333 break;
334 default:
335 break;
336 }
337
338 // Free the item wrapper (contents now owned by result)
339 lle_pool_free(item);
340
341 return LLE_SUCCESS;
342}

344lle_result_t lle_completion_result_add(lle_completion_result_t *result,
345 const char *text, const char *suffix,
346 lle_completion_type_t type,
347 int32_t relevance_score) {
348 if (!result) {
349 return LLE_ERROR_INVALID_PARAMETER;
350 }
351
352 lle_completion_item_t *item = NULL;
353 lle_result_t res = lle_completion_item_create(
354 result->memory_pool, text, suffix, type, relevance_score, &item);
355
356 if (res != LLE_SUCCESS) {
357 return res;
358 }
359
360 return lle_completion_result_add_item(result, item);
361}

363// Comparison function for sorting
364static int compare_completion_items(const void *a, const void *b) {
365 const lle_completion_item_t *item_a = (const lle_completion_item_t *)a;
366 const lle_completion_item_t *item_b = (const lle_completion_item_t *)b;
367
368 // First sort by type (category)
369 if (item_a->type != item_b->type) {
370 return item_a->type - item_b->type;
371 }
372
373 // Within same type, sort by relevance score (descending)
374 if (item_a->relevance_score != item_b->relevance_score) {
375 return item_b->relevance_score - item_a->relevance_score;
376 }
377
378 // Finally, alphabetical order
379 return strcmp(item_a->text, item_b->text);
380}

382lle_result_t lle_completion_result_sort(lle_completion_result_t *result) {
383 if (!result) {
384 return LLE_ERROR_INVALID_PARAMETER;
385 }
386
387 if (result->count <= 1) {
388 return LLE_SUCCESS;
389 }
390
391 qsort(result->items, result->count, sizeof(lle_completion_item_t),
392 compare_completion_items);
393
394 return LLE_SUCCESS;
395}

397lle_result_t lle_completion_result_free(lle_completion_result_t *result) {
398 if (!result) {
399 return LLE_ERROR_INVALID_PARAMETER;
400 }
401
402 // Free all items
403 for (size_t i = 0; i < result->count; i++) {
404 lle_completion_item_t *item = &result->items[i];
405 if (item->owns_text && item->text) {
406 lle_pool_free(item->text);
407 }
408 if (item->owns_suffix && item->suffix) {
409 lle_pool_free(item->suffix);
410 }
411 if (item->owns_description && item->description) {
412 lle_pool_free(item->description);
413 }
414 }
415
416 lle_pool_free(result->items);
417 lle_pool_free(result);
418
419 return LLE_SUCCESS;
420}

459lle_completion_type_t lle_completion_classify_text(const char *text,
460 bool is_command_position) {
461 if (!text) {
462 return LLE_COMPLETION_TYPE_UNKNOWN;
463 }
464
465 // Check for variable
466 if (text[0] == '$') {
467 return LLE_COMPLETION_TYPE_VARIABLE;
468 }
469
470 // If not at command position, likely a file/directory
471 if (!is_command_position) {
472 if (lle_completion_is_directory(text)) {
473 return LLE_COMPLETION_TYPE_DIRECTORY;
474 }
475 return LLE_COMPLETION_TYPE_FILE;
476 }
477
478 // At command position - classify command type
479 if (lle_completion_is_alias(text)) {
480 return LLE_COMPLETION_TYPE_ALIAS;
481 }
482
483 if (lle_completion_is_builtin(text)) {
484 return LLE_COMPLETION_TYPE_BUILTIN;
485 }
486
487 // Check if it's a path
488 if (strchr(text, '/') != NULL) {
489 if (lle_completion_is_directory(text)) {
490 return LLE_COMPLETION_TYPE_DIRECTORY;
491 }
492 return LLE_COMPLETION_TYPE_FILE;
493 }
494
495 // Assume it's a command from PATH
496 return LLE_COMPLETION_TYPE_COMMAND;
497}
