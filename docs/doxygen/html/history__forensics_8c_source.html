46/**
47 * Get terminal device name
48 *
49 * @param buffer Output buffer for terminal name
50 * @param size Buffer size
51 * @return true on success, false on failure
52 */
53static bool get_terminal_name(char *buffer, size_t size) {
54 if (!buffer || size == 0) {
55 return false;
56 }
57
58 /* Try ttyname() first */
59 const char *tty = ttyname(STDIN_FILENO);
60 if (tty) {
61 strncpy(buffer, tty, size - 1);
62 buffer[size - 1] = '\0';
63 return true;
64 }
65
66 /* Fallback: check if connected to terminal */
67 if (isatty(STDIN_FILENO)) {
68 strncpy(buffer, "unknown_tty", size - 1);
69 buffer[size - 1] = '\0';
70 return true;
71 }
72
73 /* Not a terminal */
74 strncpy(buffer, "not_a_tty", size - 1);
75 buffer[size - 1] = '\0';
76 return false;
77}

99/**
100 * Get high-precision timestamp in nanoseconds
101 *
102 * Uses CLOCK_MONOTONIC for consistent timing measurements.
103 *
104 * @return Timestamp in nanoseconds since arbitrary epoch
105 */
106uint64_t lle_forensic_get_timestamp_ns(void) {
107 struct timespec ts;
108
109 if (clock_gettime(CLOCK_MONOTONIC, &ts) != 0) {
110 /* Fallback to lower precision */
111 return (uint64_t)time(NULL) * 1000000000ULL;
112 }
113
114 return (uint64_t)ts.tv_sec * 1000000000ULL + (uint64_t)ts.tv_nsec;
115}

117/**
118 * Capture current forensic context
119 *
120 * Captures process, user, terminal, and timing information for forensic
121 * tracking.
122 *
123 * @param context Output structure for forensic data
124 * @return LLE_SUCCESS or error code
125 */
126lle_result_t lle_forensic_capture_context(lle_forensic_context_t *context) {
127 if (!context) {
128 return LLE_ERROR_INVALID_PARAMETER;
129 }
130
131 /* Initialize structure */
132 memset(context, 0, sizeof(lle_forensic_context_t));
133
134 /* Capture process information */
135 context->process_id = getpid();
136 context->session_id = getsid(0);
137 context->user_id = getuid();
138 context->group_id = getgid();
139
140 /* Capture terminal name */
141 char tty_buffer[256];
142 if (get_terminal_name(tty_buffer, sizeof(tty_buffer))) {
143 context->terminal_name = pool_strdup(tty_buffer);
144 } else {
145 context->terminal_name = NULL;
146 }
147
148 /* Capture working directory */
149 char cwd_buffer[4096];
150 if (getcwd(cwd_buffer, sizeof(cwd_buffer)) != NULL) {
151 context->working_directory = pool_strdup(cwd_buffer);
152 } else {
153 context->working_directory = NULL;
154 }
155
156 /* Capture high-precision timestamp */
157 context->timestamp_ns = lle_forensic_get_timestamp_ns();
158
159 return LLE_SUCCESS;
160}

162/**
163 * Apply forensic context to history entry
164 *
165 * Populates forensic metadata fields in a history entry from context.
166 *
167 * @param entry History entry to populate
168 * @param context Forensic context to apply
169 * @return LLE_SUCCESS or error code
170 */
171lle_result_t
172lle_forensic_apply_to_entry(lle_history_entry_t *entry,
173 const lle_forensic_context_t *context) {
174 if (!entry || !context) {
175 return LLE_ERROR_INVALID_PARAMETER;
176 }
177
178 /* Apply process information */
179 entry->process_id = context->process_id;
180 entry->session_id = context->session_id;
181 entry->user_id = context->user_id;
182 entry->group_id = context->group_id;
183
184 /* Apply terminal name */
185 if (context->terminal_name) {
186 entry->terminal_name = pool_strdup(context->terminal_name);
187 }
188
189 /* Apply working directory (if not already set) */
190 if (!entry->working_directory && context->working_directory) {
191 entry->working_directory = pool_strdup(context->working_directory);
192 }
193
194 /* Initialize timing fields */
195 entry->start_time_ns = context->timestamp_ns;
196 entry->end_time_ns = 0; /* Will be set by mark_end */
197 entry->duration_ms = 0;
198
199 /* Initialize usage tracking */
200 entry->usage_count = 0;
201 entry->last_access_time =
202 context->timestamp_ns / 1000000000ULL; /* Convert to seconds */
203
204 return LLE_SUCCESS;
205}

255/**
256 * Record command execution end time and calculate duration
257 *
258 * Captures end time and calculates execution duration in milliseconds.
259 *
260 * @param entry History entry
261 * @return LLE_SUCCESS or error code
262 */
263lle_result_t lle_forensic_mark_end(lle_history_entry_t *entry) {
264 if (!entry) {
265 return LLE_ERROR_INVALID_PARAMETER;
266 }
267
268 /* Capture end time */
269 entry->end_time_ns = lle_forensic_get_timestamp_ns();
270
271 /* Calculate duration in milliseconds */
272 if (entry->end_time_ns > entry->start_time_ns) {
273 uint64_t duration_ns = entry->end_time_ns - entry->start_time_ns;
274 entry->duration_ms = (uint32_t)(duration_ns / 1000000ULL);
275 } else {
276 /* Clock issue or zero duration */
277 entry->duration_ms = 0;
278 }
279
280 return LLE_SUCCESS;
281}
