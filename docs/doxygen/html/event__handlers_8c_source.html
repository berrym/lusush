18/*
19 * Register event handler
20 */
21lle_result_t lle_event_handler_register(lle_event_system_t *system,
22 lle_event_kind_t type,
23 lle_event_handler_fn handler,
24 void *user_data, const char *name) {
25 if (!system || !handler || !name) {
26 return LLE_ERROR_INVALID_PARAMETER;
27 }
28
29 pthread_mutex_lock(&system->system_mutex);
30
31 /* Check if we need to grow handler array */
32 if (system->handler_count >= system->handler_capacity) {
33 size_t new_capacity = system->handler_capacity * 2;
34 lle_event_handler_t **new_handlers =
35 lle_pool_alloc(sizeof(lle_event_handler_t *) * new_capacity);
36
37 if (!new_handlers) {
38 pthread_mutex_unlock(&system->system_mutex);
39 return LLE_ERROR_OUT_OF_MEMORY;
40 }
41
42 /* Copy existing handlers */
43 memcpy(new_handlers, system->handlers,
44 sizeof(lle_event_handler_t *) * system->handler_count);
45
46 /* Free old array */
47 lle_pool_free(system->handlers);
48
49 system->handlers = new_handlers;
50 system->handler_capacity = new_capacity;
51 }
52
53 /* Allocate new handler */
54 lle_event_handler_t *h = lle_pool_alloc(sizeof(lle_event_handler_t));
55 if (!h) {
56 pthread_mutex_unlock(&system->system_mutex);
57 return LLE_ERROR_OUT_OF_MEMORY;
58 }
59
60 /* Initialize handler */
61 h->event_type = type;
62 h->handler = handler;
63 h->user_data = user_data;
64 strncpy(h->name, name, sizeof(h->name) - 1);
65 h->name[sizeof(h->name) - 1] = '\0';
66
67 /* Add to array */
68 system->handlers[system->handler_count++] = h;
69
70 pthread_mutex_unlock(&system->system_mutex);
71
72 return LLE_SUCCESS;
73}

75/*
76 * Unregister specific handler
77 */
78lle_result_t lle_event_handler_unregister(lle_event_system_t *system,
79 lle_event_kind_t type,
80 const char *name) {
81 if (!system || !name) {
82 return LLE_ERROR_INVALID_PARAMETER;
83 }
84
85 pthread_mutex_lock(&system->system_mutex);
86
87 /* Find and remove handler */
88 for (size_t i = 0; i < system->handler_count; i++) {
89 lle_event_handler_t *h = system->handlers[i];
90
91 if (h->event_type == type && strcmp(h->name, name) == 0) {
92 /* Free handler */
93 lle_pool_free(h);
94
95 /* Shift remaining handlers down */
96 for (size_t j = i; j < system->handler_count - 1; j++) {
97 system->handlers[j] = system->handlers[j + 1];
98 }
99
100 system->handler_count--;
101
102 pthread_mutex_unlock(&system->system_mutex);
103 return LLE_SUCCESS;
104 }
105 }
106
107 pthread_mutex_unlock(&system->system_mutex);
108 return LLE_ERROR_NOT_FOUND;
109}

172/*
173 * Dispatch event to all registered handlers (Phase 1 + Phase 2C with filtering
174 * and hooks)
175 */
176lle_result_t lle_event_dispatch(lle_event_system_t *system,
177 lle_event_t *event) {
178 if (!system || !event) {
179 return LLE_ERROR_INVALID_PARAMETER;
180 }
181
182 /* Phase 2C: Apply event filters */
183 if (system->filter_system) {
184 lle_filter_result_t filter_result =
185 lle_event_filter_apply(system, event);
186 if (filter_result == LLE_FILTER_BLOCK) {
187 return LLE_SUCCESS; /* Event blocked by filter, but not an error */
188 }
189 /* FILTER_PASS, FILTER_TRANSFORM, or FILTER_ERROR: continue dispatch */
190 }
191
192 /* Phase 2C: Call pre-dispatch hook */
193 if (system->pre_dispatch_hook) {
194 lle_result_t hook_result =
195 system->pre_dispatch_hook(event, system->pre_dispatch_data);
196 if (hook_result != LLE_SUCCESS) {
197 return hook_result; /* Hook rejected event */
198 }
199 }
200
201 /* Phase 2C: Update system state to PROCESSING */
202 lle_system_state_t previous_state = system->current_state;
203 system->current_state = LLE_STATE_PROCESSING;
204
205 pthread_mutex_lock(&system->system_mutex);
206
207 /* Find all handlers for this event type */
208 lle_result_t dispatch_result = LLE_SUCCESS;
209 size_t dispatched = 0;
210
211 for (size_t i = 0; i < system->handler_count; i++) {
212 lle_event_handler_t *h = system->handlers[i];
213
214 if (h->event_type == event->type) {
215 /* Call handler */
216 lle_result_t result = h->handler(event, h->user_data);
217
218 /* Track last error (but continue with other handlers) */
219 if (result != LLE_SUCCESS) {
220 dispatch_result = result;
221 }
222
223 dispatched++;
224 }
225 }
226
227 pthread_mutex_unlock(&system->system_mutex);
228
229 /* Update statistics */
230 if (dispatched > 0) {
231 __atomic_fetch_add(&system->events_dispatched, 1, __ATOMIC_SEQ_CST);
232 }
233
234 /* Phase 2C: Restore previous system state */
235 system->current_state = previous_state;
236
237 /* Phase 2C: Call post-dispatch hook */
238 if (system->post_dispatch_hook) {
239 system->post_dispatch_hook(event, dispatch_result,
240 system->post_dispatch_data);
241 }
242
243 return dispatch_result;
244}

246/*
247 * Process events from queue (up to max_events)
248 */
249lle_result_t lle_event_process_queue(lle_event_system_t *system,
250 uint32_t max_events) {
251 if (!system) {
252 return LLE_ERROR_INVALID_PARAMETER;
253 }
254
255 if (!system->active) {
256 return LLE_ERROR_NOT_INITIALIZED;
257 }
258
259 uint32_t processed = 0;
260 lle_result_t last_result = LLE_SUCCESS;
261
262 while (processed < max_events) {
263 /* Dequeue event */
264 lle_event_t *event = NULL;
265 lle_result_t result = lle_event_dequeue(system, &event);
266
267 if (result == LLE_ERROR_QUEUE_EMPTY) {
268 /* No more events */
269 break;
270 }
271
272 if (result != LLE_SUCCESS) {
273 last_result = result;
274 break;
275 }
276
277 /* Dispatch event */
278 result = lle_event_dispatch(system, event);
279 if (result != LLE_SUCCESS) {
280 last_result = result;
281 }
282
283 /* Destroy event */
284 lle_event_destroy(system, event);
285
286 processed++;
287 }
288
289 return last_result;
290}
