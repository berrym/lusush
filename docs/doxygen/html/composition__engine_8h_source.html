16 * - High-performance composition with intelligent caching
17 * - Memory-safe composition management
18 * - Event-driven coordination between layers
19 *
20 * Design Principles:
21 * - No modification of existing layer outputs

24 * - Performance-optimized composition algorithms
25 * - Memory-safe with proper resource management

596/**
597 * Composition result with cursor position tracking.
598 *
599 * This structure extends the basic composition output with cursor position

601 * This is required for LLE to get terminal-ready output with correct cursor

603 */
604typedef struct {

606 composed_output[COMPOSITION_ENGINE_MAX_OUTPUT_SIZE]; // Composed content
607 size_t cursor_screen_row; // Cursor row (0-based)
608 size_t cursor_screen_column; // Cursor column (0-based)
609 bool cursor_found; // Cursor position was calculated
610 size_t terminal_width; // Terminal width used for wrapping
611} composition_with_cursor_t;

613/**
614 * Compose layers with cursor position tracking.
615 *
616 * This function performs composition while tracking the screen position of
617 * a cursor at a specific byte offset in the command buffer. It uses incremental
618 * cursor tracking (the proven approach from Replxx/Fish/ZLE) to handle:
619 * - Multi-byte UTF-8 characters (correct visual width)
620 * - Wide characters (CJK - 2 columns)
621 * - ANSI escape sequences (0 columns)
622 * - Tab expansion (to next multiple of 8)
623 * - Line wrapping (when content exceeds terminal width)
624 *
625 * The cursor position is calculated by walking through the command buffer
626 * character-by-character during composition, tracking the visual (x, y)
627 * position. When the byte offset matches cursor_byte_offset, the current
628 * position is recorded.
629 *
630 * This is the architecturally correct approach for LLE integration:
631 * - LLE passes cursor as byte offset in its buffer
632 * - Composition engine calculates screen position during rendering
633 * - Display controller uses pre-calculated position for terminal sequences
634 * - LLE never needs terminal knowledge (maintains architectural purity)
635 *
636 * @param engine The composition engine
637 * @param cursor_byte_offset Cursor position as byte offset in command text
638 * (0-based)
639 * @param terminal_width Terminal width for line wrapping calculation
640 * @param result Output structure with composed content and cursor position
641 * @return COMPOSITION_ENGINE_SUCCESS on success, error code on failure
642 *
643 * @note This is the CORRECT approach researched from modern line editors
644 * @note Do NOT use division/modulo formulas (breaks on UTF-8/ANSI/tabs/wide
645 * chars)
646 * @see docs/development/TERMINAL_CONTROL_WRAPPING_DESIGN.md
647 * @see docs/development/MODERN_EDITOR_WRAPPING_RESEARCH.md
648 */
649composition_engine_error_t composition_engine_compose_with_cursor(
650 composition_engine_t *engine, size_t cursor_byte_offset, int terminal_width,
651 composition_with_cursor_t *result);
