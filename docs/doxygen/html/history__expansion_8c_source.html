1/**
2 * @file history_expansion.c
3 * @brief LLE History System - History Expansion Implementation (Spec 09 Phase 3
4 * Day 10)
5 *
6 * Implements bash-compatible history expansion for the LLE history system:
7 * - !! - Repeat last command
8 * - !n - Repeat command number n
9 * - !-n - Repeat command n positions back
10 * - !string - Repeat most recent command starting with string
11 * - !?string - Repeat most recent command containing string
12 * - ^old^new - Quick substitution in last command
13 *
14 * Behavior (matches bash):
15 * - Expansion occurs before command execution
16 * - Failed expansion prints error and aborts command
17 * - Space prefix disables expansion (configurable)
18 * - :p modifier prints without executing
19 * - :s/old/new/ performs substitution
20 *
21 * @date 2025-11-01
22 * @author LLE Implementation Team
23 */

113/**
114 * Find the next history expansion marker in a string
115 *
116 * @param str String to search
117 * @param start_pos Starting position
118 * @return Position of expansion marker, or -1 if not found
119 */
120static ssize_t find_expansion_marker(const char *str, size_t start_pos) {
121 if (!str)
122 return -1;
123
124 size_t len = strlen(str);
125 for (size_t i = start_pos; i < len; i++) {
126 if (str[i] == '!') {
127 /* Check if it's escaped */
128 if (i > 0 && str[i - 1] == '\\') {
129 continue; /* Escaped, not an expansion */
130 }
131 /* Check if it's escaped or in quotes - basic implementation */
132 return (ssize_t)i;
133 } else if (str[i] == '^' && i == 0) {
134 /* Quick substitution must be at start */
135 return (ssize_t)i;
136 }
137 }
138
139 return -1;
140}

142/**
143 * Parse history number from string (handles !n and !-n)
144 *
145 * @param str String starting with number or -number
146 * @param number Output for parsed number
147 * @param is_relative Output - true if relative (!-n), false if absolute (!n)
148 * @param consumed Output - number of characters consumed
149 * @return true on success, false on parse error
150 */
151static bool parse_history_number(const char *str, int64_t *number,
152 bool *is_relative, size_t *consumed) {
153 if (!str || !number || !is_relative || !consumed) {
154 return false;
155 }
156
157 *consumed = 0;
158 *is_relative = false;
159
160 /* Check for relative notation (!-n) */
161 if (str[0] == '-') {
162 *is_relative = true;
163 str++;
164 (*consumed)++;
165 }
166
167 /* Parse the number */
168 char *endptr;
169 errno = 0;
170 long parsed = strtol(str, &endptr, 10);
171
172 if (errno != 0 || endptr == str) {
173 return false; /* Parse error */
174 }
175
176 *number = parsed;
177 *consumed += (endptr - str);
178
179 return true;
180}

182/**
183 * Extract string argument from expansion (for !string and !?string)
184 *
185 * @param str String after ! or !?
186 * @param output Output buffer for extracted string
187 * @param max_len Maximum length of output buffer
188 * @return Number of characters consumed, or 0 on error
189 */
190static size_t extract_expansion_string(const char *str, char *output,
191 size_t max_len) {
192 if (!str || !output || max_len == 0) {
193 return 0;
194 }
195
196 size_t i = 0;
197 while (str[i] != '\0' && i < max_len - 1) {
198 /* Stop at whitespace, special chars, or command terminators */
199 if (isspace(str[i]) || str[i] == ';' || str[i] == '|' ||
200 str[i] == '&' || str[i] == '>' || str[i] == '<' || str[i] == '(' ||
201 str[i] == ')' || str[i] == '\n') {
202 break;
203 }
204 output[i] = str[i];
205 i++;
206 }
207
208 output[i] = '\0';
209 return i;
210}

212/**
213 * Perform quick substitution (^old^new)
214 *
215 * @param last_command Last command from history
216 * @param old_pattern Pattern to replace
217 * @param new_pattern Replacement pattern
218 * @param result Output buffer for result
219 * @param max_len Maximum length of result buffer
220 * @return true on success, false on error
221 */
222static bool perform_quick_substitution(const char *last_command,
223 const char *old_pattern,
224 const char *new_pattern, char *result,
225 size_t max_len) {
226 if (!last_command || !old_pattern || !new_pattern || !result ||
227 max_len == 0) {
228 return false;
229 }
230
231 /* Find the old pattern in the last command */
232 const char *match_pos = strstr(last_command, old_pattern);
233 if (!match_pos) {
234 /* Pattern not found - substitution fails */
235 return false;
236 }
237
238 /* Build the substituted command */
239 size_t prefix_len = match_pos - last_command;
240 size_t old_len = strlen(old_pattern);
241 size_t new_len = strlen(new_pattern);
242 size_t suffix_len = strlen(match_pos + old_len);
243
244 /* Check if result will fit */
245 if (prefix_len + new_len + suffix_len >= max_len) {
246 /* Result too long */
247 return false;
248 }
249
250 /* Copy prefix */
251 memcpy(result, last_command, prefix_len);
252
253 /* Copy replacement */
254 memcpy(result + prefix_len, new_pattern, new_len);
255
256 /* Copy suffix */
257 memcpy(result + prefix_len + new_len, match_pos + old_len, suffix_len);
258
259 /* Null terminate */
260 result[prefix_len + new_len + suffix_len] = '\0';
261
262 return true;
263}

265/**
266 * Expand a single history reference
267 *
268 * @param expansion_str The expansion string (without leading !)
269 * @param result Output for expansion result
270 * @return LLE_SUCCESS or error code
271 */
272static lle_result_t expand_single_reference(const char *expansion_str,
273 lle_expansion_result_t *result) {
274 if (!expansion_str || !result || !g_expansion_ctx.history_core) {
275 return LLE_ERROR_INVALID_PARAMETER;
276 }
277
278 memset(result, 0, sizeof(lle_expansion_result_t));
279
280 /* Handle !! (repeat last command) */
281 if (expansion_str[0] == '!') {
282 result->type = EXPANSION_TYPE_LAST;
283 result->expansion_end = 2; /* !! is 2 characters */
284
285 lle_history_entry_t *entry = NULL;
286 lle_result_t res = lle_history_bridge_get_by_reverse_index(0, &entry);
287
288 if (res != LLE_SUCCESS || !entry) {
289 /* Error - details in return code */
290 return LLE_ERROR_NOT_FOUND;
291 }
292
293 result->expanded_command = pool_strdup(entry->command);
294 if (!result->expanded_command) {
295 return LLE_ERROR_OUT_OF_MEMORY;
296 }
297
298 return LLE_SUCCESS;
299 }
300
301 /* Handle !n or !-n (number reference) */
302 if (isdigit(expansion_str[0]) || expansion_str[0] == '-') {
303 int64_t number;
304 bool is_relative;
305 size_t consumed;
306
307 if (!parse_history_number(expansion_str, &number, &is_relative,
308 &consumed)) {
309 /* Error - details in return code */
310 return LLE_ERROR_INVALID_PARAMETER;
311 }
312
313 result->type =
314 is_relative ? EXPANSION_TYPE_RELATIVE : EXPANSION_TYPE_NUMBER;
315 result->expansion_end = consumed + 1; /* +1 for the ! */
316
317 lle_history_entry_t *entry = NULL;
318 lle_result_t res;
319
320 if (is_relative) {
321 /* !-n means n commands back (0 = most recent) */
322 if (number < 0) {
323 /* Error - details in return code */
324 return LLE_ERROR_INVALID_PARAMETER;
325 }
326 res =
327 lle_history_bridge_get_by_reverse_index((size_t)number, &entry);
328 } else {
329 /* !n means entry ID n */
330 res = lle_history_bridge_get_by_number((uint64_t)number, &entry);
331 }
332
333 if (res != LLE_SUCCESS || !entry) {
334 char err[128];
335 snprintf(err, sizeof(err), "!%s%" PRId64 ": event not found",
336 is_relative ? "-" : "", number);
337 /* Error - details in return code */
338 return LLE_ERROR_NOT_FOUND;
339 }
340
341 result->expanded_command = pool_strdup(entry->command);
342 if (!result->expanded_command) {
343 return LLE_ERROR_OUT_OF_MEMORY;
344 }
345
346 return LLE_SUCCESS;
347 }
348
349 /* Handle !?string (substring search) */
350 if (expansion_str[0] == '?') {
351 result->type = EXPANSION_TYPE_SUBSTRING;
352
353 char search_str[256];
354 size_t consumed = extract_expansion_string(
355 expansion_str + 1, search_str, sizeof(search_str));
356
357 if (consumed == 0) {
358 /* Error - details in return code */
359 return LLE_ERROR_INVALID_PARAMETER;
360 }
361
362 result->expansion_end = consumed + 2; /* +1 for !, +1 for ? */
363
364 /* Search for command containing string */
365 lle_history_search_results_t *search_results =
366 lle_history_search_substring(g_expansion_ctx.history_core,
367 search_str, 1);
368
369 if (!search_results ||
370 lle_history_search_results_get_count(search_results) == 0) {
371 if (search_results) {
372 lle_history_search_results_destroy(search_results);
373 }
374 char err[256];
375 snprintf(err, sizeof(err), "!?%.230s: event not found", search_str);
376 /* Error - details in return code */
377 return LLE_ERROR_NOT_FOUND;
378 }
379
380 const lle_search_result_t *first_result =
381 lle_history_search_results_get(search_results, 0);
382 result->expanded_command = pool_strdup(first_result->command);
383
384 lle_history_search_results_destroy(search_results);
385
386 if (!result->expanded_command) {
387 return LLE_ERROR_OUT_OF_MEMORY;
388 }
389
390 return LLE_SUCCESS;
391 }
392
393 /* Handle !string (prefix search) */
394 result->type = EXPANSION_TYPE_PREFIX;
395
396 char search_str[256];
397 size_t consumed =
398 extract_expansion_string(expansion_str, search_str, sizeof(search_str));
399
400 if (consumed == 0) {
401 /* Error - details in return code */
402 return LLE_ERROR_INVALID_PARAMETER;
403 }
404
405 result->expansion_end = consumed + 1; /* +1 for ! */
406
407 /* Search for most recent command starting with string */
408 lle_history_search_results_t *search_results =
409 lle_history_search_prefix(g_expansion_ctx.history_core, search_str, 1);
410
411 if (!search_results ||
412 lle_history_search_results_get_count(search_results) == 0) {
413 if (search_results) {
414 lle_history_search_results_destroy(search_results);
415 }
416 char err[256];
417 snprintf(err, sizeof(err), "!%.235s: event not found", search_str);
418 /* Error - details in return code */
419 return LLE_ERROR_NOT_FOUND;
420 }
421
422 const lle_search_result_t *first_result =
423 lle_history_search_results_get(search_results, 0);
424 result->expanded_command = pool_strdup(first_result->command);
425
426 lle_history_search_results_destroy(search_results);
427
428 if (!result->expanded_command) {
429 return LLE_ERROR_OUT_OF_MEMORY;
430 }
431
432 return LLE_SUCCESS;
433}

471/**
472 * Check if a command line contains history expansion
473 *
474 * @param command Command line to check
475 * @return true if expansion is present, false otherwise
476 */
477bool lle_history_expansion_needed(const char *command) {
478 if (!command || command[0] == '\0') {
479 return false;
480 }
481
482 /* Check for space prefix disabling expansion */
483 if (g_expansion_ctx.space_disables_expansion && isspace(command[0])) {
484 return false;
485 }
486
487 /* Check for quick substitution (^old^new) */
488 if (command[0] == '^') {
489 return true;
490 }
491
492 /* Check for ! expansion */
493 return (find_expansion_marker(command, 0) >= 0);
494}

496/**
497 * Expand history references in a command line
498 *
499 * Expands all history references (!!, !n, !-n, !string, etc.) in the command.
500 * The result must be freed by the caller using lle_pool_free().
501 *
502 * @param command Original command with history references
503 * @param expanded Output pointer for expanded command (allocated, caller must
504 * free)
505 * @return LLE_SUCCESS or error code
506 */
507lle_result_t lle_history_expand_line(const char *command, char **expanded) {
508 if (!command || !expanded) {
509 return LLE_ERROR_INVALID_PARAMETER;
510 }
511
512 if (!g_expansion_ctx.history_core) {
513 return LLE_ERROR_NOT_INITIALIZED;
514 }
515
516 *expanded = NULL;
517
518 /* Check for recursion depth */
519 if (g_expansion_ctx.recursion_depth >= EXPANSION_MAX_DEPTH) {
520 /* Error - details in return code */
521 return LLE_ERROR_INVALID_STATE;
522 }
523
524 /* Check if expansion is needed */
525 if (!lle_history_expansion_needed(command)) {
526 *expanded = pool_strdup(command);
527 return *expanded ? LLE_SUCCESS : LLE_ERROR_OUT_OF_MEMORY;
528 }
529
530 /* Handle quick substitution (^old^new) */
531 if (command[0] == '^') {
532 /* Parse ^old^new format */
533 const char *p = command + 1;
534 char old_pattern[256] = {0};
535 char new_pattern[256] = {0};
536
537 /* Extract old pattern */
538 size_t i = 0;
539 while (*p != '\0' && *p != '^' && i < sizeof(old_pattern) - 1) {
540 old_pattern[i++] = *p++;
541 }
542
543 if (*p != '^') {
544 /* Error - details in return code */
545 return LLE_ERROR_INVALID_PARAMETER;
546 }
547
548 p++; /* Skip second ^ */
549
550 /* Extract new pattern */
551 i = 0;
552 while (*p != '\0' && *p != ' ' && *p != '\n' &&
553 i < sizeof(new_pattern) - 1) {
554 new_pattern[i++] = *p++;
555 }
556
557 /* Get last command */
558 lle_history_entry_t *last_entry = NULL;
559 lle_result_t res =
560 lle_history_bridge_get_by_reverse_index(0, &last_entry);
561
562 if (res != LLE_SUCCESS || !last_entry) {
563 /* Error - details in return code */
564 return LLE_ERROR_NOT_FOUND;
565 }
566
567 /* Perform substitution */
568 char result[EXPANSION_MAX_LENGTH];
569 if (!perform_quick_substitution(last_entry->command, old_pattern,
570 new_pattern, result, sizeof(result))) {
571 return LLE_ERROR_INVALID_PARAMETER;
572 }
573
574 *expanded = pool_strdup(result);
575 return *expanded ? LLE_SUCCESS : LLE_ERROR_OUT_OF_MEMORY;
576 }
577
578 /* Handle ! expansions */
579 char result[EXPANSION_MAX_LENGTH];
580 size_t result_pos = 0;
581 size_t cmd_pos = 0;
582 size_t cmd_len = strlen(command);
583
584 g_expansion_ctx.recursion_depth++;
585
586 while (cmd_pos < cmd_len && result_pos < EXPANSION_MAX_LENGTH - 1) {
587 ssize_t expansion_pos = find_expansion_marker(command + cmd_pos, 0);
588
589 if (expansion_pos < 0) {
590 /* No more expansions, copy rest of string */
591 size_t remaining = cmd_len - cmd_pos;
592 if (result_pos + remaining >= EXPANSION_MAX_LENGTH) {
593 g_expansion_ctx.recursion_depth--;
594 /* Error - details in return code */
595 return LLE_ERROR_BUFFER_OVERFLOW;
596 }
597 memcpy(result + result_pos, command + cmd_pos, remaining);
598 result_pos += remaining;
599 break;
600 }
601
602 /* Copy text before expansion */
603 if (expansion_pos > 0) {
604 memcpy(result + result_pos, command + cmd_pos, expansion_pos);
605 result_pos += expansion_pos;
606 cmd_pos += expansion_pos;
607 }
608
609 /* Process the expansion */
610 lle_expansion_result_t exp_result;
611 lle_result_t res =
612 expand_single_reference(command + cmd_pos + 1, &exp_result);
613
614 if (res != LLE_SUCCESS) {
615 g_expansion_ctx.recursion_depth--;
616 return res;
617 }
618
619 /* Copy expanded text */
620 size_t expanded_len = strlen(exp_result.expanded_command);
621 if (result_pos + expanded_len >= EXPANSION_MAX_LENGTH) {
622 lle_pool_free(exp_result.expanded_command);
623 g_expansion_ctx.recursion_depth--;
624 /* Error - details in return code */
625 return LLE_ERROR_BUFFER_OVERFLOW;
626 }
627
628 memcpy(result + result_pos, exp_result.expanded_command, expanded_len);
629 result_pos += expanded_len;
630 cmd_pos += exp_result.expansion_end;
631
632 lle_pool_free(exp_result.expanded_command);
633 }
634
635 result[result_pos] = '\0';
636 g_expansion_ctx.recursion_depth--;
637
638 *expanded = pool_strdup(result);
639 return *expanded ? LLE_SUCCESS : LLE_ERROR_OUT_OF_MEMORY;
640}
