21/*
22 * Initialize enhanced statistics
23 */
24lle_result_t lle_event_enhanced_stats_init(lle_event_system_t *system) {
25 if (!system) {
26 return LLE_ERROR_INVALID_PARAMETER;
27 }
28
29 /* Already initialized */
30 if (system->enhanced_stats) {
31 return LLE_SUCCESS;
32 }
33
34 /* Allocate enhanced stats structure */
35 lle_event_enhanced_stats_t *stats =
36 lle_pool_alloc(sizeof(lle_event_enhanced_stats_t));
37 if (!stats) {
38 return LLE_ERROR_OUT_OF_MEMORY;
39 }
40
41 memset(stats, 0, sizeof(lle_event_enhanced_stats_t));
42
43 /* Allocate per-type statistics array */
44 stats->type_stats_capacity = LLE_ENHANCED_STATS_INITIAL_CAPACITY;
45 stats->type_stats = lle_pool_alloc(sizeof(lle_event_type_stats_t) *
46 stats->type_stats_capacity);
47 if (!stats->type_stats) {
48 lle_pool_free(stats);
49 return LLE_ERROR_OUT_OF_MEMORY;
50 }
51
52 memset(stats->type_stats, 0,
53 sizeof(lle_event_type_stats_t) * stats->type_stats_capacity);
54 stats->type_stats_count = 0;
55
56 /* Initialize cycle stats with sentinel values */
57 stats->min_cycle_time = UINT64_MAX;
58 stats->max_cycle_time = 0;
59
60 /* Initialize mutex */
61 pthread_mutex_init(&stats->stats_mutex, NULL);
62
63 system->enhanced_stats = stats;
64 return LLE_SUCCESS;
65}

67/*
68 * Destroy enhanced statistics
69 */
70void lle_event_enhanced_stats_destroy(lle_event_system_t *system) {
71 if (!system || !system->enhanced_stats) {
72 return;
73 }
74
75 lle_event_enhanced_stats_t *stats = system->enhanced_stats;
76
77 /* Destroy mutex */
78 pthread_mutex_destroy(&stats->stats_mutex);
79
80 /* Free type stats array */
81 if (stats->type_stats) {
82 lle_pool_free(stats->type_stats);
83 }
84
85 /* Free stats structure */
86 lle_pool_free(stats);
87
88 system->enhanced_stats = NULL;
89}

91/*
92 * Get per-type statistics for a specific event type
93 */
94lle_result_t
95lle_event_enhanced_stats_get_type(lle_event_system_t *system,
96 lle_event_kind_t type,
97 lle_event_type_stats_t *stats_out) {
98 if (!system || !stats_out) {
99 return LLE_ERROR_INVALID_PARAMETER;
100 }
101
102 if (!system->enhanced_stats) {
103 return LLE_ERROR_NOT_INITIALIZED;
104 }
105
106 lle_event_enhanced_stats_t *stats = system->enhanced_stats;
107
108 pthread_mutex_lock(&stats->stats_mutex);
109
110 /* Search for type */
111 for (size_t i = 0; i < stats->type_stats_count; i++) {
112 if (stats->type_stats[i].event_type == type) {
113 memcpy(stats_out, &stats->type_stats[i],
114 sizeof(lle_event_type_stats_t));
115 pthread_mutex_unlock(&stats->stats_mutex);
116 return LLE_SUCCESS;
117 }
118 }
119
120 pthread_mutex_unlock(&stats->stats_mutex);
121
122 /* Type not found - return zeros */
123 memset(stats_out, 0, sizeof(lle_event_type_stats_t));
124 stats_out->event_type = type;
125 stats_out->min_processing_time = 0; /* Override sentinel */
126
127 return LLE_SUCCESS;
128}

272/*
273 * Set processing configuration
274 */
275lle_result_t
276lle_event_processing_set_config(lle_event_system_t *system,
277 const lle_event_processing_config_t *config) {
278 if (!system || !config) {
279 return LLE_ERROR_INVALID_PARAMETER;
280 }
281
282 pthread_mutex_lock(&system->system_mutex);
283
284 /* If enabling detailed stats and not yet initialized, initialize now */
285 if (config->record_detailed_stats && !system->enhanced_stats) {
286 lle_result_t result = lle_event_enhanced_stats_init(system);
287 if (result != LLE_SUCCESS) {
288 pthread_mutex_unlock(&system->system_mutex);
289 return result;
290 }
291 }
292
293 /* Copy configuration */
294 system->processing_config = *config;
295
296 pthread_mutex_unlock(&system->system_mutex);
297
298 return LLE_SUCCESS;
299}
