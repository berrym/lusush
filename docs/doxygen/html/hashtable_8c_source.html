41void *lle_hashtable_key_copy_pooled(const void *key) {
42 lle_hashtable_memory_context_t *ctx = lle_get_current_memory_context();
43
44 if (!key || !ctx || !ctx->pool) {
45 /* Fallback to regular malloc */
46 const char *key_str = (const char *)key;
47 size_t len = strlen(key_str) + 1;
48 char *new_key = malloc(len);
49 if (new_key) {
50 memcpy(new_key, key_str, len);
51 }
52 return new_key;
53 }
54
55 const char *key_str = (const char *)key;
56 size_t key_len = strlen(key_str) + 1;
57 char *new_key = lusush_pool_alloc(key_len);
58 if (!new_key) {
59 return NULL;
60 }
61
62 memcpy(new_key, key_str, key_len);
63
64 /* Update statistics */
65 ctx->allocations++;
66 ctx->bytes_allocated += key_len;
67
68 return new_key;
69}

213lle_result_t lle_hashtable_registry_init(lle_hashtable_registry_t **registry) {
214 if (!registry) {
215 return LLE_ERROR_INVALID_PARAMETER;

218 lle_hashtable_registry_t *reg = calloc(1, sizeof(lle_hashtable_registry_t));

225 if (!reg->hashtables) {
226 free(reg);

231 free(reg->hashtables);

236 reg->count = 0;
237
238 *registry = reg;
239 return LLE_SUCCESS;

256lle_result_t lle_hashtable_registry_add(lle_hashtable_registry_t *registry,
257 lle_strstr_hashtable_t *hashtable) {
258 if (!registry || !hashtable) {
259 return LLE_ERROR_INVALID_PARAMETER;
260 }
261
262 pthread_mutex_lock(&registry->lock);
263
264 /* Check if we need to grow the registry */
265 if (registry->count >= registry->capacity) {
266 size_t new_capacity = registry->capacity * 2;
267 lle_strstr_hashtable_t **new_hashtables =
268 realloc(registry->hashtables,
269 new_capacity * sizeof(lle_strstr_hashtable_t *));
270
271 if (!new_hashtables) {
272 pthread_mutex_unlock(&registry->lock);
273 return LLE_ERROR_OUT_OF_MEMORY;
274 }
275
276 registry->hashtables = new_hashtables;
277 registry->capacity = new_capacity;
278 }
279
280 registry->hashtables[registry->count++] = hashtable;
281
282 pthread_mutex_unlock(&registry->lock);
283 return LLE_SUCCESS;
284}

286lle_result_t lle_hashtable_registry_remove(lle_hashtable_registry_t *registry,
287 lle_strstr_hashtable_t *hashtable) {
288 if (!registry || !hashtable) {
289 return LLE_ERROR_INVALID_PARAMETER;
290 }
291
292 pthread_mutex_lock(&registry->lock);
293
294 /* Find and remove hashtable */
295 for (size_t i = 0; i < registry->count; i++) {
296 if (registry->hashtables[i] == hashtable) {
297 /* Shift remaining entries */
298 for (size_t j = i; j < registry->count - 1; j++) {
299 registry->hashtables[j] = registry->hashtables[j + 1];
300 }
301 registry->count--;
302 pthread_mutex_unlock(&registry->lock);
303 return LLE_SUCCESS;
304 }
305 }
306
307 pthread_mutex_unlock(&registry->lock);
308 return LLE_ERROR_NOT_FOUND;
309}

396lle_result_t
397lle_hashtable_factory_create_strstr(lle_hashtable_factory_t *factory,
398 const lle_hashtable_config_t *config,
399 lle_strstr_hashtable_t **hashtable) {
400 if (!factory || !hashtable) {
401 return LLE_ERROR_INVALID_PARAMETER;
402 }
403
404 /* Use provided config or factory default */
405 const lle_hashtable_config_t *cfg =
406 config ? config : factory->default_config;
407
408 /* Allocate wrapper structure */
409 lle_strstr_hashtable_t *ht_wrapper =
410 calloc(1, sizeof(lle_strstr_hashtable_t));
411 if (!ht_wrapper) {
412 return LLE_ERROR_OUT_OF_MEMORY;
413 }
414
415 /* Store configuration */
416 ht_wrapper->config = calloc(1, sizeof(lle_hashtable_config_t));
417 if (!ht_wrapper->config) {
418 free(ht_wrapper);
419 return LLE_ERROR_OUT_OF_MEMORY;
420 }
421 memcpy(ht_wrapper->config, cfg, sizeof(lle_hashtable_config_t));
422
423 /* Create memory context if using memory pool */
424 if (cfg->use_memory_pool && cfg->memory_pool) {
425 ht_wrapper->mem_ctx = calloc(1, sizeof(lle_hashtable_memory_context_t));
426 if (!ht_wrapper->mem_ctx) {
427 free(ht_wrapper->config);
428 free(ht_wrapper);
429 return LLE_ERROR_OUT_OF_MEMORY;
430 }
431
432 ht_wrapper->mem_ctx->pool = cfg->memory_pool;
433 ht_wrapper->mem_ctx->hashtable_name = cfg->hashtable_name;
434
435 /* Set as current context for memory callbacks */
436 lle_set_current_memory_context(ht_wrapper->mem_ctx);
437 }
438
439 /* Create underlying libhashtable */
440 uint32_t flags = 0;
441 if (cfg->random_seed) {
442 flags |= HT_SEED_RANDOM;
443 }
444
445 ht_wrapper->ht = ht_strstr_create(flags);
446 if (!ht_wrapper->ht) {
447 if (ht_wrapper->mem_ctx) {
448 free(ht_wrapper->mem_ctx);
449 }
450 free(ht_wrapper->config);
451 free(ht_wrapper);
452 return LLE_ERROR_OUT_OF_MEMORY;
453 }
454
455 /* Note: libhashtable callbacks are set during creation, not after.
456 * For now, memory pool integration will be added in a future phase. */
457
458 /* Initialize performance metrics if enabled */
459 if (cfg->performance_monitoring) {
460 ht_wrapper->metrics =
461 calloc(1, sizeof(lle_hashtable_performance_metrics_t));
462 if (!ht_wrapper->metrics) {
463 ht_strstr_destroy(ht_wrapper->ht);
464 if (ht_wrapper->mem_ctx) {
465 free(ht_wrapper->mem_ctx);
466 }
467 free(ht_wrapper->config);
468 free(ht_wrapper);
469 return LLE_ERROR_OUT_OF_MEMORY;
470 }
471 }
472
473 /* Initialize thread safety if enabled (Phase 2) */
474 if (cfg->thread_safe) {
475 ht_wrapper->lock = calloc(1, sizeof(pthread_rwlock_t));
476 if (!ht_wrapper->lock) {
477 if (ht_wrapper->metrics) {
478 free(ht_wrapper->metrics);
479 }
480 ht_strstr_destroy(ht_wrapper->ht);
481 if (ht_wrapper->mem_ctx) {
482 free(ht_wrapper->mem_ctx);
483 }
484 free(ht_wrapper->config);
485 free(ht_wrapper);
486 return LLE_ERROR_OUT_OF_MEMORY;
487 }
488
489 if (pthread_rwlock_init(ht_wrapper->lock, NULL) != 0) {
490 free(ht_wrapper->lock);
491 if (ht_wrapper->metrics) {
492 free(ht_wrapper->metrics);
493 }
494 ht_strstr_destroy(ht_wrapper->ht);
495 if (ht_wrapper->mem_ctx) {
496 free(ht_wrapper->mem_ctx);
497 }
498 free(ht_wrapper->config);
499 free(ht_wrapper);
500 return LLE_ERROR_SYSTEM_CALL;
501 }
502
503 ht_wrapper->is_concurrent = true;
504 }
505
506 ht_wrapper->name = cfg->hashtable_name;
507 ht_wrapper->entry_count = 0; /* Initialize entry count */
508
509 /* Register with factory registry */
510 lle_hashtable_registry_add(factory->registry, ht_wrapper);
511
512 /* Clear memory context */
513 if (cfg->use_memory_pool) {
514 lle_set_current_memory_context(NULL);
515 }
516
517 *hashtable = ht_wrapper;
518 return LLE_SUCCESS;
519}

588/* ============================================================================
589 * STRING-TO-STRING HASHTABLE OPERATIONS (Phase 1 + Phase 2)
590 * ============================================================================
591 *
592 * THREAD SAFETY NOTE:
593 * While this wrapper provides external locking (rwlock) around all operations,
594 * libhashtable's internal linked list implementation for hash collision
595 * handling is not inherently thread-safe. Under high concurrency with many hash
596 * collisions, some entries may be lost even with external locking (typically
597 * 90-95% success rate).
598 *
599 * This is a known limitation of libhashtable's architecture. For scenarios
600 * requiring 100% reliability under concurrent writes, consider:
601 * 1. Using a different hashtable implementation
602 * 2. Serializing all writes at application level
603 * 3. Using separate hashtable instances per thread
604 */
605
606lle_result_t lle_strstr_hashtable_insert(lle_strstr_hashtable_t *ht,
607 const char *key, const char *value) {
608 if (!ht || !key) {
609 return LLE_ERROR_INVALID_PARAMETER;
610 }
611
612 uint64_t start_time = 0;
613 if (ht->metrics) {
614 start_time = lle_get_current_time_microseconds();
615 }
616
617 /* Set memory context if using memory pool */
618 if (ht->mem_ctx) {
619 lle_set_current_memory_context(ht->mem_ctx);
620 }
621
622 /* Acquire write lock if thread-safe (Phase 2) */
623 if (ht->is_concurrent && ht->lock) {
624 pthread_rwlock_wrlock(ht->lock);
625 }
626
627 /* Check if key exists (to track new inserts vs updates) */
628 bool key_exists = (ht_strstr_get(ht->ht, key) != NULL);
629
630 /* Perform insertion */
631 ht_strstr_insert(ht->ht, key, value);
632
633 /* Update entry count if this was a new insertion */
634 if (!key_exists) {
635 ht->entry_count++;
636 }
637
638 /* Update metrics - MUST happen inside lock to prevent race conditions */
639 if (ht->metrics) {
640 uint64_t end_time = lle_get_current_time_microseconds();
641 uint64_t duration = end_time - start_time;
642
643 ht->metrics->insert_operations++;
644 ht->metrics->total_insert_time_us += duration;
645
646 if (duration > ht->metrics->max_insert_time_us) {
647 ht->metrics->max_insert_time_us = duration;
648 }
649
650 ht->metrics->avg_insert_time_us =
651 ht->metrics->total_insert_time_us / ht->metrics->insert_operations;
652 }
653
654 /* Release lock if thread-safe */
655 if (ht->is_concurrent && ht->lock) {
656 pthread_rwlock_unlock(ht->lock);
657 }
658
659 /* Clear memory context */
660 if (ht->mem_ctx) {
661 lle_set_current_memory_context(NULL);
662 }
663
664 return LLE_SUCCESS;
665}

667const char *lle_strstr_hashtable_lookup(lle_strstr_hashtable_t *ht,
668 const char *key) {
669 if (!ht || !key) {
670 return NULL;
671 }
672
673 uint64_t start_time = 0;
674 if (ht->metrics) {
675 start_time = lle_get_current_time_microseconds();
676 }
677
678 /* Acquire read lock if thread-safe (Phase 2) */
679 if (ht->is_concurrent && ht->lock) {
680 pthread_rwlock_rdlock(ht->lock);
681 }
682
683 /* Perform lookup */
684 const char *result = ht_strstr_get(ht->ht, key);
685
686 /* Update metrics - MUST happen inside lock to prevent race conditions */
687 if (ht->metrics) {
688 uint64_t end_time = lle_get_current_time_microseconds();
689 uint64_t duration = end_time - start_time;
690
691 ht->metrics->lookup_operations++;
692 ht->metrics->total_lookup_time_us += duration;
693
694 if (duration > ht->metrics->max_lookup_time_us) {
695 ht->metrics->max_lookup_time_us = duration;
696 }
697
698 ht->metrics->avg_lookup_time_us =
699 ht->metrics->total_lookup_time_us / ht->metrics->lookup_operations;
700 }
701
702 /* Release lock if thread-safe */
703 if (ht->is_concurrent && ht->lock) {
704 pthread_rwlock_unlock(ht->lock);
705 }
706
707 return result;
708}

710lle_result_t lle_strstr_hashtable_delete(lle_strstr_hashtable_t *ht,
711 const char *key) {
712 if (!ht || !key) {
713 return LLE_ERROR_INVALID_PARAMETER;
714 }
715
716 uint64_t start_time = 0;
717 if (ht->metrics) {
718 start_time = lle_get_current_time_microseconds();
719 }
720
721 /* Set memory context if using memory pool */
722 if (ht->mem_ctx) {
723 lle_set_current_memory_context(ht->mem_ctx);
724 }
725
726 /* Acquire write lock if thread-safe (Phase 2) */
727 if (ht->is_concurrent && ht->lock) {
728 pthread_rwlock_wrlock(ht->lock);
729 }
730
731 /* Check if key exists before deletion */
732 bool key_exists = (ht_strstr_get(ht->ht, key) != NULL);
733
734 /* Perform deletion - libhashtable uses remove, not delete */
735 ht_strstr_remove(ht->ht, key);
736 bool deleted = key_exists;
737
738 /* Update entry count if something was actually deleted */
739 if (deleted) {
740 ht->entry_count--;
741 }
742
743 /* Update metrics - MUST happen inside lock to prevent race conditions */
744 if (ht->metrics) {
745 uint64_t end_time = lle_get_current_time_microseconds();

811void lle_strstr_hashtable_clear(lle_strstr_hashtable_t *ht) {
812 if (!ht) {
813 return;
814 }
815
816 /* Set memory context if using memory pool */
817 if (ht->mem_ctx) {
818 lle_set_current_memory_context(ht->mem_ctx);
819 }
820
821 /* Acquire write lock if thread-safe (Phase 2) */
822 if (ht->is_concurrent && ht->lock) {
823 pthread_rwlock_wrlock(ht->lock);
824 }
825
826 /* libhashtable doesn't have clear, enumerate and remove all entries */
827 ht_enum_t *enumerator = ht_strstr_enum_create(ht->ht);
828 if (enumerator) {
829 const char *key, *value;
830 /* Collect keys first to avoid modifying during enumeration */
831 char **keys = NULL;
832 size_t key_count = 0;
833 size_t key_capacity = 16;
834
835 keys = malloc(key_capacity * sizeof(char *));
836 if (keys) {
837 while (ht_strstr_enum_next(enumerator, &key, &value)) {
838 if (key_count >= key_capacity) {
839 key_capacity *= 2;
840 char **new_keys =
841 realloc(keys, key_capacity * sizeof(char *));
842 if (!new_keys)
843 break;
844 keys = new_keys;
845 }
846 keys[key_count++] = strdup(key);
847 }
848
849 /* Now remove all keys */
850 for (size_t i = 0; i < key_count; i++) {
851 ht_strstr_remove(ht->ht, keys[i]);
852 free(keys[i]);
853 }
854 free(keys);
855 }
856 ht_strstr_enum_destroy(enumerator);
857 }
858
859 /* Reset entry count */
860 ht->entry_count = 0;
861
862 /* Release lock if thread-safe */
863 if (ht->is_concurrent && ht->lock) {
864 pthread_rwlock_unlock(ht->lock);
865 }
866
867 /* Clear memory context */
868 if (ht->mem_ctx) {
869 lle_set_current_memory_context(NULL);
870 }
871}
