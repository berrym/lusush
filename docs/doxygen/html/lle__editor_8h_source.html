63/**
64 * Editor context structure
65 *
66 * This is the central structure that provides access to all LLE subsystems.
67 * It is passed to every keybinding action function to provide access to:
68 * - Buffer management for text editing
69 * - History system for command history
70 * - Kill ring for yank/kill operations
71 * - Keybinding manager for key processing
72 * - Display controller for output
73 * - Completion system for tab completion
74 * - And other editor subsystems
75 *
76 * Design:
77 * - All fields are pointers to allow lazy initialization
78 * - NULL fields indicate subsystem not available (graceful degradation)
79 * - Memory managed through unified memory pool
80 * - Thread-safe when used from single editing thread
81 */
82typedef struct lle_editor {
83 /* Core editing subsystems */
84 lle_buffer_t *buffer; /* Text buffer */
85 lle_cursor_manager_t *cursor_manager; /* Cursor position management */
86 lle_kill_ring_t *kill_ring; /* Kill/yank ring */
87 lle_keybinding_manager_t *keybinding_manager; /* Key bindings */
88 lle_change_tracker_t *change_tracker; /* Undo/redo change tracking */
89
90 /* History and search */
91 lle_history_core_t *history_system; /* Command history */
92 lle_history_buffer_integration_t
93 *history_buffer_integration; /* History<->buffer bridge */
94 size_t history_navigation_pos; /* Current position in history navigation (0
95 = current line) */
96 bool history_search_active; /* Interactive search active */
97 int history_search_direction; /* 1=forward, -1=reverse */
98
99 /* Unique-only navigation tracking (for lle_dedup_navigation_unique) */
100 uint32_t
101 *history_nav_seen_hashes; /* Hash set of commands seen this session */
102 size_t history_nav_seen_count; /* Number of seen hashes */
103 size_t history_nav_seen_capacity; /* Capacity of seen hash array */
104
105 /* Display and output */
106 lle_display_controller_t *display_controller; /* Display management */
107
108 /* Completion */
109 struct lle_completion_system
110 *completion_system; /* Tab completion (Spec 12) */
111
112 /* Editing mode and state */
113 lle_editing_mode_t editing_mode; /* Current editing mode */
114 lle_vi_state_t *vi_state; /* Vi mode state (if enabled) */
115 lle_keymap_mode_t current_keymap; /* Active keymap */
116
117 /* Special modes */
118 bool quoted_insert_mode; /* Ctrl-Q/Ctrl-V mode */
119 bool eof_requested; /* EOF signal (Ctrl-D on empty line) */
120 bool abort_requested; /* Abort signal (Ctrl-G) */
121
122 /* Extensibility and Widgets */
123 lle_widget_registry_t *widget_registry; /* Widget system registry */
124 lle_widget_hooks_manager_t *widget_hooks_manager; /* Widget hooks manager */
125 lle_plugin_manager_t *plugin_manager; /* Plugin system (future) */
126 lle_script_integration_t
127 *script_integration; /* Script integration (future) */
128 void *history_edit_callbacks; /* History edit callbacks (opaque) */
129
130 /* Memory management */
131 lusush_memory_pool_t *lusush_pool; /* Lusush memory pool (source) */
132 lle_memory_pool_t *lle_pool; /* LLE memory pool (wrapper) */
133
134 /* Editor statistics */
135 uint64_t total_keystrokes; /* Total keys processed */
136 uint64_t command_count; /* Commands executed */
137 uint64_t edit_session_start; /* Session start timestamp */
138
139} lle_editor_t;
