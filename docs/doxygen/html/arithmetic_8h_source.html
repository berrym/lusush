19/**
20 * arithm_expand:
21 * Evaluate arithmetic expression with full POSIX operator support
22 *
23 * @param orig_expr: Arithmetic expression string (with or without $(( ))
24 * wrapper)
25 * @return: String representation of the result, or NULL on error
26 *
27 * Supported operators:
28 * - Basic: + - * / %
29 * - Comparison: == != < <= > >=
30 * - Logical: && || !
31 * - Bitwise: & | ^ ~ << >>
32 * - Assignment: = += -= *= /= %= <<= >>= &= ^= |=
33 * - Increment/Decrement: ++ --
34 * - Exponentiation: **
35 * - Ternary: ? :
36 * - Parentheses for grouping
37 * - Variables and numeric literals
38 * - Octal (0123) and hexadecimal (0x123) numbers
39 */
40char *arithm_expand(const char *orig_expr);

42/**
43 * arithm_expand_with_executor:
44 * Evaluate arithmetic expression with executor context for scoped
45 * variables
46 *
47 * @param executor: Executor context for scoped variable resolution
48 * @param orig_expr: Arithmetic expression string (with or without $(( ))
49 * wrapper)
50 * @return: String representation of the result, or NULL on error
51 *
52 * This function is identical to arithm_expand but uses the executor's
53 * symbol table for variable resolution, allowing access to function
54 * parameters like $1, $2, etc.
55 */
56typedef struct executor executor_t;
57char *arithm_expand_with_executor(executor_t *executor, const char *orig_expr);
