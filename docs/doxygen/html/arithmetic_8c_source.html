33// Stack item for arithmetic evaluation
34typedef struct {
35 enum { ITEM_LONG_INT = 1, ITEM_VAR_PTR = 2 } type;
36
37 union {
38 ssize_t val;
39 char *var_name; // Store variable name instead of pointer
40 };
41 void *executor_context; // Store executor context for scoped variable
42 // resolution
43} stack_item_t;

45// Operator associativity

69// Cleanup function for arithmetic context
70static void arithm_context_cleanup(arithm_context_t *ctx) {
71 if (!ctx) {
72 return;
73 }
74
75 // Free any variable names stored in the number stack
76 for (int i = 0; i < ctx->nnumstack; i++) {
77 if (ctx->numstack[i].type == ITEM_VAR_PTR &&
78 ctx->numstack[i].var_name) {
79 free(ctx->numstack[i].var_name);
80 ctx->numstack[i].var_name = NULL;
81 }
82 }
83
84 if (ctx->error_message) {
85 free(ctx->error_message);
86 ctx->error_message = NULL;
87 }
88}

402static ssize_t eval_exp(stack_item_t *a1, stack_item_t *a2) {
403 ssize_t base = long_value(a1);
404 ssize_t exp = long_value(a2);
405 if (exp < 0) {
406 arithm_set_error("negative exponent not supported");
407 return 0;
408 }
409
410 ssize_t result = 1;
411 for (ssize_t i = 0; i < exp; i++) {
412 result *= base;
413 }
414 return result;
415}

488// Get long value from stack item
489static ssize_t long_value(stack_item_t *item) {
490 switch (item->type) {
491 case ITEM_LONG_INT:
492 return item->val;
493 case ITEM_VAR_PTR: {
494 if (item->var_name) {
495 // Use executor context if available for scoped variable resolution
496 if (item->executor_context) {
497 // Use proper executor structure from executor.h
498 executor_t *exec = (executor_t *)item->executor_context;
499
500 if (exec && exec->symtable) {
501 char *value =
502 symtable_get_var(exec->symtable, item->var_name);
503 if (value) {
504 return atol(value);
505 }
506 }
507 }
508
509 // Fallback to global manager
510 symtable_manager_t *manager = symtable_get_global_manager();
511 if (manager) {
512 char *value = symtable_get_var(manager, item->var_name);
513 if (value) {
514 return atol(value);
515 }
516 }
517 }
518 return 0;
519 }
520 default:
521 return 0;
522 }

527 if (ctx->nopstack >= MAXOPSTACK) {
528 ctx->errflag = true;
529 arithm_set_error("operator stack overflow");
530 return;

532 ctx->opstack[ctx->nopstack++] = op;

537 ctx->errflag = true;

539 return NULL;
540 }
541 return ctx->opstack[--ctx->nopstack];
542}

586// Get operator from expression
587static op_t *get_op(const char *expr) {
588 // Check for increment/decrement operators first (they are 2-char)
589 if (expr[0] == '+' && expr[1] == '+') {
590 return OP_PREINC; // Will be handled as pre/post in main parsing
591 } else if (expr[0] == '-' && expr[1] == '-') {
592 return OP_PREDEC; // Will be handled as pre/post in main parsing
593 }
594
595 // Check two-character operators first to avoid conflicts
596 for (op_t *op = operators; op->op; op++) {
597 if (op->chars == 2) {
598 if (op->op == CH_EQ && expr[0] == '=' && expr[1] == '=') {
599 return op;
600 } else if (op->op == CH_NE && expr[0] == '!' && expr[1] == '=') {
601 return op;
602 } else if (op->op == CH_LE && expr[0] == '<' && expr[1] == '=') {
603 return op;
604 } else if (op->op == CH_GE && expr[0] == '>' && expr[1] == '=') {
605 return op;
606 } else if (op->op == CH_LSH && expr[0] == '<' && expr[1] == '<') {
607 return op;
608 } else if (op->op == CH_RSH && expr[0] == '>' && expr[1] == '>') {
609 return op;
610 } else if (op->op == CH_AND && expr[0] == '&' && expr[1] == '&') {
611 return op;
612 } else if (op->op == CH_OR && expr[0] == '|' && expr[1] == '|') {
613 return op;
614 } else if (op->op == CH_EXP && expr[0] == '*' && expr[1] == '*') {
615 return op;
616 } else if (op->op == CH_ADDEQ && expr[0] == '+' && expr[1] == '=') {
617 return op;
618 } else if (op->op == CH_SUBEQ && expr[0] == '-' && expr[1] == '=') {
619 return op;
620 } else if (op->op == CH_MULEQ && expr[0] == '*' && expr[1] == '=') {
621 return op;
622 } else if (op->op == CH_DIVEQ && expr[0] == '/' && expr[1] == '=') {
623 return op;
624 } else if (op->op == CH_MODEQ && expr[0] == '%' && expr[1] == '=') {
625 return op;
626 }
627 }
628 }
629
630 // Then check single-character operators
631 for (op_t *op = operators; op->op; op++) {
632 if (op->chars == 1 && *expr == op->op) {
633 return op;
634 }
635 }
636
637 return NULL;
638}

711// Shunting yard algorithm implementation
712static void shunt_op(arithm_context_t *ctx, op_t *op) {
713 if (op->op == '(') {
714 push_opstack(ctx, op);
715 } else if (op->op == ')') {
716 while (ctx->nopstack > 0 &&
717 ctx->opstack[ctx->nopstack - 1]->op != '(') {
718 op_t *pop_op = pop_opstack(ctx);
719 if (ctx->errflag) {
720 return;
721 }
722
723 stack_item_t a1 = pop_numstack(ctx);
724 if (ctx->errflag) {
725 return;
726 }
727
728 if (pop_op->unary) {
729 push_numstackl(ctx, pop_op->eval(&a1, NULL));
730 if (arithm_error_flag) {
731 ctx->errflag = true;
732 return;
733 }
734 } else {
735 stack_item_t a2 = pop_numstack(ctx);
736 if (ctx->errflag) {
737 return;
738 }
739 push_numstackl(ctx, pop_op->eval(&a2, &a1));
740 if (arithm_error_flag) {
741 ctx->errflag = true;
742 return;
743 }
744 }
745 if (ctx->errflag) {
746 return;
747 }
748 }
749
750 if (ctx->nopstack > 0 && ctx->opstack[ctx->nopstack - 1]->op == '(') {
751 pop_opstack(ctx); // Remove the '('
752 } else {
753 ctx->errflag = true;
754 arithm_set_error("mismatched parentheses");
755 }
756 } else {
757 while (ctx->nopstack > 0 &&
758 ctx->opstack[ctx->nopstack - 1]->op != '(' &&
759 ((op->assoc == ASSOC_LEFT &&
760 op->prec >= ctx->opstack[ctx->nopstack - 1]->prec) ||
761 (op->assoc == ASSOC_RIGHT &&
762 op->prec > ctx->opstack[ctx->nopstack - 1]->prec))) {
763
764 op_t *pop_op = pop_opstack(ctx);
765 if (ctx->errflag) {
766 return;
767 }
768
769 stack_item_t a1 = pop_numstack(ctx);
770 if (ctx->errflag) {
771 return;
772 }
773
774 if (pop_op->unary) {
775 push_numstackl(ctx, pop_op->eval(&a1, NULL));
776 if (arithm_error_flag) {
777 ctx->errflag = true;
778 return;
779 }
780 } else {
781 stack_item_t a2 = pop_numstack(ctx);
782 if (ctx->errflag) {
783 return;
784 }
785 push_numstackl(ctx, pop_op->eval(&a2, &a1));
786 if (arithm_error_flag) {
787 ctx->errflag = true;
788 return;
789 }
790 }
791 if (ctx->errflag) {
792 return;
793 }
794 }
795
796 push_opstack(ctx, op);
797 }
798}

834
835static char *arithm_expand_internal(void *executor, const char *orig_expr) {
836 if (!orig_expr) {
837 return strdup("0");
838 }
839
840 // Initialize context
841 arithm_context_t ctx = {0};
842 ctx.executor = executor; // Store executor context in arithmetic context
843 arithm_clear_error();
844
845 // Parse expression and remove $(( )) wrapper if present
846 const char *expr;
847 char *cleaned_expr = NULL;
848
849 if (strncmp(orig_expr, "$((", 3) == 0) {
850 size_t len = strlen(orig_expr);
851 if (len >= 5 && orig_expr[len - 2] == ')' &&
852 orig_expr[len - 1] == ')') {
853 // Extract expression between $(( and ))
854 size_t expr_len = len - 5;
855 cleaned_expr = malloc(expr_len + 1);
856 if (!cleaned_expr) {
857 arithm_set_error("memory allocation failed");
858 return NULL;
859 }
860 strncpy(cleaned_expr, orig_expr + 3, expr_len);
861 cleaned_expr[expr_len] = '\0';
862 expr = cleaned_expr;
863 } else {
864 arithm_set_error(
865 "malformed arithmetic expression: missing closing ))");
866 return NULL;
867 }
868 } else {
869 expr = orig_expr;
870 }
871
872 // Tokenize and evaluate expression using clearer state management
873 const char *current = expr;
874 op_t start_op = {'X', 0, ASSOC_NONE, 0, 0, NULL};
875 op_t *last_op = &start_op;
876
877 while (*current && !ctx.errflag) {
878 // Skip whitespace
879 while (*current && isspace(*current)) {
880 current++;
881 }
882
883 if (!*current) {
884 break;
885 }
886
887 // Try to parse an operator first
888 op_t *op = get_op(current);
889 if (op) {
890 // Handle increment/decrement operators
891 if (op == OP_PREINC || op == OP_PREDEC) {
892 // Check if this should be post-increment/decrement
893 // If the last token was a variable or closing paren, it's
894 // post-increment
895 if (ctx.nnumstack > 0 &&
896 (ctx.numstack[ctx.nnumstack - 1].type == ITEM_VAR_PTR ||
897 (last_op && last_op->op == ')'))) {
898 // Convert to post-increment/decrement
899 if (op == OP_PREINC) {
900 op = OP_POSTINC;
901 } else {
902 op = OP_POSTDEC;
903 }
904 }
905 }
906
907 // Handle unary operators
908 if (last_op && (last_op == &start_op || last_op->op != ')')) {
909 if (op->op == '-') {
910 op = OP_UMINUS;
911 } else if (op->op == '+') {
912 op = OP_UPLUS;
913 } else if (op->op != '(' && !op->unary) {
914 arithm_set_error("illegal use of binary operator");
915 break;
916 }
917 }
918
919 shunt_op(&ctx, op);
920 if (ctx.errflag) {
921 break;
922 }
923 last_op = op;
924 current += op->chars;
925 } else if (isdigit(*current)) {
926 // Parse number
927 int nchars;
928 ssize_t num = get_num(current, &nchars);
929 push_numstackl(&ctx, num);
930 if (ctx.errflag) {
931 break;
932 }
933 last_op = NULL;
934 current += nchars;
935 } else if (*current == '$' && *(current + 1) == '(') {
936 // Handle command substitution $(command) in arithmetic expressions
937 // Find the matching closing parenthesis
938 int paren_count = 1;
939 const char *start = current + 2; // Skip $(
940 const char *end = start;
941
942 while (*end && paren_count > 0) {
943 if (*end == '(') {
944 paren_count++;
945 } else if (*end == ')') {
946 paren_count--;
947 }
948 if (paren_count > 0) {
949 end++;
950 }
951 }
952
953 if (paren_count == 0 && *end == ')') {
954 // Extract command and execute it
955 size_t cmd_len = end - start;
956 char *command = malloc(cmd_len + 1);
957 if (command) {
958 strncpy(command, start, cmd_len);
959 command[cmd_len] = '\0';
960
961 // Simple implementation: handle basic echo commands
962 char *trimmed = command;
963 while (*trimmed && isspace(*trimmed)) {
964 trimmed++;
965 }
966
967 // Check if it's a simple echo number command
968 if (strncmp(trimmed, "echo ", 5) == 0) {
969 char *num_str = trimmed + 5;
970 while (*num_str && isspace(*num_str)) {
971 num_str++;
972 }
973 if (isdigit(*num_str) ||
974 (*num_str == '-' && isdigit(*(num_str + 1)))) {
975 long val = strtol(num_str, NULL, 10);
976 push_numstackl(&ctx, val);
977 } else {
978 push_numstackl(&ctx, 0);
979 }
980 } else {
981 // For other commands, default to 0 for now
982 push_numstackl(&ctx, 0);
983 }
984
985 free(command);
986 } else {
987 push_numstackl(&ctx, 0);
988 }
989
990 current = end + 1; // Skip past the closing )
991 } else {
992 // Malformed command substitution
993 arithm_set_error(
994 "malformed command substitution in arithmetic");
995 break;
996 }
997
998 if (ctx.errflag) {
999 break;
1000 }
1001 last_op = NULL;
1002 } else if (*current == '$' && valid_name_char(*(current + 1))) {
1003 // Handle $variable syntax in arithmetic expressions
1004 current++; // Skip the '$'
1005 int nchars;
1006 char *var_name = get_var_name_with_context(&ctx, current, &nchars);
1007 if (var_name) {
1008 push_numstackv_with_context(&ctx, var_name);
1009 free(var_name);
1010 } else {
1011 push_numstackl(&ctx, 0); // Undefined variable = 0
1012 }
1013 if (ctx.errflag) {
1014 break;
1015 }
1016 last_op = NULL;
1017 current += nchars;
1018 } else if (valid_name_char(*current)) {
1019 // Parse variable name
1020 int nchars;
1021 char *var_name = get_var_name_with_context(&ctx, current, &nchars);
1022 if (var_name) {
1023 push_numstackv_with_context(&ctx, var_name);
1024 free(var_name);
1025 } else {
1026 push_numstackl(&ctx, 0); // Undefined variable = 0
1027 }
1028 if (ctx.errflag) {
1029 break;
1030 }
1031 last_op = NULL;
1032 current += nchars;
1033 } else {
1034 arithm_set_error("syntax error in arithmetic expression");
1035 break;
1036 }
1037 }
1038
1039 // Process remaining operators
1040 while (ctx.nopstack > 0 && !ctx.errflag) {
1041 op_t *op = pop_opstack(&ctx);
1042 if (ctx.errflag) {
1043 break;
1044 }
1045
1046 stack_item_t a1 = pop_numstack(&ctx);
1047 if (ctx.errflag) {
1048 break;
1049 }
1050
1051 if (op->unary) {
1052 push_numstackl(&ctx, op->eval(&a1, NULL));
1053 if (arithm_error_flag) {
1054 ctx.errflag = true;
1055 break;
1056 }
1057 } else {
1058 stack_item_t a2 = pop_numstack(&ctx);
1059 if (ctx.errflag) {
1060 break;
1061 }
1062 push_numstackl(&ctx, op->eval(&a2, &a1));
1063 if (arithm_error_flag) {
1064 ctx.errflag = true;
1065 break;
1066 }
1067 }
1068 if (ctx.errflag) {
1069 break;
1070 }
1071 }
1072
1073 // Clean up
1074 if (cleaned_expr) {
1075 free(cleaned_expr);
1076 }
1077
1078 // Check for errors
1079 if (ctx.errflag || arithm_error_flag) {
1080 arithm_context_cleanup(&ctx);
1081 return NULL;
1082 }
1083
1084 // Should have exactly one result
1085 if (ctx.nnumstack != 1) {
1086 arithm_set_error("invalid arithmetic expression");
1087 arithm_context_cleanup(&ctx);
1088 return NULL;
1089 }
1090
1091 // Format result
1092 ssize_t result = long_value(&ctx.numstack[0]);
1093 char *result_str = malloc(32);
1094 if (!result_str) {
1095 arithm_set_error("memory allocation failed");
1096 arithm_context_cleanup(&ctx);
1097 return NULL;
1098 }
1099
1100 snprintf(result_str, 32, "%ld", result);
1101 arithm_context_cleanup(&ctx);
1102 return result_str;
1103}
