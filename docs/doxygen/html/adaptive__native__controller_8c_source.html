128 if (!state) {
129 return NULL;
130 }
131
132 /* Get terminal dimensions */
133 struct winsize ws;
134 if (ioctl(STDOUT_FILENO, TIOCGWINSZ, &ws) == 0) {
135 state->terminal_width = ws.ws_col;
136 state->terminal_height = ws.ws_row;
137 } else {
138 state->terminal_width = 80;
139 state->terminal_height = 24;
140 }
141
142 state->cursor_row = 0;
143 state->cursor_col = 0;
144 state->current_fg_color = -1;
145 state->current_bg_color = -1;
146
147 return state;
148}

201/**
202 * Enter raw mode.
203 */
204static lle_result_t lle_native_enter_raw_mode(lle_native_controller_t *native) {
205 if (native->raw_mode_active) {
206 return LLE_SUCCESS;
207 }
208
209 /* Save original terminal settings */
210 if (!native->termios_saved) {
211 if (tcgetattr(STDIN_FILENO, &native->original_termios) != 0) {
212 return LLE_ERROR_TERMINAL_ABSTRACTION;
213 }
214 native->termios_saved = true;
215 }
216
217 /* Configure raw mode */
218 native->raw_termios = native->original_termios;
219 native->raw_termios.c_iflag &= ~(BRKINT | ICRNL | INPCK | ISTRIP | IXON);
220 native->raw_termios.c_oflag &= ~(OPOST);
221 native->raw_termios.c_cflag |= (CS8);
222 native->raw_termios.c_lflag &= ~(ECHO | ICANON | IEXTEN | ISIG);
223 native->raw_termios.c_cc[VMIN] = 1;
224 native->raw_termios.c_cc[VTIME] = 0;
225
226 /* Apply raw mode */
227 if (tcsetattr(STDIN_FILENO, TCSAFLUSH, &native->raw_termios) != 0) {
228 return LLE_ERROR_TERMINAL_ABSTRACTION;
229 }
230
231 native->raw_mode_active = true;
232 native->perf_stats->raw_mode_toggles++;
233
234 return LLE_SUCCESS;
235}
236
237/**
238 * Exit raw mode.
239 */
240static lle_result_t lle_native_exit_raw_mode(lle_native_controller_t *native) {
241 if (!native->raw_mode_active || !native->termios_saved) {
242 return LLE_SUCCESS;
243 }
244
245 /* Restore original terminal settings */
246 if (tcsetattr(STDIN_FILENO, TCSAFLUSH, &native->original_termios) != 0) {
247 return LLE_ERROR_TERMINAL_ABSTRACTION;
248 }
249
250 native->raw_mode_active = false;
251 native->perf_stats->raw_mode_toggles++;
252
253 return LLE_SUCCESS;
254}

261/**
262 * Append to output buffer.
263 */
264static lle_result_t lle_native_buffer_append(lle_native_controller_t *native,
265 const char *data, size_t length) {
266
267 /* Ensure capacity */
268 size_t needed = native->buffer_used + length;
269 if (needed >= native->buffer_capacity) {
270 size_t new_capacity = native->buffer_capacity;
271 while (new_capacity < needed + 1) {
272 new_capacity *= 2;
273 }
274 char *new_buffer = realloc(native->output_buffer, new_capacity);
275 if (!new_buffer) {
276 return LLE_ERROR_OUT_OF_MEMORY;
277 }
278 native->output_buffer = new_buffer;
279 native->buffer_capacity = new_capacity;
280 }
281
282 memcpy(native->output_buffer + native->buffer_used, data, length);
283 native->buffer_used += length;
284 native->output_buffer[native->buffer_used] = '\0';
285
286 return LLE_SUCCESS;
287}

289/**
290 * Flush output buffer.
291 */
292static lle_result_t lle_native_buffer_flush(lle_native_controller_t *native) {
293 if (native->buffer_used == 0) {
294 return LLE_SUCCESS;
295 }
296
297 ssize_t written =
298 write(STDOUT_FILENO, native->output_buffer, native->buffer_used);
299 if (written < 0) {
300 return LLE_ERROR_TERMINAL_ABSTRACTION;
301 }
302
303 native->perf_stats->total_bytes_written += written;
304 native->buffer_used = 0;
305
306 return LLE_SUCCESS;
307}

328 char seq[32];

447/**

450static lle_result_t lle_apply_capability_optimizations(
451 lle_native_controller_t *native,
452 const lle_terminal_detection_result_t *detection) {
453
454 native->optimization_flags = LLE_NATIVE_OPT_NONE;
455
456 if (detection->supports_cursor_queries) {
457 native->optimization_flags |= LLE_NATIVE_OPT_FAST_CURSOR_QUERIES;
458 }
459
460 if (detection->supports_256_colors) {
461 native->optimization_flags |= LLE_NATIVE_OPT_EXTENDED_COLOR_CACHE;
462 }
463
464 if (detection->capability_level >= LLE_CAPABILITY_FULL) {
465 native->optimization_flags |= LLE_NATIVE_OPT_ADVANCED_SEQUENCES;
466 }
467
468 if (detection->supports_mouse) {
469 native->optimization_flags |= LLE_NATIVE_OPT_MOUSE_TRACKING;
470 }
471
472 if (detection->supports_bracketed_paste) {
473 native->optimization_flags |= LLE_NATIVE_OPT_BRACKETED_PASTE;
474 }
475
476 return LLE_SUCCESS;
477}

479/**
480 * Calculate optimal buffer size based on terminal size.
481 */
482static size_t lle_calculate_optimal_buffer_size(
483 const lle_terminal_detection_result_t *detection) {
484
485 /* Base size: 16KB */
486 size_t base_size = 16384;
487
488 /* Adjust based on capability level */
489 switch (detection->capability_level) {
490 case LLE_CAPABILITY_PREMIUM:
491 return base_size * 2;
492 case LLE_CAPABILITY_FULL:
493 return base_size;
494 case LLE_CAPABILITY_STANDARD:
495 return base_size / 2;
496 default:
497 return base_size / 4;
498 }
499}

506/**
507 * Initialize native terminal controller.
508 */
509lle_result_t
510lle_initialize_native_controller(lle_adaptive_context_t *context,
511 lusush_memory_pool_t *memory_pool) {
512
513 lle_native_controller_t *native =
514 calloc(1, sizeof(lle_native_controller_t));
515 if (!native) {
516 return LLE_ERROR_OUT_OF_MEMORY;
517 }
518
519 /* Create terminal state */
520 native->terminal_state = lle_terminal_state_create();
521 if (!native->terminal_state) {
522 free(native);
523 return LLE_ERROR_OUT_OF_MEMORY;
524 }
525
526 /* Create capabilities structure */
527 native->capabilities = calloc(1, sizeof(lle_native_capabilities_t));
528 if (!native->capabilities) {
529 lle_terminal_state_destroy(native->terminal_state);
530 free(native);
531 return LLE_ERROR_OUT_OF_MEMORY;
532 }
533
534 /* Configure capabilities from detection */
535 native->capabilities->has_colors =
536 context->detection_result->supports_colors;
537 native->capabilities->has_256_colors =
538 context->detection_result->supports_256_colors;
539 native->capabilities->has_truecolor =
540 context->detection_result->supports_truecolor;
541 native->capabilities->has_cursor_positioning =
542 context->detection_result->supports_cursor_positioning;
543 native->capabilities->has_cursor_queries =
544 context->detection_result->supports_cursor_queries;
545 native->capabilities->has_mouse_support =
546 context->detection_result->supports_mouse;
547 native->capabilities->has_bracketed_paste =
548 context->detection_result->supports_bracketed_paste;
549 native->capabilities->has_unicode =
550 context->detection_result->supports_unicode;
551
552 /* Apply capability optimizations */
553 lle_result_t result =
554 lle_apply_capability_optimizations(native, context->detection_result);
555 if (result != LLE_SUCCESS) {
556 free(native->capabilities);
557 lle_terminal_state_destroy(native->terminal_state);
558 free(native);
559 return result;
560 }
561
562 /* Create output buffer */
563 native->buffer_capacity =
564 lle_calculate_optimal_buffer_size(context->detection_result);
565 native->output_buffer = malloc(native->buffer_capacity);
566 if (!native->output_buffer) {
567 free(native->capabilities);
568 lle_terminal_state_destroy(native->terminal_state);
569 free(native);
570 return LLE_ERROR_OUT_OF_MEMORY;
571 }
572
573 /* Create performance statistics */
574 native->perf_stats = lle_terminal_performance_stats_create();
575 if (!native->perf_stats) {
576 free(native->output_buffer);
577 free(native->capabilities);
578 lle_terminal_state_destroy(native->terminal_state);
579 free(native);
580 return LLE_ERROR_OUT_OF_MEMORY;
581 }
582
583 native->memory_pool = memory_pool;
584 context->controller.native = native;
585
586 return LLE_SUCCESS;
587}

609/**
610 * Read line using native controller.
611 */
612lle_result_t lle_native_read_line(lle_native_controller_t *native,
613 const char *prompt, char **line) {
614
615 if (!native || !prompt || !line) {
616 return LLE_ERROR_INVALID_PARAMETER;
617 }
618
619 /* Enter raw mode */
620 lle_result_t result = lle_native_enter_raw_mode(native);
621 if (result != LLE_SUCCESS) {
622 return result;
623 }
624
625 /* Display prompt with color */
626 lle_native_buffer_clear(native);
627 if (native->capabilities->has_colors) {
628 lle_native_set_fg_color(native, 2); /* Green */
629 }
630 lle_native_buffer_append(native, prompt, strlen(prompt));
631 if (native->capabilities->has_colors) {
632 lle_native_reset_formatting(native);
633 }
634 lle_native_buffer_flush(native);
635
636 /* Read line using simple fgets for now (full implementation would use raw
637 * input) */
638 lle_native_exit_raw_mode(native);
639
640 char buffer[4096];
641 if (!fgets(buffer, sizeof(buffer), stdin)) {
642 if (feof(stdin)) {
643 return LLE_ERROR_OUT_OF_MEMORY; /* EOF treated as error */
644 }
645 return LLE_ERROR_INPUT_PARSING;
646 }
647
648 /* Remove trailing newline */
649 size_t len = strlen(buffer);
650 if (len > 0 && buffer[len - 1] == '\n') {
651 buffer[len - 1] = '\0';
652 }
653
654 *line = strdup(buffer);
655 if (!*line) {
656 return LLE_ERROR_OUT_OF_MEMORY;
657 }
658
659 return LLE_SUCCESS;
660}
