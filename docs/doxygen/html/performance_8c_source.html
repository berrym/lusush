54/**
55 * @brief Update running statistics with new measurement
56 */
57static void lle_perf_stats_update(lle_perf_statistics_t *stats,
58 uint64_t duration_ns, bool success) {
59 stats->call_count++;
60
61 if (success) {
62 stats->completed_count++;
63 stats->consecutive_successes++;
64 if (stats->consecutive_successes > stats->max_consecutive_successes) {
65 stats->max_consecutive_successes = stats->consecutive_successes;
66 }
67 } else {
68 stats->error_count++;
69 stats->consecutive_successes = 0;
70 }
71
72 stats->total_duration_ns += duration_ns;
73
74 if (duration_ns < stats->min_duration_ns) {
75 stats->min_duration_ns = duration_ns;
76 }
77
78 if (duration_ns > stats->max_duration_ns) {
79 stats->max_duration_ns = duration_ns;
80 }
81
82 /* Update mean */
83 if (stats->completed_count > 0) {
84 stats->mean_duration_ns =
85 stats->total_duration_ns / stats->completed_count;
86 }
87
88 /* Calculate reliability score */
89 if (stats->call_count > 0) {
90 stats->reliability_score =
91 (double)stats->completed_count / (double)stats->call_count;
92 }
93}

100/**
101 * @brief Initialize performance monitor
102 */
103lle_result_t lle_perf_monitor_init(lle_performance_monitor_t *monitor,
104 lle_performance_config_t *config) {
105 if (!monitor) {
106 return LLE_ERROR_INVALID_PARAMETER;
107 }
108
109 /* Zero-initialize the entire structure */
110 memset(monitor, 0, sizeof(lle_performance_monitor_t));
111
112 /* Set configuration from config or defaults */
113 if (config) {
114 monitor->monitoring_enabled = config->monitoring_enabled;
115 monitor->monitoring_level = config->monitoring_level;
116 monitor->warning_threshold_ns = config->warning_threshold_ns;
117 monitor->critical_threshold_ns = config->critical_threshold_ns;
118
119 /* Initialize filters from default */
120 monitor->active_filters = config->default_filters;
121 } else {
122 /* Use defaults */
123 monitor->monitoring_enabled = true;
124 monitor->monitoring_level = LLE_PERF_MONITORING_NORMAL;
125 monitor->warning_threshold_ns = LLE_PERF_TARGET_RESPONSE_TIME_NS;
126 monitor->critical_threshold_ns = LLE_PERF_TARGET_RESPONSE_TIME_NS * 2;
127 }
128
129 /* Initialize all operation statistics */
130 lle_perf_stats_init(&monitor->global_stats);
131 for (int i = 0; i < LLE_PERF_OPERATION_COUNT; i++) {
132 lle_perf_stats_init(&monitor->operation_stats[i]);
133 }
134
135 /* Initialize history ring buffer */
136 monitor->history_ring.capacity = 100; /* Fixed size for Phase 1 */
137 monitor->history_ring.entries =
138 calloc(monitor->history_ring.capacity, sizeof(lle_perf_statistics_t));
139 if (!monitor->history_ring.entries) {
140 return LLE_ERROR_OUT_OF_MEMORY;
141 }
142
143 monitor->measurement_count = 0;
144 monitor->measurement_index = 0;
145 monitor->total_operations = 0;
146
147 return LLE_SUCCESS;
148}

170/**
171 * @brief Calculate time difference in nanoseconds
172 */
173uint64_t lle_perf_timespec_diff_ns(const struct timespec *start,
174 const struct timespec *end) {
175 uint64_t start_ns =
176 (uint64_t)start->tv_sec * 1000000000ULL + (uint64_t)start->tv_nsec;
177 uint64_t end_ns =
178 (uint64_t)end->tv_sec * 1000000000ULL + (uint64_t)end->tv_nsec;
179
180 if (end_ns > start_ns) {
181 return end_ns - start_ns;
182 }
183 return 0;
184}

186/**
187 * @brief Check if operation should be monitored based on filters
188 */
189bool lle_perf_should_monitor_operation(lle_performance_monitor_t *monitor,
190 lle_perf_operation_type_t op_type) {
191 if (!monitor || !monitor->monitoring_enabled) {
192 return false;
193 }
194
195 /* Check monitoring level */
196 if (monitor->monitoring_level == LLE_PERF_MONITORING_DISABLED) {
197 return false;
198 }
199
200 /* Check operation filter */
201 if (monitor->active_filters.filter_by_operation) {
202 if (op_type >= LLE_PERF_OPERATION_COUNT) {
203 return false;
204 }
205
206 uint32_t op_bit = 1U << op_type;
207 if (!(monitor->active_filters.operation_mask & op_bit)) {
208 return false;
209 }
210 }
211
212 return true;
213}

233/**
234 * @brief Start a performance measurement
235 */
236lle_result_t lle_perf_measurement_start(
237 lle_performance_monitor_t *monitor, lle_perf_operation_type_t op_type,
238 const char *op_name, lle_performance_context_t *context,
239 lle_perf_measurement_id_t *measurement_id) {
240 if (!monitor || !op_name || !measurement_id) {
241 return LLE_ERROR_INVALID_PARAMETER;
242 }
243
244 /* Check if monitoring is enabled */
245 if (!monitor->monitoring_enabled) {
246 *measurement_id = (lle_perf_measurement_id_t)-1;
247 return LLE_SUCCESS;
248 }
249
250 /* Check if we should monitor this operation */
251 if (!lle_perf_should_monitor_operation(monitor, op_type)) {
252 *measurement_id = (lle_perf_measurement_id_t)-1;
253 return LLE_SUCCESS;
254 }
255
256 /* Validate operation type */
257 if (op_type >= LLE_PERF_OPERATION_COUNT) {
258 return LLE_ERROR_INVALID_PARAMETER;
259 }
260
261 /* Allocate measurement slot (atomic circular buffer) */
262 uint32_t index =
263 __atomic_fetch_add(&monitor->measurement_index, 1, __ATOMIC_RELAXED);
264 index %= LLE_PERF_MAX_MEASUREMENTS;
265
266 lle_performance_measurement_t *measurement = &monitor->measurements[index];
267
268 /* Initialize measurement */
269 memset(measurement, 0, sizeof(lle_performance_measurement_t));
270 measurement->operation_type = op_type;
271 measurement->operation_name = op_name;
272
273 if (context) {
274 measurement->context = *context;
275 }
276
277 measurement->thread_id = perf_get_thread_id();
278 measurement->is_critical_path = lle_perf_is_critical_path(op_type);
279
280 /* High-precision timing start */
281 if (clock_gettime(CLOCK_MONOTONIC, &measurement->start_time) != 0) {
282 return LLE_ERROR_SYSTEM_CALL;
283 }
284
285 /* Increment total operations counter */
286 __atomic_fetch_add(&monitor->total_operations, 1, __ATOMIC_RELAXED);
287
288 *measurement_id = index;
289 return LLE_SUCCESS;
290}

292/**
293 * @brief End a performance measurement
294 */
295lle_result_t lle_perf_measurement_end(lle_performance_monitor_t *monitor,
296 lle_perf_measurement_id_t measurement_id,
297 lle_perf_result_t result_code) {
298 if (!monitor) {
299 return LLE_ERROR_INVALID_PARAMETER;
300 }
301
302 /* Check for disabled monitoring sentinel */
303 if (measurement_id == (lle_perf_measurement_id_t)-1) {
304 return LLE_SUCCESS;
305 }
306
307 /* Validate measurement ID */
308 if (measurement_id >= LLE_PERF_MAX_MEASUREMENTS) {
309 return LLE_ERROR_INVALID_PARAMETER;
310 }
311
312 lle_performance_measurement_t *measurement =
313 &monitor->measurements[measurement_id];
314
315 /* High-precision timing end */
316 if (clock_gettime(CLOCK_MONOTONIC, &measurement->end_time) != 0) {
317 return LLE_ERROR_SYSTEM_CALL;
318 }
319
320 /* Calculate duration */
321 measurement->duration_ns = lle_perf_timespec_diff_ns(
322 &measurement->start_time, &measurement->end_time);
323
324 /* Determine success */
325 bool success = (result_code == LLE_PERF_RESULT_SUCCESS);
326
327 /* Update global statistics */
328 lle_perf_stats_update(&monitor->global_stats, measurement->duration_ns,
329 success);
330
331 /* Update operation-specific statistics */
332 if (measurement->operation_type < LLE_PERF_OPERATION_COUNT) {
333 lle_perf_stats_update(
334 &monitor->operation_stats[measurement->operation_type],
335 measurement->duration_ns, success);
336 }
337
338 /* Check thresholds */
339 if (measurement->duration_ns >= monitor->critical_threshold_ns) {
340 lle_perf_handle_critical_threshold_exceeded(monitor, measurement);
341 } else if (measurement->duration_ns >= monitor->warning_threshold_ns) {
342 lle_perf_handle_warning_threshold_exceeded(monitor, measurement);
343 }
344
345 /* Increment measurement count */
346 __atomic_fetch_add(&monitor->measurement_count, 1, __ATOMIC_RELAXED);
347
348 return LLE_SUCCESS;
349}

378/**
379 * @brief Calculate advanced statistics (percentiles, std dev)
380 */
381lle_result_t lle_perf_calculate_statistics(lle_performance_monitor_t *monitor,
382 lle_perf_operation_type_t op_type,
383 lle_perf_statistics_t *stats) {
384 if (!monitor || !stats) {
385 return LLE_ERROR_INVALID_PARAMETER;
386 }
387
388 /* First get basic statistics */
389 lle_result_t result = lle_perf_get_statistics(monitor, op_type, stats);
390 if (result != LLE_SUCCESS) {
391 return result;
392 }
393
394 /* Collect duration samples for percentile calculation */
395 uint64_t *durations = calloc(LLE_PERF_MAX_MEASUREMENTS, sizeof(uint64_t));
396 if (!durations) {
397 return LLE_ERROR_OUT_OF_MEMORY;
398 }
399
400 size_t sample_count = 0;
401 uint64_t sum_squared_diff = 0;
402
403 /* Gather samples from measurements array */
404 for (uint32_t i = 0; i < LLE_PERF_MAX_MEASUREMENTS; i++) {
405 lle_performance_measurement_t *m = &monitor->measurements[i];
406
407 /* Filter by operation type */
408 if (op_type != LLE_PERF_OP_COUNT && m->operation_type != op_type) {
409 continue;
410 }
411
412 /* Skip if no duration recorded */
413 if (m->duration_ns == 0) {
414 continue;
415 }
416
417 durations[sample_count++] = m->duration_ns;
418
419 /* Calculate variance for standard deviation */
420 if (stats->mean_duration_ns > 0) {
421 int64_t diff =
422 (int64_t)m->duration_ns - (int64_t)stats->mean_duration_ns;
423 sum_squared_diff += (uint64_t)(diff * diff);
424 }
425 }
426
427 if (sample_count == 0) {
428 free(durations);
429 return LLE_SUCCESS; /* No samples, return basic stats */
430 }
431
432 /* Sort durations for percentile calculation */
433 for (size_t i = 0; i < sample_count - 1; i++) {
434 for (size_t j = i + 1; j < sample_count; j++) {
435 if (durations[i] > durations[j]) {
436 uint64_t temp = durations[i];
437 durations[i] = durations[j];
438 durations[j] = temp;
439 }
440 }
441 }
442
443 /* Calculate percentiles */
444 stats->median_duration_ns = durations[sample_count / 2];
445 stats->p95_duration_ns = durations[(sample_count * 95) / 100];
446 stats->p99_duration_ns = durations[(sample_count * 99) / 100];
447
448 /* Calculate standard deviation */
449 if (sample_count > 1) {
450 uint64_t variance = sum_squared_diff / (sample_count - 1);
451 /* Simple integer square root */
452 uint64_t sqrt_var = 0;
453 uint64_t bit = 1ULL << 30;
454 while (bit > variance) {
455 bit >>= 2;
456 }
457 while (bit != 0) {
458 if (variance >= sqrt_var + bit) {
459 variance -= sqrt_var + bit;
460 sqrt_var = (sqrt_var >> 1) + bit;
461 } else {
462 sqrt_var >>= 1;
463 }
464 bit >>= 2;
465 }
466 stats->standard_deviation_ns = sqrt_var;
467 }
468
469 free(durations);
470 return LLE_SUCCESS;
471}

473/**
474 * @brief Record statistics in history ring buffer
475 */
476lle_result_t
477lle_perf_history_record(lle_performance_monitor_t *monitor,
478 lle_performance_measurement_t *measurement) {
479 if (!monitor || !measurement) {
480 return LLE_ERROR_INVALID_PARAMETER;
481 }
482
483 if (!monitor->history_ring.entries) {
484 return LLE_ERROR_INVALID_STATE;
485 }
486
487 /* Convert measurement to statistics and add to ring buffer */
488 uint32_t write_index = monitor->history_ring.head;
489
490 /* Create stats from measurement */
491 lle_perf_statistics_t stats = {0};
492 stats.call_count = 1;
493 stats.total_duration_ns = measurement->duration_ns;
494 stats.min_duration_ns = measurement->duration_ns;
495 stats.max_duration_ns = measurement->duration_ns;
496 stats.mean_duration_ns = measurement->duration_ns;
497
498 monitor->history_ring.entries[write_index] = stats;
499
500 /* Update head */
501 monitor->history_ring.head =
502 (write_index + 1) % monitor->history_ring.capacity;
503
504 /* Update size and tail */
505 if (monitor->history_ring.size < monitor->history_ring.capacity) {
506 monitor->history_ring.size++;
507 } else {
508 /* Ring buffer is full, move tail */
509 monitor->history_ring.tail =
510 (monitor->history_ring.tail + 1) % monitor->history_ring.capacity;
511 }
512
513 monitor->history_ring.total_entries_recorded++;
514
515 return LLE_SUCCESS;
516}
