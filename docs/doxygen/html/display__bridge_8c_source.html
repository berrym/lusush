52/**
53 * @brief Initialize display bridge for LLE-Lusush communication
54 *
55 * Creates and initializes a display bridge that connects the LLE editing
56 * system with the Lusush display controller. The bridge manages event flow,
57 * render request queuing, and state synchronization.
58 *
59 * @param bridge Output pointer to receive initialized bridge
60 * @param editor LLE editor context (opaque pointer)
61 * @param display Lusush display controller
62 * @param memory_pool Memory pool for allocations
63 * @return LLE_SUCCESS on success, error code on failure
64 *
65 * SPEC COMPLIANCE: Section 3.2 "Display Bridge Implementation"
66 * ERROR HANDLING: All allocation failures handled, proper cleanup on error
67 */
68lle_result_t lle_display_bridge_init(lle_display_bridge_t **bridge,
69 void *editor,
70 display_controller_t *display,
71 lle_memory_pool_t *memory_pool) {
72 lle_result_t result = LLE_SUCCESS;
73 lle_display_bridge_t *br = NULL;
74
75 /* Step 1: Validate parameters */
76 if (!bridge) {
77 return LLE_ERROR_INVALID_PARAMETER;
78 }
79 /* Note: editor can be NULL - it will be set later per readline call */
80 if (!display || !memory_pool) {
81 return LLE_ERROR_INVALID_PARAMETER;
82 }
83
84 /* Step 2: Allocate bridge structure from memory pool */
85 br = lle_pool_alloc(sizeof(lle_display_bridge_t));
86 if (!br) {
87 return LLE_ERROR_OUT_OF_MEMORY;
88 }
89 memset(br, 0, sizeof(lle_display_bridge_t));
90
91 /* Step 3: Connect to LLE systems */
92 /* NOTE: editor is opaque until lle_editor_t is fully defined in future spec
93 */
94 /* For now, we store the reference but don't dereference it */
95 br->lle_event_manager =
96 editor; /* Store editor as event manager reference */
97 br->active_buffer = NULL; /* Will be set when editor type is defined */
98 br->cursor_pos = NULL; /* Will be set when editor type is defined */
99
100 /* Step 4: Connect to Lusush display systems */
101 /* Store display controller reference for terminal info access */
102 br->display_controller = display;
103
104 /* Get composition engine and event system from display controller */
105 br->composition_engine = display->compositor;
106
107 if (!br->composition_engine) {
108 lle_pool_free(br);
109 return LLE_ERROR_INVALID_STATE;
110 }
111
112 /* Get event system from compositor */
113 br->layer_events = br->composition_engine->event_system;
114
115 /* Get command_layer from composition_engine */
116 br->command_layer = (void *)br->composition_engine->command_layer;
117 if (!br->command_layer) {
118 lle_pool_free(br);
119 return LLE_ERROR_INVALID_STATE;
120 }
121
122 /* Step 5: Initialize synchronization state */
123 br->sync_state = LLE_DISPLAY_SYNC_IDLE;
124 br->force_full_render = true; /* Initial render must be complete */
125 br->render_skip_count = 0;
126 br->consecutive_errors = 0;
127
128 /* Step 6: Initialize render request queue */
129 result = lle_render_queue_init(&br->render_queue, memory_pool);
130 if (result != LLE_SUCCESS) {
131 lle_pool_free(br);
132 return result;
133 }
134
135 /* Step 7: Initialize display difference tracking */
136 result = lle_display_diff_init(&br->diff_tracker, memory_pool);
137 if (result != LLE_SUCCESS) {
138 lle_render_queue_cleanup(br->render_queue);
139 lle_pool_free(br);
140 return result;
141 }
142
143 /* Step 8: Initialize error context for bridge */
144 br->error_context = LLE_CREATE_ERROR_CONTEXT(
145 LLE_SUCCESS, "Display Bridge initialization", "display_bridge");
146 if (!br->error_context) {
147 lle_display_diff_cleanup(br->diff_tracker);
148 lle_render_queue_cleanup(br->render_queue);
149 lle_pool_free(br);
150 return LLE_ERROR_OUT_OF_MEMORY;
151 }
152
153 /* Step 9: Set initial timestamp */
154 if (clock_gettime(CLOCK_MONOTONIC, &br->last_render_time) != 0) {
155 /* If CLOCK_MONOTONIC fails, use realtime as fallback */
156 clock_gettime(CLOCK_REALTIME, &br->last_render_time);
157 }
158
159 /* Success - return initialized bridge */
160 *bridge = br;
161 return LLE_SUCCESS;
162}

166 *
167 * Releases all resources associated with the display bridge including
168 * render queue, diff tracker, and error context. The bridge pointer

170 *
171 * @param bridge Display bridge to clean up
172 * @return LLE_SUCCESS on success, error code on failure

174 * SPEC COMPLIANCE: Standard cleanup pattern
175 * ERROR HANDLING: Handles NULL gracefully, cleans up in reverse order

178 if (!bridge) {
179 return LLE_ERROR_INVALID_PARAMETER;
180 }
181
182 /* Clean up in reverse order of initialization */
183
184 /* Step 1: Clean up error context */
185 if (bridge->error_context) {
186 lle_pool_free(bridge->error_context);
187 bridge->error_context = NULL;
188 }
189
190 /* Step 2: Clean up display diff tracker */

195
196 /* Step 3: Clean up render request queue */

199 bridge->render_queue = NULL;
200 }
201
202 /* Step 4: Clear references (don't free - not owned by bridge) */
203 bridge->display_controller = NULL;
204 bridge->composition_engine = NULL;
205 bridge->layer_events = NULL;
206 bridge->command_layer = NULL;
207 bridge->lle_event_manager = NULL;
208 bridge->active_buffer = NULL;
209 bridge->cursor_pos = NULL;
210
211 /* Note: The bridge structure itself is not freed here.
212 * It should be freed by the caller using the same memory pool
213 * that was used to allocate it. */
214
215 return LLE_SUCCESS;
216}

237/**
238 * @brief Send rendered output to Lusush display system
239 *
240 * This is the critical bridge function that takes LLE's rendered output
241 * and sends it through Lusush's layered display system.
242 *
243 * Architecture:
244 * 1. Calculate cursor screen position using incremental tracking (Replxx
245 * approach)
246 * 2. Update render_output with cursor screen coordinates
247 * 3. Update command_layer with rendered text
248 * 4. command_layer publishes REDRAW_NEEDED event
249 * 5. display_controller handles event and renders to terminal
250 *
251 * CRITICAL: Per MODERN_EDITOR_WRAPPING_RESEARCH.md, we use incremental cursor
252 * tracking during rendering (like Replxx, Fish, ZLE) NOT division/modulo
253 * calculation.
254 *
255 * @param bridge Display bridge instance
256 * @param render_output Rendered output from LLE render system
257 * @param cursor Cursor position
258 * @return LLE_SUCCESS on success, error code on failure
259 */
260lle_result_t lle_display_bridge_send_output(lle_display_bridge_t *bridge,
261 lle_render_output_t *render_output,
262 lle_cursor_position_t *cursor) {
263 if (!bridge) {
264 return LLE_ERROR_INVALID_PARAMETER;
265 }
266
267 /* Get command layer */
268 command_layer_t *cmd_layer = (command_layer_t *)bridge->command_layer;
269 if (!cmd_layer) {
270 return LLE_ERROR_INVALID_STATE;
271 }
272
273 /* Extract command text and cursor position from render output */
274 const char *command_text = "";
275 size_t cursor_pos = 0;
276
277 if (render_output && render_output->content) {
278 command_text = render_output->content;
279 }
280
281 /* Calculate cursor screen position using incremental tracking
282 *
283 * Per MODERN_EDITOR_WRAPPING_RESEARCH.md, modern editors (Replxx, Fish,
284 * ZLE) calculate cursor position incrementally during rendering, NOT via
285 * division/modulo.
286 *
287 * This handles:
288 * - Multi-byte UTF-8 characters
289 * - Wide characters (2 columns)
290 * - Line wrapping at terminal boundaries
291 * - ANSI escape sequences (0 width)
292 */
293 if (cursor && cursor->position_valid && render_output) {
294 cursor_pos = cursor->byte_offset;
295
296 /* Get actual terminal width from display controller */
297 size_t terminal_width = 80; /* Default fallback */
298 if (bridge->display_controller &&
299 bridge->display_controller->terminal_ctrl) {
300 int width = bridge->display_controller->terminal_ctrl->capabilities
301 .terminal_width;
302 if (width > 0) {
303 terminal_width = (size_t)width;
304 }
305 }
306
307 /* Get prompt width from composition engine
308 * Per MODERN_EDITOR_WRAPPING_RESEARCH.md, we start at prompt_indent
309 * like Replxx */
310 size_t prompt_width = 0;
311 if (bridge->composition_engine &&
312 bridge->composition_engine->prompt_layer) {
313 prompt_layer_t *prompt_layer =
314 bridge->composition_engine->prompt_layer;
315 char prompt_buffer[4096] = {0}; /* PROMPT_LAYER_MAX_CONTENT_SIZE */
316 prompt_layer_get_rendered_content(prompt_layer, prompt_buffer,
317 sizeof(prompt_buffer));
318
319 /* Calculate visual width (excluding ANSI codes and readline
320 * markers) */
321 bool in_escape = false;
322 for (const char *p = prompt_buffer; *p; p++) {
323 if (*p == '\001' || *p == '\002')
324 continue; /* Skip readline markers */
325 if (*p == '\033') {
326 in_escape = true;
327 continue;
328 }
329 if (in_escape) {
330 if ((*p >= 'A' && *p <= 'Z') || (*p >= 'a' && *p <= 'z') ||
331 *p == 'm') {
332 in_escape = false;
333 }
334 continue;
335 }
336 if (*p >= 32)
337 prompt_width++;
338 }
339 }
340
341 /* IMPORTANT: Per Replxx approach, LLE calculates cursor position
342 * starting from prompt_width (like Replxx's prompt_indent), so first
343 * line has less space available. This gives us ABSOLUTE screen
344 * coordinates, not relative to command start. */
345
346 /* Calculate cursor screen position using incremental tracking
347 * Starting from prompt_width (where command actually starts on screen)
348 */
349 size_t cursor_row = 0, cursor_col = 0;
350 calculate_cursor_screen_position(
351 command_text, cursor_pos,
352 prompt_width, /* Start at prompt position - gives absolute
353 coordinates */
354 terminal_width, &cursor_row, &cursor_col);
355
356 /* DEBUG: Log what we calculated */
357 /* Temporarily disabled - interferes with display
358 fprintf(stderr, "[BRIDGE_DEBUG] Calculated cursor position: row=%zu,
359 col=%zu (cursor_pos=%zu, prompt_width=%zu, term_width=%zu)\n",
360 cursor_row, cursor_col, cursor_pos, prompt_width,
361 terminal_width); fflush(stderr);
362 */
363
364 /* Store in render output for display system to use */
365 render_output->cursor_screen_row = cursor_row;
366 render_output->cursor_screen_column = cursor_col;
367 render_output->cursor_position_valid = true;
368
369 /* Also store in command layer for display_controller to access */
370 cmd_layer->cursor_screen_row = cursor_row;
371 cmd_layer->cursor_screen_column = cursor_col;
372 cmd_layer->cursor_screen_position_valid = true;
373 } else if (render_output) {
374 render_output->cursor_position_valid = false;
375 cmd_layer->cursor_screen_position_valid = false;
376 }
377
378 /* Update command layer with new text and cursor position
379 * This performs syntax highlighting and publishes REDRAW_NEEDED event */
380 command_layer_error_t error =
381 command_layer_set_command(cmd_layer, command_text, cursor_pos);
382
383 if (error != COMMAND_LAYER_SUCCESS) {
384 bridge->consecutive_errors++;
385 return LLE_ERROR_DISPLAY_INTEGRATION;
386 }
387
388 /* Process layer events to trigger display update
389 * This causes display_controller to handle REDRAW_NEEDED event
390 */
391 if (bridge->layer_events) {
392 layer_events_process_pending(bridge->layer_events, 10, 0);
393 }
394
395 bridge->consecutive_errors = 0;
396 bridge->sync_state = LLE_DISPLAY_SYNC_COMPLETE;
397
398 return LLE_SUCCESS;
399}

405/**
406 * @brief Initialize render request queue
407 *
408 * Creates and initializes a queue for managing pending render requests.
409 * The queue uses a circular buffer with thread-safe access.
410 *
411 * @param queue Output pointer to receive initialized queue
412 * @param memory_pool Memory pool for allocations
413 * @return LLE_SUCCESS on success, error code on failure
414 */
415static lle_result_t lle_render_queue_init(lle_coord_queue_t **queue,
416 lle_memory_pool_t *memory_pool) {
417 lle_coord_queue_t *q = NULL;
418
419 if (!queue || !memory_pool) {
420 return LLE_ERROR_INVALID_PARAMETER;
421 }
422
423 /* Allocate queue structure */
424 q = lle_pool_alloc(sizeof(lle_coord_queue_t));
425 if (!q) {
426 return LLE_ERROR_OUT_OF_MEMORY;
427 }
428 memset(q, 0, sizeof(lle_coord_queue_t));
429
430 /* Set initial capacity (reasonable default) */
431 q->capacity = 16; /* Start with 16 entries, will grow if needed */
432
433 /* Allocate request array */
434 q->requests = lle_pool_alloc(sizeof(lle_render_request_t) * q->capacity);
435 if (!q->requests) {
436 lle_pool_free(q);
437 return LLE_ERROR_OUT_OF_MEMORY;
438 }
439 memset(q->requests, 0, sizeof(lle_render_request_t) * q->capacity);
440
441 /* Initialize queue state */
442 q->count = 0;
443 q->head = 0;
444 q->tail = 0;
445
446 /* Initialize mutex for thread safety */
447 if (pthread_mutex_init(&q->lock, NULL) != 0) {
448 lle_pool_free(q->requests);
449 lle_pool_free(q);
450 return LLE_ERROR_SYSTEM_CALL;
451 }
452
453 *queue = q;
454 return LLE_SUCCESS;
455}

535/**
536 * @brief Calculate cursor screen position using incremental tracking
537 *
538 * This implements the Replxx/Fish/ZLE approach for cursor positioning:
539 * Walk through the text character by character, tracking (x, y) position,
540 * and when we reach the cursor byte offset, that's our screen position.
541 *
542 * This correctly handles:
543 * - Line wrapping at terminal boundaries
544 * - Multi-byte UTF-8 characters (1 column)
545 * - Wide characters like CJK (2 columns)
546 * - ANSI escape sequences (0 columns)
547 * - Tab expansion
548 *
549 * See: docs/development/MODERN_EDITOR_WRAPPING_RESEARCH.md
550 *
551 * @param text The rendered text buffer
552 * @param cursor_byte_offset Byte offset of cursor in text
553 * @param prompt_visual_width Visual width of prompt (where command starts)
554 * @param terminal_width Terminal width in columns
555 * @param out_row Output: cursor row (0-based)
556 * @param out_col Output: cursor column (0-based)
557 */
558static void calculate_cursor_screen_position(const char *text,
559 size_t cursor_byte_offset,
560 size_t prompt_visual_width,
561 size_t terminal_width,
562 size_t *out_row, size_t *out_col) {
563 if (!text || !out_row || !out_col || terminal_width == 0) {
564 if (out_row)
565 *out_row = 0;
566 if (out_col)
567 *out_col = prompt_visual_width;
568 return;
569 }
570
571 /* Start position: after the prompt on row 0 */
572 size_t x = prompt_visual_width;
573 size_t y = 0;
574 size_t bytes_processed = 0;
575 size_t text_len = strlen(text);
576
577 /* Walk through text character by character */
578 for (size_t i = 0; i < text_len;) {
579 /* CRITICAL: Check cursor position BEFORE processing next character
580 * This ensures we capture the position before the character at
581 * cursor_byte_offset */
582 if (bytes_processed == cursor_byte_offset) {
583 *out_row = y;
584 *out_col = x;
585 return;
586 }
587
588 unsigned char ch = (unsigned char)text[i];
589 size_t start_i = i; /* Remember start position for byte counting */
590
591 /* Handle ANSI escape sequences (don't advance x position) */
592 if (ch == '\033' || ch == '\x1b') {
593 /* Skip entire escape sequence */
594 i++;
595 if (i < text_len && text[i] == '[') {
596 i++;
597 /* Skip until we find the terminator */
598 while (i < text_len) {
599 char c = text[i++];
600 /* Common CSI terminators */
601 if ((c >= 'A' && c <= 'Z') || (c >= 'a' && c <= 'z') ||
602 c == 'm' || c == 'H' || c == 'J' || c == 'K' ||
603 c == 'G' || c == 'f' || c == 's' || c == 'u') {
604 break;
605 }
606 }
607 }
608 /* Count actual bytes consumed by escape sequence */
609 bytes_processed += (i - start_i);
610 continue;
611 }
612
613 /* Handle newlines (move to start of next line) */
614 if (ch == '\n') {
615 x = 0;
616 y++;
617 i++;
618 bytes_processed++;
619 continue;
620 }
621
622 /* Handle carriage return */
623 if (ch == '\r') {
624 x = 0;
625 i++;
626 bytes_processed++;
627 continue;
628 }
629
630 /* Handle tab (expand to next tab stop based on config.tab_width) */
631 if (ch == '\t') {
632 int tw = config.tab_width > 0 ? config.tab_width : 4;
633 size_t tab_width = tw - (x % tw);
634 x += tab_width;
635 /* Handle wrapping for tabs */
636 if (x >= terminal_width) {
637 y += x / terminal_width;
638 x = x % terminal_width;
639 }
640 i++;
641 bytes_processed++;
642 continue;
643 }
644
645 /* GRAPHEME-AWARE WIDTH CALCULATION (Phase 2 Fix)
646 *
647 * Instead of processing individual UTF-8 codepoints, we now process
648 * entire grapheme clusters as atomic units. This correctly handles:
649 * - CJK characters (2 columns)
650 * - Emoji (2 columns)
651 * - Combining marks (0 additional columns)
652 * - ZWJ sequences (rendered as single unit)
653 * - Regional Indicator pairs (flags)
654 * - Emoji with skin tone modifiers
655 */
656
657 /* Find the end of this grapheme cluster
658 * CRITICAL: Must advance by UTF-8 character boundaries, not individual
659 * bytes! lle_is_grapheme_boundary() requires valid UTF-8 character
660 * starts.
661 */
662 const char *grapheme_start = text + i;
663 const char *grapheme_end = grapheme_start;
664
665 /* Scan forward by UTF-8 characters until we hit a grapheme boundary */
666 do {
667 /* Advance to next UTF-8 character */
668 int char_len =
669 lle_utf8_sequence_length((unsigned char)*grapheme_end);
670 if (char_len <= 0 || grapheme_end + char_len > text + text_len) {
671 /* Invalid UTF-8 or end of string - treat as single byte */
672 grapheme_end++;
673 break;
674 }
675 grapheme_end += char_len;
676
677 /* Check if this is a grapheme boundary */
678 if (grapheme_end >= text + text_len ||
679 lle_is_grapheme_boundary(grapheme_end, text, text + text_len)) {
680 break;
681 }
682 } while (grapheme_end < text + text_len);
683
684 size_t grapheme_bytes = grapheme_end - grapheme_start;
685
686 /* Calculate visual width of this grapheme cluster
687 *
688 * Strategy:
689 * 1. Decode first codepoint of grapheme (base character)
690 * 2. Use wcwidth() on base character for width
691 * 3. Treat entire grapheme cluster as that width
692 *
693 * This handles:
694 * - Base emoji + modifier base determines width
695 * - Base char + combining mark base determines width
696 * - ZWJ sequences first emoji determines width
697 */
698 uint32_t base_codepoint = 0;
699 int decode_result = lle_utf8_decode_codepoint(
700 grapheme_start, grapheme_bytes, &base_codepoint);
701
702 int visual_width = 1; /* Default to 1 column */
703
704 if (decode_result > 0) {
705 /* Successfully decoded first codepoint - get its width */
706 wchar_t wc = (wchar_t)base_codepoint;
707 int wc_width = wcwidth(wc);
708
709 if (wc_width >= 0) {
710 /* wcwidth() returned valid width (0, 1, or 2) */
711 visual_width = wc_width;
712 } else {

716 }
717 }
718

720 x += visual_width;
721

723 * IMPORTANT: Wrap happens AFTER incrementing x, before processing next
724 * char */
725 if (x >= terminal_width) {
726 x = 0;
727 y++;
728 }
729
730 /* Advance byte counter and string position by entire grapheme */
731 i += grapheme_bytes;
732 bytes_processed += grapheme_bytes;
733 }
734

736 if (bytes_processed == cursor_byte_offset) {
737 *out_row = y;
738 *out_col = x;
739 return;
740 }
741
742 /* Fallback: cursor beyond text length - place at end */
743 *out_row = y;
744 *out_col = x;
