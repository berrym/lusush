12/*
13 * Get the length of a UTF-8 sequence from its first byte
14 */
15int lle_utf8_sequence_length(unsigned char first_byte) {
16 // Single-byte character (ASCII): 0xxxxxxx
17 if ((first_byte & 0x80) == 0x00) {
18 return 1;
19 }
20
21 // Two-byte sequence: 110xxxxx
22 if ((first_byte & 0xE0) == 0xC0) {
23 return 2;
24 }
25
26 // Three-byte sequence: 1110xxxx
27 if ((first_byte & 0xF0) == 0xE0) {
28 return 3;
29 }
30
31 // Four-byte sequence: 11110xxx
32 if ((first_byte & 0xF8) == 0xF0) {
33 return 4;
34 }
35
36 // Invalid UTF-8 start byte
37 return 0;
38}

40/*
41 * Validate a complete UTF-8 sequence
42 */
43bool lle_utf8_is_valid_sequence(const char *ptr, int length) {
44 if (!ptr || length <= 0 || length > 4) {
45 return false;
46 }
47
48 unsigned char first = (unsigned char)ptr[0];
49
50 // Validate sequence length matches first byte
51 int expected_len = lle_utf8_sequence_length(first);
52 if (expected_len != length) {
53 return false;
54 }
55
56 // Single-byte ASCII
57 if (length == 1) {
58 return true;
59 }
60
61 // Multi-byte sequences - validate continuation bytes
62 for (int i = 1; i < length; i++) {
63 unsigned char byte = (unsigned char)ptr[i];
64 // Continuation bytes must be 10xxxxxx
65 if ((byte & 0xC0) != 0x80) {
66 return false;
67 }
68 }
69
70 // Additional validation for overlong encodings and invalid ranges
71 uint32_t codepoint = 0;
72
73 if (length == 2) {
74 // Two-byte: 110xxxxx 10xxxxxx
75 codepoint = ((first & 0x1F) << 6) | (ptr[1] & 0x3F);
76 // Must be >= 0x80 (not overlong)
77 if (codepoint < 0x80) {
78 return false;
79 }
80 } else if (length == 3) {
81 // Three-byte: 1110xxxx 10xxxxxx 10xxxxxx
82 codepoint =
83 ((first & 0x0F) << 12) | ((ptr[1] & 0x3F) << 6) | (ptr[2] & 0x3F);
84 // Must be >= 0x800 (not overlong)
85 // Must not be in surrogate range (0xD800-0xDFFF)
86 if (codepoint < 0x800 || (codepoint >= 0xD800 && codepoint <= 0xDFFF)) {
87 return false;
88 }
89 } else if (length == 4) {
90 // Four-byte: 11110xxx 10xxxxxx 10xxxxxx 10xxxxxx
91 codepoint = ((first & 0x07) << 18) | ((ptr[1] & 0x3F) << 12) |
92 ((ptr[2] & 0x3F) << 6) | (ptr[3] & 0x3F);
93 // Must be >= 0x10000 (not overlong)
94 // Must be <= 0x10FFFF (maximum Unicode codepoint)
95 if (codepoint < 0x10000 || codepoint > 0x10FFFF) {
96 return false;
97 }
98 }
99
100 return true;
101}

103/*
104 * Validate an entire UTF-8 string
105 */
106bool lle_utf8_is_valid(const char *text, size_t length) {
107 if (!text) {
108 return false;
109 }
110
111 const char *ptr = text;
112 const char *end = text + length;
113
114 while (ptr < end) {
115 int seq_len = lle_utf8_sequence_length((unsigned char)*ptr);
116 if (seq_len == 0 || ptr + seq_len > end) {
117 return false;
118 }
119
120 if (!lle_utf8_is_valid_sequence(ptr, seq_len)) {
121 return false;
122 }
123
124 ptr += seq_len;
125 }
126
127 return true;
128}

130/*
131 * Decode a UTF-8 sequence to a Unicode codepoint
132 */
133int lle_utf8_decode_codepoint(const char *ptr, size_t max_len,
134 uint32_t *codepoint) {
135 if (!ptr || !codepoint || max_len == 0) {
136 return -1;
137 }
138
139 unsigned char first = (unsigned char)ptr[0];
140 int seq_len = lle_utf8_sequence_length(first);
141
142 if (seq_len == 0 || (size_t)seq_len > max_len) {
143 return -1;
144 }
145
146 // Validate the sequence
147 if (!lle_utf8_is_valid_sequence(ptr, seq_len)) {
148 return -1;
149 }
150
151 // Decode based on length
152 if (seq_len == 1) {
153 // ASCII
154 *codepoint = first;
155 } else if (seq_len == 2) {
156 // Two-byte sequence
157 *codepoint = ((first & 0x1F) << 6) | (ptr[1] & 0x3F);
158 } else if (seq_len == 3) {
159 // Three-byte sequence
160 *codepoint =
161 ((first & 0x0F) << 12) | ((ptr[1] & 0x3F) << 6) | (ptr[2] & 0x3F);
162 } else if (seq_len == 4) {
163 // Four-byte sequence
164 *codepoint = ((first & 0x07) << 18) | ((ptr[1] & 0x3F) << 12) |
165 ((ptr[2] & 0x3F) << 6) | (ptr[3] & 0x3F);
166 }
167
168 return seq_len;
169}

171/*
172 * Encode a Unicode codepoint to UTF-8
173 */
174int lle_utf8_encode_codepoint(uint32_t codepoint, char *buffer) {
175 if (!buffer) {
176 return 0;
177 }
178
179 // Validate codepoint range
180 if (codepoint > 0x10FFFF || (codepoint >= 0xD800 && codepoint <= 0xDFFF)) {
181 return 0; // Invalid codepoint
182 }
183
184 if (codepoint <= 0x7F) {
185 // Single-byte (ASCII)
186 buffer[0] = (char)codepoint;
187 return 1;
188 } else if (codepoint <= 0x7FF) {
189 // Two-byte sequence
190 buffer[0] = (char)(0xC0 | (codepoint >> 6));
191 buffer[1] = (char)(0x80 | (codepoint & 0x3F));
192 return 2;
193 } else if (codepoint <= 0xFFFF) {
194 // Three-byte sequence
195 buffer[0] = (char)(0xE0 | (codepoint >> 12));
196 buffer[1] = (char)(0x80 | ((codepoint >> 6) & 0x3F));
197 buffer[2] = (char)(0x80 | (codepoint & 0x3F));
198 return 3;
199 } else {
200 // Four-byte sequence
201 buffer[0] = (char)(0xF0 | (codepoint >> 18));
202 buffer[1] = (char)(0x80 | ((codepoint >> 12) & 0x3F));
203 buffer[2] = (char)(0x80 | ((codepoint >> 6) & 0x3F));
204 buffer[3] = (char)(0x80 | (codepoint & 0x3F));
205 return 4;
206 }
207}

293 * Convert codepoint index to grapheme cluster index
294 */
295int lle_utf8_codepoint_to_grapheme_index(const char *text, size_t cp_index,
296 size_t *grapheme_index) {
297 if (!text || !grapheme_index) {
298 return -1;
299 }
300
301 const char *ptr = text;
302 const char *start = text;
303 size_t cp_count = 0;
304 size_t grapheme_count = 0;
305
306 // Count grapheme clusters up to the specified codepoint
307 while (cp_count <= cp_index) {
308 int seq_len = lle_utf8_sequence_length((unsigned char)*ptr);
309 if (seq_len == 0) {
310 return -1; // Invalid UTF-8
311 }
312
313 // Check if this is a grapheme boundary
314 if (lle_is_grapheme_boundary(ptr, start, ptr + seq_len)) {
315 if (cp_count < cp_index) {
316 grapheme_count++;
317 }
318 }
319
320 if (cp_count == cp_index) {
321 break;
322 }
323
324 ptr += seq_len;
325 cp_count++;
326 }
327
328 *grapheme_index = grapheme_count;
329 return 0;
330}
