112lle_result_t lle_multiline_parser_split_lines(lle_multiline_parser_t *parser,
113 const char *command_text,
114 size_t command_length,
115 lle_parsed_line_t ***lines,
116 size_t *line_count) {
117 if (!parser || !parser->active || !command_text || !lines || !line_count) {
118 return LLE_ERROR_INVALID_PARAMETER;
119 }
120
121 /* Count lines first */
122 size_t count = 1;
123 for (size_t i = 0; i < command_length; i++) {
124 if (command_text[i] == '\n') {
125 count++;
126 }
127 }
128
129 if (count > parser->config.max_lines) {
130 return LLE_ERROR_INVALID_PARAMETER;
131 }
132
133 /* Allocate line array */
134 lle_parsed_line_t **line_array =
135 lle_pool_alloc(count * sizeof(lle_parsed_line_t *));
136 if (!line_array) {
137 return LLE_ERROR_OUT_OF_MEMORY;
138 }
139
140 /* Split into lines */
141 size_t line_idx = 0;
142 size_t line_start = 0;
143
144 for (size_t i = 0; i <= command_length; i++) {
145 if (i == command_length || command_text[i] == '\n') {
146 size_t line_len = i - line_start;
147
148 lle_parsed_line_t *line = create_parsed_line(
149 parser, &command_text[line_start], line_len, line_idx);
150
151 if (!line) {
152 return LLE_ERROR_OUT_OF_MEMORY;
153 }
154
155 line_array[line_idx] = line;
156 line_idx++;
157 line_start = i + 1;
158 }
159 }
160
161 *lines = line_array;
162 *line_count = count;
163
164 return LLE_SUCCESS;
165}

167lle_result_t lle_multiline_parser_parse(lle_multiline_parser_t *parser,
168 const char *command_text,
169 size_t command_length,
170 lle_multiline_parse_result_t **result) {
171 if (!parser || !parser->active || !command_text || !result) {
172 return LLE_ERROR_INVALID_PARAMETER;
173 }
174
175 /* Allocate parse result */
176 lle_multiline_parse_result_t *parse_result =
177 lle_pool_alloc(sizeof(lle_multiline_parse_result_t));
178 if (!parse_result) {
179 return LLE_ERROR_OUT_OF_MEMORY;
180 }
181
182 memset(parse_result, 0, sizeof(lle_multiline_parse_result_t));
183
184 /* Split into lines */
185 lle_parsed_line_t **lines = NULL;
186 size_t line_count = 0;
187
188 lle_result_t res = lle_multiline_parser_split_lines(
189 parser, command_text, command_length, &lines, &line_count);
190
191 if (res != LLE_SUCCESS) {
192 return res;
193 }
194
195 /* Build linked list of lines */
196 if (line_count > 0) {
197 parse_result->first_line = lines[0];
198 parse_result->last_line = lines[line_count - 1];
199
200 for (size_t i = 0; i < line_count - 1; i++) {
201 lines[i]->next = lines[i + 1];
202 }
203 }
204
205 parse_result->line_count = line_count;
206 parse_result->total_length = command_length;
207
208 /* Analyze command structure */
209 lle_command_structure_t *structure = NULL;
210 res = lle_structure_analyzer_analyze(parser->analyzer, command_text,
211 command_length, &structure);
212
213 if (res == LLE_SUCCESS) {
214 parse_result->structure = structure;
215 parse_result->is_complete = structure->is_complete;
216 parse_result->has_syntax_error = structure->has_syntax_error;
217 } else {
218 parse_result->structure = NULL;
219 parse_result->is_complete = false;
220 parse_result->has_syntax_error = true;
221 }
222
223 /* Check for expected closing keyword if incomplete */
224 if (!parse_result->is_complete) {
225 lle_keyword_type_t expected = LLE_KEYWORD_NONE;
226 lle_structure_analyzer_check_complete(
227 parser->analyzer, command_text, command_length,
228 &parse_result->is_complete, &expected);
229 parse_result->expected_keyword = expected;
230 }
231
232 *result = parse_result;
233 return LLE_SUCCESS;
234}

260static lle_parsed_line_t *create_parsed_line(lle_multiline_parser_t *parser,
261 const char *content, size_t length,
262 size_t line_number) {
263 lle_parsed_line_t *line = lle_pool_alloc(sizeof(lle_parsed_line_t));
264 if (!line) {
265 return NULL;
266 }
267
268 memset(line, 0, sizeof(lle_parsed_line_t));
269
270 /* Allocate and copy content */
271 line->content = lle_pool_alloc(length + 1);
272 if (!line->content) {
273 return NULL;
274 }
275
276 memcpy(line->content, content, length);
277 line->content[length] = '\0';
278 line->length = length;
279 line->line_number = line_number;
280
281 /* Calculate indentation if configured */
282 if (parser->config.preserve_indentation) {
283 line->indent_level = calculate_indent_level(content, length);
284 }
285
286 /* Check for continuation */
287 if (parser->config.detect_continuations) {
288 line->has_continuation = has_backslash_continuation(content, length);
289 }
290
291 /* Detect primary keyword (simple detection) */
292 line->primary_keyword = LLE_KEYWORD_NONE;
293 line->keyword_count = 0;
294
295 /* Skip leading whitespace to find first word */
296 size_t start = 0;
297 while (start < length && isspace(content[start])) {
298 start++;
299 }
300
301 if (start < length) {
302 /* Check for common keywords */
303 const char *keywords[] = {"for", "while", "until", "if",
304 "case", "function", NULL};
305 const lle_keyword_type_t types[] = {
306 LLE_KEYWORD_FOR, LLE_KEYWORD_WHILE, LLE_KEYWORD_UNTIL,
307 LLE_KEYWORD_IF, LLE_KEYWORD_CASE, LLE_KEYWORD_FUNCTION};
308
309 for (size_t i = 0; keywords[i] != NULL; i++) {
310 size_t kw_len = strlen(keywords[i]);
311 if (start + kw_len <= length &&
312 strncmp(&content[start], keywords[i], kw_len) == 0 &&
313 (start + kw_len == length ||
314 isspace(content[start + kw_len]))) {
315 line->primary_keyword = types[i];
316 line->keyword_count = 1;
317 break;
318 }
319 }
320 }
321
322 return line;
323}

331static bool has_backslash_continuation(const char *line, size_t length) {
332 if (length == 0) {
333 return false;
334 }
335
336 /* Skip trailing whitespace */
337 size_t end = length;
338 while (end > 0 && isspace(line[end - 1])) {
339 end--;
340 }
341
342 if (end == 0) {
343 return false;
344 }
345
346 /* Check for backslash */
347 if (line[end - 1] == '\\') {
348 /* Count preceding backslashes to handle escapes */
349 size_t backslash_count = 1;
350 size_t pos = end - 1;
351
352 while (pos > 0 && line[pos - 1] == '\\') {
353 backslash_count++;
354 pos--;
355 }
356
357 /* Odd number of backslashes means continuation */
358 return (backslash_count % 2) == 1;
359 }
360
361 return false;
362}

364static size_t calculate_indent_level(const char *line, size_t length) {
365 size_t indent = 0;
366
367 for (size_t i = 0; i < length; i++) {
368 if (line[i] == ' ') {
369 indent++;
370 } else if (line[i] == '\t') {
371 indent += 4; /* Tab = 4 spaces */
372 } else {
373 break; /* Non-whitespace found */
374 }
375 }
376
377 return indent;
378}
