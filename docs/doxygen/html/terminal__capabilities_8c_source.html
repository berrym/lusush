26/*
27 * Detect terminal type from TERM environment variable
28 */
29static lle_terminal_type_t detect_terminal_type(const char *term_env) {
30 if (!term_env) {
31 return LLE_TERMINAL_GENERIC;
32 }
33
34 /* Check for specific terminal types (order matters) */
35
36 /* Modern terminals first */
37 if (strstr(term_env, "alacritty"))
38 return LLE_TERMINAL_ALACRITTY;
39 if (strstr(term_env, "kitty"))
40 return LLE_TERMINAL_KITTY;
41 if (strstr(term_env, "iterm"))
42 return LLE_TERMINAL_ITERM2;
43
44 /* Terminal multiplexers */
45 if (strstr(term_env, "tmux"))
46 return LLE_TERMINAL_TMUX;
47 if (strstr(term_env, "screen"))
48 return LLE_TERMINAL_SCREEN;
49
50 /* Traditional terminals */
51 if (strstr(term_env, "xterm"))
52 return LLE_TERMINAL_XTERM;
53 if (strstr(term_env, "rxvt"))
54 return LLE_TERMINAL_RXVT;
55 if (strstr(term_env, "konsole"))
56 return LLE_TERMINAL_KONSOLE;
57 if (strstr(term_env, "gnome"))
58 return LLE_TERMINAL_GNOME_TERMINAL;
59
60 /* Console */
61 if (strstr(term_env, "linux"))
62 return LLE_TERMINAL_LINUX_CONSOLE;
63
64 /* macOS */
65 if (strstr(term_env, "nsterm"))
66 return LLE_TERMINAL_DARWIN_TERMINAL;
67
68 return LLE_TERMINAL_GENERIC;
69}

110/*
111 * Detect color capabilities from terminfo and environment
112 */
113static void detect_color_capabilities(lle_terminal_capabilities_t *caps) {
114 /* Initialize terminfo */
115 int err;
116 if (setupterm(NULL, STDOUT_FILENO, &err) != OK) {
117 /* Terminfo unavailable - use conservative defaults */
118 caps->supports_ansi_colors = false;
119 caps->supports_256_colors = false;
120 caps->supports_truecolor = false;
121 caps->detected_color_depth = 0;
122 return;
123 }
124
125 /* Query color capability from terminfo */
126 int colors = query_terminfo_num("colors");
127
128 if (colors >= 256) {
129 caps->supports_ansi_colors = true;
130 caps->supports_256_colors = true;
131 caps->detected_color_depth = 8; /* 8-bit color */
132 } else if (colors >= 8) {
133 caps->supports_ansi_colors = true;
134 caps->supports_256_colors = false;
135 caps->detected_color_depth = 4; /* 4-bit color (16 colors) */
136 } else {
137 caps->supports_ansi_colors = false;
138 caps->supports_256_colors = false;
139 caps->detected_color_depth = 0;
140 }
141
142 /* Check for truecolor via environment (not in standard terminfo) */
143 const char *colorterm = getenv("COLORTERM");
144 if (colorterm && (strcmp(colorterm, "truecolor") == 0 ||
145 strcmp(colorterm, "24bit") == 0)) {
146 caps->supports_truecolor = true;
147 caps->detected_color_depth = 24; /* 24-bit truecolor */
148 } else {
149 caps->supports_truecolor = false;
150 }
151
152 /* Some modern terminals support truecolor even without COLORTERM */
153 if (!caps->supports_truecolor) {
154 if (caps->terminal_type_enum == LLE_TERMINAL_ALACRITTY ||
155 caps->terminal_type_enum == LLE_TERMINAL_KITTY ||
156 caps->terminal_type_enum == LLE_TERMINAL_ITERM2) {
157 caps->supports_truecolor = true;
158 caps->detected_color_depth = 24;
159 }
160 }
161}

178 caps->terminal_type_enum == LLE_TERMINAL_ALACRITTY ||

180 caps->terminal_type_enum == LLE_TERMINAL_ITERM2 ||

182 caps->supports_strikethrough = true;

238/*
239 * Detect terminal window size using ioctl
240 */
241static void detect_terminal_geometry(lle_terminal_capabilities_t *caps) {
242 struct winsize ws;
243
244 /* Try ioctl first */
245 if (ioctl(STDOUT_FILENO, TIOCGWINSZ, &ws) == 0 && ws.ws_col > 0 &&
246 ws.ws_row > 0) {
247 caps->terminal_width = ws.ws_col;
248 caps->terminal_height = ws.ws_row;
249 } else {
250 /* Fallback to environment variables */
251 const char *env_cols = getenv("COLUMNS");
252 const char *env_lines = getenv("LINES");
253
254 caps->terminal_width = (env_cols && *env_cols)
255 ? (size_t)atoi(env_cols)
256 : 80; /* Default 80 columns */
257 caps->terminal_height = (env_lines && *env_lines)
258 ? (size_t)atoi(env_lines)
259 : 24; /* Default 24 lines */
260 }
261
262 /* Sanity checks - ensure reasonable minimums */
263 if (caps->terminal_width < 20)
264 caps->terminal_width = 80;
265 if (caps->terminal_height < 5)
266 caps->terminal_height = 24;
267}

269/*
270 * Estimate terminal latency and performance characteristics
271 */
272static void
273detect_performance_characteristics(lle_terminal_capabilities_t *caps) {
274 /* Estimated round-trip latency in milliseconds */
275 switch (caps->terminal_type_enum) {
276 case LLE_TERMINAL_ALACRITTY:
277 case LLE_TERMINAL_KITTY:
278 /* GPU-accelerated terminals - very fast */
279 caps->estimated_latency_ms = 5;
280 caps->supports_fast_updates = true;
281 break;
282
283 case LLE_TERMINAL_ITERM2:
284 case LLE_TERMINAL_GNOME_TERMINAL:
285 /* Modern terminals - fast */
286 caps->estimated_latency_ms = 10;
287 caps->supports_fast_updates = true;
288 break;
289
290 case LLE_TERMINAL_XTERM:
291 case LLE_TERMINAL_RXVT:
292 case LLE_TERMINAL_KONSOLE:
293 /* Traditional terminals - moderate */
294 caps->estimated_latency_ms = 15;
295 caps->supports_fast_updates = true;
296 break;
297
298 case LLE_TERMINAL_SCREEN:
299 case LLE_TERMINAL_TMUX:
300 /* Terminal multiplexers - slower (additional layer) */
301 caps->estimated_latency_ms = 20;
302 caps->supports_fast_updates = false;
303 break;
304
305 case LLE_TERMINAL_LINUX_CONSOLE:
306 /* Console - slower */
307 caps->estimated_latency_ms = 30;
308 caps->supports_fast_updates = false;
309 break;
310
311 default:
312 /* Conservative default */
313 caps->estimated_latency_ms = 15;
314 caps->supports_fast_updates = true;
315 break;
316 }
317}

319/*
320 * Set optimization flags based on terminal capabilities
321 */
322static void set_optimization_flags(lle_terminal_capabilities_t *caps) {
323 caps->optimizations = LLE_OPT_NONE;
324
325 /* Fast cursor positioning for GPU-accelerated terminals */
326 if (caps->terminal_type_enum == LLE_TERMINAL_ALACRITTY ||
327 caps->terminal_type_enum == LLE_TERMINAL_KITTY ||
328 caps->terminal_type_enum == LLE_TERMINAL_ITERM2) {
329 caps->optimizations |= LLE_OPT_FAST_CURSOR;
330 }
331
332 /* Batch updates for slower terminals and multiplexers */
333 if (caps->terminal_type_enum == LLE_TERMINAL_SCREEN ||
334 caps->terminal_type_enum == LLE_TERMINAL_TMUX ||
335 caps->terminal_type_enum == LLE_TERMINAL_LINUX_CONSOLE) {
336 caps->optimizations |= LLE_OPT_BATCH_UPDATES;
337 }
338
339 /* Incremental drawing for fast terminals */
340 if (caps->supports_fast_updates) {
341 caps->optimizations |= LLE_OPT_INCREMENTAL_DRAW;
342 }
343
344 /* Unicode awareness */
345 if (caps->supports_unicode) {
346 caps->optimizations |= LLE_OPT_UNICODE_AWARE;
347 }
348}

355/*
356 * Main capability detection entry point
357 *
358 * Detects all terminal capabilities at initialization time. This is called
359 * ONCE at startup and the results are cached for the lifetime of the program.
360 */
361lle_result_t
362lle_capabilities_detect_environment(lle_terminal_capabilities_t **caps,
363 lle_unix_interface_t *unix_iface) {
364 if (!caps) {
365 return LLE_ERROR_INVALID_PARAMETER;
366 }
367
368 /* Note: unix_iface may be NULL - not required for capability detection */
369 (void)unix_iface; /* Unused in Phase 1 */
370
371 /* Allocate capabilities structure */
372 lle_terminal_capabilities_t *c =
373 calloc(1, sizeof(lle_terminal_capabilities_t));
374 if (!c) {
375 return LLE_ERROR_OUT_OF_MEMORY;
376 }
377
378 /* Already zeroed by calloc */
379
380 /* Detect TTY status */
381 c->is_tty = detect_is_tty();
382
383 /* Get environment variables */
384 const char *term_env = getenv("TERM");
385 c->terminal_type = term_env ? strdup(term_env) : strdup("unknown");
386 if (!c->terminal_type) {
387 lle_pool_free(c);
388 return LLE_ERROR_OUT_OF_MEMORY;
389 }
390
391 const char *term_program = detect_terminal_program();
392 c->terminal_program = strdup(term_program);
393 if (!c->terminal_program) {
394 free((char *)c->terminal_type);
395 lle_pool_free(c);
396 return LLE_ERROR_OUT_OF_MEMORY;
397 }
398
399 /* Detect terminal type enum */
400 c->terminal_type_enum = detect_terminal_type(term_env);
401
402 /* Detect all capabilities */
403 detect_color_capabilities(c);
404 detect_text_attributes(c);
405 detect_advanced_features(c);
406 detect_terminal_geometry(c);
407 detect_performance_characteristics(c);
408 set_optimization_flags(c);
409
410 *caps = c;
411 return LLE_SUCCESS;
412}

434/*
435 * Update terminal geometry (called after SIGWINCH)
436 *
437 * This is the ONLY function that queries terminal state at runtime.
438 * It's called in response to window resize signals.
439 *
440 * Note: Header signature takes width/height by value (not pointers).
441 * This function updates internal caps structure, not output parameters.
442 */
443lle_result_t lle_capabilities_update_geometry(lle_terminal_capabilities_t *caps,
444 size_t width, size_t height) {
445 if (!caps) {
446 return LLE_ERROR_INVALID_PARAMETER;
447 }
448
449 /* If width/height provided, use them directly (from signal handler) */
450 if (width > 0 && height > 0) {
451 caps->terminal_width = width;
452 caps->terminal_height = height;
453 } else {
454 /* Otherwise, re-detect geometry via ioctl */
455 detect_terminal_geometry(caps);
456 }
457
458 /* Sanity checks */
459 if (caps->terminal_width < 20)
460 caps->terminal_width = 80;
461 if (caps->terminal_height < 5)
462 caps->terminal_height = 24;
463
464 return LLE_SUCCESS;
465}
