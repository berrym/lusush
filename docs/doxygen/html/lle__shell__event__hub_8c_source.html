21/**
22 * @brief Check if event debug logging is enabled

24 * Set LUSUSH_EVENT_DEBUG=1 to enable verbose event system logging.
25 * Logs event firing, handler dispatch, and silent failure conditions.

27static inline bool lle_event_debug_enabled(void) {
28 static int cached = -1;
29 if (cached == -1) {
30 const char *env = getenv("LUSUSH_EVENT_DEBUG");
31 cached = (env && env[0] == '1') ? 1 : 0;
32 }
33 return cached == 1;

52uint64_t lle_shell_event_get_timestamp_us(void) {
53 struct timespec ts;
54 if (clock_gettime(CLOCK_MONOTONIC, &ts) == 0) {
55 return (uint64_t)ts.tv_sec * 1000000ULL + (uint64_t)ts.tv_nsec / 1000;
56 }
57 return 0;
58}

78lle_result_t lle_shell_event_hub_create(lle_shell_event_hub_t **hub_out) {
79 if (!hub_out) {
80 return LLE_ERROR_INVALID_PARAMETER;
81 }
82
83 lle_shell_event_hub_t *hub = calloc(1, sizeof(lle_shell_event_hub_t));
84 if (!hub) {
85 return LLE_ERROR_OUT_OF_MEMORY;
86 }
87
88 /* Initialize handler counts */
89 for (int i = 0; i < LLE_SHELL_EVENT_TYPE_COUNT; i++) {
90 hub->handler_counts[i] = 0;
91 }
92
93 /* Capture current working directory */
94 if (getcwd(hub->current_dir, sizeof(hub->current_dir)) == NULL) {
95 hub->current_dir[0] = '\0';
96 }
97
98 hub->command_start_time_us = 0;
99 hub->current_command[0] = '\0';
100 hub->initialized = true;
101 hub->events_fired = 0;
102 hub->directory_changes = 0;
103 hub->commands_executed = 0;
104
105 *hub_out = hub;
106 return LLE_SUCCESS;
107}

123lle_result_t lle_shell_event_hub_register(lle_shell_event_hub_t *hub,
124 lle_shell_event_type_t event_type,
125 lle_shell_event_handler_t handler,
126 void *user_data, const char *name) {
127
128 if (!hub || !handler) {
129 return LLE_ERROR_INVALID_PARAMETER;
130 }
131
132 if (event_type >= LLE_SHELL_EVENT_TYPE_COUNT) {
133 return LLE_ERROR_INVALID_PARAMETER;
134 }
135
136 size_t count = hub->handler_counts[event_type];
137 if (count >= LLE_SHELL_EVENT_MAX_HANDLERS) {
138 return LLE_ERROR_RESOURCE_EXHAUSTED;
139 }
140
141 /* Add handler at next available slot */
142 hub->handlers[event_type][count].handler = handler;
143 hub->handlers[event_type][count].user_data = user_data;
144 hub->handlers[event_type][count].name = name;
145 hub->handler_counts[event_type]++;
146
147 return LLE_SUCCESS;
148}

150lle_result_t lle_shell_event_hub_unregister(lle_shell_event_hub_t *hub,
151 lle_shell_event_type_t event_type,
152 const char *name) {
153
154 if (!hub || !name) {
155 return LLE_ERROR_INVALID_PARAMETER;
156 }
157
158 if (event_type >= LLE_SHELL_EVENT_TYPE_COUNT) {
159 return LLE_ERROR_INVALID_PARAMETER;
160 }
161
162 size_t count = hub->handler_counts[event_type];
163 for (size_t i = 0; i < count; i++) {
164 if (hub->handlers[event_type][i].name &&
165 strcmp(hub->handlers[event_type][i].name, name) == 0) {
166 /* Found - shift remaining handlers down */
167 for (size_t j = i; j < count - 1; j++) {
168 hub->handlers[event_type][j] = hub->handlers[event_type][j + 1];
169 }
170 hub->handler_counts[event_type]--;
171 return LLE_SUCCESS;
172 }
173 }
174
175 return LLE_ERROR_NOT_FOUND;
176}

183void lle_shell_event_hub_fire(lle_shell_event_hub_t *hub,
184 lle_shell_event_type_t event_type,
185 void *event_data) {
186
187 if (!hub || !hub->initialized) {
188 if (lle_event_debug_enabled()) {
189 fprintf(stderr,
190 "[LLE_EVENT] WARN: lle_shell_event_hub_fire: "
191 "hub=%p initialized=%d\n",
192 (void *)hub, hub ? hub->initialized : 0);
193 }
194 return;
195 }
196
197 if (event_type >= LLE_SHELL_EVENT_TYPE_COUNT) {
198 if (lle_event_debug_enabled()) {
199 fprintf(stderr,
200 "[LLE_EVENT] WARN: lle_shell_event_hub_fire: "
201 "invalid event_type=%d (max=%d)\n",
202 event_type, LLE_SHELL_EVENT_TYPE_COUNT);
203 }
204 return;
205 }
206
207 /* Update statistics */
208 hub->events_fired++;
209
210 /* Call all registered handlers */
211 size_t count = hub->handler_counts[event_type];
212 if (lle_event_debug_enabled()) {
213 fprintf(stderr, "[LLE_EVENT] Firing %s to %zu handlers\n",
214 lle_shell_event_type_name(event_type), count);
215 }
216
217 for (size_t i = 0; i < count; i++) {
218 lle_shell_event_handler_t handler =
219 hub->handlers[event_type][i].handler;
220 void *user_data = hub->handlers[event_type][i].user_data;
221 if (handler) {
222 if (lle_event_debug_enabled()) {
223 fprintf(stderr, "[LLE_EVENT] -> handler[%zu]: %s\n", i,
224 hub->handlers[event_type][i].name
225 ? hub->handlers[event_type][i].name
226 : "(unnamed)");
227 }
228 handler(event_data, user_data);
229 }
230 }
231}

238void lle_fire_directory_changed(const char *old_dir, const char *new_dir) {
239 /* Get global shell integration */
240 if (!g_lle_integration || !g_lle_integration->event_hub) {
241 if (lle_event_debug_enabled()) {
242 fprintf(stderr,
243 "[LLE_EVENT] WARN: lle_fire_directory_changed: "
244 "g_lle_integration=%p event_hub=%p\n",
245 (void *)g_lle_integration,
246 g_lle_integration ? (void *)g_lle_integration->event_hub
247 : NULL);
248 }
249 return;
250 }
251
252 lle_shell_event_hub_t *hub = g_lle_integration->event_hub;
253 char new_dir_buf[PATH_MAX];
254
255 /* Use provided old_dir or hub's tracked current_dir */
256 const char *actual_old_dir = old_dir ? old_dir : hub->current_dir;
257
258 /* Get new directory if not provided */
259 const char *actual_new_dir = new_dir;
260 if (!actual_new_dir) {
261 if (getcwd(new_dir_buf, sizeof(new_dir_buf)) != NULL) {
262 actual_new_dir = new_dir_buf;
263 } else {
264 if (lle_event_debug_enabled()) {
265 fprintf(stderr,
266 "[LLE_EVENT] WARN: lle_fire_directory_changed: "
267 "getcwd() failed: %s\n",
268 strerror(errno));
269 }
270 return; /* Can't determine new directory */
271 }
272 }
273
274 /* Create event data */
275 lle_directory_changed_event_t event = {.old_dir = actual_old_dir,
276 .new_dir = actual_new_dir};
277
278 /* Update hub's current directory tracking */
279 strncpy(hub->current_dir, actual_new_dir, sizeof(hub->current_dir) - 1);
280 hub->current_dir[sizeof(hub->current_dir) - 1] = '\0';
281
282 /* Update statistics */
283 hub->directory_changes++;
284
285 /* Fire to registered handlers */
286 lle_shell_event_hub_fire(hub, LLE_SHELL_EVENT_DIRECTORY_CHANGED, &event);
287}

289void lle_fire_pre_command(const char *command, bool is_background) {
290 /* Get global shell integration */
291 if (!g_lle_integration || !g_lle_integration->event_hub) {
292 if (lle_event_debug_enabled()) {
293 fprintf(stderr,
294 "[LLE_EVENT] WARN: lle_fire_pre_command: "
295 "g_lle_integration=%p event_hub=%p\n",
296 (void *)g_lle_integration,
297 g_lle_integration ? (void *)g_lle_integration->event_hub
298 : NULL);
299 }
300 return;
301 }
302
303 if (lle_event_debug_enabled()) {
304 fprintf(stderr,
305 "[LLE_EVENT] lle_fire_pre_command: cmd='%.40s%s' bg=%d\n",
306 command ? command : "(null)",
307 (command && strlen(command) > 40) ? "..." : "", is_background);
308 }
309
310 lle_shell_event_hub_t *hub = g_lle_integration->event_hub;
311
312 /* Record command start time */
313 hub->command_start_time_us = lle_shell_event_get_timestamp_us();
314
315 /* Store command for post-command event */
316 if (command) {
317 strncpy(hub->current_command, command,
318 sizeof(hub->current_command) - 1);
319 hub->current_command[sizeof(hub->current_command) - 1] = '\0';
320 } else {
321 hub->current_command[0] = '\0';
322 }
323
324 /* Create event data */
325 lle_pre_command_event_t event = {.command = command,
326 .is_background = is_background};
327
328 /* Fire to registered handlers */
329 lle_shell_event_hub_fire(hub, LLE_SHELL_EVENT_PRE_COMMAND, &event);
330}

332void lle_fire_post_command(const char *command, int exit_code,
333 uint64_t duration_us) {
334 /* Get global shell integration */
335 if (!g_lle_integration || !g_lle_integration->event_hub) {
336 if (lle_event_debug_enabled()) {
337 fprintf(stderr,
338 "[LLE_EVENT] WARN: lle_fire_post_command: "
339 "g_lle_integration=%p event_hub=%p exit=%d\n",
340 (void *)g_lle_integration,
341 g_lle_integration ? (void *)g_lle_integration->event_hub
342 : NULL,
343 exit_code);
344 }
345 return;
346 }
347
348 if (lle_event_debug_enabled()) {
349 fprintf(stderr,
350 "[LLE_EVENT] lle_fire_post_command: exit=%d duration=%lluus\n",
351 exit_code, (unsigned long long)duration_us);
352 }
353
354 lle_shell_event_hub_t *hub = g_lle_integration->event_hub;
355
356 /* Calculate duration if not provided */
357 uint64_t actual_duration = duration_us;
358 if (actual_duration == 0 && hub->command_start_time_us > 0) {
359 uint64_t end_time = lle_shell_event_get_timestamp_us();
360 actual_duration = end_time - hub->command_start_time_us;
361 }
362
363 /* Use stored command if not provided */
364 const char *actual_command = command ? command : hub->current_command;
365
366 /* Create event data */
367 lle_post_command_event_t event = {.command = actual_command,
368 .exit_code = exit_code,
369 .duration_us = actual_duration};
370
371 /* Update statistics */
372 hub->commands_executed++;
373
374 /* Reset command tracking */
375 hub->command_start_time_us = 0;
376 hub->current_command[0] = '\0';
377
378 /* Fire to registered handlers */
379 lle_shell_event_hub_fire(hub, LLE_SHELL_EVENT_POST_COMMAND, &event);
380}
