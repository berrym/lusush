33lle_result_t lle_multiline_context_init(lle_multiline_context_t **ctx,
34 lusush_memory_pool_t *memory_pool) {
35 if (!ctx || !memory_pool) {
36 return LLE_ERROR_INVALID_PARAMETER;
37 }
38
39 /* Allocate context structure */
40 lle_multiline_context_t *context =
41 lle_pool_alloc(sizeof(lle_multiline_context_t));
42 if (!context) {
43 return LLE_ERROR_OUT_OF_MEMORY;
44 }
45
46 /* Initialize fields */
47 memset(context, 0, sizeof(lle_multiline_context_t));
48 context->memory_pool = memory_pool;
49
50 /* Allocate and initialize core parser state */
51 continuation_state_t *core_state =
52 lle_pool_alloc(sizeof(continuation_state_t));
53 if (!core_state) {
54 lle_pool_free(context);
55 return LLE_ERROR_OUT_OF_MEMORY;
56 }
57
58 continuation_state_init(core_state);
59 context->core_state = core_state;
60
61 /* Initialize LLE-specific fields */
62 context->current_construct = NULL;
63 context->construct_start_line = 0;
64 context->construct_start_offset = 0;
65 context->nesting_level = 0;
66 context->construct_complete = true;
67 context->needs_continuation = false;
68 context->expected_terminator = NULL;
69 context->cache_key = 0;
70 context->cache_valid = false;
71
72 *ctx = context;
73 return LLE_SUCCESS;
74}

141static const char *get_construct_name(const continuation_state_t *state) {
142 if (!state) {
143 return NULL;
144 }
145
146 /* Check quote states first (highest priority) */
147 if (state->in_single_quote) {
148 return "single quote";
149 }
150 if (state->in_double_quote) {
151 return "double quote";
152 }
153 if (state->in_backtick) {
154 return "backtick";
155 }
156
157 /* Check here document */
158 if (state->in_here_doc) {
159 return "here document";
160 }
161
162 /* Check control structures */
163 if (state->in_function_definition) {
164 return "function definition";
165 }
166 if (state->in_case_statement) {
167 return "case statement";
168 }
169 if (state->in_if_statement) {
170 return "if statement";
171 }
172 if (state->in_while_loop) {
173 return "while loop";
174 }
175 if (state->in_for_loop) {
176 return "for loop";
177 }
178 if (state->in_until_loop) {
179 return "until loop";
180 }
181
182 /* Check brackets/braces */
183 if (state->brace_count > 0) {
184 return "brace group";
185 }
186 if (state->paren_count > 0) {
187 return "subshell";
188 }
189 if (state->bracket_count > 0) {
190 return "bracket expression";
191 }
192
193 /* Check command substitution */
194 if (state->in_command_substitution) {
195 return "command substitution";
196 }
197 if (state->in_arithmetic) {
198 return "arithmetic expansion";
199 }
200
201 /* Check continuation */
202 if (state->has_continuation) {
203 return "line continuation";
204 }
205
206 return NULL;
207}

209/**
210 * @brief Helper to calculate nesting level from parser state
211 */
212static uint8_t get_nesting_level(const continuation_state_t *state) {
213 if (!state) {
214 return 0;
215 }
216
217 uint8_t level = 0;
218
219 /* Count nesting from various constructs */
220 level += state->paren_count;
221 level += state->brace_count;
222 level += state->bracket_count;
223 level += state->compound_command_depth;
224
225 /* Quote states don't nest but contribute to depth */
226 if (state->in_single_quote || state->in_double_quote ||
227 state->in_backtick) {
228 level += 1;
229 }
230
231 return level;
232}

234lle_result_t lle_multiline_analyze_line(lle_multiline_context_t *ctx,
235 const char *line, size_t length) {
236 (void)length; /* Reserved for length-aware analysis */
237 if (!ctx || !line) {
238 return LLE_ERROR_INVALID_PARAMETER;
239 }
240
241 continuation_state_t *state = (continuation_state_t *)ctx->core_state;
242 if (!state) {
243 return LLE_ERROR_INVALID_STATE;
244 }
245
246 /* Delegate core parsing to shared parser */
247 continuation_analyze_line(line, state);
248
249 /* Extract and cache LLE-specific state */
250 const char *construct = get_construct_name(state);
251 if (construct) {
252 /* Allocate and copy construct name if changed */
253 if (!ctx->current_construct ||
254 strcmp(ctx->current_construct, construct) != 0) {
255 if (ctx->current_construct) {
256 lle_pool_free(ctx->current_construct);
257 }
258 size_t len = strlen(construct) + 1;
259 ctx->current_construct = lle_pool_alloc(len);
260 if (ctx->current_construct) {
261 memcpy(ctx->current_construct, construct, len);
262 }
263 }
264 } else {
265 /* No construct - clear */
266 if (ctx->current_construct) {
267 lle_pool_free(ctx->current_construct);
268 ctx->current_construct = NULL;
269 }
270 }
271
272 /* Update nesting level */
273 ctx->nesting_level = get_nesting_level(state);
274
275 /* Update completion state */
276 ctx->construct_complete = continuation_is_complete(state);
277 ctx->needs_continuation = continuation_needs_continuation(state);
278
279 /* Invalidate cache since state changed */
280 ctx->cache_valid = false;
281
282 return LLE_SUCCESS;
283}

370/**
371 * @brief Helper to convert continuation_state_t to lle_multiline_state_t
372 */
373static lle_multiline_state_t
374convert_to_lle_state(const continuation_state_t *state) {
375 if (!state) {
376 return LLE_MULTILINE_STATE_NONE;
377 }
378
379 /* Priority order matches input_continuation.c logic */
380 if (state->in_single_quote) {
381 return LLE_MULTILINE_STATE_QUOTE_SINGLE;
382 }
383 if (state->in_double_quote) {
384 return LLE_MULTILINE_STATE_QUOTE_DOUBLE;
385 }
386 if (state->in_backtick) {
387 return LLE_MULTILINE_STATE_QUOTE_BACKTICK;
388 }
389 if (state->in_here_doc) {
390 return LLE_MULTILINE_STATE_HEREDOC;
391 }
392 /* Control structures - map to generic states since specific ones don't
393 * exist */
394 if (state->in_if_statement || state->in_case_statement ||
395 state->in_while_loop || state->in_for_loop || state->in_until_loop ||
396 state->in_function_definition) {
397 return LLE_MULTILINE_STATE_BRACE; /* Use brace as generic control
398 structure */
399 }
400 if (state->brace_count > 0) {
401 return LLE_MULTILINE_STATE_BRACE;
402 }
403 if (state->paren_count > 0) {
404 return LLE_MULTILINE_STATE_PAREN;
405 }
406 if (state->bracket_count > 0) {
407 return LLE_MULTILINE_STATE_BRACKET;
408 }
409 if (state->has_continuation) {
410 return LLE_MULTILINE_STATE_BACKSLASH;
411 }
412
413 return LLE_MULTILINE_STATE_NONE;
414}

416lle_result_t
417lle_multiline_manager_analyze_buffer(lle_multiline_manager_t *manager,
418 lle_buffer_t *buffer) {
419 if (!manager || !buffer) {
420 return LLE_ERROR_INVALID_PARAMETER;
421 }
422
423 lle_result_t result = LLE_SUCCESS;
424
425 /* Step 1: Initialize or reset multiline context */
426 if (!buffer->multiline_ctx) {
427 result = lle_multiline_context_init(&buffer->multiline_ctx,
428 manager->memory_pool);
429 if (result != LLE_SUCCESS) {
430 return result;
431 }
432 } else {
433 result = lle_multiline_context_reset(buffer->multiline_ctx);
434 if (result != LLE_SUCCESS) {
435 return result;
436 }
437 }
438
439 /* Step 2: Analyze each line in the buffer */
440 if (buffer->line_count > 0) {
441 /* Buffer has line structure - analyze each line */
442 for (size_t line_idx = 0; line_idx < buffer->line_count; line_idx++) {
443 lle_line_info_t *line = &buffer->lines[line_idx];
444
445 /* Extract line content */
446 if (line->start_offset + line->length > buffer->length) {
447 return LLE_ERROR_BUFFER_OVERFLOW;
448 }
449
450 const char *line_content =
451 (const char *)(buffer->data + line->start_offset);
452 size_t line_length = line->length;
453
454 /* Analyze line */
455 result = lle_multiline_analyze_line(buffer->multiline_ctx,
456 line_content, line_length);
457 if (result != LLE_SUCCESS) {
458 return result;
459 }
460
461 /* Update line multiline state */
462 continuation_state_t *state =
463 (continuation_state_t *)buffer->multiline_ctx->core_state;
464 line->ml_state = convert_to_lle_state(state);
465
466 /* Update line flags */
467 if (buffer->multiline_ctx->needs_continuation) {
468 line->flags |= LLE_LINE_FLAG_CONTINUATION;
469 } else {
470 line->flags &= ~LLE_LINE_FLAG_CONTINUATION;
471 }
472
473 manager->line_updates++;
474 }
475 } else if (buffer->length > 0) {
476 /* Buffer has text but no line structure - treat entire buffer as one
477 * line */
478 result = lle_multiline_analyze_line(
479 buffer->multiline_ctx, (const char *)buffer->data, buffer->length);
480 if (result != LLE_SUCCESS) {
481 return result;
482 }
483 manager->line_updates++;
484 }
485
486 /* Step 3: Update buffer-wide multiline status */
487 buffer->multiline_active = !buffer->multiline_ctx->construct_complete;
488
489 /* Update statistics */
490 manager->analysis_count++;
491
492 return LLE_SUCCESS;
493}

495lle_result_t lle_multiline_manager_update_line_state(
496 lle_multiline_manager_t *manager, lle_buffer_t *buffer, size_t line_index) {
497 if (!manager || !buffer) {
498 return LLE_ERROR_INVALID_PARAMETER;
499 }
500
501 if (line_index >= buffer->line_count) {
502 return LLE_ERROR_INVALID_PARAMETER; /* Use existing error code for out
503 of bounds */
504 }
505
506 /* For single-line update, we need to re-analyze from the beginning
507 * because multiline state is cumulative. This is acceptable for
508 * typical shell inputs (<100 lines). */
509 return lle_multiline_manager_analyze_buffer(manager, buffer);
510}
