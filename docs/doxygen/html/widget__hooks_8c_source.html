37 lle_widget_registry_t *registry,
38 lle_memory_pool_t *memory_pool) {
39 if (!manager || !registry || !memory_pool) {

44 lle_widget_hooks_manager_t *mgr =

46 if (!mgr) {

70 if (!manager) {

74 /* Free all hook registrations */

95lle_result_t lle_widget_hook_register(lle_widget_hooks_manager_t *manager,
96 lle_widget_hook_type_t hook_type,
97 const char *widget_name) {
98 if (!manager || !widget_name) {
99 return LLE_ERROR_INVALID_PARAMETER;
100 }
101
102 if (hook_type >= LLE_HOOK_COUNT) {
103 return LLE_ERROR_INVALID_PARAMETER;
104 }
105
106 /* Lookup widget in registry */
107 lle_widget_t *widget = lle_widget_lookup(manager->registry, widget_name);
108 if (!widget) {
109 return LLE_ERROR_NOT_FOUND;
110 }
111
112 /* Check if widget already registered for this hook */
113 lle_hook_registration_t *existing = manager->hooks[hook_type];
114 while (existing) {
115 if (existing->widget == widget) {
116 return LLE_ERROR_ALREADY_EXISTS;
117 }
118 existing = existing->next;
119 }
120
121 /* Allocate registration structure */
122 lle_hook_registration_t *reg =
123 lle_pool_alloc(sizeof(lle_hook_registration_t));
124 if (!reg) {
125 return LLE_ERROR_OUT_OF_MEMORY;
126 }
127
128 /* Initialize registration */
129 reg->widget = widget;
130 reg->hook_type = hook_type;
131 reg->trigger_count = 0;
132 reg->enabled = true;
133 reg->next = NULL;
134
135 /* Add to end of list (maintains registration order) */
136 if (!manager->hooks[hook_type]) {
137 /* First registration for this hook */
138 manager->hooks[hook_type] = reg;
139 } else {
140 /* Find end of list */
141 lle_hook_registration_t *last = manager->hooks[hook_type];
142 while (last->next) {
143 last = last->next;
144 }
145 last->next = reg;
146 }
147
148 return LLE_SUCCESS;
149}

151lle_result_t lle_widget_hook_unregister(lle_widget_hooks_manager_t *manager,
152 lle_widget_hook_type_t hook_type,
153 const char *widget_name) {
154 if (!manager || !widget_name) {
155 return LLE_ERROR_INVALID_PARAMETER;
156 }
157
158 if (hook_type >= LLE_HOOK_COUNT) {
159 return LLE_ERROR_INVALID_PARAMETER;
160 }
161
162 /* Lookup widget in registry */
163 lle_widget_t *widget = lle_widget_lookup(manager->registry, widget_name);
164 if (!widget) {
165 return LLE_ERROR_NOT_FOUND;
166 }
167
168 /* Find and remove registration */
169 lle_hook_registration_t *reg = manager->hooks[hook_type];
170 lle_hook_registration_t *prev = NULL;
171
172 while (reg) {
173 if (reg->widget == widget) {
174 /* Found the registration - remove it */
175 if (prev) {
176 prev->next = reg->next;
177 } else {
178 manager->hooks[hook_type] = reg->next;
179 }
180
181 lle_pool_free(reg);
182 return LLE_SUCCESS;
183 }
184
185 prev = reg;
186 reg = reg->next;
187 }
188
189 return LLE_ERROR_NOT_FOUND;
190}

192lle_result_t lle_widget_hook_trigger(lle_widget_hooks_manager_t *manager,
193 lle_widget_hook_type_t hook_type,
194 lle_editor_t *editor) {
195 if (!manager || !editor) {
196 return LLE_ERROR_INVALID_PARAMETER;
197 }
198
199 if (hook_type >= LLE_HOOK_COUNT) {
200 return LLE_ERROR_INVALID_PARAMETER;
201 }
202
203 /* Check if hooks are globally enabled */
204 if (!manager->hooks_enabled) {
205 return LLE_SUCCESS;
206 }
207
208 /* Update statistics */
209 manager->total_hooks_triggered++;
210 manager->hook_trigger_counts[hook_type]++;
211
212 /* Execute all widgets registered for this hook */
213 lle_hook_registration_t *reg = manager->hooks[hook_type];
214
215 while (reg) {
216 /* Check if registration is enabled */
217 if (!reg->enabled) {
218 reg = reg->next;
219 continue;
220 }
221
222 /* Execute widget (error-resilient: continue even if widget fails) */
223 lle_widget_execute_direct(reg->widget, editor);
224
225 /* Note: Widget errors are logged but do not stop hook execution.
226 * This ensures that one misbehaving widget doesn't break the entire
227 * hook chain. Individual widget failures can be diagnosed via widget
228 * execution statistics and logs. */
229
230 /* Update registration statistics */
231 reg->trigger_count++;
232
233 reg = reg->next;
234 }
235
236 /* Hook triggering always succeeds (error-resilient design) */
237 return LLE_SUCCESS;
238}
