26/**
27 * Get the byte length of a UTF-8 character at the given position.
28 * Returns 1 for ASCII characters, 2-4 for multi-byte sequences.
29 * Returns 1 for invalid sequences (safe fallback).
30 */
31static int utf8_char_len(const char *p) {
32 if (!p || !*p)
33 return 0;
34
35 unsigned char c = (unsigned char)*p;
36
37 // ASCII (0x00-0x7F) - single byte
38 if (c < 0x80)
39 return 1;
40
41 // Use LLE's UTF-8 sequence length detection
42 int len = lle_utf8_sequence_length(c);
43
44 // Validate the sequence
45 if (len > 1 && lle_utf8_is_valid_sequence(p, len)) {
46 return len;
47 }
48
49 // Invalid sequence - skip one byte
50 return 1;
51}

92/**
93 * Pop a context from the stack if it matches the expected type
94 * Returns true if popped successfully
95 */
96static bool context_stack_pop(continuation_state_t *state,
97 continuation_context_type_t expected) {
98 if (!state || state->context_stack_depth <= 0)
99 return false;
100
101 // Check if top matches expected (for proper nesting validation)
102 continuation_context_type_t top =
103 state->context_stack[state->context_stack_depth - 1];
104 if (top == expected) {
105 state->context_stack_depth--;
106 return true;
107 }
108
109 // For loops, accept any loop type (for/while/until all end with 'done')
110 if (expected == CONTEXT_FOR || expected == CONTEXT_WHILE ||
111 expected == CONTEXT_UNTIL) {
112 if (top == CONTEXT_FOR || top == CONTEXT_WHILE ||
113 top == CONTEXT_UNTIL) {
114 state->context_stack_depth--;
115 return true;
116 }
117 }
118
119 return false;
120}

152bool continuation_is_control_keyword(const char *word) {
153 if (!word)
154 return false;
155
156 const char *keywords[] = {"if", "then", "else", "elif", "fi",
157 "case", "esac", "while", "until", "do",
158 "done", "for", "in", "function", "select",
159 "{", "}"};
160
161 for (size_t i = 0; i < sizeof(keywords) / sizeof(keywords[0]); i++) {
162 if (strcmp(word, keywords[i]) == 0) {
163 return true;
164 }
165 }
166 return false;
167}

169bool continuation_is_terminator(const char *line) {
170 if (!line)
171 return false;
172
173 // Skip whitespace
174 while (*line && isspace(*line))
175 line++;
176
177 // Check for terminators
178 return (strncmp(line, "fi", 2) == 0 || strncmp(line, "done", 4) == 0 ||
179 strncmp(line, "esac", 4) == 0 || strcmp(line, "}") == 0);
180}

186void continuation_analyze_line(const char *line, continuation_state_t *state) {
187 if (!line || !state)
188 return;
189
190 const char *p = line;
191 char word[256] = {0};
192 int word_pos = 0;
193
194 while (*p) {
195 unsigned char uc = (unsigned char)*p;
196 char c = *p;
197
198 // UTF-8 multi-byte sequence handling:
199 // If this is a non-ASCII byte (high bit set), it's part of a UTF-8
200 // multi-byte character. Skip the entire sequence since shell syntax
201 // characters are all ASCII. This prevents misinterpreting UTF-8
202 // continuation bytes as shell metacharacters.
203 if (uc >= 0x80) {
204 int char_len = utf8_char_len(p);
205
206 // If we're collecting a word, flush it first (UTF-8 chars break
207 // words for keyword detection purposes - keywords are ASCII only)
208 if (word_pos > 0) {
209 word[word_pos] = '\0';
210 // Keywords are ASCII-only, so no need to check here
211 word_pos = 0;
212 memset(word, 0, sizeof(word));
213 }
214
215 // Skip the entire UTF-8 sequence
216 p += (char_len > 0) ? char_len : 1;
217 continue;
218 }
219
220 // Handle escape sequences
221 if (state->escaped) {
222 state->escaped = false;
223 p++;
224 continue;
225 }
226
227 if (c == '\\') {
228 state->escaped = true;
229 p++;
230 continue;
231 }
232
233 // Handle quotes
234 if (c == '\'' && !state->in_double_quote && !state->in_backtick) {
235 state->in_single_quote = !state->in_single_quote;
236 if (state->in_single_quote) {
237 state->quote_count++;
238 }
239 } else if (c == '"' && !state->in_single_quote) {
240 state->in_double_quote = !state->in_double_quote;
241 if (state->in_double_quote) {
242 state->double_quote_count++;
243 }
244 } else if (c == '`' && !state->in_single_quote) {
245 state->in_backtick = !state->in_backtick;
246 if (state->in_backtick) {
247 state->backtick_count++;
248 }
249 }
250
251 // Skip if we're in quotes
252 if (state->in_single_quote || state->in_double_quote ||
253 state->in_backtick) {
254 p++;
255 continue;
256 }
257
258 // Handle parentheses, braces, brackets
259 if (c == '(') {
260 state->paren_count++;
261 } else if (c == ')') {
262 state->paren_count--;
263 } else if (c == '{') {
264 state->brace_count++;
265 } else if (c == '}') {
266 state->brace_count--;
267 } else if (c == '[') {
268 state->bracket_count++;
269 } else if (c == ']') {
270 state->bracket_count--;
271 }
272
273 // Handle here document detection
274 if (c == '<' && *(p + 1) == '<' && !state->in_here_doc) {
275 // Found <<, look for delimiter
276 const char *delim_start = p + 2;
277
278 // Skip optional '-' for <<-
279 if (*delim_start == '-') {
280 delim_start++;
281 }
282
283 // Skip whitespace
284 while (*delim_start == ' ' || *delim_start == '\t') {
285 delim_start++;
286 }
287
288 // Extract delimiter (up to end of line or whitespace)
289 const char *delim_end = delim_start;
290 while (*delim_end && *delim_end != '\n' && *delim_end != ' ' &&
291 *delim_end != '\t') {
292 delim_end++;
293 }
294
295 if (delim_end > delim_start) {
296 // Handle quoted delimiters - strip surrounding quotes
297 const char *actual_delim_start = delim_start;
298 const char *actual_delim_end = delim_end;
299
300 // Check for single or double quotes
301 if ((*delim_start == '\'' || *delim_start == '"') &&
302 delim_end > delim_start + 1 &&
303 *(delim_end - 1) == *delim_start) {
304 // Strip quotes
305 actual_delim_start++;
306 actual_delim_end--;
307 }
308
309 // Found a delimiter, enter here document mode
310 state->in_here_doc = true;
311 if (state->here_doc_delimiter) {
312 free(state->here_doc_delimiter);
313 }
314 size_t delim_len = actual_delim_end - actual_delim_start;
315 state->here_doc_delimiter = malloc(delim_len + 1);
316 if (state->here_doc_delimiter) {
317 strncpy(state->here_doc_delimiter, actual_delim_start,
318 delim_len);
319 state->here_doc_delimiter[delim_len] = '\0';
320 }
321 }
322 }
323
324 // Check if current line is a here document delimiter (ends here doc)
325 if (state->in_here_doc && state->here_doc_delimiter) {
326 // Check if this entire line matches the delimiter
327 const char *line_start = line;
328 while (*line_start == ' ' || *line_start == '\t') {
329 line_start++; // Skip leading whitespace
330 }
331
332 if (strncmp(line_start, state->here_doc_delimiter,
333 strlen(state->here_doc_delimiter)) == 0) {
334 // Check if delimiter is followed by end of line or whitespace
335 const char *after_delim =
336 line_start + strlen(state->here_doc_delimiter);
337 if (*after_delim == '\0' || *after_delim == '\n' ||
338 *after_delim == ' ' || *after_delim == '\t') {
339 // This line is the delimiter, end here document
340 state->in_here_doc = false;
341 free(state->here_doc_delimiter);
342 state->here_doc_delimiter = NULL;
343 }
344 }
345 }
346
347 // Collect words for keyword analysis
348 if (isalnum(c) || c == '_') {
349 if (word_pos < (int)sizeof(word) - 1) {
350 word[word_pos++] = c;
351 }
352 } else if (c == '{' || c == '}') {
353 // Handle { and } as single-character keywords
354 if (word_pos > 0) {
355 // Process any accumulated word first
356 word[word_pos] = '\0';
357
358 // Check for control keywords
359 if (continuation_is_control_keyword(word)) {
360
361 if (strcmp(word, "if") == 0) {
362 state->in_if_statement = true;
363 state->compound_command_depth++;
364 context_stack_push(state, CONTEXT_IF);
365 } else if (strcmp(word, "while") == 0) {
366 state->in_while_loop = true;
367 state->compound_command_depth++;
368 context_stack_push(state, CONTEXT_WHILE);
369 } else if (strcmp(word, "for") == 0) {
370 state->in_for_loop = true;
371 state->compound_command_depth++;
372 context_stack_push(state, CONTEXT_FOR);
373 } else if (strcmp(word, "until") == 0) {
374 state->in_until_loop = true;
375 state->compound_command_depth++;
376 context_stack_push(state, CONTEXT_UNTIL);
377 } else if (strcmp(word, "case") == 0) {
378 state->in_case_statement = true;
379 state->compound_command_depth++;
380 context_stack_push(state, CONTEXT_CASE);
381 } else if (strcmp(word, "function") == 0) {
382 state->in_function_definition = true;
383 state->compound_command_depth++;
384 context_stack_push(state, CONTEXT_FUNCTION);
385 } else if (strcmp(word, "fi") == 0) {
386 state->in_if_statement = false;
387 state->has_continuation = false;
388 if (state->compound_command_depth > 0) {
389 state->compound_command_depth--;
390 }
391 context_stack_pop(state, CONTEXT_IF);
392 } else if (strcmp(word, "done") == 0) {
393 state->in_while_loop = false;
394 state->in_for_loop = false;
395 state->in_until_loop = false;
396 state->has_continuation = false;
397 if (state->compound_command_depth > 0) {
398 state->compound_command_depth--;
399 }
400 context_stack_pop_loop(state);
401 } else if (strcmp(word, "esac") == 0) {
402 state->in_case_statement = false;
403 state->has_continuation = false;
404 if (state->compound_command_depth > 0) {
405 state->compound_command_depth--;
406 }
407 context_stack_pop(state, CONTEXT_CASE);
408 }
409 }
410
411 word_pos = 0;
412 memset(word, 0, sizeof(word));
413 }
414
415 // Now handle the { or } character as a single-character keyword
416 if (c == '{') {
417 // Only increment depth if not already in a function definition
418 // (function keyword already incremented depth)
419 if (!state->in_function_definition) {
420 state->compound_command_depth++;
421 context_stack_push(state, CONTEXT_BRACE_GROUP);
422 }
423 } else if (c == '}') {
424 if (state->compound_command_depth > 0) {
425 state->compound_command_depth--;
426 }
427 context_stack_pop(state, CONTEXT_BRACE_GROUP);
428 // Also try popping function context
429 if (state->context_stack_depth > 0 &&
430 context_stack_top(state) == CONTEXT_FUNCTION) {
431 context_stack_pop(state, CONTEXT_FUNCTION);
432 }
433 if (state->compound_command_depth == 0) {
434 state->in_function_definition = false;
435 }
436 }
437 } else {
438 if (word_pos > 0) {
439 word[word_pos] = '\0';
440
441 // Check for control keywords
442 if (continuation_is_control_keyword(word)) {
443
444 if (strcmp(word, "if") == 0) {
445 state->in_if_statement = true;
446 state->compound_command_depth++;
447 context_stack_push(state, CONTEXT_IF);
448 } else if (strcmp(word, "while") == 0) {
449 state->in_while_loop = true;
450 state->compound_command_depth++;
451 context_stack_push(state, CONTEXT_WHILE);
452 } else if (strcmp(word, "for") == 0) {
453 state->in_for_loop = true;
454 state->compound_command_depth++;
455 context_stack_push(state, CONTEXT_FOR);
456 } else if (strcmp(word, "until") == 0) {
457 state->in_until_loop = true;
458 state->compound_command_depth++;
459 context_stack_push(state, CONTEXT_UNTIL);
460 } else if (strcmp(word, "case") == 0) {
461 state->in_case_statement = true;
462 state->compound_command_depth++;
463 context_stack_push(state, CONTEXT_CASE);
464 } else if (strcmp(word, "function") == 0) {
465 state->in_function_definition = true;
466 state->compound_command_depth++;
467 context_stack_push(state, CONTEXT_FUNCTION);
468 } else if (strcmp(word, "{") == 0) {
469 state->compound_command_depth++;
470 context_stack_push(state, CONTEXT_BRACE_GROUP);
471 } else if (strcmp(word, "fi") == 0) {
472 state->in_if_statement = false;
473 state->has_continuation = false;
474 if (state->compound_command_depth > 0) {
475 state->compound_command_depth--;
476 }
477 context_stack_pop(state, CONTEXT_IF);
478 } else if (strcmp(word, "done") == 0) {
479 state->in_while_loop = false;
480 state->in_for_loop = false;
481 state->in_until_loop = false;
482 state->has_continuation = false;
483 if (state->compound_command_depth > 0) {
484 state->compound_command_depth--;
485 }
486 context_stack_pop_loop(state);
487 } else if (strcmp(word, "esac") == 0) {
488 state->in_case_statement = false;
489 state->has_continuation = false;
490 if (state->compound_command_depth > 0) {
491 state->compound_command_depth--;
492 }
493 context_stack_pop(state, CONTEXT_CASE);
494 } else if (strcmp(word, "}") == 0) {
495 if (state->compound_command_depth > 0) {
496 state->compound_command_depth--;
497 }
498 context_stack_pop(state, CONTEXT_BRACE_GROUP);
499 if (state->compound_command_depth == 0) {
500 state->in_function_definition = false;
501 }
502 }
503 }
504
505 word_pos = 0;
506 memset(word, 0, sizeof(word));
507 }
508 }
509
510 p++;
511 }
512
513 // Check if line ends with unescaped backslash (line continuation)
514 // Must check after loop completes to handle escaped flag correctly
515 if (state->escaped) {
516 state->has_continuation = true;
517 state->escaped = false; // Reset for next line
518 }
519
520 // Check if line ends with pipe character (requires continuation)
521 // Need to check backwards from end, skipping whitespace
522 const char *end = line + strlen(line);
523 while (end > line && isspace(*(end - 1))) {
524 end--;
525 }
526 if (end > line && *(end - 1) == '|') {
527 // Line ends with pipe - needs continuation
528 state->has_continuation = true;
529 }
530
531 // Handle remaining word at end of line
532 if (word_pos > 0) {
533 word[word_pos] = '\0';
534
535 if (continuation_is_control_keyword(word)) {
536 // Handle keywords found at end of line
537 if (strcmp(word, "then") == 0 || strcmp(word, "do") == 0) {
538 state->has_continuation = true;
539 } else if (strcmp(word, "done") == 0) {
540 state->in_while_loop = false;
541 state->in_for_loop = false;
542 state->in_until_loop = false;
543 state->has_continuation = false;
544 if (state->compound_command_depth > 0) {
545 state->compound_command_depth--;
546 }
547 context_stack_pop_loop(state);
548 } else if (strcmp(word, "esac") == 0) {
549 state->in_case_statement = false;
550 state->has_continuation = false;
551 if (state->compound_command_depth > 0) {
552 state->compound_command_depth--;
553 }
554 context_stack_pop(state, CONTEXT_CASE);
555 } else if (strcmp(word, "}") == 0) {
556 if (state->compound_command_depth > 0) {
557 state->compound_command_depth--;
558 }
559 context_stack_pop(state, CONTEXT_BRACE_GROUP);
560 if (state->compound_command_depth == 0) {
561 state->in_function_definition = false;
562 }
563 } else if (strcmp(word, "fi") == 0) {
564 state->in_if_statement = false;
565 state->has_continuation = false;
566 if (state->compound_command_depth > 0) {
567 state->compound_command_depth--;
568 }
569 context_stack_pop(state, CONTEXT_IF);
570 }
571 }
572 }

579bool continuation_is_complete(const continuation_state_t *state) {
580 if (!state)
581 return true;
582
583 // Check for unmatched quotes
584 if (state->in_single_quote || state->in_double_quote ||
585 state->in_backtick) {
586 return false;
587 }
588
589 // Check for unmatched parentheses, braces, brackets
590 if (state->paren_count > 0 || state->brace_count > 0 ||
591 state->bracket_count > 0) {
592 return false;
593 }
594
595 // Check for incomplete compound commands
596 if (state->compound_command_depth > 0) {
597 return false;
598 }
599
600 // Check for line continuation
601 if (state->has_continuation) {
602 return false;
603 }
604
605 // Check for incomplete control structures
606 if (state->in_if_statement || state->in_while_loop || state->in_for_loop ||
607 state->in_until_loop || state->in_case_statement ||
608 state->in_function_definition) {
609 return false;
610 }
611
612 // Check for here documents
613 if (state->in_here_doc) {
614 return false;
615 }
616
617 return true;
618}

628const char *continuation_get_prompt(const continuation_state_t *state) {
629 if (!state)
630 return "> ";
631
632 // Use PS2 from symbol table, with fallback
633 const char *ps2 = symtable_get_global_default("PS2", "> ");
634
635 // Quote state takes highest priority (not tracked in context stack)
636 if (state->in_single_quote || state->in_double_quote) {
637 return "quote> ";
638 }
639
640 // Use context stack for proper nested construct tracking
641 // The top of the stack represents the innermost active construct
642 continuation_context_type_t current_ctx = context_stack_top(state);
643
644 switch (current_ctx) {
645 case CONTEXT_IF:
646 return "if> ";
647 case CONTEXT_WHILE:
648 return "while> ";
649 case CONTEXT_FOR:
650 return "for> ";
651 case CONTEXT_UNTIL:
652 return "until> ";
653 case CONTEXT_CASE:
654 return "case> ";
655 case CONTEXT_FUNCTION:
656 return "func> ";
657 case CONTEXT_BRACE_GROUP:
658 return "brace> ";
659 case CONTEXT_NONE:
660 default:
661 break;
662 }
663
664 // Fallback to legacy boolean flags for backwards compatibility
665 // (in case context stack isn't being used consistently)
666 if (state->in_function_definition) {
667 return "func> ";
668 } else if (state->in_if_statement) {
669 return "if> ";
670 } else if (state->in_while_loop) {
671 return "while> ";
672 } else if (state->in_for_loop) {
673 return "for> ";
674 } else if (state->in_until_loop) {
675 return "until> ";
676 } else if (state->in_case_statement) {
677 return "case> ";
678 }
679
680 return ps2;
681}
