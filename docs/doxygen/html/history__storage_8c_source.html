42/**
43 * Acquire exclusive lock on history file
44 *
45 * Uses flock() for advisory locking to prevent concurrent writes
46 * from multiple shell instances.
47 *
48 * @param fd File descriptor
49 * @return LLE_SUCCESS or error code
50 */
51static lle_result_t lle_history_file_lock(int fd) {
52 if (fd < 0) {
53 return LLE_ERROR_INVALID_PARAMETER;
54 }
55
56 /* Try to acquire exclusive lock with timeout */
57 struct timespec timeout = {.tv_sec = 1, .tv_nsec = 0};
58 int attempts = 5;
59
60 while (attempts > 0) {
61 if (flock(fd, LOCK_EX | LOCK_NB) == 0) {
62 return LLE_SUCCESS;
63 }
64
65 if (errno != EWOULDBLOCK) {
66 return LLE_ERROR_SYSTEM_CALL;
67 }
68
69 /* Wait before retry */
70 nanosleep(&timeout, NULL);
71 attempts--;
72 }
73
74 return LLE_ERROR_TIMEOUT;
75}

100/**
101 * Escape special characters in string for TSV format
102 *
103 * Escapes: \t -> \\t, \n -> \\n, \\ -> \\\\
104 *
105 * @param str Input string
106 * @param output Output buffer
107 * @param output_size Output buffer size
108 * @return LLE_SUCCESS or error code
109 */
110static lle_result_t lle_escape_string(const char *str, char *output,
111 size_t output_size) {
112 if (!str || !output || output_size == 0) {
113 return LLE_ERROR_INVALID_PARAMETER;
114 }
115
116 size_t out_idx = 0;
117 size_t in_idx = 0;
118
119 while (str[in_idx] != '\0' && out_idx < output_size - 1) {
120 if (str[in_idx] == '\t') {
121 if (out_idx + 2 >= output_size)
122 break;
123 output[out_idx++] = '\\';
124 output[out_idx++] = 't';
125 } else if (str[in_idx] == '\n') {
126 if (out_idx + 2 >= output_size)
127 break;
128 output[out_idx++] = '\\';
129 output[out_idx++] = 'n';
130 } else if (str[in_idx] == '\\') {
131 if (out_idx + 2 >= output_size)
132 break;
133 output[out_idx++] = '\\';
134 output[out_idx++] = '\\';
135 } else {
136 output[out_idx++] = str[in_idx];
137 }
138 in_idx++;
139 }
140
141 output[out_idx] = '\0';
142 return LLE_SUCCESS;
143}

146 * Unescape special characters from TSV format
147 *
148 * @param str Input string
149 * @param output Output buffer
150 * @param output_size Output buffer size

153static lle_result_t lle_unescape_string(const char *str, char *output,

159 size_t out_idx = 0;

161

164 if (str[in_idx + 1] == 't') {
165 output[out_idx++] = '\t';
166 in_idx += 2;
167 } else if (str[in_idx + 1] == 'n') {
168 output[out_idx++] = '\n';
169 in_idx += 2;
170 } else if (str[in_idx + 1] == '\\') {
171 output[out_idx++] = '\\';
172 in_idx += 2;
173 } else {
174 output[out_idx++] = str[in_idx++];
175 }
176 } else {
177 output[out_idx++] = str[in_idx++];
178 }
179 }
180
181 output[out_idx] = '\0';
182 return LLE_SUCCESS;

185/**
186 * Format history entry as TSV line
187 *
188 * Format: TIMESTAMP\tCOMMAND\tEXIT_CODE\tWORKING_DIR\n
189 *
190 * @param entry Entry to format
191 * @param line Output buffer
192 * @param line_size Buffer size
193 * @return LLE_SUCCESS or error code
194 */
195static lle_result_t lle_history_format_entry(const lle_history_entry_t *entry,
196 char *line, size_t line_size) {
197 if (!entry || !line || line_size == 0) {
198 return LLE_ERROR_INVALID_PARAMETER;
199 }
200
201 /* Allocate temporary buffers for escaped strings */
202 char *escaped_cmd = lle_pool_alloc(LLE_HISTORY_MAX_COMMAND_LENGTH * 2);
203 char *escaped_wd = lle_pool_alloc(LLE_HISTORY_MAX_PATH_LENGTH * 2);
204
205 if (!escaped_cmd || !escaped_wd) {
206 if (escaped_cmd)
207 lle_pool_free(escaped_cmd);
208 if (escaped_wd)
209 lle_pool_free(escaped_wd);
210 return LLE_ERROR_OUT_OF_MEMORY;
211 }
212
213 /* Escape command and working directory */
214 lle_escape_string(entry->command, escaped_cmd,
215 LLE_HISTORY_MAX_COMMAND_LENGTH * 2);
216
217 const char *wd = entry->working_directory ? entry->working_directory : "";
218 lle_escape_string(wd, escaped_wd, LLE_HISTORY_MAX_PATH_LENGTH * 2);
219
220 /* Format line */
221 int written = snprintf(line, line_size, "%lu\t%s\t%d\t%s\n",
222 (unsigned long)entry->timestamp, escaped_cmd,
223 entry->exit_code, escaped_wd);
224
225 lle_pool_free(escaped_cmd);
226 lle_pool_free(escaped_wd);
227
228 if (written < 0 || (size_t)written >= line_size) {
229 return LLE_ERROR_BUFFER_OVERFLOW;
230 }
231
232 return LLE_SUCCESS;
233}

235/**
236 * Parse TSV line into history entry
237 *
238 * @param line Input line
239 * @param entry Output entry pointer
240 * @param memory_pool Memory pool
241 * @return LLE_SUCCESS or error code
242 */
243static lle_result_t lle_history_parse_line(const char *line,
244 lle_history_entry_t **entry,
245 lle_memory_pool_t *memory_pool) {
246 if (!line || !entry) {
247 return LLE_ERROR_INVALID_PARAMETER;
248 }
249
250 /* Skip comments and empty lines */
251 if (line[0] == '#' || line[0] == '\0' || line[0] == '\n') {
252 *entry = NULL;
253 return LLE_SUCCESS;
254 }
255
256 /* Parse fields */
257 uint64_t timestamp = 0;
258 char cmd_buffer[LLE_HISTORY_MAX_COMMAND_LENGTH * 2];
259 int exit_code = 0;
260 char wd_buffer[LLE_HISTORY_MAX_PATH_LENGTH * 2];
261
262 int parsed =
263 sscanf(line, "%lu\t%[^\t]\t%d\t%[^\n]", (unsigned long *)&timestamp,
264 cmd_buffer, &exit_code, wd_buffer);
265
266 if (parsed < 2) {
267 /* Malformed line - skip it */
268 *entry = NULL;
269 return LLE_SUCCESS;
270 }
271
272 /* Unescape command */
273 char unescaped_cmd[LLE_HISTORY_MAX_COMMAND_LENGTH];
274 lle_unescape_string(cmd_buffer, unescaped_cmd, sizeof(unescaped_cmd));
275
276 /* Create entry */
277 lle_result_t result =
278 lle_history_entry_create(entry, unescaped_cmd, memory_pool);
279 if (result != LLE_SUCCESS) {
280 return result;
281 }
282
283 (*entry)->timestamp = timestamp;
284 (*entry)->exit_code = exit_code;
285
286 /* Set working directory if present */
287 if (parsed >= 4 && wd_buffer[0] != '\0') {
288 char unescaped_wd[LLE_HISTORY_MAX_PATH_LENGTH];
289 lle_unescape_string(wd_buffer, unescaped_wd, sizeof(unescaped_wd));
290
291 /* Free the working_directory allocated by lle_history_entry_create */
292 if ((*entry)->working_directory) {
293 lle_pool_free((*entry)->working_directory);
294 }
295
296 size_t wd_len = strlen(unescaped_wd) + 1;
297 (*entry)->working_directory = lle_pool_alloc(wd_len);
298 if ((*entry)->working_directory) {
299 memcpy((*entry)->working_directory, unescaped_wd, wd_len);
300 }
301 }
302
303 return LLE_SUCCESS;
304}

311/**
312 * Save all history entries to file
313 *
314 * @param core History core
315 * @param file_path Path to history file
316 * @return LLE_SUCCESS or error code
317 */
318lle_result_t lle_history_save_to_file(lle_history_core_t *core,
319 const char *file_path) {
320 if (!core || !file_path) {
321 return LLE_ERROR_INVALID_PARAMETER;
322 }
323
324 pthread_rwlock_rdlock(&core->lock);
325
326 /* Open file for writing */
327 int fd = open(file_path, O_WRONLY | O_CREAT | O_TRUNC, 0600);
328 if (fd < 0) {
329 pthread_rwlock_unlock(&core->lock);
330 return LLE_ERROR_IO_ERROR;
331 }
332
333 /* Acquire lock */
334 lle_result_t result = lle_history_file_lock(fd);
335 if (result != LLE_SUCCESS) {
336 close(fd);
337 pthread_rwlock_unlock(&core->lock);
338 return result;
339 }
340
341 /* Write header */
342 char header[256];
343 snprintf(header, sizeof(header), "%s%s\n# Generated: %lu\n# Entries: %zu\n",
344 LLE_HISTORY_MAGIC_HEADER, LLE_HISTORY_FILE_VERSION_STR,
345 (unsigned long)time(NULL), core->entry_count);
346
347 if (write(fd, header, strlen(header)) < 0) {
348 lle_history_file_unlock(fd);
349 close(fd);
350 pthread_rwlock_unlock(&core->lock);
351 return LLE_ERROR_IO_ERROR;
352 }
353
354 /* Write entries */
355 char *line_buffer = lle_pool_alloc(LLE_HISTORY_MAX_LINE_LENGTH);
356 if (!line_buffer) {
357 lle_history_file_unlock(fd);
358 close(fd);
359 pthread_rwlock_unlock(&core->lock);
360 return LLE_ERROR_OUT_OF_MEMORY;
361 }
362
363 for (size_t i = 0; i < core->entry_count; i++) {
364 lle_history_entry_t *entry = core->entries[i];
365 if (!entry)
366 continue;
367
368 result = lle_history_format_entry(entry, line_buffer,
369 LLE_HISTORY_MAX_LINE_LENGTH);
370 if (result != LLE_SUCCESS) {
371 continue; /* Skip malformed entries */
372 }
373
374 if (write(fd, line_buffer, strlen(line_buffer)) < 0) {
375 lle_pool_free(line_buffer);
376 lle_history_file_unlock(fd);
377 close(fd);
378 pthread_rwlock_unlock(&core->lock);
379 return LLE_ERROR_IO_ERROR;
380 }
381 }
382
383 lle_pool_free(line_buffer);
384
385 /* Update statistics */
386 core->stats.save_count++;
387
388 /* Release lock and close */
389 lle_history_file_unlock(fd);
390 close(fd);
391
392 pthread_rwlock_unlock(&core->lock);
393
394 return LLE_SUCCESS;
395}

397/**
398 * Append single entry to history file
399 *
400 * For incremental saves without rewriting entire file.
401 *
402 * @param entry Entry to append
403 * @param file_path Path to history file
404 * @return LLE_SUCCESS or error code
405 */
406lle_result_t lle_history_append_entry(const lle_history_entry_t *entry,
407 const char *file_path) {
408 if (!entry || !file_path) {
409 return LLE_ERROR_INVALID_PARAMETER;
410 }
411
412 /* Open file for appending */
413 int fd = open(file_path, O_WRONLY | O_CREAT | O_APPEND, 0600);
414 if (fd < 0) {
415 return LLE_ERROR_IO_ERROR;
416 }
417
418 /* Acquire lock */
419 lle_result_t result = lle_history_file_lock(fd);
420 if (result != LLE_SUCCESS) {
421 close(fd);
422 return result;
423 }
424
425 /* Format and write entry */
426 char *line_buffer = lle_pool_alloc(LLE_HISTORY_MAX_LINE_LENGTH);
427 if (!line_buffer) {
428 lle_history_file_unlock(fd);
429 close(fd);
430 return LLE_ERROR_OUT_OF_MEMORY;
431 }
432
433 result = lle_history_format_entry(entry, line_buffer,
434 LLE_HISTORY_MAX_LINE_LENGTH);
435 if (result != LLE_SUCCESS) {
436 lle_pool_free(line_buffer);
437 lle_history_file_unlock(fd);
438 close(fd);
439 return result;
440 }
441
442 if (write(fd, line_buffer, strlen(line_buffer)) < 0) {
443 lle_pool_free(line_buffer);
444 lle_history_file_unlock(fd);
445 close(fd);
446 return LLE_ERROR_IO_ERROR;
447 }
448
449 lle_pool_free(line_buffer);
450
451 /* Release lock and close */
452 lle_history_file_unlock(fd);
453 close(fd);
454
455 return LLE_SUCCESS;

463/**
464 * Load history entries from file
465 *
466 * @param core History core
467 * @param file_path Path to history file
468 * @return LLE_SUCCESS or error code
469 */
470lle_result_t lle_history_load_from_file(lle_history_core_t *core,
471 const char *file_path) {
472 if (!core || !file_path) {
473 return LLE_ERROR_INVALID_PARAMETER;
474 }
475
476 /* Check if file exists */
477 struct stat st;
478 if (stat(file_path, &st) != 0) {
479 /* File doesn't exist - not an error, just empty history */
480 return LLE_SUCCESS;
481 }
482
483 /* Open file for reading */
484 FILE *fp = fopen(file_path, "r");
485 if (!fp) {
486 return LLE_ERROR_IO_ERROR;
487 }
488
489 pthread_rwlock_wrlock(&core->lock);
490
491 /* Read and parse lines */
492 char *line_buffer = lle_pool_alloc(LLE_HISTORY_MAX_LINE_LENGTH);
493 if (!line_buffer) {
494 fclose(fp);
495 pthread_rwlock_unlock(&core->lock);
496 return LLE_ERROR_OUT_OF_MEMORY;
497 }
498
499 size_t loaded_count = 0;
500 (void)loaded_count; /* Reserved for load statistics */
501 size_t skipped_count = 0;
502 (void)skipped_count; /* Reserved for skip statistics */
503
504 while (fgets(line_buffer, LLE_HISTORY_MAX_LINE_LENGTH, fp) != NULL) {
505 lle_history_entry_t *entry = NULL;
506 lle_result_t result =
507 lle_history_parse_line(line_buffer, &entry, core->memory_pool);
508
509 if (result != LLE_SUCCESS) {
510 skipped_count++;
511 continue;
512 }
513
514 if (!entry) {
515 /* Comment or empty line */
516 continue;
517 }
518
519 /* Manually add to history without calling lle_history_add_entry */
520 /* (to avoid deadlock - we already have write lock) */
521
522 /* Check capacity */
523 if (core->entry_count >= core->entry_capacity) {
524 result = lle_history_expand_capacity(core);
525 if (result != LLE_SUCCESS) {
526 lle_history_entry_destroy(entry, core->memory_pool);
527 break;
528 }
529 }
530
531 /* Assign entry ID */
532 entry->entry_id = core->next_entry_id++;
533
534 /* Add to array */
535 core->entries[core->entry_count] = entry;
536
537 /* Update linked list */
538 if (core->last_entry) {
539 core->last_entry->next = entry;
540 entry->prev = core->last_entry;
541 } else {
542 core->first_entry = entry;
543 }
544 core->last_entry = entry;
545
546 core->entry_count++;
547
548 /* Add to index if enabled */
549 if (core->entry_lookup) {
550 lle_history_index_insert(core->entry_lookup, entry->entry_id,
551 entry);
552 }
553
554 /* Update statistics */
555 core->stats.total_entries++;
556 core->stats.active_entries++;
557
558 loaded_count++;
559 }
560
561 lle_pool_free(line_buffer);
562 fclose(fp);
563
564 /* Update statistics */
565 core->stats.load_count++;
566
567 pthread_rwlock_unlock(&core->lock);
568
569 return LLE_SUCCESS;
570}
