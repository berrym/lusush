79lle_result_t
80lle_history_buffer_bridge_get_buffer_text(lle_history_buffer_bridge_t *bridge,
81 lle_buffer_t *buffer, char **text,
82 size_t *length) {
83 if (!bridge || !bridge->active || !buffer || !text || !length) {
84 return LLE_ERROR_INVALID_PARAMETER;
85 }
86
87 /* Get buffer text directly from buffer structure */
88 const char *buffer_text = buffer->data;
89 if (!buffer_text) {
90 *text = NULL;
91 *length = 0;
92 return LLE_SUCCESS;
93 }
94
95 size_t text_len = buffer->length;
96
97 /* Allocate and copy */
98 char *text_copy = lle_pool_alloc(text_len + 1);
99 if (!text_copy) {
100 return LLE_ERROR_OUT_OF_MEMORY;
101 }
102
103 memcpy(text_copy, buffer_text, text_len);
104 text_copy[text_len] = '\0';
105
106 *text = text_copy;
107 *length = text_len;
108
109 return LLE_SUCCESS;
110}

113lle_history_buffer_bridge_set_buffer_text(lle_history_buffer_bridge_t *bridge,
114 lle_buffer_t *buffer,
115 const char *text, size_t length) {
116 if (!bridge || !bridge->active || !buffer || !text) {
117 return LLE_ERROR_INVALID_PARAMETER;
118 }
119
120 /* Clear buffer first */
121 lle_result_t result = lle_buffer_clear(buffer);
122 if (result != LLE_SUCCESS) {
123 return result;
124 }
125
126 /* Insert text at beginning */
127 if (length > 0) {
128 result = lle_buffer_insert_text(buffer, 0, text, length);
129 if (result != LLE_SUCCESS) {
130 return result;
131 }
132 }
133
134 return LLE_SUCCESS;
135}

147lle_result_t lle_history_buffer_bridge_load_to_buffer(
148 lle_history_buffer_bridge_t *bridge, size_t history_index,
149 lle_buffer_t *buffer, const lle_transfer_options_t *options,
150 lle_transfer_result_t *result) {
151 if (!bridge || !bridge->active || !buffer) {
152 return LLE_ERROR_INVALID_PARAMETER;
153 }
154
155 /* Get default options if not provided */
156 lle_transfer_options_t default_options;
157 if (!options) {
158 lle_history_buffer_bridge_get_default_options(&default_options);
159 options = &default_options;
160 }
161
162 /* Initialize result */
163 if (result) {
164 memset(result, 0, sizeof(lle_transfer_result_t));
165 result->entry_index = history_index;
166 }
167
168 /* Get history entry text */
169 char *entry_text = NULL;
170 size_t entry_length = 0;
171 lle_result_t res = get_history_entry_text(
172 bridge->history_core, history_index, &entry_text, &entry_length);
173
174 if (res != LLE_SUCCESS) {
175 if (result)
176 result->success = false;
177 return res;
178 }
179
180 /* Apply reconstruction if requested */
181 char *final_text = entry_text;
182 size_t final_length = entry_length;
183
184 if (options->apply_reconstruction && bridge->reconstruction) {
185 lle_reconstructed_command_t *reconstructed = NULL;
186 res = lle_reconstruction_engine_reconstruct(
187 bridge->reconstruction, entry_text, entry_length, &reconstructed);
188
189 if (res == LLE_SUCCESS && reconstructed) {
190 final_text = reconstructed->text;
191 final_length = reconstructed->length;
192 }
193 }
194
195 /* Parse multiline structure if requested */
196 bool is_multiline = false;
197 if (options->parse_multiline && bridge->parser) {
198 lle_multiline_parse_result_t *parse_result = NULL;
199 res = lle_multiline_parser_parse(bridge->parser, final_text,
200 final_length, &parse_result);
201
202 if (res == LLE_SUCCESS && parse_result) {
203 is_multiline = (parse_result->line_count > 1);
204 if (result)
205 result->is_multiline = is_multiline;
206 }
207 }
208
209 /* Set buffer text */
210 res = lle_history_buffer_bridge_set_buffer_text(bridge, buffer, final_text,
211 final_length);
212
213 if (result) {
214 result->success = (res == LLE_SUCCESS);
215 result->bytes_transferred = final_length;
216 result->buffer_position = 0;
217 }
218
219 return res;
220}

222lle_result_t lle_history_buffer_bridge_save_from_buffer(
223 lle_history_buffer_bridge_t *bridge, lle_buffer_t *buffer,
224 const lle_transfer_options_t *options, lle_transfer_result_t *result) {
225 if (!bridge || !bridge->active || !buffer) {
226 return LLE_ERROR_INVALID_PARAMETER;
227 }
228
229 /* Get default options if not provided */
230 lle_transfer_options_t default_options;
231 if (!options) {
232 lle_history_buffer_bridge_get_default_options(&default_options);
233 options = &default_options;
234 }
235
236 /* Initialize result */
237 if (result) {
238 memset(result, 0, sizeof(lle_transfer_result_t));
239 }
240
241 /* Get buffer text */
242 char *buffer_text = NULL;
243 size_t buffer_length = 0;
244 lle_result_t res = lle_history_buffer_bridge_get_buffer_text(
245 bridge, buffer, &buffer_text, &buffer_length);
246
247 if (res != LLE_SUCCESS) {
248 if (result)
249 result->success = false;
250 return res;
251 }
252
253 /* Add to history */
254 if (buffer_text && buffer_length > 0) {
255 uint64_t entry_id = 0;
256 res = lle_history_add_entry(bridge->history_core, buffer_text, -1,
257 &entry_id);
258
259 if (res == LLE_SUCCESS && result) {
260 result->success = true;
261 result->bytes_transferred = buffer_length;
262
263 /* Get the index of the newly added entry */
264 size_t entry_count = 0;
265 if (lle_history_get_entry_count(bridge->history_core,
266 &entry_count) == LLE_SUCCESS) {
267 result->entry_index = entry_count - 1;
268 }
269 }
270 }
271
272 return res;
273}

292 }
293
294 memcpy(text_copy, entry->command, cmd_len);
295 text_copy[cmd_len] = '\0';
296
297 *text = text_copy;
298 *length = cmd_len;

300 return LLE_SUCCESS;
