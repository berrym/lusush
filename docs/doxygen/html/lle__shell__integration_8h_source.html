1/**
2 * @file lle_shell_integration.h
3 * @brief LLE Shell Integration - Centralized LLE initialization and lifecycle
4 *
5 * This module provides the shell-level LLE initialization system that:
6 * - Initializes LLE subsystems in proper sequence
7 * - Manages the global LLE editor instance
8 * - Provides three-tier reset hierarchy (Soft/Hard/Nuclear)
9 * - Tracks initialization state and error recovery
10 * - Coordinates with shell event hub for lifecycle events
11 *
12 * Initialization sequence:
13 * 1. Verify global_memory_pool exists
14 * 2. Verify terminal detection complete
15 * 3. Create shell event hub
16 * 4. Create and configure LLE editor
17 * 5. Initialize history system
18 * 6. Register atexit handler for cleanup
19 *
20 * Reset hierarchy:
21 * - Soft Reset: Abort current line, return to prompt
22 * - Hard Reset: Save history, destroy/recreate editor
23 * - Nuclear Reset: Hard reset + terminal reset sequences
24 *
25 * Specification: docs/lle_specification/26_initialization_system_complete.md
26 * Date: 2025-01-16
27 */
28

125/**
126 * @brief Initialize LLE shell integration
127 *
128 * Called during shell startup (in init.c) after display integration.
129 * Initializes all LLE subsystems in proper sequence:
130 * 1. Verifies prerequisites (memory pool, terminal)
131 * 2. Creates shell event hub
132 * 3. Creates LLE editor with all subsystems
133 * 4. Loads command history
134 * 5. Registers atexit cleanup handler
135 *
136 * On failure, cleans up partial state and returns error code.
137 * The shell should fall back to GNU readline if this fails.
138 *
139 * @return LLE_SUCCESS or error code
140 */
141lle_result_t lle_shell_integration_init(void);
142
143/**
144 * @brief Shutdown LLE shell integration
145 *
146 * Called during shell exit. Saves history, destroys editor and event hub,
147 * and frees all resources. Safe to call multiple times.
148 *
149 * Also called by atexit handler if not explicitly called.
150 */
151void lle_shell_integration_shutdown(void);

183/**
184 * @brief Update the shell prompt
185 *
186 * Renders the prompt using the LLE prompt composer and updates PS1/PS2

189 * Actions:

200/* ============================================================================
201 * RESET FUNCTIONS (THREE-TIER HIERARCHY)
202 * ============================================================================
203 */
204
205/**
206 * @brief Soft reset - Abort current line
207 *
208 * Tier 1 reset: Aborts the current line editing operation and returns
209 * the user to a fresh prompt. Does not destroy any state.
210 *
211 * Triggered by:
212 * - Ctrl+G (abort)
213 * - Recoverable errors during line editing
214 *
215 * Actions:
216 * - Sets abort flag on editor
217 * - Clears current buffer
218 * - Returns empty string from lle_readline()
219 */
220void lle_soft_reset(void);
221
222/**
223 * @brief Hard reset - Destroy and recreate editor
224 *
225 * Tier 2 reset: Destroys the LLE editor and recreates it from scratch.
226 * Used when editor state may be corrupted.
227 *
228 * Triggered by:
229 * - Triple Ctrl+G within 2 seconds
230 * - 5 consecutive errors
231 * - `display lle reset` command
232 *
233 * Actions:
234 * - Saves command history
235 * - Destroys current editor
236 * - Creates new editor instance
237 * - Reloads history
238 * - Resets error counters
239 */
240void lle_hard_reset(void);
241
242/**
243 * @brief Nuclear reset - Hard reset plus terminal reset
244 *
245 * Tier 3 reset: Performs hard reset and also resets terminal state.
246 * Used when terminal may be in corrupted state.
247 *
248 * Triggered by:
249 * - `display lle reset --terminal` command
250 * - Detection of terminal corruption
251 *
252 * Actions:
253 * - Performs hard reset
254 * - Sends terminal reset sequence (ESC c)
255 * - Re-detects terminal capabilities
256 * - Reinitializes display system
257 */
258void lle_nuclear_reset(void);
