27// Variable types
28typedef enum {
29 SYMVAR_STRING, // Regular string variable
30 SYMVAR_INTEGER, // Integer variable (for arithmetic)
31 SYMVAR_ARRAY, // Array variable (bash extension)
32 SYMVAR_FUNCTION // Function definition
33} symvar_type_t;
34
35// Variable flags
36typedef enum {
37 SYMVAR_NONE = 0,
38 SYMVAR_EXPORTED = (1 << 0), // Variable is exported to environment
39 SYMVAR_READONLY = (1 << 1), // Variable is read-only
40 SYMVAR_LOCAL = (1 << 2), // Variable is local to current scope
41 SYMVAR_SPECIAL_VAR = (1 << 3), // Special system variable
42 SYMVAR_UNSET = (1 << 4) // Variable is explicitly unset
43} symvar_flags_t;
44
45// Scope types for different contexts
46typedef enum {
47 SCOPE_GLOBAL, // Global shell scope
48 SCOPE_FUNCTION, // Function local scope
49 SCOPE_LOOP, // Loop iteration scope (for/while)
50 SCOPE_SUBSHELL, // Subshell scope
51 SCOPE_CONDITIONAL // Conditional execution scope (if/case)
52} scope_type_t;
53
54// Variable entry structure
55struct symvar {
56 char *name; // Variable name
57 char *value; // Variable value (string representation)
58 symvar_type_t type; // Variable type
59 symvar_flags_t flags; // Variable flags
60 size_t scope_level; // Scope level where defined
61 symvar_t *next; // Next variable in hash chain
62};
63
64// Enhanced symbol table scope structure using libhashtable
65struct symtable_scope_enhanced {
66 scope_type_t scope_type; // Type of scope
67 size_t level; // Scope nesting level
68 ht_strstr_t *vars_ht; // libhashtable ht_strstr_t for variables
69 symtable_scope_t *parent; // Parent scope
70 char *scope_name; // Name of scope (for debugging)
71};
