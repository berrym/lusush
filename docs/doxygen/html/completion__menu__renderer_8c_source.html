43/**
44 * Calculate visual width of string, skipping ANSI escape sequences
45 *
46 * Uses lle_utf8_string_width() for proper Unicode display width calculation:
47 * - CJK characters: 2 columns
48 * - Emoji: 2 columns
49 * - Combining marks: 0 columns
50 * - Regular ASCII/Latin: 1 column
51 *
52 * ANSI escape sequences have zero visual width but take up bytes.
53 * Format: ESC [ ... final_byte (where final_byte is 0x40-0x7E)
54 */
55static size_t visual_width(const char *str) {
56 if (!str)
57 return 0;
58
59 size_t len = strlen(str);
60 if (len == 0)
61 return 0;
62
63 /* Fast path: no ANSI escape sequences */
64 const char *esc = memchr(str, 0x1B, len);
65 if (!esc) {
66 /* No escape sequences - use lle_utf8_string_width directly */
67 return lle_utf8_string_width(str, len);
68 }
69
70 /* Slow path: string contains ANSI escape sequences
71 * Process segments between escape sequences */
72 size_t width = 0;
73 size_t i = 0;
74
75 while (i < len) {
76 unsigned char c = (unsigned char)str[i];
77
78 /* Check for ANSI escape sequence (ESC = 0x1B) */
79 if (c == 0x1B && i + 1 < len) {
80 unsigned char next = (unsigned char)str[i + 1];
81 if (next == '[') {
82 /* CSI sequence: ESC [ ... final_byte
83 * Skip until we find the final byte (0x40-0x7E) */
84 i += 2; /* Skip ESC [ */
85 while (i < len) {
86 unsigned char seq_char = (unsigned char)str[i];
87 i++;
88 if (seq_char >= 0x40 && seq_char <= 0x7E) {
89 break; /* Found final byte, sequence complete */
90 }
91 }
92 continue; /* Don't add to width, continue to next character */
93 }
94 /* Other escape sequences (ESC + single char) */
95 i += 2;
96 continue;
97 }
98
99 /* Find length of text segment until next escape or end */
100 size_t segment_start = i;
101 while (i < len && (unsigned char)str[i] != 0x1B) {
102 i++;
103 }
104
105 /* Calculate width of this segment using proper UTF-8 width */
106 if (i > segment_start) {
107 width +=
108 lle_utf8_string_width(str + segment_start, i - segment_start);
109 }
110 }
111
112 return width;
113}

115/**
116 * Pad string to width
117 */
118static size_t pad_string(char *dest, size_t dest_size, const char *src,
119 size_t target_width) {
120 if (!dest || !src || dest_size == 0)
121 return 0;
122
123 size_t src_width = visual_width(src);
124 size_t written = 0;
125
126 // Copy source
127 size_t src_len = strlen(src);
128 if (src_len >= dest_size) {
129 src_len = dest_size - 1;
130 }
131 memcpy(dest, src, src_len);
132 written = src_len;
133
134 // Add padding if needed
135 if (src_width < target_width && written < dest_size - 1) {
136 size_t padding = target_width - src_width;
137 if (padding > dest_size - written - 1) {
138 padding = dest_size - written - 1;
139 }
140 memset(dest + written, ' ', padding);
141 written += padding;
142 }
143
144 dest[written] = '\0';
145 return written;
146}

166size_t
167lle_menu_renderer_calculate_column_width(const lle_completion_item_t *items,
168 size_t count, size_t terminal_width,
169 size_t max_columns) {
170 (void)max_columns; // Reserved for future use
171
172 if (!items || count == 0) {
173 return LLE_MENU_RENDERER_MIN_COL_WIDTH;
174 }
175
176 // Find longest item
177 size_t max_len = 0;
178 for (size_t i = 0; i < count; i++) {
179 if (items[i].text) {
180 size_t len = visual_width(items[i].text);
181
182 // Account for type indicator if present
183 if (items[i].type_indicator) {
184 len += strlen(items[i].type_indicator);
185 }
186
187 if (len > max_len) {
188 max_len = len;
189 }
190 }
191 }
192
193 // Add padding
194 max_len += LLE_MENU_RENDERER_COL_PADDING;
195
196 // Ensure minimum width
197 if (max_len < LLE_MENU_RENDERER_MIN_COL_WIDTH) {
198 max_len = LLE_MENU_RENDERER_MIN_COL_WIDTH;
199 }
200
201 // Ensure we can fit at least one column
202 if (max_len > terminal_width) {
203 max_len = terminal_width;
204 }
205
206 return max_len;
207}

252lle_result_t lle_menu_renderer_format_item(const lle_completion_item_t *item,
253 bool is_selected,
254 bool show_indicator,
255 const char *selection_prefix,
256 char *output, size_t output_size) {
257 if (!item || !output || output_size == 0) {
258 return LLE_ERROR_INVALID_PARAMETER;
259 }
260
261 char formatted[256];
262 size_t pos = 0;
263
264 // Add selection prefix if selected
265 if (is_selected && selection_prefix) {
266 size_t prefix_len = strlen(selection_prefix);
267 if (prefix_len < sizeof(formatted) - pos) {
268 strcpy(formatted + pos, selection_prefix);
269 pos += prefix_len;
270 }
271 } else if (selection_prefix) {
272 // Add spaces to align non-selected items
273 size_t prefix_len = strlen(selection_prefix);
274 if (prefix_len < sizeof(formatted) - pos) {
275 memset(formatted + pos, ' ', prefix_len);
276 pos += prefix_len;
277 }
278 }
279
280 // Add item text
281 if (item->text && pos < sizeof(formatted) - 1) {
282 size_t text_len = strlen(item->text);
283 if (text_len > sizeof(formatted) - pos - 1) {
284 text_len = sizeof(formatted) - pos - 1;
285 }
286 memcpy(formatted + pos, item->text, text_len);
287 pos += text_len;
288 }
289
290 // Add type indicator
291 if (show_indicator && item->type_indicator && pos < sizeof(formatted) - 1) {
292 size_t ind_len = strlen(item->type_indicator);
293 if (ind_len > sizeof(formatted) - pos - 1) {
294 ind_len = sizeof(formatted) - pos - 1;
295 }
296 memcpy(formatted + pos, item->type_indicator, ind_len);
297 pos += ind_len;
298 }
299
300 formatted[pos] = '\0';
301
302 // Apply highlighting if selected
303 int written;
304 if (is_selected) {
305 written =
306 snprintf(output, output_size, "%s%s%s", LLE_MENU_SELECTION_START,
307 formatted, LLE_MENU_SELECTION_END);
308 } else {
309 written = snprintf(output, output_size, "%s", formatted);
310 }
311
312 if (written < 0 || (size_t)written >= output_size) {
313 return LLE_ERROR_BUFFER_OVERFLOW;
314 }
315
316 return LLE_SUCCESS;
317}

319size_t
320lle_menu_renderer_estimate_size(const lle_completion_menu_state_t *state,
321 const lle_menu_render_options_t *options) {
322 (void)options; // Reserved for future use
323
324 if (!state || !state->result) {
325 return 1024; // Default estimate
326 }
327
328 size_t visible = state->visible_count;
329 if (visible > state->result->count - state->first_visible) {
330 visible = state->result->count - state->first_visible;
331 }
332
333 // Rough estimate:
334 // - Average 20 chars per item
335 // - Category headers: 30 chars each (max 8 types)
336 // - ANSI codes: ~20 chars per item
337 // - Newlines and padding
338
339 size_t estimate = visible * 40 + 8 * 30 + 256;
340
341 if (estimate > LLE_MENU_RENDERER_MAX_OUTPUT) {
342 estimate = LLE_MENU_RENDERER_MAX_OUTPUT;
343 }
344
345 return estimate;
346}

348lle_result_t
349lle_completion_menu_render(const lle_completion_menu_state_t *state,
350 const lle_menu_render_options_t *options,
351 char *output, size_t output_size,
352 lle_menu_render_stats_t *stats) {
353 if (!state || !state->result || !output || output_size == 0) {
354 return LLE_ERROR_INVALID_PARAMETER;
355 }
356
357 if (!state->menu_active || state->result->count == 0) {
358 output[0] = '\0';
359 if (stats) {
360 memset(stats, 0, sizeof(*stats));
361 }
362 return LLE_SUCCESS;
363 }
364
365 // Use default options if not provided
366 lle_menu_render_options_t default_opts;
367 if (!options) {
368 default_opts = lle_menu_renderer_default_options(80);
369 options = &default_opts;
370 }
371
372 // Initialize stats
373 lle_menu_render_stats_t local_stats = {0};
374
375 // Get visible range
376 size_t start_idx = state->first_visible;
377 size_t end_idx = start_idx + state->visible_count;
378 if (end_idx > state->result->count) {
379 end_idx = state->result->count;
380 }
381
382 const lle_completion_item_t *items = state->result->items;
383 size_t output_pos = 0;
384 size_t rows_used = 0;
385
386 // Use pre-calculated column layout from menu state if available
387 // This ensures stable layout during navigation (no column shifting)
388 size_t col_width = 0;
389 size_t columns = 1;
390 if (options->use_multi_column) {
391 // Prefer cached layout from state (set by
392 // lle_completion_menu_update_layout)
393 if (state->column_width > 0 && state->num_columns > 0) {
394 col_width = state->column_width;
395 columns = state->num_columns;
396 } else {
397 // Fallback: calculate fresh if state doesn't have layout info
398 col_width = lle_menu_renderer_calculate_column_width(
399 items + start_idx, end_idx - start_idx, options->terminal_width,
400 LLE_MENU_RENDERER_MAX_COLS);
401 columns = lle_menu_renderer_calculate_columns(
402 options->terminal_width, col_width,
403 LLE_MENU_RENDERER_COL_PADDING);
404 }
405 local_stats.columns_used = columns;
406 }
407
408 lle_completion_type_t current_category = LLE_COMPLETION_TYPE_UNKNOWN;
409 size_t column = 0;
410
411 for (size_t i = start_idx; i < end_idx && rows_used < options->max_rows;
412 i++) {
413 const lle_completion_item_t *item = &items[i];
414
415 // Category header if type changed
416 if (options->show_category_headers && item->type != current_category) {
417 // Finish current row if mid-row
418 if (column > 0 && output_pos < output_size - 1) {
419 output[output_pos++] = '\n';
420 rows_used++;
421 column = 0;
422 }
423
424 // Check row limit
425 if (rows_used >= options->max_rows) {
426 local_stats.truncated = true;
427 break;
428 }
429
430 // Format category header
431 char header[128];
432 lle_result_t result = lle_menu_renderer_format_category_header(
433 item->type, header, sizeof(header),
434 true // Use bold
435 );
436
437 if (result == LLE_SUCCESS) {
438 size_t header_len = strlen(header);
439 if (output_pos + header_len + 1 < output_size) {
440 memcpy(output + output_pos, header, header_len);
441 output_pos += header_len;
442 output[output_pos++] = '\n';
443 rows_used++;
444 local_stats.categories_shown++;
445 current_category = item->type;
446 }
447 }
448 }
449
450 // Check row limit before rendering item
451 if (rows_used >= options->max_rows) {
452 local_stats.truncated = true;
453 break;
454 }
455
456 // Format item
457 char formatted[256];
458 bool is_selected = (i == state->selected_index);
459 lle_result_t result = lle_menu_renderer_format_item(
460 item, is_selected && options->highlight_selection,
461 options->show_type_indicators,
462 options->use_multi_column ? NULL : options->selection_prefix,
463 formatted, sizeof(formatted));
464
465 if (result != LLE_SUCCESS) {
466 continue; // Skip this item
467 }
468
469 // Add to output with column layout
470 if (options->use_multi_column) {
471 // Pad to column width
472 char padded[256];
473 pad_string(padded, sizeof(padded), formatted, col_width);
474
475 size_t padded_len = strlen(padded);
476 if (output_pos + padded_len + 1 < output_size) {
477 memcpy(output + output_pos, padded, padded_len);
478 output_pos += padded_len;
479
480 column++;
481 if (column >= columns) {
482 // End of row
483 output[output_pos++] = '\n';
484 rows_used++;
485 column = 0;
486 } else {
487 // Add column separator (included in padding)
488 }
489 }
490 } else {
491 // Single column layout
492 size_t formatted_len = strlen(formatted);
493 if (output_pos + formatted_len + 1 < output_size) {
494 memcpy(output + output_pos, formatted, formatted_len);
495 output_pos += formatted_len;
496 output[output_pos++] = '\n';
497 rows_used++;
498 }
499 }
500
501 local_stats.items_rendered++;
502 }
503
504 // Finish last row if needed
505 if (column > 0 && output_pos < output_size - 1) {
506 output[output_pos++] = '\n';
507 rows_used++;
508 }
509
510 // Null terminate
511 if (output_pos < output_size) {
512 output[output_pos] = '\0';
513 } else {
514 output[output_size - 1] = '\0';
515 local_stats.truncated = true;
516 }
517
518 local_stats.rows_used = rows_used;
519
520 // Copy stats if requested
521 if (stats) {
522 *stats = local_stats;
523 }
524
525 return LLE_SUCCESS;
526}
