114/**
115 * Check if command expects only directory arguments
116 */
117static bool is_directory_only_command(const char *command_name) {
118 if (!command_name) {
119 return false;
120 }
121
122 /*
123 * POSIX commands that only accept directory arguments.
124 * Note: pushd/popd are bash extensions, not implemented in lusush.
125 */
126 static const char *dir_commands[] = {"cd", "rmdir", NULL};
127
128 for (const char **cmd = dir_commands; *cmd != NULL; cmd++) {
129 if (strcmp(command_name, *cmd) == 0) {
130 return true;
131 }
132 }
133
134 return false;
135}

137/**
138 * File/directory source - command-aware
139 *
140 * For commands like cd, rmdir: only show directories
141 * For other commands: show both files and directories
142 */
143static lle_result_t file_source_generate(lle_memory_pool_t *pool,
144 const lle_context_analyzer_t *context,
145 const char *prefix,
146 lle_completion_result_t *result) {
147 /* Check if command expects directories only */
148 if (context && is_directory_only_command(context->command_name)) {
149 return lle_completion_source_directories(pool, prefix, result);
150 }
151
152 /* Default: show both files and directories */
153 return lle_completion_source_files(pool, prefix, result);
154}

186lle_result_t lle_source_manager_create(lle_memory_pool_t *pool,
187 lle_source_manager_t **out_manager) {
188 if (!pool || !out_manager) {
189 return LLE_ERROR_INVALID_PARAMETER;
190 }
191
192 lle_source_manager_t *manager = lle_pool_alloc(sizeof(*manager));
193 if (!manager) {
194 return LLE_ERROR_OUT_OF_MEMORY;
195 }
196
197 manager->num_sources = 0;
198 manager->pool = pool;
199
200 /* Register default sources */
201 lle_result_t res;
202
203 res = lle_source_manager_register(manager, LLE_SOURCE_BUILTINS, "builtins",
204 builtin_source_generate,
205 builtin_source_applicable);
206 if (res != LLE_SUCCESS) {
207 return res;
208 }
209
210 res = lle_source_manager_register(manager, LLE_SOURCE_ALIASES, "aliases",
211 alias_source_generate,
212 alias_source_applicable);
213 if (res != LLE_SUCCESS) {
214 return res;
215 }
216
217 res = lle_source_manager_register(
218 manager, LLE_SOURCE_EXTERNAL_COMMANDS, "external_commands",
219 external_command_source_generate, external_command_source_applicable);
220 if (res != LLE_SUCCESS) {
221 return res;
222 }
223
224 res = lle_source_manager_register(manager, LLE_SOURCE_FILES, "files",
225 file_source_generate,
226 file_source_applicable);
227 if (res != LLE_SUCCESS) {
228 return res;
229 }
230
231 res = lle_source_manager_register(manager, LLE_SOURCE_VARIABLES,
232 "variables", variable_source_generate,
233 variable_source_applicable);
234 if (res != LLE_SUCCESS) {
235 return res;
236 }
237
238 res = lle_source_manager_register(manager, LLE_SOURCE_HISTORY, "history",
239 history_source_generate,
240 history_source_applicable);
241 if (res != LLE_SUCCESS) {
242 return res;
243 }
244
245 *out_manager = manager;
246 return LLE_SUCCESS;
247}

258lle_result_t
259lle_source_manager_register(lle_source_manager_t *manager,
260 lle_source_type_t type, const char *name,
261 lle_source_generate_fn generate_fn,
262 lle_source_applicable_fn applicable_fn) {
263 if (!manager || !name || !generate_fn) {
264 return LLE_ERROR_INVALID_PARAMETER;
265 }
266
267 if (manager->num_sources >= MAX_COMPLETION_SOURCES) {
268 return LLE_ERROR_BUFFER_OVERFLOW; /* Array full */
269 }
270
271 lle_completion_source_t *source = lle_pool_alloc(sizeof(*source));
272 if (!source) {
273 return LLE_ERROR_OUT_OF_MEMORY;
274 }
275
276 source->type = type;
277 source->name = name;
278 source->generate = generate_fn;
279 source->is_applicable = applicable_fn;
280 source->user_data = NULL;
281
282 manager->sources[manager->num_sources++] = source;
283 return LLE_SUCCESS;
284}

286lle_result_t lle_source_manager_query(lle_source_manager_t *manager,
287 const lle_context_analyzer_t *context,
288 const char *prefix,
289 lle_completion_result_t *result) {
290 if (!manager || !context || !prefix || !result) {
291 return LLE_ERROR_INVALID_PARAMETER;
292 }
293
294 /* Query all applicable sources */
295 for (size_t i = 0; i < manager->num_sources; i++) {
296 lle_completion_source_t *source = manager->sources[i];
297
298 /* Check if source is applicable for this context */
299 if (source->is_applicable && !source->is_applicable(context)) {
300 continue; /* Skip this source */
301 }
302
303 /* Query source for completions */
304 lle_result_t res =
305 source->generate(manager->pool, context, prefix, result);
306
307 /* Continue even if source fails - other sources may succeed */
308 (void)res;
309 }
310
311 return LLE_SUCCESS;
312}
