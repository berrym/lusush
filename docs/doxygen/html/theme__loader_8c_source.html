33/**
34 * @brief Read entire file into buffer
35 */
36static lle_result_t read_file_contents(const char *filepath, char **content,
37 size_t *content_len) {
38 FILE *fp = fopen(filepath, "r");
39 if (!fp) {
40 return LLE_ERROR_IO_ERROR;
41 }
42
43 /* Get file size */
44 if (fseek(fp, 0, SEEK_END) != 0) {
45 fclose(fp);
46 return LLE_ERROR_IO_ERROR;
47 }
48
49 long file_size = ftell(fp);
50 if (file_size < 0) {
51 fclose(fp);
52 return LLE_ERROR_IO_ERROR;
53 }
54
55 if ((size_t)file_size > LLE_THEME_FILE_MAX_SIZE) {
56 fclose(fp);
57 return LLE_ERROR_BUFFER_OVERFLOW;
58 }
59
60 if (fseek(fp, 0, SEEK_SET) != 0) {
61 fclose(fp);
62 return LLE_ERROR_IO_ERROR;
63 }
64
65 /* Allocate buffer */
66 char *buffer = malloc((size_t)file_size + 1);
67 if (!buffer) {
68 fclose(fp);
69 return LLE_ERROR_OUT_OF_MEMORY;
70 }
71
72 /* Read file */
73 size_t bytes_read = fread(buffer, 1, (size_t)file_size, fp);
74 fclose(fp);
75
76 if (bytes_read != (size_t)file_size) {
77 free(buffer);
78 return LLE_ERROR_IO_ERROR;
79 }
80
81 buffer[bytes_read] = '\0';
82 *content = buffer;
83 *content_len = bytes_read;
84
85 return LLE_SUCCESS;
86}

88/**
89 * @brief Check if a path has the .toml extension
90 */
91static bool has_toml_extension(const char *path) {
92 size_t len = strlen(path);
93 size_t ext_len = strlen(LLE_THEME_FILE_EXTENSION);
94
95 if (len < ext_len) {
96 return false;
97 }
98
99 return strcasecmp(path + len - ext_len, LLE_THEME_FILE_EXTENSION) == 0;

120/**
121 * @brief Create directory recursively
122 */
123static lle_result_t mkdir_recursive(const char *path) {
124 char tmp[LLE_THEME_PATH_MAX];
125 char *p = NULL;
126 size_t len;
127
128 snprintf(tmp, sizeof(tmp), "%s", path);
129 len = strlen(tmp);
130
131 if (tmp[len - 1] == '/') {
132 tmp[len - 1] = '\0';
133 }
134
135 for (p = tmp + 1; *p; p++) {
136 if (*p == '/') {
137 *p = '\0';
138 if (mkdir(tmp, 0755) != 0 && errno != EEXIST) {
139 return LLE_ERROR_PERMISSION_DENIED;
140 }
141 *p = '/';
142 }
143 }
144
145 if (mkdir(tmp, 0755) != 0 && errno != EEXIST) {
146 return LLE_ERROR_PERMISSION_DENIED;
147 }
148
149 return LLE_SUCCESS;
150}

157/**
158 * @brief Load a theme from a file path
159 */
160lle_result_t lle_theme_load_from_file(const char *filepath, lle_theme_t *theme,
161 lle_theme_load_result_t *result) {
162 if (!filepath || !theme) {
163 return LLE_ERROR_INVALID_PARAMETER;
164 }
165
166 /* Initialize result if provided */
167 if (result) {
168 memset(result, 0, sizeof(*result));
169 snprintf(result->filepath, sizeof(result->filepath), "%s", filepath);
170 }
171
172 /* Read file contents */
173 char *content = NULL;
174 size_t content_len = 0;
175 lle_result_t status = read_file_contents(filepath, &content, &content_len);
176
177 if (status != LLE_SUCCESS) {
178 if (result) {
179 result->status = status;
180 snprintf(result->error_msg, sizeof(result->error_msg),
181 "Failed to read file: %s", strerror(errno));
182 }
183 return status;
184 }
185
186 /* Parse the content */
187 status = lle_theme_load_from_string(content, theme, result);
188 free(content);
189
190 if (status == LLE_SUCCESS) {
191 /* Mark theme source as USER (loaded from file) */
192 theme->source = LLE_THEME_SOURCE_USER;
193
194 if (result) {
195 result->status = LLE_SUCCESS;
196 snprintf(result->theme_name, sizeof(result->theme_name), "%s",
197 theme->name);
198 }
199 }
200
201 return status;
202}

204/**
205 * @brief Load a theme from a string
206 */
207lle_result_t lle_theme_load_from_string(const char *content, lle_theme_t *theme,
208 lle_theme_load_result_t *result) {
209 if (!content || !theme) {
210 return LLE_ERROR_INVALID_PARAMETER;
211 }
212
213 /* Initialize theme */
214 memset(theme, 0, sizeof(*theme));
215 lle_symbol_set_init_unicode(&theme->symbols);
216
217 /* Initialize parser */
218 lle_theme_parser_t parser;
219 lle_result_t status = lle_theme_parser_init(&parser, content);
220
221 if (status != LLE_SUCCESS) {
222 if (result) {
223 result->status = status;
224 snprintf(result->error_msg, sizeof(result->error_msg),
225 "Parser initialization failed");
226 }
227 return status;
228 }
229
230 /* Parse into theme */
231 status = lle_theme_parser_parse_to_theme(&parser, theme);
232
233 if (status != LLE_SUCCESS) {
234 if (result) {
235 result->status = status;
236 result->error_line = lle_theme_parser_error_line(&parser);
237 result->error_column = lle_theme_parser_error_column(&parser);
238 snprintf(result->error_msg, sizeof(result->error_msg), "%s",
239 lle_theme_parser_error(&parser));
240 }
241 return status;
242 }
243
244 /* Validate theme */
245 char validation_error[256];
246 status = lle_theme_parser_validate(theme, validation_error,
247 sizeof(validation_error));
248
249 if (status != LLE_SUCCESS) {
250 if (result) {
251 result->status = status;
252 snprintf(result->error_msg, sizeof(result->error_msg),
253 "Validation failed: %.230s", validation_error);
254 }
255 return status;
256 }
257
258 if (result) {
259 result->status = LLE_SUCCESS;
260 snprintf(result->theme_name, sizeof(result->theme_name), "%s",
261 theme->name);
262 }
263
264 return LLE_SUCCESS;
265}

267/**
268 * @brief Load all themes from a directory
269 */
270size_t lle_theme_load_directory(const char *dirpath,
271 lle_theme_registry_t *registry,
272 lle_theme_batch_result_t *result) {
273 if (!dirpath || !registry) {
274 return 0;
275 }
276
277 /* Initialize result if provided */
278 if (result) {
279 result->total_files = 0;
280 result->loaded_count = 0;
281 result->failed_count = 0;
282 result->skipped_count = 0;
283 }
284
285 /* Open directory */
286 DIR *dir = opendir(dirpath);
287 if (!dir) {
288 return 0;
289 }
290
291 size_t loaded = 0;
292 size_t result_idx = 0;
293 struct dirent *entry;
294
295 while ((entry = readdir(dir)) != NULL) {
296 /* Skip non-TOML files */
297 if (!has_toml_extension(entry->d_name)) {
298 continue;
299 }
300
301 if (result) {
302 result->total_files++;
303 }
304
305 /* Build full path */
306 char filepath[LLE_THEME_PATH_MAX];
307 snprintf(filepath, sizeof(filepath), "%s/%s", dirpath, entry->d_name);
308
309 /* Check if it's a regular file */
310 struct stat st;
311 if (stat(filepath, &st) != 0 || !S_ISREG(st.st_mode)) {
312 continue;
313 }
314
315 /* Allocate theme */
316 lle_theme_t *theme = calloc(1, sizeof(lle_theme_t));
317 if (!theme) {
318 if (result) {
319 result->failed_count++;
320 }
321 continue;
322 }
323
324 /* Get result slot if available */
325 lle_theme_load_result_t *load_result = NULL;
326 if (result && result->results &&
327 result_idx < result->results_capacity) {
328 load_result = &result->results[result_idx++];
329 }
330
331 /* Load theme */
332 lle_result_t status =
333 lle_theme_load_from_file(filepath, theme, load_result);
334
335 if (status != LLE_SUCCESS) {
336 free(theme);
337 if (result) {
338 result->failed_count++;
339 }
340 continue;
341 }
342
343 /* Check if theme already exists */
344 if (lle_theme_registry_find(registry, theme->name)) {
345 if (result) {
346 result->skipped_count++;
347 }
348 if (load_result) {
349 load_result->status = LLE_ERROR_ALREADY_EXISTS;
350 snprintf(load_result->error_msg, sizeof(load_result->error_msg),
351 "Theme '%s' already exists", theme->name);
352 }
353 free(theme);
354 continue;
355 }
356
357 /* Register theme */
358 status = lle_theme_registry_register(registry, theme);
359 if (status != LLE_SUCCESS) {
360 free(theme);
361 if (result) {
362 result->failed_count++;
363 }
364 if (load_result) {
365 load_result->status = status;
366 snprintf(load_result->error_msg, sizeof(load_result->error_msg),
367 "Failed to register theme");
368 }
369 continue;
370 }
371
372 loaded++;
373 if (result) {
374 result->loaded_count++;
375 }
376 }
377
378 closedir(dir);
379 return loaded;
380}

443/**
444 * @brief Reload a specific theme from its file
445 */
446lle_result_t lle_theme_reload_by_name(lle_theme_registry_t *registry,
447 const char *name) {
448 if (!registry || !name) {
449 return LLE_ERROR_INVALID_PARAMETER;
450 }
451
452 /* Find existing theme */
453 lle_theme_t *existing = lle_theme_registry_find(registry, name);
454 if (!existing) {
455 return LLE_ERROR_NOT_FOUND;
456 }
457
458 /* Only reload user themes */
459 if (existing->source != LLE_THEME_SOURCE_USER) {
460 return LLE_ERROR_INVALID_STATE;
461 }
462
463 /* Search for theme file */
464 char user_dir[LLE_THEME_PATH_MAX];
465 char filepath[LLE_THEME_PATH_MAX];
466
467 if (lle_theme_get_user_dir(user_dir, sizeof(user_dir)) == LLE_SUCCESS) {
468 /* Limit components to prevent truncation: path + "/" + name + ext */
469 snprintf(filepath, sizeof(filepath), "%.4020s/%.64s%s", user_dir, name,
470 LLE_THEME_FILE_EXTENSION);
471
472 if (lle_theme_file_exists(filepath)) {
473 /* Load into temporary theme */
474 lle_theme_t temp_theme;
475 lle_result_t status =
476 lle_theme_load_from_file(filepath, &temp_theme, NULL);
477
478 if (status == LLE_SUCCESS) {
479 /* Copy updated values to existing theme */
480 bool was_active = existing->is_active;
481 memcpy(existing, &temp_theme, sizeof(lle_theme_t));
482 existing->is_active = was_active;
483 existing->source = LLE_THEME_SOURCE_USER;
484 return LLE_SUCCESS;
485 }
486 return status;
487 }
488 }
489
490 return LLE_ERROR_NOT_FOUND;
491}

498/**
499 * @brief Escape a string for TOML output
500 */
501static size_t toml_escape_string(const char *input, char *output,
502 size_t output_size) {
503 if (!input || !output || output_size == 0) {
504 return 0;
505 }
506
507 size_t out_len = 0;
508 output[out_len++] = '"';
509
510 for (size_t i = 0; input[i] != '\0' && out_len < output_size - 2; i++) {
511 char c = input[i];
512 switch (c) {
513 case '\n':
514 if (out_len + 2 < output_size - 1) {
515 output[out_len++] = '\\';
516 output[out_len++] = 'n';
517 }
518 break;
519 case '\t':
520 if (out_len + 2 < output_size - 1) {
521 output[out_len++] = '\\';
522 output[out_len++] = 't';
523 }
524 break;
525 case '\r':
526 if (out_len + 2 < output_size - 1) {
527 output[out_len++] = '\\';
528 output[out_len++] = 'r';
529 }
530 break;
531 case '\\':
532 if (out_len + 2 < output_size - 1) {
533 output[out_len++] = '\\';
534 output[out_len++] = '\\';
535 }
536 break;
537 case '"':
538 if (out_len + 2 < output_size - 1) {
539 output[out_len++] = '\\';
540 output[out_len++] = '"';
541 }
542 break;
543 default:
544 output[out_len++] = c;
545 break;
546 }
547 }
548
549 output[out_len++] = '"';
550 output[out_len] = '\0';
551 return out_len;
552}

554/**
555 * @brief Format a color as TOML inline table
556 */
557static size_t format_color_toml(const lle_color_t *color, char *output,
558 size_t output_size) {
559 if (!color || !output || output_size == 0) {
560 return 0;
561 }
562
563 if (color->mode == LLE_COLOR_MODE_NONE) {
564 return 0;
565 }
566
567 char fg_str[64] = {0};
568
569 switch (color->mode) {
570 case LLE_COLOR_MODE_BASIC: {
571 const char *names[] = {"black", "red", "green", "yellow",
572 "blue", "magenta", "cyan", "white"};
573 if (color->value.basic < 8) {
574 snprintf(fg_str, sizeof(fg_str), "\"%s\"",
575 names[color->value.basic]);
576 }
577 break;
578 }
579 case LLE_COLOR_MODE_256:
580 snprintf(fg_str, sizeof(fg_str), "%d", color->value.palette);
581 break;
582 case LLE_COLOR_MODE_TRUE:
583 snprintf(fg_str, sizeof(fg_str), "\"#%02x%02x%02x\"",
584 color->value.rgb.r, color->value.rgb.g, color->value.rgb.b);
585 break;
586 default:
587 return 0;
588 }
589
590 /* Build inline table */
591 int len = snprintf(output, output_size, "{ fg = %s", fg_str);
592
593 if (color->bold && (size_t)len < output_size - 20) {
594 len +=
595 snprintf(output + len, output_size - (size_t)len, ", bold = true");
596 }
597 if (color->italic && (size_t)len < output_size - 20) {
598 len += snprintf(output + len, output_size - (size_t)len,
599 ", italic = true");
600 }
601 if (color->underline && (size_t)len < output_size - 20) {
602 len += snprintf(output + len, output_size - (size_t)len,
603 ", underline = true");
604 }
605 if (color->dim && (size_t)len < output_size - 20) {
606 len +=
607 snprintf(output + len, output_size - (size_t)len, ", dim = true");
608 }
609
610 if ((size_t)len < output_size - 2) {
611 len += snprintf(output + len, output_size - (size_t)len, " }");
612 }
613
614 return (size_t)len;
615}

641 * @brief Export a theme to TOML format
642 */
643size_t lle_theme_export_to_toml(const lle_theme_t *theme, char *output,
644 size_t output_size) {
645 if (!theme || !output || output_size == 0) {
646 return 0;
647 }
648
649 size_t pos = 0;
650 char escaped[LLE_THEME_PARSER_STRING_MAX * 2];
651 char color_buf[128];
652
653#define APPEND(...) \
654 do { \
655 int written = snprintf(output + pos, output_size - pos, __VA_ARGS__); \
656 if (written > 0 && (size_t)written < output_size - pos) { \
657 pos += (size_t)written; \
658 } \
659 } while (0)
660
661 /* Header comment */
662 APPEND("# LLE Theme File\n");
663 APPEND("# Generated by lusush theme system\n\n");
664
665 /* [theme] section */
666 APPEND("[theme]\n");
667
668 toml_escape_string(theme->name, escaped, sizeof(escaped));
669 APPEND("name = %s\n", escaped);
670
671 if (strlen(theme->description) > 0) {
672 toml_escape_string(theme->description, escaped, sizeof(escaped));
673 APPEND("description = %s\n", escaped);
674 }
675
676 if (strlen(theme->author) > 0) {
677 toml_escape_string(theme->author, escaped, sizeof(escaped));
678 APPEND("author = %s\n", escaped);
679 }
680
681 if (strlen(theme->version) > 0) {
682 toml_escape_string(theme->version, escaped, sizeof(escaped));
683 APPEND("version = %s\n", escaped);
684 }
685
686 APPEND("category = \"%s\"\n", category_to_string(theme->category));
687
688 if (strlen(theme->inherits_from) > 0) {
689 toml_escape_string(theme->inherits_from, escaped, sizeof(escaped));
690 APPEND("inherits_from = %s\n", escaped);
691 }
692
693 APPEND("\n");
694
695 /* [capabilities] section */
696 APPEND("[capabilities]\n");
697 if (theme->capabilities & LLE_THEME_CAP_UNICODE) {
698 APPEND("unicode = true\n");
699 }

701 APPEND("multiline = true\n");

713 APPEND("nerd_font = true\n");
714 }

716 APPEND("right_prompt = true\n");
717 }
718 if (theme->capabilities & LLE_THEME_CAP_TRUE_COLOR) {
719 APPEND("true_color = true\n");
720 }
721 if (theme->capabilities & LLE_THEME_CAP_256_COLOR) {
722 APPEND("256_color = true\n");

725 APPEND("ascii_fallback = true\n");
726 }
727 APPEND("\n");

730 APPEND("[layout]\n");
731
732 if (strlen(theme->layout.ps1_format) > 0) {
733 toml_escape_string(theme->layout.ps1_format, escaped, sizeof(escaped));
734 APPEND("ps1 = %s\n", escaped);
735 }
736

738 toml_escape_string(theme->layout.ps2_format, escaped, sizeof(escaped));
739 APPEND("ps2 = %s\n", escaped);

741

743 toml_escape_string(theme->layout.rps1_format, escaped, sizeof(escaped));
744 APPEND("rprompt = %s\n", escaped);
745 }
746
747 if (strlen(theme->layout.transient_format) > 0) {
748 toml_escape_string(theme->layout.transient_format, escaped,
749 sizeof(escaped));
750 APPEND("transient = %s\n", escaped);
751 }
752
753 if (theme->layout.newline_before) {
754 APPEND("newline_before = true\n");
755 }
756 if (theme->layout.newline_after) {
757 APPEND("newline_after = true\n");
758 }
759 if (theme->layout.enable_multiline) {
760 APPEND("multiline = true\n");
761 }

763 APPEND("compact = true\n");
764 }

768 if (theme->enabled_segment_count > 0) {
769 APPEND("[segments]\n");
770 APPEND("enabled = [");

781 APPEND("[colors]\n");
782
783#define EXPORT_COLOR(name, field) \

933 const char *filepath) {
934 if (!theme || !filepath) {
935 return LLE_ERROR_INVALID_PARAMETER;
936 }
937
938 /* Generate TOML content */
939 char *buffer = malloc(LLE_THEME_FILE_MAX_SIZE);
940 if (!buffer) {
941 return LLE_ERROR_OUT_OF_MEMORY;
942 }
943
944 size_t content_len =
945 lle_theme_export_to_toml(theme, buffer, LLE_THEME_FILE_MAX_SIZE);
946 if (content_len == 0) {
947 free(buffer);
948 return LLE_ERROR_INVALID_STATE;
949 }
950
951 /* Write to file */
952 FILE *fp = fopen(filepath, "w");
953 if (!fp) {
954 free(buffer);
955 return LLE_ERROR_IO_ERROR;
956 }
957
958 size_t written = fwrite(buffer, 1, content_len, fp);
959 fclose(fp);
960 free(buffer);
961
962 if (written != content_len) {
963 return LLE_ERROR_IO_ERROR;
964 }
965
966 return LLE_SUCCESS;
967}
