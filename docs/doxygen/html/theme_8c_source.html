39/**
40 * @brief Cleanup the theme registry
41 */
42void lle_theme_registry_cleanup(lle_theme_registry_t *registry) {
43 if (!registry || !registry->initialized) {
44 return;
45 }
46
47 for (size_t i = 0; i < registry->count; i++) {
48 if (registry->themes[i]) {
49 lle_theme_free(registry->themes[i]);
50 registry->themes[i] = NULL;

52 }

55 registry->active_theme = NULL;
56 registry->initialized = false;

59/**
60 * @brief Register a theme with the registry
61 */
62lle_result_t lle_theme_registry_register(lle_theme_registry_t *registry,
63 lle_theme_t *theme) {
64 if (!registry || !theme || !registry->initialized) {
65 return LLE_ERROR_INVALID_PARAMETER;
66 }
67
68 if (strlen(theme->name) == 0) {
69 return LLE_ERROR_INVALID_PARAMETER;
70 }
71
72 if (registry->count >= LLE_THEME_REGISTRY_MAX) {
73 return LLE_ERROR_BUFFER_OVERFLOW;
74 }
75
76 /* Check for duplicate name */
77 for (size_t i = 0; i < registry->count; i++) {
78 if (strcmp(registry->themes[i]->name, theme->name) == 0) {
79 return LLE_ERROR_INVALID_STATE; /* Duplicate */
80 }
81 }
82
83 /* Resolve inheritance if specified */
84 if (strlen(theme->inherits_from) > 0) {
85 lle_result_t result = lle_theme_resolve_inheritance(registry, theme);
86 if (result != LLE_SUCCESS) {
87 return result;
88 }
89 }
90
91 /* Register theme */
92 registry->themes[registry->count] = theme;
93 registry->count++;
94
95 /* Update statistics */
96 if (theme->source == LLE_THEME_SOURCE_BUILTIN) {
97 registry->builtin_count++;
98 } else {
99 registry->user_count++;
100 }
101
102 return LLE_SUCCESS;
103}

221/**
222 * @brief Resolve theme inheritance
223 */
224lle_result_t lle_theme_resolve_inheritance(lle_theme_registry_t *registry,
225 lle_theme_t *theme) {
226 if (!registry || !theme) {
227 return LLE_ERROR_INVALID_PARAMETER;
228 }
229
230 if (strlen(theme->inherits_from) == 0) {
231 return LLE_SUCCESS; /* No inheritance */
232 }
233
234 /* Find parent theme */
235 lle_theme_t *parent =
236 lle_theme_registry_find(registry, theme->inherits_from);
237 if (!parent) {
238 return LLE_ERROR_NOT_FOUND;
239 }
240
241 /* Check for inheritance cycle (max depth 10) */
242 lle_theme_t *ancestor = parent;
243 for (int depth = 0; depth < 10 && ancestor; depth++) {
244 if (strcmp(ancestor->name, theme->name) == 0) {
245 return LLE_ERROR_INVALID_STATE; /* Cycle detected */
246 }
247 ancestor = ancestor->parent;
248 }
249
250 /* Link parent */
251 theme->parent = parent;
252
253 /* Inherit colors (only if child's is unset) */
254 lle_color_scheme_t *child = &theme->colors;
255 const lle_color_scheme_t *par = &parent->colors;
256
257#define INHERIT_COLOR(field) \
258 if (child->field.mode == LLE_COLOR_MODE_NONE) { \
259 child->field = par->field; \
260 }
261
262 INHERIT_COLOR(primary);
263 INHERIT_COLOR(secondary);
264 INHERIT_COLOR(success);
265 INHERIT_COLOR(warning);
266 INHERIT_COLOR(error);
267 INHERIT_COLOR(info);
268 INHERIT_COLOR(text);
269 INHERIT_COLOR(text_dim);
270 INHERIT_COLOR(text_bright);
271 INHERIT_COLOR(border);
272 INHERIT_COLOR(background);
273 INHERIT_COLOR(highlight);
274 INHERIT_COLOR(git_clean);
275 INHERIT_COLOR(git_dirty);
276 INHERIT_COLOR(git_staged);
277 INHERIT_COLOR(git_untracked);
278 INHERIT_COLOR(git_branch);
279 INHERIT_COLOR(git_ahead);
280 INHERIT_COLOR(git_behind);
281 INHERIT_COLOR(path_home);
282 INHERIT_COLOR(path_root);
283 INHERIT_COLOR(path_normal);
284 INHERIT_COLOR(path_separator);
285 INHERIT_COLOR(status_ok);
286 INHERIT_COLOR(status_error);
287 INHERIT_COLOR(status_running);
288
289#undef INHERIT_COLOR
290
291 /* Inherit symbols if empty */
292#define INHERIT_SYMBOL(field) \
293 if (strlen(theme->symbols.field) == 0) { \
294 memcpy(theme->symbols.field, parent->symbols.field, \
295 sizeof(theme->symbols.field)); \
296 }
297
298 INHERIT_SYMBOL(prompt);
299 INHERIT_SYMBOL(prompt_root);
300 INHERIT_SYMBOL(continuation);
301 INHERIT_SYMBOL(separator_left);
302 INHERIT_SYMBOL(separator_right);
303 INHERIT_SYMBOL(branch);
304 INHERIT_SYMBOL(staged);
305 INHERIT_SYMBOL(unstaged);
306 INHERIT_SYMBOL(untracked);
307 INHERIT_SYMBOL(ahead);
308 INHERIT_SYMBOL(behind);
309 INHERIT_SYMBOL(stash);
310 INHERIT_SYMBOL(conflict);
311 INHERIT_SYMBOL(directory);
312 INHERIT_SYMBOL(home);
313 INHERIT_SYMBOL(error);
314 INHERIT_SYMBOL(success);
315 INHERIT_SYMBOL(time);
316 INHERIT_SYMBOL(jobs);
317
318#undef INHERIT_SYMBOL
319
320 /* Inherit layout if not set */
321 if (strlen(theme->layout.ps1_format) == 0) {
322 memcpy(theme->layout.ps1_format, parent->layout.ps1_format,
323 sizeof(theme->layout.ps1_format));
324 }
325 if (strlen(theme->layout.ps2_format) == 0) {
326 memcpy(theme->layout.ps2_format, parent->layout.ps2_format,
327 sizeof(theme->layout.ps2_format));
328 }
329 if (strlen(theme->layout.rps1_format) == 0) {
330 memcpy(theme->layout.rps1_format, parent->layout.rps1_format,
331 sizeof(theme->layout.rps1_format));
332 }
333 if (strlen(theme->layout.transient_format) == 0) {
334 memcpy(theme->layout.transient_format, parent->layout.transient_format,
335 sizeof(theme->layout.transient_format));
336 }
337
338 /* Inherit capabilities (additive for inheritable flags) */
339 theme->capabilities |= (parent->capabilities & LLE_THEME_CAP_INHERITABLE);
340
341 return LLE_SUCCESS;
342}

381/**
382 * @brief Generate ANSI escape sequence for a color
383 */
384size_t lle_color_to_ansi(const lle_color_t *color, bool foreground,
385 char *output, size_t output_size) {
386 if (!color || !output || output_size == 0) {
387 return 0;
388 }
389
390 int base = foreground ? 30 : 40;
391 int len = 0;
392
393 switch (color->mode) {
394 case LLE_COLOR_MODE_NONE:
395 output[0] = '\0';
396 return 0;
397
398 case LLE_COLOR_MODE_BASIC:
399 len = snprintf(output, output_size, "\033[%dm",
400 base + color->value.basic);
401 break;
402
403 case LLE_COLOR_MODE_256:
404 len = snprintf(output, output_size, "\033[%d;5;%dm",
405 foreground ? 38 : 48, color->value.palette);
406 break;
407
408 case LLE_COLOR_MODE_TRUE:
409 len = snprintf(output, output_size, "\033[%d;2;%d;%d;%dm",
410 foreground ? 38 : 48, color->value.rgb.r,
411 color->value.rgb.g, color->value.rgb.b);
412 break;
413 }
414
415 /* Add attributes if foreground */
416 if (foreground && len > 0 && (size_t)len < output_size - 20) {
417 char attrs[32] = "";
418 if (color->bold) {
419 strcat(attrs, "\033[1m");
420 }
421 if (color->dim) {
422 strcat(attrs, "\033[2m");
423 }
424 if (color->italic) {
425 strcat(attrs, "\033[3m");
426 }
427 if (color->underline) {
428 strcat(attrs, "\033[4m");
429 }
430 if (strlen(attrs) > 0) {
431 size_t remaining = output_size - (size_t)len;
432 size_t attrs_len = strlen(attrs);
433 if (attrs_len < remaining) {
434 memcpy(output + len, attrs, attrs_len + 1);
435 len += (int)attrs_len;
436 }
437 }
438 }
439
440 return (size_t)len;
441}

443/**
444 * @brief Downgrade color to match terminal capabilities
445 *
446 * Converts truecolor to 256-color or basic ANSI as needed.
447 */
448lle_color_t lle_color_downgrade(const lle_color_t *color, bool has_truecolor,
449 bool has_256color) {
450 if (!color) {
451 lle_color_t none = {0};
452 return none;
453 }
454
455 /* If terminal supports the color mode, return as-is */
456 if (color->mode == LLE_COLOR_MODE_NONE) {
457 return *color;
458 }
459 if (color->mode == LLE_COLOR_MODE_BASIC) {
460 return *color; /* Basic colors always supported */
461 }
462 if (color->mode == LLE_COLOR_MODE_256 && has_256color) {
463 return *color;
464 }
465 if (color->mode == LLE_COLOR_MODE_TRUE && has_truecolor) {
466 return *color;
467 }
468
469 /* Need to downgrade */
470 lle_color_t result = *color;
471
472 if (color->mode == LLE_COLOR_MODE_TRUE) {
473 if (has_256color) {
474 /* Downgrade truecolor to 256-color approximation */
475 /* Use 6x6x6 color cube (colors 16-231) */
476 uint8_t r = color->value.rgb.r;
477 uint8_t g = color->value.rgb.g;
478 uint8_t b = color->value.rgb.b;
479
480 /* Map 0-255 to 0-5 */
481 uint8_t r6 = (r * 6) / 256;
482 uint8_t g6 = (g * 6) / 256;
483 uint8_t b6 = (b * 6) / 256;
484
485 result.mode = LLE_COLOR_MODE_256;
486 result.value.palette = 16 + (36 * r6) + (6 * g6) + b6;
487 } else {
488 /* Downgrade to basic 8-color */
489 uint8_t r = color->value.rgb.r;
490 uint8_t g = color->value.rgb.g;
491 uint8_t b = color->value.rgb.b;
492
493 /* Simple threshold-based mapping */
494 int bright = (r > 127 || g > 127 || b > 127) ? 1 : 0;
495 int basic = 0;
496
497 if (r > 127)
498 basic |= 1; /* Red bit */
499 if (g > 127)
500 basic |= 2; /* Green bit */
501 if (b > 127)
502 basic |= 4; /* Blue bit */
503
504 result.mode = LLE_COLOR_MODE_BASIC;
505 result.value.basic = (uint8_t)basic;
506 result.bold = bright ? true : result.bold;
507 }
508 } else if (color->mode == LLE_COLOR_MODE_256 && !has_256color) {
509 /* Downgrade 256-color to basic 8-color */
510 uint8_t idx = color->value.palette;
511
512 if (idx < 8) {
513 /* Standard colors 0-7 map directly */
514 result.mode = LLE_COLOR_MODE_BASIC;
515 result.value.basic = idx;
516 } else if (idx < 16) {
517 /* Bright colors 8-15 map to standard + bold */
518 result.mode = LLE_COLOR_MODE_BASIC;
519 result.value.basic = idx - 8;
520 result.bold = true;
521 } else if (idx < 232) {
522 /* 6x6x6 color cube (16-231) */
523 int cube_idx = idx - 16;
524 int r = cube_idx / 36;
525 int g = (cube_idx / 6) % 6;
526 int b = cube_idx % 6;
527
528 int basic = 0;
529 if (r >= 3)
530 basic |= 1;
531 if (g >= 3)
532 basic |= 2;
533 if (b >= 3)
534 basic |= 4;
535
536 result.mode = LLE_COLOR_MODE_BASIC;
537 result.value.basic = (uint8_t)basic;
538 result.bold = (r >= 4 || g >= 4 || b >= 4);
539 } else {
540 /* Grayscale 232-255 */
541 int gray = idx - 232; /* 0-23 */
542 if (gray < 8) {
543 result.mode = LLE_COLOR_MODE_BASIC;
544 result.value.basic = LLE_COLOR_BLACK;
545 } else if (gray < 16) {
546 result.mode = LLE_COLOR_MODE_BASIC;
547 result.value.basic = LLE_COLOR_WHITE;
548 result.dim = true;
549 } else {
550 result.mode = LLE_COLOR_MODE_BASIC;
551 result.value.basic = LLE_COLOR_WHITE;
552 }
553 }
554 }
555
556 return result;
557}

713/**
714 * @brief Create the powerline theme
715 *
716 * Powerline-inspired prompt with separator glyphs.
717 * Requires a powerline-patched font (Nerd Fonts, Powerline fonts, etc.)
718 *
719 * Note: True powerline with colored background segments would require
720 * additional composer support for background colors. This theme provides
721 * a powerline-style look with foreground colors and separator symbols.
722 *
