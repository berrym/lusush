1/**
2 * @file widget_hooks.h
3 * @brief LLE Widget Hooks Manager
4 *
5 * Manages hook points in the editor lifecycle where widgets can be
6 * automatically triggered. Inspired by ZSH's hook system.
7 *
8 * Specification: docs/lle_specification/07_extensibility_framework_complete.md
9 * Section: 5 - Widget Hooks Integration
10 *
11 * Hook Types (ZSH-inspired):
12 * - line-init: Called when line editing starts (zle-line-init)
13 * - line-finish: Called when line editing ends (zle-line-finish)
14 * - buffer-modified: Called when buffer content changes
15 * - pre-command: Called before command execution (precmd)
16 * - post-command: Called after command execution
17 * - completion-start: Called when tab completion is triggered
18 * - completion-end: Called when completion finishes
19 * - history-search: Called when history search starts
20 * - terminal-resize: Called when terminal is resized
21 *
22 * Example Usage:
23 *
24 * // Initialize hooks manager
25 * lle_widget_hooks_manager_t *manager;
26 * lle_widget_hooks_manager_init(&manager, widget_registry, memory_pool);
27 *
28 * // Register widget for hook
29 * lle_widget_hook_register(manager, LLE_HOOK_BUFFER_MODIFIED, "my-widget");
30 *
31 * // Trigger hook (called by system)
32 * lle_widget_hook_trigger(manager, LLE_HOOK_BUFFER_MODIFIED, editor);
33 */

54/**
55 * Widget hook types (ZSH-inspired)
56 *
57 * These represent lifecycle points where widgets can be automatically
58 * triggered without explicit invocation.
59 *
60 * Lifecycle order for line editing:
61 * LINE_INIT -> [editing] -> LINE_ACCEPTED -> LINE_FINISH -> PRE_COMMAND
62 */
63typedef enum {
64 LLE_HOOK_LINE_INIT, /**< Start of line editing (zle-line-init) */
65 LLE_HOOK_LINE_ACCEPTED, /**< Line accepted, before display finalized */
66 LLE_HOOK_LINE_FINISH, /**< End of line editing (zle-line-finish) */
67 LLE_HOOK_BUFFER_MODIFIED, /**< Buffer content changed */
68 LLE_HOOK_PRE_COMMAND, /**< Before command execution (precmd) */
69 LLE_HOOK_POST_COMMAND, /**< After command execution */
70 LLE_HOOK_COMPLETION_START, /**< Tab completion triggered */
71 LLE_HOOK_COMPLETION_END, /**< Completion finished */
72 LLE_HOOK_HISTORY_SEARCH, /**< History search started */
73 LLE_HOOK_TERMINAL_RESIZE, /**< Terminal resized */
74 LLE_HOOK_COUNT /**< Total hook types (not a valid hook) */
75} lle_widget_hook_type_t;
76
77/**
78 * Hook registration structure
79 *
80 * Represents a widget registered for a specific hook type.
81 * Multiple widgets can be registered for the same hook.
82 */
83typedef struct lle_hook_registration {
84 lle_widget_t *widget; /**< Widget to execute */
85 lle_widget_hook_type_t hook_type; /**< Hook type */
86 uint64_t trigger_count; /**< Times triggered */
87 bool enabled; /**< Registration enabled */
88 struct lle_hook_registration *next; /**< Next registration in list */
89} lle_hook_registration_t;
90
91/**
92 * Widget hooks manager structure
93 *
94 * Manages all hook registrations and triggers. Maintains separate lists
95 * for each hook type for efficient triggering.
96 */
97typedef struct lle_widget_hooks_manager {
98 lle_widget_registry_t *registry; /**< Widget registry reference */
99 lle_hook_registration_t *hooks[LLE_HOOK_COUNT]; /**< Hooks per type */
100 lle_memory_pool_t *memory_pool; /**< Memory pool */
101 uint64_t total_hooks_triggered; /**< Total triggers across all hooks */
102 uint64_t hook_trigger_counts[LLE_HOOK_COUNT]; /**< Triggers per hook type */
103 bool hooks_enabled; /**< Global enable/disable */
104} lle_widget_hooks_manager_t;

148/**
149 * Register widget for hook
150 *
151 * Registers a widget to be executed when the specified hook is triggered.
152 * Multiple widgets can be registered for the same hook and will be executed
153 * in registration order.
154 *
155 * The widget must already be registered in the widget registry. The widget
156 * name is looked up at registration time and the widget pointer is cached
157 * for efficiency.
158 *
159 * @param manager Hooks manager (must not be NULL)
160 * @param hook_type Type of hook (must be < LLE_HOOK_COUNT)
161 * @param widget_name Name of widget to trigger (must not be NULL)
162 * @return LLE_SUCCESS on success
163 * @return LLE_ERROR_INVALID_PARAMETER if manager or widget_name is NULL
164 * @return LLE_ERROR_INVALID_PARAMETER if hook_type >= LLE_HOOK_COUNT
165 * @return LLE_ERROR_NOT_FOUND if widget does not exist in registry
166 * @return LLE_ERROR_ALREADY_EXISTS if widget already registered for this hook
167 * @return LLE_ERROR_OUT_OF_MEMORY if allocation fails
168 */
169lle_result_t lle_widget_hook_register(lle_widget_hooks_manager_t *manager,
170 lle_widget_hook_type_t hook_type,
171 const char *widget_name);
172
173/**
174 * Unregister widget from hook
175 *
176 * Removes widget from the specified hook. The widget will no longer be
177 * executed when the hook is triggered.
178 *
179 * @param manager Hooks manager (must not be NULL)
180 * @param hook_type Type of hook (must be < LLE_HOOK_COUNT)
181 * @param widget_name Name of widget to remove (must not be NULL)
182 * @return LLE_SUCCESS on success
183 * @return LLE_ERROR_INVALID_PARAMETER if manager or widget_name is NULL
184 * @return LLE_ERROR_INVALID_PARAMETER if hook_type >= LLE_HOOK_COUNT
185 * @return LLE_ERROR_NOT_FOUND if widget not registered for this hook
186 */
187lle_result_t lle_widget_hook_unregister(lle_widget_hooks_manager_t *manager,
188 lle_widget_hook_type_t hook_type,
189 const char *widget_name);

191/**
192 * Trigger all widgets registered for a hook
193 *
194 * This is the main function called when a hook point is reached in the
195 * editor lifecycle. It executes all widgets registered for the specified
196 * hook type in registration order.
197 *
198 * Widget execution is error-resilient: if one widget fails, the remaining
199 * widgets are still executed. The first error encountered is returned.
200 *
201 * Called by:
202 * - input_widget_hooks.c for input-related hooks
203 * - lle_readline.c for line editing hooks
204 * - completion system for completion hooks
205 * - history system for history hooks
206 *
207 * @param manager Hooks manager (must not be NULL)
208 * @param hook_type Type of hook to trigger (must be < LLE_HOOK_COUNT)
209 * @param editor Editor context (must not be NULL)
210 * @return LLE_SUCCESS if all widgets executed successfully
211 * @return LLE_ERROR_INVALID_PARAMETER if manager, editor is NULL or hook_type
212 * invalid
213 * @return First error code from widget execution (other widgets still executed)
214 */
215lle_result_t lle_widget_hook_trigger(lle_widget_hooks_manager_t *manager,
216 lle_widget_hook_type_t hook_type,
217 lle_editor_t *editor);
