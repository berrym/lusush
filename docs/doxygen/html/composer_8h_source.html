1/**
2 * @file composer.h
3 * @brief LLE Prompt Composer - Template/Segment/Theme Integration
4 *
5 * Specification: Spec 25 - Prompt Composer
6 * Version: 1.1.0
7 *
8 * The prompt composer integrates the template engine, segment registry,
9 * and theme registry to render complete prompts. It provides the callbacks
10 * needed by the template engine to resolve segments, check visibility,
11 * and apply theme colors.
12 *
13 * Shell Event Integration (Spec 26):
14 * The composer registers handlers with the shell event hub to automatically
15 * respond to directory changes, pre-command, and post-command events.
16 * This enables event-driven cache invalidation instead of time-based polling.
17 */
18

67/**
68 * @brief Prompt composer state
69 *
70 * Central coordinator that connects template engine, segment registry,
71 * and theme registry for unified prompt rendering.
72 *
73 * Shell Event Integration:
74 * The composer maintains a reference to the shell event hub and registers
75 * handlers for directory changed, pre-command, and post-command events.
76 * This enables automatic cache invalidation and context updates.
77 */
78typedef struct lle_prompt_composer {
79 lle_segment_registry_t *segments; /**< Segment registry */
80 lle_theme_registry_t *themes; /**< Theme registry */
81 lle_prompt_context_t context; /**< Current prompt context */
82 lle_composer_config_t config; /**< Composer configuration */
83 bool initialized; /**< Composer is initialized */
84

86 struct lle_shell_event_hub *shell_event_hub; /**< Shell event hub ref */
87 bool events_registered; /**< Event handlers registered */
88 bool needs_regeneration; /**< Prompt needs to be re-rendered */
89

91 lle_transient_state_t transient; /**< Transient prompt tracking */
