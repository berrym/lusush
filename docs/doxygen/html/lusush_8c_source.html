46/**
47 * @brief Detect if command line ends with background operator
48 *
49 * Checks if command ends with '&' but not '&&' (logical AND).
50 * Used by Spec 26 shell event hub to set is_background flag correctly.
51 *
52 * @param command The command line to check
53 * @return true if command will run in background, false otherwise
54 */
55static bool is_background_command(const char *command) {
56 if (!command) {
57 return false;
58 }
59
60 size_t len = strlen(command);
61 if (len == 0) {
62 return false;
63 }
64
65 /* Find last non-whitespace character */
66 size_t i = len;
67 while (i > 0 && isspace((unsigned char)command[i - 1])) {
68 i--;
69 }
70
71 if (i == 0) {
72 return false;
73 }
74
75 /* Check if ends with '&' */
76 if (command[i - 1] != '&') {
77 return false;
78 }
79
80 /* Make sure it's not '&&' (logical AND) */
81 if (i >= 2 && command[i - 2] == '&') {
82 return false;
83 }
84
85 return true;
86}

95int main(int argc, char **argv) {
96 FILE *in = NULL; // input file stream pointer
97 char *line = NULL; // pointer to a line of input read
98
99 // Initialize special shell variables
100 shell_pid = getpid();
101 shell_argc = argc;
102 shell_argv = argv;
103 last_exit_status = 0;
104
105 // Perform startup tasks
106 init(argc, argv, &in);
107
108 // Handle command mode (-c option)
109 if (shell_opts.command_mode && shell_opts.command_string) {
110 // Print the command if verbose mode is enabled
111 if (shell_opts.verbose) {
112 fprintf(stderr, "%s\n", shell_opts.command_string);
113 }
114
115 /**
116 * @brief Fire pre-command event for command mode (Spec 26)
117 *
118 * Command mode (-c) must also fire shell events for consistency
119 * with the event system. This enables proper tracking even for
120 * single-command invocations.
121 */
122 uint64_t cmd_start_us = 0;
123 if (g_lle_integration) {
124 struct timespec ts;
125 clock_gettime(CLOCK_MONOTONIC, &ts);
126 cmd_start_us =
127 (uint64_t)ts.tv_sec * 1000000ULL + (uint64_t)ts.tv_nsec / 1000;
128 bool is_bg = is_background_command(shell_opts.command_string);
129 lle_fire_pre_command(shell_opts.command_string, is_bg);
130 }
131
132 // Execute the command string and exit
133 int exit_status = parse_and_execute(shell_opts.command_string);
134
135 /**
136 * @brief Fire post-command event for command mode (Spec 26)
137 */
138 if (g_lle_integration) {
139 struct timespec ts;
140 clock_gettime(CLOCK_MONOTONIC, &ts);
141 uint64_t cmd_end_us =
142 (uint64_t)ts.tv_sec * 1000000ULL + (uint64_t)ts.tv_nsec / 1000;
143 lle_fire_post_command(shell_opts.command_string, exit_status,
144 cmd_end_us - cmd_start_us);
145 }
146
147 // Flush output buffers before exit to ensure all output is displayed
148 fflush(stdout);
149 fflush(stderr);
150
151 // Clean up and exit
152 free(shell_opts.command_string);
153
154 // Execute EXIT traps before terminating
155 execute_exit_traps();
156 exit(exit_status);
157 }
158
159 // Read input (buffering complete syntactic units) until user exits
160 // or EOF is read from either stdin or input file
161 while (!exit_flag) {
162 // Read complete command(s) using unified input system
163 // This ensures consistent parsing behavior between interactive and
164 // non-interactive modes
165 line = get_unified_input(in);
166
167 if (line == NULL) {
168 // Check if this was due to SIGINT (Ctrl+C) rather than real EOF
169 if (check_and_clear_sigint_flag()) {
170 // SIGINT received - just continue to show new prompt
171 // The signal handler already printed newline
172 continue;
173 }
174 // Real EOF encountered - print newline to prevent shell prompt
175 // artifacts
176 if (is_interactive_shell()) {
177 printf("\n");
178 fflush(stdout);
179 }
180 exit_flag = true;
181 continue;
182 }
183
184 // Add command to history if in interactive mode (handled by readline)
185 // History is automatically managed by the readline integration
186
187 /**
188 * @brief Fire pre-command event for LLE shell integration (Spec 26)
189 *
190 * Records command start time and notifies handlers before execution.
191 * Detects background commands (ending with &) to set is_background
192 * flag.
193 */
194 uint64_t cmd_start_us = 0;
195 if (g_lle_integration) {
196 struct timespec ts;
197 clock_gettime(CLOCK_MONOTONIC, &ts);
198 cmd_start_us =
199 (uint64_t)ts.tv_sec * 1000000ULL + (uint64_t)ts.tv_nsec / 1000;
200 bool is_bg = is_background_command(line);
201 lle_fire_pre_command(line, is_bg);
202 }
203
204 // Execute using unified modern parser and store exit status
205 int exit_status = parse_and_execute(line);
206 last_exit_status = exit_status;
207 set_exit_status(exit_status);
208
209 /* Fire post-command event for LLE shell integration (Spec 26)
210 * Provides exit code and execution duration for prompt and history.
211 */
212 if (g_lle_integration) {
213 struct timespec ts;
214 clock_gettime(CLOCK_MONOTONIC, &ts);
215 uint64_t cmd_end_us =
216 (uint64_t)ts.tv_sec * 1000000ULL + (uint64_t)ts.tv_nsec / 1000;
217 lle_fire_post_command(line, exit_status, cmd_end_us - cmd_start_us);
218 }
219
220 // Post-command display integration for layered display caching
221 // Enables the layered display system to handle post-command prompt
222 // rendering and achieve >75% cache hit rate targets
223 display_integration_post_command_update(line);
224
225 // Check notify option (-b): asynchronous background job notification
226 if (shell_opts.notify && global_executor) {
227 executor_update_job_status(global_executor);
228 }
229
230 // Check onecmd option (-t): exit after executing one command
231 if (shell_opts.onecmd) {
232 exit_flag = true;
233 }
234
235 if (is_interactive_shell()) {
236 free(line);
237 } else {
238 free_input_buffers();
239 }
240 }
241
242 if (in) {
243 fclose(in);
244 }
245
246 // Execute logout scripts if this is a login shell
247 if (is_login_shell()) {
248 config_execute_logout_scripts();
249 }
250
251 // Cleanup is handled by atexit() handlers registered in init.c
252 // This prevents double cleanup when exit() command is used
253
254 // Cleanup global executor before exit
255 if (global_executor) {
256 executor_free(global_executor);
257 global_executor = NULL;
258 }
259
260 // Execute EXIT traps before shell terminates normally
261 execute_exit_traps();
262
263 // Exit with the status of the last command executed (POSIX requirement)
264 exit(last_exit_status);
265}

267int parse_and_execute(const char *command) {
268 // Use global persistent executor for all commands to maintain function
269 // definitions
270 if (!global_executor) {
271 global_executor = executor_new();
272 if (!global_executor) {
273 return 1;
274 }
275 }
276
277 int exit_status = executor_execute_command_line(global_executor, command);
278
279 // Print error messages to stderr if there were any errors
280 if (executor_has_error(global_executor)) {
281 fprintf(stderr, "lusush: %s\n", executor_error(global_executor));
282 }
283
284 return exit_status;
285}
