53lle_result_t lle_edit_cache_create(lle_edit_cache_t **cache,
54 lle_memory_pool_t *memory_pool,

97lle_result_t lle_edit_cache_lookup(lle_edit_cache_t *cache,
98 size_t history_index,
99 lle_edit_cache_entry_t **entry) {
100 if (!cache || !cache->active || !entry) {
101 return LLE_ERROR_INVALID_PARAMETER;
102 }
103
104 *entry = NULL;
105
106 /* Search for entry */
107 lle_edit_cache_entry_t *current = cache->head;
108 while (current) {
109 if (current->history_index == history_index) {
110 /* Check if expired */
111 if (cache->config.entry_ttl_ms > 0 &&
112 is_expired(current, cache->config.entry_ttl_ms)) {
113 /* Remove expired entry */
114 remove_from_list(cache, current);
115 cache->stats.current_entries--;
116 cache->stats.expirations++;
117 cache->stats.misses++;
118 return LLE_SUCCESS;
119 }
120
121 /* Cache hit */
122 *entry = current;
123 cache->stats.hits++;
124
125 /* Update access metadata */
126 if (cache->config.track_access) {
127 clock_gettime(CLOCK_MONOTONIC, &current->last_accessed);
128 current->access_count++;
129 }
130
131 /* Move to head (most recently used) */
132 move_to_head(cache, current);
133
134 return LLE_SUCCESS;
135 }
136 current = current->next;
137 }
138
139 /* Cache miss */
140 cache->stats.misses++;
141
142 return LLE_SUCCESS;
143}

145lle_result_t lle_edit_cache_insert(lle_edit_cache_t *cache,
146 size_t history_index, uint64_t entry_id,
147 const char *original_text,
148 size_t original_length,
149 const char *reconstructed_text,
150 size_t reconstructed_length) {
151 if (!cache || !cache->active || !original_text) {
152 return LLE_ERROR_INVALID_PARAMETER;
153 }
154
155 /* Check if entry already exists */
156 lle_edit_cache_entry_t *existing = NULL;
157 lle_edit_cache_lookup(cache, history_index, &existing);
158
159 if (existing) {
160 /* Update existing entry */
161 /* For simplicity, we'll invalidate and re-insert */
162 lle_edit_cache_invalidate(cache, history_index);
163 }
164
165 /* Check if we need to evict */
166 if (cache->stats.current_entries >= cache->config.max_entries) {
167 evict_lru(cache);
168 }
169
170 /* Allocate new entry */
171 lle_edit_cache_entry_t *new_entry =
172 lle_pool_alloc(sizeof(lle_edit_cache_entry_t));
173 if (!new_entry) {
174 return LLE_ERROR_OUT_OF_MEMORY;
175 }
176
177 memset(new_entry, 0, sizeof(lle_edit_cache_entry_t));
178
179 new_entry->history_index = history_index;
180 new_entry->entry_id = entry_id;
181
182 /* Copy original text */
183 new_entry->original_text = lle_pool_alloc(original_length + 1);
184 if (!new_entry->original_text) {
185 return LLE_ERROR_OUT_OF_MEMORY;
186 }
187 memcpy(new_entry->original_text, original_text, original_length);
188 new_entry->original_text[original_length] = '\0';
189 new_entry->original_length = original_length;
190
191 /* Copy reconstructed text if provided */
192 if (reconstructed_text && reconstructed_length > 0) {
193 new_entry->reconstructed_text =
194 lle_pool_alloc(reconstructed_length + 1);
195 if (!new_entry->reconstructed_text) {
196 return LLE_ERROR_OUT_OF_MEMORY;
197 }
198 memcpy(new_entry->reconstructed_text, reconstructed_text,
199 reconstructed_length);
200 new_entry->reconstructed_text[reconstructed_length] = '\0';
201 new_entry->reconstructed_length = reconstructed_length;
202 }
203
204 /* Set timestamps */
205 clock_gettime(CLOCK_MONOTONIC, &new_entry->cached_at);
206 new_entry->last_accessed = new_entry->cached_at;
207 new_entry->access_count = 0;
208
209 /* Add to head of list */
210 new_entry->next = cache->head;
211 new_entry->prev = NULL;
212
213 if (cache->head) {
214 cache->head->prev = new_entry;
215 }
216 cache->head = new_entry;
217
218 if (!cache->tail) {
219 cache->tail = new_entry;
220 }
221
222 cache->stats.current_entries++;
223
224 return LLE_SUCCESS;
225}

227lle_result_t lle_edit_cache_invalidate(lle_edit_cache_t *cache,
228 size_t history_index) {
229 if (!cache || !cache->active) {
230 return LLE_ERROR_INVALID_PARAMETER;
231 }
232
233 /* Find and remove entry */
234 lle_edit_cache_entry_t *current = cache->head;
235 while (current) {
236 if (current->history_index == history_index) {
237 remove_from_list(cache, current);
238 cache->stats.current_entries--;
239 return LLE_SUCCESS;
240 }
241 current = current->next;
242 }
243
244 return LLE_SUCCESS;

272lle_result_t lle_edit_cache_evict_expired(lle_edit_cache_t *cache,
273 size_t *expired_count) {
274 if (!cache || !cache->active) {
275 return LLE_ERROR_INVALID_PARAMETER;
276 }
277
278 if (cache->config.entry_ttl_ms == 0) {
279 if (expired_count)
280 *expired_count = 0;
281 return LLE_SUCCESS; /* No TTL configured */
282 }
283
284 size_t count = 0;
285 lle_edit_cache_entry_t *current = cache->head;
286
287 while (current) {
288 lle_edit_cache_entry_t *next = current->next;
289
290 if (is_expired(current, cache->config.entry_ttl_ms)) {
291 remove_from_list(cache, current);
292 cache->stats.current_entries--;
293 cache->stats.expirations++;
294 count++;
295 }
296
297 current = next;
298 }
299
300 if (expired_count) {
301 *expired_count = count;
302 }
303
304 return LLE_SUCCESS;
305}

309static void move_to_head(lle_edit_cache_t *cache,
310 lle_edit_cache_entry_t *entry) {
311 if (cache->head == entry) {
312 return; /* Already at head */
313 }
314
315 /* Remove from current position */
316 if (entry->prev) {
317 entry->prev->next = entry->next;
318 }
319 if (entry->next) {
320 entry->next->prev = entry->prev;
321 }
322 if (cache->tail == entry) {
323 cache->tail = entry->prev;
324 }
325
326 /* Add to head */
327 entry->next = cache->head;
328 entry->prev = NULL;
329
330 if (cache->head) {
331 cache->head->prev = entry;
332 }
333 cache->head = entry;
334
335 if (!cache->tail) {
336 cache->tail = entry;
337 }
338}

340static void remove_from_list(lle_edit_cache_t *cache,
341 lle_edit_cache_entry_t *entry) {
342 if (entry->prev) {
343 entry->prev->next = entry->next;
344 } else {
345 cache->head = entry->next;
346 }
347
348 if (entry->next) {
349 entry->next->prev = entry->prev;
350 } else {
351 cache->tail = entry->prev;
352 }
353}

355static void evict_lru(lle_edit_cache_t *cache) {
356 if (!cache->tail) {
357 return; /* Cache is empty */
358 }
359
360 lle_edit_cache_entry_t *lru = cache->tail;
361 remove_from_list(cache, lru);
362 cache->stats.current_entries--;
363 cache->stats.evictions++;
364}

366static bool is_expired(lle_edit_cache_entry_t *entry, uint32_t ttl_ms) {
367 struct timespec now;
368 clock_gettime(CLOCK_MONOTONIC, &now);
369
370 int64_t age_ms = timespec_diff_ms(&entry->cached_at, &now);
371
372 return (age_ms >= (int64_t)ttl_ms);
373}

375static int64_t timespec_diff_ms(const struct timespec *start,
376 const struct timespec *end) {
377 int64_t sec_diff = end->tv_sec - start->tv_sec;
378 int64_t nsec_diff = end->tv_nsec - start->tv_nsec;
379
380 return (sec_diff * 1000) + (nsec_diff / 1000000);
381}
