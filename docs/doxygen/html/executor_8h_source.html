21// Function parameter definition
22typedef struct function_param {
23 char *name; // Parameter name
24 char *default_value; // Default value (NULL if required)
25 bool is_required; // True if parameter is required
26 struct function_param *next; // Next parameter in list
27} function_param_t;

29// Function definition storage
30typedef struct function_def {
31 char *name; // Function name
32 node_t *body; // Function body AST
33 function_param_t *params; // Parameter list (NULL for no params)
34 int param_count; // Number of parameters
35 struct function_def *next; // Next function in list
36} function_def_t;

41// Process in a job
42typedef struct process {

49// Job control structure
50typedef struct job {
51 int job_id;
52 pid_t pgid;
53 job_state_t state;
54 bool foreground;
55 process_t *processes;

60// Loop control states
61typedef enum {
62 LOOP_NORMAL, // Normal execution
63 LOOP_BREAK, // Break out of loop
64 LOOP_CONTINUE // Continue to next iteration
65} loop_control_t;

67// Execution context for maintaining state
68typedef struct executor {
69 bool interactive; // Interactive mode flag
70 bool debug; // Debug mode flag
71 int exit_status; // Last command exit status
72 const char *error_message; // Last error message
73 bool has_error; // Error flag
74 symtable_manager_t *symtable; // Symbol table manager
75 function_def_t *functions; // Function definition table
76 job_t *jobs; // Job control list
77 int next_job_id; // Next job ID to assign
78 pid_t shell_pgid; // Shell process group ID
79 loop_control_t loop_control; // Loop control state
80 int loop_depth; // Current loop nesting depth
81
82 // Script execution context for debugging
83 char *current_script_file; // Current script file being executed
84 int current_script_line; // Current line number in script
85 bool in_script_execution; // True if executing from script file
86
87 // Expansion error tracking
88 bool expansion_error; // True if error occurred during expansion
89 int expansion_exit_status; // Exit status from expansion errors
90
91} executor_t;
