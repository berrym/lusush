63/**
64 * @brief Initialize render controller
65 *
66 * Creates and initializes a render controller that manages all rendering
67 * operations for the LLE display integration. Initializes buffer renderer,
68 * cursor renderer, cache, dirty tracker, metrics, and configuration.
69 *
70 * @param controller Output pointer to receive initialized controller
71 * @param bridge Display bridge for communication with Lusush
72 * @param memory_pool Memory pool for allocations
73 * @return LLE_SUCCESS on success, error code on failure
74 *
75 * SPEC COMPLIANCE: Section 3.4 "Rendering System"
76 * ERROR HANDLING: All allocation failures handled, proper cleanup on error
77 */
78lle_result_t lle_render_controller_init(lle_render_controller_t **controller,
79 lle_display_bridge_t *bridge,
80 lle_memory_pool_t *memory_pool) {
81 lle_result_t result = LLE_SUCCESS;
82 lle_render_controller_t *ctrl = NULL;
83
84 /* Step 1: Validate parameters */
85 if (!controller) {
86 return LLE_ERROR_INVALID_PARAMETER;
87 }
88 if (!bridge || !memory_pool) {
89 return LLE_ERROR_INVALID_PARAMETER;
90 }
91
92 /* Step 2: Allocate controller structure */
93 ctrl = lle_pool_alloc(sizeof(lle_render_controller_t));
94 if (!ctrl) {
95 return LLE_ERROR_OUT_OF_MEMORY;
96 }
97 memset(ctrl, 0, sizeof(lle_render_controller_t));
98
99 /* Step 3: Store references */
100 ctrl->bridge = bridge;
101 ctrl->memory_pool = memory_pool;
102
103 /* Step 4: Initialize buffer renderer */
104 result =
105 lle_buffer_renderer_init_internal(&ctrl->buffer_renderer, memory_pool);
106 if (result != LLE_SUCCESS) {
107 lle_pool_free(ctrl);
108 return result;
109 }
110
111 /* Step 5: Initialize cursor renderer */
112 result =
113 lle_cursor_renderer_init_internal(&ctrl->cursor_renderer, memory_pool);
114 if (result != LLE_SUCCESS) {
115 lle_buffer_renderer_cleanup(ctrl->buffer_renderer);
116 lle_pool_free(ctrl);
117 return result;
118 }
119
120 /* Step 6: Initialize frame scheduler */
121 result = lle_frame_scheduler_init_internal(&ctrl->scheduler, memory_pool);
122 if (result != LLE_SUCCESS) {
123 lle_cursor_renderer_cleanup(ctrl->cursor_renderer);
124 lle_buffer_renderer_cleanup(ctrl->buffer_renderer);
125 lle_pool_free(ctrl);
126 return result;
127 }
128
129 /* Step 7: Initialize render cache */
130 result = lle_render_cache_init_internal(&ctrl->cache, memory_pool);
131 if (result != LLE_SUCCESS) {
132 lle_frame_scheduler_cleanup(ctrl->scheduler);
133 lle_cursor_renderer_cleanup(ctrl->cursor_renderer);
134 lle_buffer_renderer_cleanup(ctrl->buffer_renderer);
135 lle_pool_free(ctrl);
136 return result;
137 }
138
139 /* Step 8: Initialize render metrics */
140 result = lle_render_metrics_init_internal(&ctrl->metrics, memory_pool);
141 if (result != LLE_SUCCESS) {
142 lle_render_cache_cleanup(ctrl->cache);
143 lle_frame_scheduler_cleanup(ctrl->scheduler);
144 lle_cursor_renderer_cleanup(ctrl->cursor_renderer);
145 lle_buffer_renderer_cleanup(ctrl->buffer_renderer);
146 lle_pool_free(ctrl);
147 return result;
148 }
149
150 /* Step 9: Initialize render configuration */
151 result = lle_render_config_init_internal(&ctrl->config, memory_pool);
152 if (result != LLE_SUCCESS) {
153 lle_render_metrics_cleanup(ctrl->metrics);
154 lle_render_cache_cleanup(ctrl->cache);
155 lle_frame_scheduler_cleanup(ctrl->scheduler);
156 lle_cursor_renderer_cleanup(ctrl->cursor_renderer);
157 lle_buffer_renderer_cleanup(ctrl->buffer_renderer);
158 lle_pool_free(ctrl);
159 return result;
160 }
161
162 /* Note: pipeline will be initialized when rendering operations are
163 * implemented */
164 ctrl->pipeline = NULL;
165
166 /* Success - return initialized controller */
167 *controller = ctrl;
168 return LLE_SUCCESS;
169}

171/**
172 * @brief Clean up render controller
173 *
174 * Releases all resources associated with the render controller including
175 * renderers, cache, metrics, and configuration. The controller pointer
176 * becomes invalid after this call.
177 *
178 * @param controller Render controller to clean up
179 * @return LLE_SUCCESS on success, error code on failure
180 *
181 * SPEC COMPLIANCE: Standard cleanup pattern
182 * ERROR HANDLING: Handles NULL gracefully, cleans up in reverse order
183 */
184lle_result_t
185lle_render_controller_cleanup(lle_render_controller_t *controller) {
186 if (!controller) {
187 return LLE_ERROR_INVALID_PARAMETER;
188 }
189
190 /* Clean up in reverse order of initialization */
191
192 /* Step 1: Clean up configuration */
193 if (controller->config) {
194 lle_render_config_cleanup(controller->config);
195 controller->config = NULL;
196 }
197
198 /* Step 2: Clean up metrics */
199 if (controller->metrics) {
200 lle_render_metrics_cleanup(controller->metrics);
201 controller->metrics = NULL;
202 }
203
204 /* Step 3: Clean up render cache */
205 if (controller->cache) {
206 lle_render_cache_cleanup(controller->cache);
207 controller->cache = NULL;
208 }
209
210 /* Step 4: Clean up frame scheduler */
211 if (controller->scheduler) {
212 lle_frame_scheduler_cleanup(controller->scheduler);
213 controller->scheduler = NULL;
214 }
215
216 /* Step 5: Clean up cursor renderer */
217 if (controller->cursor_renderer) {
218 lle_cursor_renderer_cleanup(controller->cursor_renderer);
219 controller->cursor_renderer = NULL;
220 }
221
222 /* Step 6: Clean up buffer renderer */
223 if (controller->buffer_renderer) {
224 lle_buffer_renderer_cleanup(controller->buffer_renderer);
225 controller->buffer_renderer = NULL;
226 }
227
228 /* Step 7: Clean up pipeline (if initialized) */
229 if (controller->pipeline) {
230 /* Pipeline cleanup will be implemented in future phase */
231 controller->pipeline = NULL;
232 }
233
234 /* Clear references (not owned by controller) */
235 controller->bridge = NULL;
236 controller->memory_pool = NULL;
237
238 /* Note: The controller structure itself is not freed here.
239 * It should be freed by the caller using the same memory pool
240 * that was used to allocate it. */
241
242 return LLE_SUCCESS;
243}

576/**
577 * @brief Render buffer content to display output
578 *
579 * Converts buffer content to rendered display output with cursor positioning.
580 * This is the main rendering entry point that coordinates buffer-to-display
581 * conversion with performance optimization.
582 *
583 * @param controller Render controller managing the rendering process
584 * @param buffer Buffer containing content to render
585 * @param cursor Current cursor position
586 * @param output Output pointer to receive rendered output
587 * @return LLE_SUCCESS on success, error code on failure
588 *
589 * SPEC COMPLIANCE: Section 3.4 "Rendering System"
590 * PERFORMANCE: Uses render metrics for timing tracking
591 */
592lle_result_t lle_render_buffer_content(lle_render_controller_t *controller,
593 lle_buffer_t *buffer,
594 lle_cursor_position_t *cursor,
595 lle_render_output_t **output) {
596 struct timespec start_time, end_time;
597
598 /* Step 1: Performance monitoring start */
599 clock_gettime(CLOCK_MONOTONIC, &start_time);
600
601 /* Step 2: Validate input parameters */
602 if (!controller || !buffer || !cursor || !output) {
603 return LLE_ERROR_INVALID_PARAMETER;
604 }
605
606 /* Step 3: Allocate render output structure */
607 lle_render_output_t *render_out =
608 lle_pool_alloc(sizeof(lle_render_output_t));
609 if (!render_out) {
610 return LLE_ERROR_OUT_OF_MEMORY;
611 }
612 memset(render_out, 0, sizeof(lle_render_output_t));
613
614 /* Step 4: Estimate required output size (buffer length + ANSI codes
615 * overhead) */
616 size_t estimated_size =
617 buffer->length + 256; /* Extra space for ANSI codes */
618
619 /* Step 5: Allocate output content buffer */
620 render_out->content = lle_pool_alloc(estimated_size);
621 if (!render_out->content) {
622 lle_pool_free(render_out);
623 return LLE_ERROR_OUT_OF_MEMORY;
624 }
625 render_out->content_capacity = estimated_size;
626
627 /* Step 6: Full render - copy buffer content */
628 if (buffer->length > 0) {
629 memcpy(render_out->content, buffer->data, buffer->length);
630 render_out->content[buffer->length] =
631 '\0'; /* CRITICAL: Null-terminate for string functions */
632 render_out->content_length = buffer->length;
633 } else {
634 render_out->content[0] = '\0';
635 render_out->content_length = 0;
636 }
637
638 /* Step 7: Set render timestamp */
639 clock_gettime(CLOCK_MONOTONIC, &end_time);
640 render_out->timestamp = (uint64_t)end_time.tv_sec * 1000000ULL +
641 (uint64_t)end_time.tv_nsec / 1000ULL;
642
643 /* Step 8: Update render metrics */
644 uint64_t render_time_ns =
645 (end_time.tv_sec - start_time.tv_sec) * 1000000000ULL +
646 (end_time.tv_nsec - start_time.tv_nsec);
647
648 controller->metrics->total_renders++;
649 controller->metrics->full_renders++;
650
651 /* Update average full render time */
652 if (controller->metrics->full_renders == 1) {
653 controller->metrics->avg_full_render_time_ns = render_time_ns;
654 } else {
655 controller->metrics->avg_full_render_time_ns =
656 (controller->metrics->avg_full_render_time_ns *
657 (controller->metrics->full_renders - 1) +
658 render_time_ns) /
659 controller->metrics->full_renders;
660 }
661
662 /* Update global min/max render times */
663 if (render_time_ns < controller->metrics->min_render_time_ns ||
664 controller->metrics->min_render_time_ns == 0) {
665 controller->metrics->min_render_time_ns = render_time_ns;
666 }
667 if (render_time_ns > controller->metrics->max_render_time_ns) {
668 controller->metrics->max_render_time_ns = render_time_ns;
669 }
670
671 /* Update average render time (all renders) */
672 if (controller->metrics->total_renders == 1) {
673 controller->metrics->avg_render_time_ns = render_time_ns;
674 } else {
675 controller->metrics->avg_render_time_ns =
676 (controller->metrics->avg_render_time_ns *
677 (controller->metrics->total_renders - 1) +
678 render_time_ns) /
679 controller->metrics->total_renders;
680 }
681
682 /* Step 9: Return rendered output */
683 *output = render_out;
684 return LLE_SUCCESS;
685}

687/**
688 * @brief Render cursor position to terminal escape codes
689 *
690 * Generates ANSI escape sequences to position the cursor at the specified
691 * location. Handles coordinate translation from buffer positions to screen
692 * positions.
693 *
694 * @param controller Render controller
695 * @param cursor Cursor position to render
696 * @param output Output buffer for escape codes
697 * @param output_size Size of output buffer
698 * @param bytes_written Number of bytes written to output
699 * @return LLE_SUCCESS on success, error code on failure
700 *
701 * SPEC COMPLIANCE: Section 3.4.2 "Cursor Rendering"
702 */
703lle_result_t lle_render_cursor_position(lle_render_controller_t *controller,
704 lle_cursor_position_t *cursor,
705 char *output, size_t output_size,
706 size_t *bytes_written) {
707 /* Step 1: Validate parameters */
708 if (!controller || !cursor || !output || !bytes_written) {
709 return LLE_ERROR_INVALID_PARAMETER;
710 }
711 if (output_size < 32) { /* Minimum space for ANSI cursor positioning */
712 return LLE_ERROR_INVALID_PARAMETER;
713 }
714
715 /* Step 2: Check cursor visibility */
716 if (!controller->cursor_renderer->cursor_visible) {
717 /* Cursor hidden - generate hide cursor sequence */
718 int written = snprintf(output, output_size, "\033[?25l");
719 if (written < 0 || (size_t)written >= output_size) {
720 return LLE_ERROR_INVALID_PARAMETER;
721 }

738 }
739

741

743 /* For basic implementation, no additional state tracking needed */

746}
