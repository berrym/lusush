1/**
2 * @file layer_events.h
3 * @brief Layer events system - Inter-layer communication protocol
4 *
5 * Provides event-driven communication between display layers using a
6 * publisher/subscriber pattern for loose coupling. Manages event queues
7 * and processing for the layered display architecture.
8 *
9 * @author Michael Berry <trismegustis@gmail.com>
10 * @copyright Copyright (C) 2021-2026 Michael Berry
11 * @license GPL-3.0-or-later
12 *
13 * Key Features:
14 * - Publisher/Subscriber pattern for loose coupling
15 * - Event queue management and processing
16 * - Performance-optimized event handling
17 * - Memory-safe event data management
18 * - Event debugging and monitoring
19 * - Priority-based event processing
20 *
21 * Design Principles:
22 * - No direct layer dependencies
23 * - Asynchronous event processing
24 * - Type-safe event handling
25 * - Performance-optimized delivery
26 * - Comprehensive event debugging
27 */

690 * layer_event_system_t *events = layer_events_create(NULL);
691 * layer_events_init(events);

694 * layer_events_subscribe(events, LAYER_EVENT_CONTENT_CHANGED,
695 * LAYER_ID_COMMAND_LAYER, handle_content_change,
696 * NULL, LAYER_EVENT_PRIORITY_NORMAL);

698 * // Publish a content change event
699 * layer_events_publish_content_changed(events, LAYER_ID_PROMPT_LAYER,
700 * "New prompt text", 15, false);
701 *
702 * // Process events
703 * layer_events_process_pending(events, 0, 100);

705 * // Cleanup
