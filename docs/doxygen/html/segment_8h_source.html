1/**
2 * @file segment.h
3 * @brief LLE Prompt Segment System - Type Definitions and Function
4 * Declarations
5 *
6 * Specification: Spec 25 Section 5 - Segment Architecture
7 * Version: 1.0.0
8 *
9 * Segments are the building blocks of prompts - modular, reusable components
10 * that render specific pieces of information (directory, git status, etc.).
11 *
12 * Segments can be:
13 * - Synchronous: Render immediately (directory, user, host)
14 * - Asynchronous: Fetch data in background (git status)
15 * - Cached: Reuse previous output until invalidated
16 * - Conditional: Only visible when certain conditions are met
17 */

63/**
64 * @brief Segment capability flags
65 */
66typedef enum lle_segment_capability {
67 LLE_SEG_CAP_NONE = 0,
68 LLE_SEG_CAP_ASYNC = (1 << 0), /**< Requires async data fetch */
69 LLE_SEG_CAP_CACHEABLE = (1 << 1), /**< Output can be cached */
70 LLE_SEG_CAP_EXPENSIVE = (1 << 2), /**< May be slow, respect timeouts */
71 LLE_SEG_CAP_THEME_AWARE = (1 << 3), /**< Uses theme colors */
72 LLE_SEG_CAP_DYNAMIC = (1 << 4), /**< Content changes frequently */
73 LLE_SEG_CAP_OPTIONAL = (1 << 5), /**< Can be hidden if no data */
74 LLE_SEG_CAP_PROPERTIES = (1 << 6) /**< Exposes sub-properties */
75} lle_segment_capability_t;
