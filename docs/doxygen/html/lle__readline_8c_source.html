282/**
283 * @brief Update autosuggestion based on current buffer content
284 *
285 * Searches LLE history for prefix matches and stores the suggestion
286 * (the remaining text to append) in ctx->current_suggestion.
287 *
288 * Fish-style autosuggestion rules:
289 * - Only suggest when cursor is at end of buffer
290 * - Only suggest for non-empty input (>= 2 chars)
291 * - Don't suggest if input ends with space
292 * - Don't suggest in multiline mode
293 * - Search history most-recent-first for best prefix match
294 *
295 * @param ctx Readline context with buffer and editor
296 */
297static void update_autosuggestion(readline_context_t *ctx) {
298 /* Clear existing suggestion */
299 if (ctx->current_suggestion) {
300 ctx->current_suggestion[0] = '\0';
301 }
302
303 /* Bail out conditions */
304 if (!ctx || !ctx->buffer || !ctx->editor || !ctx->editor->history_system) {
305 return;
306 }
307
308 /* Check if autosuggestion regeneration is suppressed (e.g., during Ctrl+G
309 * clear) */
310 if (ctx->suppress_autosuggestion) {
311 return;
312 }
313
314 /* Only suggest when cursor at end */
315 if (ctx->buffer->cursor.byte_offset != ctx->buffer->length) {
316 return;
317 }
318
319 /* Need at least 2 characters */
320 if (ctx->buffer->length < 2) {
321 return;
322 }
323
324 const char *input = ctx->buffer->data;
325 if (!input || !*input) {
326 return;
327 }
328
329 /* Don't suggest if ends with space */
330 if (input[ctx->buffer->length - 1] == ' ') {
331 return;
332 }
333
334 /* Don't suggest in multiline mode */
335 if (strchr(input, '\n') != NULL) {
336 return;
337 }
338
339 /* Search LLE history for prefix match (most recent first) */
340 lle_history_core_t *history = ctx->editor->history_system;
341 size_t count = 0;
342 lle_history_get_entry_count(history, &count);
343
344 if (count == 0) {
345 return;
346 }
347
348 size_t input_len = ctx->buffer->length;
349
350 /* Limit history search to prevent hangs with very large histories */
351 #define MAX_SUGGESTION_SEARCH_ITERATIONS 5000
352 size_t iterations = 0;
353
354 /* Search backwards (most recent first) */
355 for (size_t i = count; i > 0 && iterations < MAX_SUGGESTION_SEARCH_ITERATIONS; i--, iterations++) {
356 /* Check watchdog every 500 iterations to allow abort on timeout */
357 if (iterations > 0 && (iterations % 500) == 0 && lle_watchdog_check()) {
358 return; /* Watchdog fired - abort suggestion search */
359 }
360
361 lle_history_entry_t *entry = NULL;
362 lle_result_t result =
363 lle_history_get_entry_by_index(history, i - 1, &entry);
364
365 if (result != LLE_SUCCESS || !entry || !entry->command) {
366 continue;
367 }
368
369 /* Skip multiline history entries - don't suggest them for single-line
370 * input This prevents suggesting "echo \"\nhello\"" when user types
371 * "echo"
372 *
373 * Note: LLE history stores newlines as escaped "\n" (two chars:
374 * backslash + n) not as actual newline bytes. Check for both formats.
375 */
376 if (strchr(entry->command, '\n') != NULL ||
377 strstr(entry->command, "\\n") != NULL) {
378 continue;
379 }
380
381 /* Check for prefix match using Unicode-aware comparison (TR#29
382 * compliant) This handles NFC normalization to ensure equivalent
383 * Unicode sequences match regardless of encoding (e.g., precomposed vs
384 * decomposed) */
385 if (lle_unicode_is_prefix(input, input_len, entry->command,
386 strlen(entry->command), NULL)) {
387 /* Found a match - get the remaining text */
388 const char *remaining = entry->command + input_len;
389
390 /* Only suggest if there's something to add */
391 if (*remaining) {
392 size_t remaining_len = strlen(remaining);
393
394 /* Ensure buffer is large enough */
395 if (remaining_len + 1 > ctx->suggestion_alloc_size) {
396 size_t new_size = remaining_len + 64;
397 char *new_buf = realloc(ctx->current_suggestion, new_size);
398 if (new_buf) {
399 ctx->current_suggestion = new_buf;
400 ctx->suggestion_alloc_size = new_size;
401 } else {
402 return; /* Allocation failed */
403 }
404 }
405
406 strcpy(ctx->current_suggestion, remaining);
407 return; /* Found best match */
408 }
409 }
410 }
411}

475/**
476 * @brief Context-aware RIGHT arrow action (Fish-style autosuggestion
477 * acceptance)
478 *
479 * Priority order:
480 * 1. If completion menu is active, navigate menu (delegate to lle_forward_char)
481 * 2. If cursor at end of buffer with autosuggestion, accept suggestion
482 * 3. Otherwise, move cursor right by one grapheme
483 *
484 * This is a context-aware action that requires readline_context_t access
485 * for the autosuggestion buffer.
486 */
487lle_result_t lle_forward_char_or_accept_suggestion(readline_context_t *ctx) {
488 if (!ctx || !ctx->buffer) {
489 return LLE_ERROR_INVALID_PARAMETER;
490 }
491
492 /* Priority 1: If completion menu is active, delegate to lle_forward_char
493 * which handles menu navigation (move right in multi-column menu) */
494 if (ctx->editor) {
495 bool menu_active = false;
496
497 if (ctx->editor->completion_system &&
498 lle_completion_system_is_menu_visible(
499 ctx->editor->completion_system)) {
500 menu_active = true;
501 }
502
503 if (menu_active) {
504 /* Delegate to lle_forward_char which handles menu navigation */
505 lle_result_t result = lle_forward_char(ctx->editor);
506 refresh_display(ctx);
507 return result;
508 }
509 }
510
511 /* Priority 2: Fish-style autosuggestion - if at end of buffer with
512 * suggestion, accept it */
513 if (ctx->buffer->cursor.byte_offset == ctx->buffer->length &&
514 has_autosuggestion(ctx)) {
515 if (accept_autosuggestion(ctx)) {
516 refresh_display(ctx);
517 return LLE_SUCCESS;
518 }
519 }
520
521 /* Priority 3: Normal behavior - move cursor right */
522 if (ctx->buffer->cursor.grapheme_index < ctx->buffer->grapheme_count &&
523 ctx->editor && ctx->editor->cursor_manager) {
524 lle_cursor_manager_move_to_byte_offset(ctx->editor->cursor_manager,
525 ctx->buffer->cursor.byte_offset);
526 lle_cursor_manager_move_by_graphemes(ctx->editor->cursor_manager, 1);
527 lle_cursor_manager_get_position(ctx->editor->cursor_manager,
528 &ctx->buffer->cursor);
529 refresh_display(ctx);
530 }
531
532 return LLE_SUCCESS;
533}

536 * @brief Context-aware END key action (Fish-style autosuggestion acceptance)
537 *
538 * If cursor is already at end of buffer AND autosuggestion is available, accept
539 * it. Otherwise, move cursor to end of line.
540 *
541 * This is a context-aware action that requires readline_context_t access
542 * for the autosuggestion buffer.
543 */
544lle_result_t lle_end_of_line_or_accept_suggestion(readline_context_t *ctx) {
545 if (!ctx || !ctx->buffer) {
546 return LLE_ERROR_INVALID_PARAMETER;
547 }
548
549 /* Fish-style: If already at end of buffer with suggestion, accept it
550 * Note: Suggestion acceptance only applies at absolute buffer end,
551 * since suggestions extend the entire command, not individual lines */
552 if (ctx->buffer->cursor.byte_offset == ctx->buffer->length &&
553 has_autosuggestion(ctx)) {
554 if (accept_autosuggestion(ctx)) {
555 refresh_display(ctx);
556 return LLE_SUCCESS;
557 }
558 }
559
560 /* Normal behavior: move cursor to end of CURRENT LINE (not buffer)
561 * This is critical for multiline mode - Ctrl+E should move to end of
562 * the current logical line, while Alt+> moves to end of buffer */
563 if (ctx->editor) {
564 lle_end_of_line(ctx->editor);
565 /* Sync buffer cursor from editor after the move */
566 if (ctx->editor->cursor_manager) {
567 lle_cursor_manager_get_position(ctx->editor->cursor_manager,
568 &ctx->buffer->cursor);
569 }
570 } else {
571 /* Fallback for no editor: move to buffer end (single-line behavior) */
572 ctx->buffer->cursor.byte_offset = ctx->buffer->length;
573 }
574
575 refresh_display(ctx);
576
577 return LLE_SUCCESS;

580/**
581 * @brief Find next word boundary in suggestion text
582 *
583 * Finds the end of the next word in the suggestion, used for partial
584 * acceptance with Ctrl+Right.
585 *
586 * @param suggestion The suggestion text to search
587 * @return Byte offset to end of next word (including trailing space if any)
588 */
589static size_t find_next_word_boundary_in_suggestion(const char *suggestion) {
590 if (!suggestion || !*suggestion) {
591 return 0;
592 }
593
594 size_t pos = 0;
595
596 /* Skip leading whitespace */
597 while (suggestion[pos] &&
598 (suggestion[pos] == ' ' || suggestion[pos] == '\t')) {
599 pos++;
600 }
601
602 /* Find end of word (non-whitespace characters) */
603 while (suggestion[pos] && suggestion[pos] != ' ' &&
604 suggestion[pos] != '\t') {
605 pos++;
606 }
607
608 /* Include one trailing space if present (feels more natural) */
609 if (suggestion[pos] == ' ') {
610 pos++;
611 }
612
613 return pos;
614}

616/**
617 * @brief Accept partial autosuggestion (one word) into buffer
618 *
619 * Inserts the next word from the suggestion into the buffer and updates
620 * the suggestion to show remaining text.
621 *
622 * @param ctx Readline context
623 * @return true if partial suggestion was accepted, false otherwise
624 */
625static bool accept_partial_autosuggestion(readline_context_t *ctx) {
626 if (!has_autosuggestion(ctx)) {
627 return false;
628 }
629
630 /* Find next word boundary */
631 size_t word_len =
632 find_next_word_boundary_in_suggestion(ctx->current_suggestion);
633 if (word_len == 0) {
634 return false;
635 }
636
637 /* Insert just the word portion at cursor (which is at end) */
638 lle_result_t result =
639 lle_buffer_insert_text(ctx->buffer, ctx->buffer->cursor.byte_offset,
640 ctx->current_suggestion, word_len);
641
642 if (result == LLE_SUCCESS) {
643 /* Sync cursor manager */
644 if (ctx->editor && ctx->editor->cursor_manager) {
645 lle_cursor_manager_move_to_byte_offset(
646 ctx->editor->cursor_manager, ctx->buffer->cursor.byte_offset);
647 }
648
649 /* Update suggestion to show remaining text */
650 size_t remaining_len = strlen(ctx->current_suggestion) - word_len;
651 if (remaining_len > 0) {
652 memmove(ctx->current_suggestion, ctx->current_suggestion + word_len,
653 remaining_len + 1); /* +1 for null terminator */
654 } else {
655 /* No more suggestion text */
656 ctx->current_suggestion[0] = '\0';
657 }
658
659 return true;
660 }

662 return false;

802 return;

811 *
812 * This must happen BEFORE rendering so the ghost text is available
813 * for display_controller to render in dc_handle_redraw_needed().

832 return;

874 }
875 }

1411 }
1412 }

1417 if (ctx->current_suggestion) {
1418 ctx->current_suggestion[0] = '\0';

1451 if (ctx->editor && ctx->editor->completion_system) {

1460 display_controller_clear_completion_menu(dc);
1461 }

1465

1467 if (ctx->current_suggestion) {
1468 ctx->current_suggestion[0] = '\0';

1497 * 1. If completion menu is visible: dismiss menu (first press)
1498 * 2. If autosuggestion is visible: clear suggestion (second press)
1499 * 3. If nothing to dismiss: no-op (does NOT abort line)

1516 ctx->editor->completion_system)) {
1517 menu_visible = true;
1518 }

1522 * Restoring original pre-completion text would require storing
1523 * the buffer state before completion started, which is not yet
1524 * implemented. Current behavior matches many shells where ESC
1525 * just dismisses the menu without reverting the text. */

1528 lle_completion_system_clear(ctx->editor->completion_system);
1529 }

1533 if (dc) {
1534 display_controller_clear_completion_menu(dc);
1535 }
1536
1537 /* Refresh display to clear menu from screen */
1538 refresh_display(ctx);
1539
1540 return LLE_SUCCESS; /* Menu dismissed, stop here */
1541 }
1542 }
1543
1544 /* Tier 2: Check if autosuggestion is visible - clear it */

1547 if (ctx->current_suggestion) {

1562 refresh_display(ctx);

1565 }

1616 while (pos > 0 && (data[pos - 1] == ' ' || data[pos - 1] == '\t')) {
1617 pos--;
1618 }
1619

1624 word_start--;
1625 }

1635 ctx->kill_buffer_size = kill_len + 1;
1636 }
1637 }

1639 if (ctx->kill_buffer && ctx->kill_buffer_size >= kill_len + 1) {
1640 memcpy(ctx->kill_buffer, data + word_start, kill_len);
1641 ctx->kill_buffer[kill_len] = '\0';

1645 begin_change_sequence(ctx, "kill-word");
