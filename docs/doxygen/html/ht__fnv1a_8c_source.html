22/**
23 * __fnv1a_hash:
24 * Return a hash key using the 64 bit FNV1A algorithm.
25 */
26static uint64_t __fnv1a_hash(const void *key, uint64_t seed, bool ignore_case) {
27 uint64_t h, c;
28
29 h = seed;
30
31 for (unsigned char *p = (unsigned char *)key; *p; p++) {
32 c = (uint64_t)(*p);
33 if (ignore_case) {
34 c = tolower(c);
35 }
36 h ^= (uint64_t)c;
37 h *= FNV1A_PRIME;
38 }
39
40 return h;
41}
