62/**
63 * Read line implementation - routes to appropriate controller.
64 */
65static lle_result_t
66lle_adaptive_interface_read_line(lle_adaptive_context_t *ctx,
67 const char *prompt, char **line) {
68
69 if (!ctx || !prompt || !line) {
70 return LLE_ERROR_INVALID_PARAMETER;
71 }
72
73 switch (ctx->mode) {
74 case LLE_ADAPTIVE_MODE_NATIVE:
75 return lle_native_read_line(ctx->controller.native, prompt, line);
76
77 case LLE_ADAPTIVE_MODE_ENHANCED:
78 return lle_display_client_read_line(ctx->controller.display_client,
79 prompt, line);
80
81 case LLE_ADAPTIVE_MODE_MULTIPLEXED:
82 return lle_multiplexer_read_line(ctx->controller.mux, prompt, line);
83
84 case LLE_ADAPTIVE_MODE_MINIMAL:
85 return lle_minimal_read_line(ctx->controller.minimal, prompt, line);
86
87 case LLE_ADAPTIVE_MODE_NONE:
88 /* Non-interactive mode - no line editing available */
89 return LLE_ERROR_FEATURE_NOT_AVAILABLE;
90 }
91
92 /* Should never reach here */
93 return LLE_ERROR_INVALID_STATE;
94}

200/**
201 * Initialize adaptive context with detected mode.
202 */
203lle_result_t lle_initialize_adaptive_context(
204 lle_adaptive_context_t **context,
205 const lle_terminal_detection_result_t *detection_result,
206 lusush_memory_pool_t *memory_pool) {
207
208 if (!context || !detection_result) {
209 return LLE_ERROR_INVALID_PARAMETER;
210 }
211
212 /* Create context */
213 lle_adaptive_context_t *ctx = calloc(1, sizeof(lle_adaptive_context_t));
214 if (!ctx) {
215 return LLE_ERROR_OUT_OF_MEMORY;
216 }
217
218 /* Clone detection result */
219 ctx->detection_result = malloc(sizeof(lle_terminal_detection_result_t));
220 if (!ctx->detection_result) {
221 free(ctx);
222 return LLE_ERROR_OUT_OF_MEMORY;
223 }
224 memcpy(ctx->detection_result, detection_result,
225 sizeof(lle_terminal_detection_result_t));
226
227 ctx->mode = detection_result->recommended_mode;
228 ctx->memory_pool = memory_pool;
229 ctx->healthy = true;
230 ctx->error_count = 0;
231
232 /* Initialize mode-specific controller */
233 lle_result_t result;
234 switch (ctx->mode) {
235 case LLE_ADAPTIVE_MODE_NATIVE:
236 result = lle_initialize_native_controller(ctx, memory_pool);
237 break;
238
239 case LLE_ADAPTIVE_MODE_ENHANCED:
240 result = lle_initialize_display_client_controller(ctx, memory_pool);
241 break;
242
243 case LLE_ADAPTIVE_MODE_MULTIPLEXED:
244 result = lle_initialize_multiplexer_controller(ctx, memory_pool);
245 break;
246
247 case LLE_ADAPTIVE_MODE_MINIMAL:
248 result = lle_initialize_minimal_controller(ctx, memory_pool);
249 break;
250
251 case LLE_ADAPTIVE_MODE_NONE:
252 /* Non-interactive mode - no controller to initialize */
253 free(ctx->detection_result);
254 free(ctx);
255 return LLE_ERROR_FEATURE_NOT_AVAILABLE;
256 }
257
258 /* Should never reach here - all cases handled */
259 if (ctx->mode < LLE_ADAPTIVE_MODE_NONE ||
260 ctx->mode > LLE_ADAPTIVE_MODE_MULTIPLEXED) {
261 free(ctx->detection_result);
262 free(ctx);
263 return LLE_ERROR_INVALID_STATE;
264 }
265
266 if (result != LLE_SUCCESS) {
267 free(ctx->detection_result);
268 free(ctx);
269 return result;
270 }
271
272 *context = ctx;
273 return LLE_SUCCESS;
274}

316/**
317 * Create adaptive interface with automatic detection.
318 */
319lle_result_t lle_create_adaptive_interface(lle_adaptive_interface_t **interface,
320 void *config) {
321 (void)config; /* Reserved for configuration-driven initialization */
322
323 if (!interface) {
324 return LLE_ERROR_INVALID_PARAMETER;
325 }
326
327 /* Perform terminal detection */
328 lle_terminal_detection_result_t *detection = NULL;
329 lle_result_t result =
330 lle_detect_terminal_capabilities_optimized(&detection);
331 if (result != LLE_SUCCESS) {
332 return result;
333 }
334
335 /* Create adaptive context */
336 lle_adaptive_context_t *context = NULL;
337 result = lle_initialize_adaptive_context(&context, detection, NULL);
338
339 if (result != LLE_SUCCESS) {
340 /* Do NOT destroy detection - the optimized version returns a cached
341 * result that is managed internally. Destroying it could cause
342 * double-free if the cache was refreshed. */
343 return result;
344 }
345
346 /* Detection result is cached by the detection system, don't free it */
347
348 /* Create interface wrapper */
349 lle_adaptive_interface_t *iface =
350 calloc(1, sizeof(lle_adaptive_interface_t));
351 if (!iface) {
352 lle_adaptive_context_destroy(context);
353 return LLE_ERROR_OUT_OF_MEMORY;
354 }
355
356 iface->adaptive_context = context;
357
358 /* Assign function pointers */
359 lle_assign_interface_functions(iface);
360
361 *interface = iface;
362 return LLE_SUCCESS;
363}

382/**
383 * Determine if shell should be interactive.
384 */
385bool lle_adaptive_should_shell_be_interactive(bool forced_interactive,
386 bool has_script_file,
387 bool stdin_mode) {
388
389 /* Script execution is never interactive */
390 if (has_script_file) {
391 return false;
392 }
393
394 /* Forced interactive always wins */
395 if (forced_interactive) {
396 return true;
397 }
398
399 /* Stdin mode typically disables interactive features */
400 if (stdin_mode) {
401 return false;
402 }
403
404 /* Use enhanced detection for final decision */
405 lle_terminal_detection_result_t *detection = NULL;
406 lle_result_t result =
407 lle_detect_terminal_capabilities_comprehensive(&detection);
408 if (result != LLE_SUCCESS) {
409 return false;
410 }
411
412 /* Interactive if mode is not NONE */
413 /* Also check force_interactive flag from signature */
414 bool interactive = (detection->recommended_mode != LLE_ADAPTIVE_MODE_NONE);
415 if (!interactive && detection->matched_signature) {
416 interactive = detection->matched_signature->force_interactive;
417 }
418
419 lle_terminal_detection_result_destroy(detection);
420 return interactive;
421}

423/**
424 * Get configuration recommendations.
425 */
426void lle_adaptive_get_recommended_config(
427 lle_adaptive_config_recommendation_t *config) {
428
429 if (!config) {
430 return;
431 }
432
433 lle_terminal_detection_result_t *detection = NULL;
434 lle_result_t result =
435 lle_detect_terminal_capabilities_comprehensive(&detection);
436 if (result != LLE_SUCCESS) {
437 /* Conservative defaults for detection failure */
438 memset(config, 0, sizeof(lle_adaptive_config_recommendation_t));
439 config->enable_lle = false;
440 config->enable_tab_completion = true;
441 config->enable_history = true;
442 config->color_support_level = 0;
443 config->recommended_mode = LLE_ADAPTIVE_MODE_NONE;
444 return;
445 }
446
447 /* Configure based on detected mode and capabilities */
448 config->enable_lle =
449 (detection->recommended_mode != LLE_ADAPTIVE_MODE_NONE);
450 config->recommended_mode = detection->recommended_mode;
451
452 switch (detection->recommended_mode) {
453 case LLE_ADAPTIVE_MODE_NATIVE:
454 case LLE_ADAPTIVE_MODE_ENHANCED:
455 config->enable_syntax_highlighting = detection->supports_colors;
456 config->enable_autosuggestions = true;
457 config->enable_tab_completion = true;
458 config->enable_history = true;
459 config->enable_multiline_editing = true;
460 config->enable_undo_redo = true;
461 break;
462
463 case LLE_ADAPTIVE_MODE_MULTIPLEXED:
464 config->enable_syntax_highlighting = detection->supports_colors;
465 config->enable_autosuggestions = true;
466 config->enable_tab_completion = true;
467 config->enable_history = true;
468 config->enable_multiline_editing =
469 detection->supports_cursor_positioning;
470 config->enable_undo_redo = detection->supports_cursor_positioning;
471 break;
472
473 case LLE_ADAPTIVE_MODE_MINIMAL:
474 config->enable_syntax_highlighting = false;
475 config->enable_autosuggestions = false;
476 config->enable_tab_completion = true;
477 config->enable_history = true;
478 config->enable_multiline_editing = false;
479 config->enable_undo_redo = false;
480 break;
481
482 case LLE_ADAPTIVE_MODE_NONE:
483 default:
484 config->enable_lle = false;
485 config->enable_syntax_highlighting = false;
486 config->enable_autosuggestions = false;
487 config->enable_tab_completion = false;
488 config->enable_history = false;
489 config->enable_multiline_editing = false;
490 config->enable_undo_redo = false;
491 break;
492 }
493
494 /* Set color support level */
495 if (detection->supports_truecolor) {
496 config->color_support_level = 3;
497 } else if (detection->supports_256_colors) {
498 config->color_support_level = 2;
499 } else if (detection->supports_colors) {
500 config->color_support_level = 1;
501 } else {
502 config->color_support_level = 0;
503 }
504
505 lle_terminal_detection_result_destroy(detection);
506}

531/**
532 * Try fallback mode if current mode fails.
533 *
534 * Implements graceful degradation hierarchy:
535 * NATIVE ENHANCED MINIMAL
536 * ENHANCED MINIMAL
537 * MULTIPLEXED NATIVE ENHANCED MINIMAL
538 * MINIMAL (no fallback available)
539 */
540lle_result_t lle_adaptive_try_fallback_mode(lle_adaptive_context_t *context) {
541 if (!context) {
542 return LLE_ERROR_INVALID_PARAMETER;
543 }
544
545 /* Determine fallback mode based on current mode */
546 lle_adaptive_mode_t fallback_mode =
547 LLE_ADAPTIVE_MODE_MINIMAL; /* Default fallback */
548 lle_adaptive_mode_t original_mode = context->mode;
549
550 switch (context->mode) {
551 case LLE_ADAPTIVE_MODE_NATIVE:
552 fallback_mode = LLE_ADAPTIVE_MODE_ENHANCED;
553 break;
554
555 case LLE_ADAPTIVE_MODE_ENHANCED:
556 fallback_mode = LLE_ADAPTIVE_MODE_MINIMAL;
557 break;
558
559 case LLE_ADAPTIVE_MODE_MULTIPLEXED:
560 /* Multiplexer failure: try native first, then enhanced */
561 fallback_mode = LLE_ADAPTIVE_MODE_NATIVE;
562 break;
563
564 case LLE_ADAPTIVE_MODE_MINIMAL:
565 /* Already at minimal - no further fallback */
566 return LLE_ERROR_FEATURE_NOT_AVAILABLE;
567
568 case LLE_ADAPTIVE_MODE_NONE:
569 /* Non-interactive - cannot fallback */
570 return LLE_ERROR_FEATURE_NOT_AVAILABLE;
571 }
572
573 /* Cleanup current controller before switching */
574 switch (original_mode) {
575 case LLE_ADAPTIVE_MODE_NATIVE:
576 lle_cleanup_native_controller(context->controller.native);
577 context->controller.native = NULL;
578 break;
579
580 case LLE_ADAPTIVE_MODE_ENHANCED:
581 lle_cleanup_display_client_controller(
582 context->controller.display_client);
583 context->controller.display_client = NULL;
584 break;
585
586 case LLE_ADAPTIVE_MODE_MULTIPLEXED:
587 lle_cleanup_multiplexer_controller(context->controller.mux);
588 context->controller.mux = NULL;
589 break;
590
591 case LLE_ADAPTIVE_MODE_MINIMAL:
592 lle_cleanup_minimal_controller(context->controller.minimal);
593 context->controller.minimal = NULL;
594 break;
595
596 default:
597 break;
598 }
599
600 /* Update mode */
601 context->mode = fallback_mode;
602
603 /* Initialize new controller */
604 lle_result_t result;
605 switch (fallback_mode) {
606 case LLE_ADAPTIVE_MODE_NATIVE:
607 result =
608 lle_initialize_native_controller(context, context->memory_pool);
609 break;
610
611 case LLE_ADAPTIVE_MODE_ENHANCED:
612 result = lle_initialize_display_client_controller(context,
613 context->memory_pool);
614 break;
615
616 case LLE_ADAPTIVE_MODE_MINIMAL:
617 result =
618 lle_initialize_minimal_controller(context, context->memory_pool);
619 break;
620
621 default:
622 return LLE_ERROR_INVALID_STATE;
623 }
624
625 if (result != LLE_SUCCESS) {
626 /* Fallback initialization failed */
627 /* If we're not already at minimal, try that as last resort */
628 if (fallback_mode != LLE_ADAPTIVE_MODE_MINIMAL) {
629 context->mode = LLE_ADAPTIVE_MODE_MINIMAL;
630 result = lle_initialize_minimal_controller(context,
631 context->memory_pool);
632 if (result != LLE_SUCCESS) {
633 /* Even minimal failed - mark context as unhealthy */
634 context->healthy = false;
635 return result;
636 }
637 } else {
638 /* Minimal failed - no options left */
639 context->healthy = false;
640 return result;
641 }
642 }
643
644 /* Reset error count on successful fallback */
645 context->error_count = 0;
646 context->healthy = true;
647
648 return LLE_SUCCESS;
649}
