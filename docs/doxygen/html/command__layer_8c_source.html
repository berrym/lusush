152command_layer_t *command_layer_create(void) {
153 command_layer_t *layer = calloc(1, sizeof(command_layer_t));
154 if (!layer) {
155 return NULL;
156 }
157
158 // Initialize magic number for validation
159 layer->magic = COMMAND_LAYER_MAGIC;
160 layer->initialized = false;
161 layer->needs_redraw = false;
162
163 // Initialize command content
164 layer->command_text[0] = '\0';
165 layer->highlighted_text[0] = '\0';
166 layer->cursor_position = 0;
167
168 // Initialize syntax highlighting state
169 layer->region_count = 0;
170 layer->syntax_config.enabled = true;
171 layer->syntax_config.use_colors = true;
172 layer->syntax_config.highlight_errors = true;
173 layer->syntax_config.cache_enabled = true;
174 layer->syntax_config.cache_expiry_ms = COMMAND_LAYER_CACHE_EXPIRY_MS;
175 layer->syntax_config.max_update_time_ms =
176 COMMAND_LAYER_TARGET_UPDATE_TIME_MS;
177
178 // Initialize performance statistics
179 memset(&layer->performance, 0, sizeof(command_performance_t));
180 layer->performance.min_update_time_ns = UINT64_MAX;
181
182 // Initialize cache
183 layer->cache_size = 0;
184 layer->cache_access_count = 0;
185 for (size_t i = 0; i < COMMAND_LAYER_CACHE_SIZE; i++) {
186 layer->cache[i].is_valid = false;
187 }
188
189 // Initialize event system integration
190 layer->event_system = NULL;
191 layer->event_subscription_id = 0;
192
193 // Initialize prompt layer integration
194 layer->prompt_layer = NULL;
195 layer->prompt_integration_enabled = false;
196
197 // Initialize completion menu state
198 layer->completion_menu_visible = false;
199 layer->completion_menu_content = NULL;
200 layer->completion_menu_content_size = 0;
201 layer->completion_menu_lines = 0;
202 layer->completion_menu_selected_index = 0;
203
204 // Initialize performance monitoring
205 clock_gettime(CLOCK_MONOTONIC, &layer->last_update_time);
206 layer->update_sequence_number = 0;
207
208 // Initialize spec-compliant syntax highlighter (Spec 11)
209 layer->spec_highlighter = NULL;
210 if (lle_syntax_highlighter_create(&layer->spec_highlighter) != 0) {
211 // Highlighter creation failed - continue without it, fall back to
212 // inline
213 layer->spec_highlighter = NULL;
214 }
215
216 return layer;
217}

273command_layer_error_t command_layer_set_command(command_layer_t *layer,
274 const char *command_text,
275 size_t cursor_pos) {
276 if (!validate_layer_state(layer)) {
277 return COMMAND_LAYER_ERROR_INVALID_PARAM;
278 }
279
280 if (!command_text) {
281 return COMMAND_LAYER_ERROR_NULL_POINTER;
282 }
283
284 size_t command_length = strlen(command_text);
285 if (command_length >= COMMAND_LAYER_MAX_COMMAND_SIZE) {
286 return COMMAND_LAYER_ERROR_COMMAND_TOO_LARGE;
287 }
288
289 if (cursor_pos > command_length) {
290 return COMMAND_LAYER_ERROR_INVALID_CURSOR_POS;
291 }
292
293 uint64_t start_time = get_current_time_ns();
294
295 // Check if command has changed
296 bool command_changed = (strcmp(layer->command_text, command_text) != 0);
297 bool cursor_changed = (layer->cursor_position != cursor_pos);
298
299 /* Always need initial render even if buffer is empty */
300 bool is_first_render = (layer->update_sequence_number == 0);
301
302 /* Check if completion menu state changed (even if command/cursor didn't)
303 * When menu is shown/hidden, we need redraw even if command text unchanged
304 */
305 bool menu_changed = false;
306 display_controller_t *dc = display_integration_get_controller();
307 if (dc) {
308 menu_changed = display_controller_check_and_clear_menu_changed(dc);
309 }
310
311 if (!command_changed && !cursor_changed && !is_first_render &&
312 !menu_changed) {
313 // No change, just update performance stats with minimal time
314 update_performance_stats(layer, get_current_time_ns() - start_time);
315 return COMMAND_LAYER_SUCCESS;
316 }
317
318 // Update command text and cursor position
319 safe_string_copy(layer->command_text, command_text,
320 sizeof(layer->command_text));
321 layer->cursor_position = cursor_pos;
322 layer->needs_redraw = true;
323 layer->update_sequence_number++;
324
325 // Check cache first for performance
326 command_cache_entry_t *cached = NULL;
327 if (layer->syntax_config.cache_enabled && command_changed) {
328 cached = find_cache_entry(layer, command_text);
329 layer->performance.cache_hits += (cached != NULL) ? 1 : 0;
330 layer->performance.cache_misses += (cached == NULL) ? 1 : 0;
331
332 // Record layer-specific cache operation
333 display_integration_record_layer_cache_operation("command_layer",
334 (cached != NULL));
335 }
336
337 if (cached && cached->is_valid) {
338 // Use cached result
339 safe_string_copy(layer->highlighted_text, cached->highlighted_text,
340 sizeof(layer->highlighted_text));
341 layer->metrics = cached->metrics;
342 layer->region_count =
343 0; // Reset regions as they're embedded in highlighted text
344 } else {
345 // Perform syntax highlighting
346 command_layer_error_t result = perform_syntax_highlighting(layer);
347 if (result != COMMAND_LAYER_SUCCESS) {
348 update_performance_stats(layer, get_current_time_ns() - start_time);
349 return result;
350 }
351
352 // Update metrics
353 result = update_command_metrics(layer);
354 if (result != COMMAND_LAYER_SUCCESS) {
355 update_performance_stats(layer, get_current_time_ns() - start_time);
356 return result;
357 }
358
359 // Add to cache if enabled
360 if (layer->syntax_config.cache_enabled && command_changed) {
361 add_to_cache(layer, command_text, layer->highlighted_text,
362 &layer->metrics);
363 }
364 }
365
366 // Update cursor position in metrics
367 layer->metrics.cursor_position = cursor_pos;
368
369 uint64_t operation_time = get_current_time_ns() - start_time;
370 update_performance_stats(layer, operation_time);
371
372 // Publish content changed event if command changed
373 if (command_changed) {
374 publish_command_event(layer, LAYER_EVENT_CONTENT_CHANGED);
375 }
376
377 // Publish cursor moved event if cursor changed
378 if (cursor_changed) {
379 publish_command_event(layer, LAYER_EVENT_CURSOR_MOVED);
380 }
381
382 // Publish redraw needed event (triggers display_controller to render)
383 // This was previously done by command_layer_update(), but that caused
384 // redundant syntax highlighting. Now we do it here after highlighting.
385 publish_command_event(layer, LAYER_EVENT_REDRAW_NEEDED);
386
387 return COMMAND_LAYER_SUCCESS;
388}

624/**
625 * Primary syntax highlighting using spec-compliant system (Spec 11)
626 * Falls back to inline implementation if spec highlighter unavailable
627 */
628static command_layer_error_t
629perform_syntax_highlighting(command_layer_t *layer) {
630 if (!layer->syntax_config.enabled) {
631 // If highlighting is disabled, just copy the command text
632 safe_string_copy(layer->highlighted_text, layer->command_text,
633 sizeof(layer->highlighted_text));
634 layer->region_count = 0;
635 return COMMAND_LAYER_SUCCESS;
636 }
637
638 uint64_t start_time = get_current_time_ns();
639
640 // Initialize highlighting output
641 layer->highlighted_text[0] = '\0';
642 layer->region_count = 0;
643
644 if (!layer->command_text[0]) {
645 // Empty command, nothing to highlight
646 return COMMAND_LAYER_SUCCESS;
647 }
648
649 // Use spec-compliant highlighter if available
650 if (layer->spec_highlighter) {
651 size_t command_len = strlen(layer->command_text);
652
653 // Tokenize using spec highlighter
654 int token_count = lle_syntax_highlight(
655 layer->spec_highlighter, layer->command_text, command_len);
656
657 if (token_count >= 0) {
658 // Get tokens from spec highlighter
659 size_t spec_token_count;
660 const lle_syntax_token_t *spec_tokens = lle_syntax_get_tokens(
661 layer->spec_highlighter, &spec_token_count);
662
663 // Build highlighted output using command layer's color scheme
664 // This preserves the command layer's theme integration
665 size_t output_pos = 0;
666 size_t last_end = 0;
667
668 for (size_t i = 0;
669 i < spec_token_count &&
670 layer->region_count < COMMAND_LAYER_MAX_HIGHLIGHT_REGIONS;
671 i++) {
672 const lle_syntax_token_t *tok = &spec_tokens[i];
673
674 // Add any whitespace/text between tokens
675 if (last_end < tok->start) {
676 size_t gap_len = tok->start - last_end;
677 if (output_pos + gap_len <
678 sizeof(layer->highlighted_text) - 1) {
679 memcpy(layer->highlighted_text + output_pos,
680 layer->command_text + last_end, gap_len);
681 output_pos += gap_len;
682 }
683 }
684
685 // Map spec token type to command layer type
686 command_token_type_t cmd_type = map_spec_token_type(tok->type);
687
688 // Skip whitespace tokens for coloring
689 if (tok->type == LLE_TOKEN_WHITESPACE) {
690 size_t tok_len = tok->end - tok->start;
691 if (output_pos + tok_len <
692 sizeof(layer->highlighted_text) - 1) {
693 memcpy(layer->highlighted_text + output_pos,
694 layer->command_text + tok->start, tok_len);
695 output_pos += tok_len;
696 }
697 last_end = tok->end;
698 continue;
699 }
700
701 // Get color from command layer's color scheme
702 const char *color = get_token_color(layer, cmd_type);
703 bool has_color =
704 layer->syntax_config.use_colors && color && color[0];
705
706 // Add color code
707 if (has_color) {
708 size_t color_len = strlen(color);
709 if (output_pos + color_len <
710 sizeof(layer->highlighted_text) - 1) {
711 memcpy(layer->highlighted_text + output_pos, color,
712 color_len);
713 output_pos += color_len;
714 }
715 }
716
717 // Add token text, re-applying color after newlines for
718 // multiline tokens
719 size_t tok_len = tok->end - tok->start;
720 const char *tok_text = layer->command_text + tok->start;
721 for (size_t j = 0;
722 j < tok_len &&
723 output_pos < sizeof(layer->highlighted_text) - 1;
724 j++) {
725 layer->highlighted_text[output_pos++] = tok_text[j];
726
727 // Re-apply color after newlines within tokens
728 if (tok_text[j] == '\n' && has_color && j + 1 < tok_len) {
729 size_t color_len = strlen(color);
730 if (output_pos + color_len <
731 sizeof(layer->highlighted_text) - 1) {
732 memcpy(layer->highlighted_text + output_pos, color,
733 color_len);
734 output_pos += color_len;
735 }
736 }
737 }
738
739 // Add reset code
740 if (has_color) {
741 const char *reset =
742 layer->syntax_config.color_scheme.reset_color;
743 size_t reset_len = strlen(reset);
744 if (output_pos + reset_len <
745 sizeof(layer->highlighted_text) - 1) {
746 memcpy(layer->highlighted_text + output_pos, reset,
747 reset_len);
748 output_pos += reset_len;
749 }
750 }
751
752 // Store region info
753 command_highlight_region_t *region =
754 &layer->highlight_regions[layer->region_count];
755 region->start = tok->start;
756 region->length = tok->end - tok->start;
757 region->token_type = cmd_type;
758 if (color) {
759 safe_string_copy(region->color_code, color,
760 sizeof(region->color_code));
761 } else {
762 region->color_code[0] = '\0';
763 }
764 layer->region_count++;
765
766 last_end = tok->end;
767 g_highlighting_stats.tokens_parsed++;
768 }
769
770 // Add any remaining text
771 if (last_end < command_len) {
772 size_t remaining = command_len - last_end;
773 if (output_pos + remaining <
774 sizeof(layer->highlighted_text) - 1) {
775 memcpy(layer->highlighted_text + output_pos,
776 layer->command_text + last_end, remaining);
777 output_pos += remaining;
778 }
779 }
780
781 layer->highlighted_text[output_pos] = '\0';
782
783 uint64_t highlighting_time = get_current_time_ns() - start_time;
784 g_highlighting_stats.highlighting_time_ns += highlighting_time;
785 g_highlighting_stats.regions_created += layer->region_count;
786
787 return COMMAND_LAYER_SUCCESS;
788 }
789 // Spec highlighting returned error - fall through to plain text
790 }
791
792 // No spec highlighter available or it failed - just copy plain text
793 // This is a graceful degradation: syntax highlighting is cosmetic,
794 // so we show unhighlighted text rather than failing
795 safe_string_copy(layer->highlighted_text, layer->command_text,
796 sizeof(layer->highlighted_text));
797 return COMMAND_LAYER_SUCCESS;
798}

850static command_cache_entry_t *find_cache_entry(command_layer_t *layer,
851 const char *command) {
852 if (!layer->syntax_config.cache_enabled || !command) {
853 return NULL;
854 }
855
856 uint32_t hash = calculate_command_hash(command);
857 uint64_t current_time = get_current_time_ns();
858
859 for (size_t i = 0; i < layer->cache_size; i++) {
860 command_cache_entry_t *entry = &layer->cache[i];
861
862 if (!entry->is_valid) {
863 continue;
864 }
865
866 // Check if cache entry has expired
867 uint64_t age_ns = current_time - entry->timestamp_ns;
868 if (age_ns > (layer->syntax_config.cache_expiry_ms *
869 NANOSECONDS_PER_MILLISECOND)) {
870 entry->is_valid = false;
871 continue;
872 }
873
874 // Check if hash and command match
875 if (entry->hash == hash && strcmp(entry->command_text, command) == 0) {
876 return entry;
877 }
878 }
879
880 return NULL;
881}

883static command_layer_error_t add_to_cache(command_layer_t *layer,
884 const char *command,
885 const char *highlighted,
886 const command_metrics_t *metrics) {
887 if (!layer->syntax_config.cache_enabled || !command || !highlighted ||
888 !metrics) {
889 return COMMAND_LAYER_SUCCESS; // Not an error, just disabled
890 }
891
892 // Find empty slot or oldest entry
893 command_cache_entry_t *entry = NULL;
894 uint64_t oldest_time = UINT64_MAX;
895 size_t oldest_index = 0;
896
897 for (size_t i = 0; i < COMMAND_LAYER_CACHE_SIZE; i++) {
898 if (!layer->cache[i].is_valid) {
899 entry = &layer->cache[i];
900 break;
901 }
902
903 if (layer->cache[i].timestamp_ns < oldest_time) {
904 oldest_time = layer->cache[i].timestamp_ns;
905 oldest_index = i;
906 }
907 }
908
909 // If no empty slot found, use oldest entry
910 if (!entry) {
911 entry = &layer->cache[oldest_index];
912 } else if (layer->cache_size < COMMAND_LAYER_CACHE_SIZE) {
913 layer->cache_size++;
914 }
915
916 // Store cache entry
917 safe_string_copy(entry->command_text, command, sizeof(entry->command_text));
918 safe_string_copy(entry->highlighted_text, highlighted,
919 sizeof(entry->highlighted_text));
920 entry->metrics = *metrics;
921 entry->timestamp_ns = get_current_time_ns();
922 entry->hash = calculate_command_hash(command);
923 entry->is_valid = true;
924
925 return COMMAND_LAYER_SUCCESS;
926}

1026static layer_events_error_t handle_layer_event(const layer_event_t *event,
1027 void *user_data) {
1028 command_layer_t *layer = (command_layer_t *)user_data;
1029
1030 if (!validate_layer_state(layer) || !event) {
1031 return LAYER_EVENTS_ERROR_INVALID_PARAM;
1032 }
1033
1034 switch (event->type) {
1035 case LAYER_EVENT_THEME_CHANGED:
1036 // Update color scheme when theme changes
1037 command_layer_create_default_colors(&layer->syntax_config.color_scheme);
1038 layer->needs_redraw = true;
1039 publish_command_event(layer, LAYER_EVENT_REDRAW_NEEDED);
1040 break;
1041
1042 case LAYER_EVENT_CONTENT_CHANGED:
1043 // Prompt layer content changed, update positioning
1044 if (layer->prompt_integration_enabled) {
1045 update_command_metrics(layer);
1046 layer->needs_redraw = true;
1047 }
1048 break;
1049
1050 case LAYER_EVENT_SIZE_CHANGED:
1051 // Terminal size changed, may affect display
1052 layer->needs_redraw = true;
1053 publish_command_event(layer, LAYER_EVENT_REDRAW_NEEDED);
1054 break;
1055
1056 default:
1057 // Ignore other events
1058 break;
1059 }
1060
1061 return LAYER_EVENTS_SUCCESS;
1062}

1065publish_command_event(command_layer_t *layer, layer_event_type_t event_type) {
1066 if (!layer->event_system) {
1067 return COMMAND_LAYER_SUCCESS; // Not an error if no event system
1068 }
1069
1070 /* Use HIGH priority for REDRAW_NEEDED events to match display_controller
1071 * subscription */
1072 layer_event_priority_t priority = (event_type == LAYER_EVENT_REDRAW_NEEDED)
1073 ? LAYER_EVENT_PRIORITY_HIGH
1074 : LAYER_EVENT_PRIORITY_NORMAL;
1075
1076 layer_events_error_t result = layer_events_publish_simple(
1077 layer->event_system, event_type, LAYER_ID_COMMAND_LAYER, 0, priority);
1078
1079 return (result == LAYER_EVENTS_SUCCESS) ? COMMAND_LAYER_SUCCESS
1080 : COMMAND_LAYER_ERROR_EVENT_SYSTEM;
1081}
