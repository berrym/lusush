2 * @file syntax_highlighting.h
3 * @brief LLE Syntax Highlighting System
4 *
5 * Provides real-time syntax highlighting for shell command input with
6 * theme integration. Core features:
7 * - Shell-specific token classification (commands, builtins, keywords, etc.)
8 * - Themeable color scheme
9 * - Command/path existence validation for visual feedback
10 * - Integration with LLE display system
11 *
12 * SPECIFICATION: docs/lle_specification/11_syntax_highlighting_complete.md
13 */

141 uint32_t path_invalid; /**< Invalid paths (typically underlined red) */
142
143 /* Operators */

145 uint32_t redirect; /**< Redirection operators */

147
148 /* Assignment */
149 uint32_t assignment; /**< Variable assignment (VAR=value) */

151 /* Other */

153 uint32_t number; /**< Numbers (typically cyan) */
154 uint32_t option; /**< Options (typically white/default) */

156 uint32_t argument; /**< Generic arguments (default) */

177 /* Text attributes (bitmask) */

180 uint8_t error_underline; /**< Underline errors */
181 uint8_t path_underline; /**< Underline paths */
182 uint8_t comment_dim; /**< Dim comments */
