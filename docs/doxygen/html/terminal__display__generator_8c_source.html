28/*
29 * Create display content structure
30 */
31lle_result_t lle_display_content_create(lle_display_content_t **content,
32 size_t line_capacity) {
33 if (!content) {
34 return LLE_ERROR_INVALID_PARAMETER;
35 }
36
37 if (line_capacity < 1) {
38 line_capacity = 10; /* Minimum capacity */
39 }
40
41 lle_display_content_t *dc = calloc(1, sizeof(lle_display_content_t));
42 if (!dc) {
43 return LLE_ERROR_OUT_OF_MEMORY;
44 }
45
46 /* Allocate lines array */
47 dc->lines = calloc(line_capacity, sizeof(lle_display_line_t));
48 if (!dc->lines) {
49 free(dc);
50 return LLE_ERROR_OUT_OF_MEMORY;
51 }
52
53 /* Initialize each line */
54 for (size_t i = 0; i < line_capacity; i++) {
55 dc->lines[i].capacity = 256;
56 dc->lines[i].content = calloc(256, 1);
57 if (!dc->lines[i].content) {
58 /* Clean up previously allocated lines */
59 for (size_t j = 0; j < i; j++) {
60 free(dc->lines[j].content);
61 }
62 free(dc->lines);
63 free(dc);
64 return LLE_ERROR_OUT_OF_MEMORY;
65 }
66 dc->lines[i].length = 0;
67 dc->lines[i].contains_cursor = false;
68 }
69
70 dc->line_count = 0;
71 dc->cursor_line = 0;
72 dc->cursor_column = 0;
73 dc->cursor_visible = true;
74 dc->is_complete_refresh = true;
75 dc->content_version = 0;
76 dc->generation_time = lle_get_current_time_microseconds();
77
78 *content = dc;
79 return LLE_SUCCESS;
80}

164/*
165 * Ensure line has sufficient capacity
166 */
167static lle_result_t ensure_line_capacity(lle_display_line_t *line,
168 size_t required) {
169 if (line->capacity >= required) {
170 return LLE_SUCCESS;
171 }
172
173 /* Calculate new capacity (grow by 1.5x) */
174 size_t new_capacity = line->capacity;
175 while (new_capacity < required) {
176 new_capacity = new_capacity + (new_capacity / 2);
177 }
178
179 char *new_content = realloc(line->content, new_capacity);
180 if (!new_content) {
181 return LLE_ERROR_OUT_OF_MEMORY;
182 }
183
184 line->content = new_content;
185 line->capacity = new_capacity;
186
187 return LLE_SUCCESS;
188}

190/*
191 * Append text to display line
192 */
193static lle_result_t append_to_line(lle_display_line_t *line, const char *text,
194 size_t length) {
195 if (!line || !text || length == 0) {
196 return LLE_ERROR_INVALID_PARAMETER;
197 }
198
199 /* Ensure capacity */
200 lle_result_t result = ensure_line_capacity(line, line->length + length + 1);
201 if (result != LLE_SUCCESS) {
202 return result;
203 }
204
205 /* Append text */
206 memcpy(line->content + line->length, text, length);
207 line->length += length;
208 line->content[line->length] = '\0';
209
210 return LLE_SUCCESS;
211}

213/*
214 * Calculate display line count for buffer with line wrapping
215 */
216static size_t calculate_display_lines(const char *buffer, size_t buffer_length,
217 size_t terminal_width,
218 size_t prompt_width) {
219 if (!buffer || terminal_width == 0) {
220 return 1;
221 }
222
223 size_t line_count = 1;
224 size_t current_column = prompt_width;
225
226 for (size_t i = 0; i < buffer_length;) {
227 char c = buffer[i];
228
229 if (c == '\n') {
230 line_count++;
231 current_column = 0;
232 i++;
233 } else if (c == '\t') {
234 current_column += 8 - (current_column % 8);
235 i++;
236 } else {
237 /* Decode UTF-8 codepoint and get its display width */
238 uint32_t codepoint = 0;
239 int bytes_in_char = lle_utf8_decode_codepoint(
240 &buffer[i], buffer_length - i, &codepoint);
241
242 if (bytes_in_char > 0) {
243 int char_width = lle_utf8_codepoint_width(codepoint);
244 current_column += char_width;
245 i += bytes_in_char;
246 } else {
247 /* Invalid UTF-8 - treat as single column */
248 current_column++;
249 i++;
250 }
251 }
252
253 /* Line wrap */
254 if (current_column >= terminal_width) {
255 line_count++;
256 current_column = 0;
257 }
258 }
259
260 return line_count;
261}

263/*
264 * Generate display content from internal state
265 *
266 * CRITICAL: This function converts authoritative internal state to
267 * display content. It NEVER queries terminal state.
268 */
269lle_result_t
270lle_display_generator_generate_content(lle_display_generator_t *generator,
271 lle_display_content_t **content) {
272 if (!generator || !content) {
273 return LLE_ERROR_INVALID_PARAMETER;
274 }
275
276 lle_internal_state_t *state = generator->internal_state;
277 lle_terminal_capabilities_t *caps = generator->capabilities;
278 (void)caps; /* Reserved for capability-aware rendering */
279
280 /* Calculate required display lines */
281 size_t required_lines = calculate_display_lines(
282 state->command_buffer->data, state->command_buffer->length,
283 state->terminal_width, state->prompt_width);
284
285 /* Create new display content */
286 lle_display_content_t *new_content = NULL;
287 lle_result_t result =
288 lle_display_content_create(&new_content, required_lines + 5);
289 if (result != LLE_SUCCESS) {
290 return result;
291 }
292
293 /* Generate display lines from buffer */
294 size_t current_line = 0;
295 size_t current_column = 0;
296 size_t buffer_pos = 0;
297
298 /* Initialize first line */
299 new_content->lines[0].length = 0;
300
301 /* Process each character in buffer */
302 while (buffer_pos < state->command_buffer->length) {
303 char c = state->command_buffer->data[buffer_pos];
304
305 /* Check if cursor is at this position */
306 if (buffer_pos == state->cursor_position) {
307 new_content->cursor_line = current_line;
308 new_content->cursor_column = current_column;
309 new_content->lines[current_line].contains_cursor = true;
310 new_content->lines[current_line].cursor_column = current_column;
311 }
312
313 /* Handle special characters */
314 if (c == '\n') {
315 /* Newline - advance to next line */
316 current_line++;
317 current_column = 0;
318 buffer_pos++;
319
320 if (current_line < required_lines + 5) {
321 new_content->lines[current_line].length = 0;
322 }
323 } else if (c == '\t') {
324 /* Tab - expand to spaces */
325 size_t spaces = 8 - (current_column % 8);
326 for (size_t i = 0;
327 i < spaces && current_column < state->terminal_width; i++) {
328 result =
329 append_to_line(&new_content->lines[current_line], " ", 1);
330 if (result != LLE_SUCCESS) {
331 lle_display_content_destroy(new_content);
332 return result;
333 }
334 current_column++;
335 }
336 buffer_pos++;
337 } else {
338 /* Regular character - decode UTF-8 and get display width */
339 uint32_t codepoint = 0;
340 int bytes_in_char = lle_utf8_decode_codepoint(
341 &state->command_buffer->data[buffer_pos],
342 state->command_buffer->length - buffer_pos, &codepoint);
343
344 if (bytes_in_char > 0) {
345 /* Append complete UTF-8 sequence to line */
346 result = append_to_line(
347 &new_content->lines[current_line],
348 &state->command_buffer->data[buffer_pos], bytes_in_char);
349 if (result != LLE_SUCCESS) {
350 lle_display_content_destroy(new_content);
351 return result;
352 }
353
354 /* Update column with actual display width */
355 int char_width = lle_utf8_codepoint_width(codepoint);
356 current_column += char_width;
357 buffer_pos += bytes_in_char;
358 } else {
359 /* Invalid UTF-8 - treat as single byte */
360 result =
361 append_to_line(&new_content->lines[current_line], &c, 1);
362 if (result != LLE_SUCCESS) {
363 lle_display_content_destroy(new_content);
364 return result;
365 }
366 current_column++;
367 buffer_pos++;
368 }
369 }
370
371 /* Handle line wrapping */
372 if (current_column >= state->terminal_width) {
373 current_line++;
374 current_column = 0;
375
376 if (current_line < required_lines + 5) {
377 new_content->lines[current_line].length = 0;
378 }
379 }
380 }
381
382 /* Handle cursor at end of buffer */
383 if (buffer_pos == state->cursor_position) {
384 new_content->cursor_line = current_line;
385 new_content->cursor_column = current_column;
386 new_content->lines[current_line].contains_cursor = true;
387 new_content->lines[current_line].cursor_column = current_column;
388 }
389
390 /* Set display content metadata */
391 new_content->line_count = current_line + 1;
392 new_content->cursor_visible = true;
393 new_content->is_complete_refresh =
394 state->command_buffer->needs_full_refresh ||
395 generator->params.force_full_refresh;
396 new_content->generation_time = lle_get_current_time_microseconds();
397 new_content->content_version = (uint32_t)state->modification_count;
398
399 /* Store previous content for delta calculation (future optimization) */
400 if (generator->previous_content) {
401 lle_display_content_destroy(generator->previous_content);
402 }
403 generator->previous_content = generator->current_content;
404 generator->current_content = new_content;
405
406 *content = new_content;
407 return LLE_SUCCESS;
408}
