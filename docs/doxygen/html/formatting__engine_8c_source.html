92 /* Keep current options */

99lle_result_t
100lle_formatting_engine_create(lle_formatting_engine_t **engine,
101 lle_memory_pool_t *memory_pool,
102 lle_structure_analyzer_t *analyzer,
103 const lle_formatting_options_t *options) {
104 if (!engine || !analyzer) /* memory_pool can be NULL */ {
105 return LLE_ERROR_INVALID_PARAMETER;
106 }
107
108 lle_formatting_engine_t *new_engine =
109 lle_pool_alloc(sizeof(lle_formatting_engine_t));
110 if (!new_engine) {
111 return LLE_ERROR_OUT_OF_MEMORY;
112 }
113
114 new_engine->memory_pool = memory_pool;
115 new_engine->analyzer = analyzer;
116
117 if (options) {
118 memcpy(&new_engine->options, options, sizeof(lle_formatting_options_t));
119 } else {
120 lle_formatting_engine_get_default_options(&new_engine->options);
121 }
122
123 new_engine->active = true;
124
125 *engine = new_engine;
126 return LLE_SUCCESS;

152lle_result_t lle_formatting_engine_normalize(lle_formatting_engine_t *engine,
153 const char *command_text,
154 size_t command_length,
155 char **normalized_text,
156 size_t *normalized_length) {
157 if (!engine || !engine->active || !command_text || !normalized_text ||
158 !normalized_length) {
159 return LLE_ERROR_INVALID_PARAMETER;
160 }
161
162 /* Allocate output buffer */
163 char *output = lle_pool_alloc(command_length + 1);
164 if (!output) {
165 return LLE_ERROR_OUT_OF_MEMORY;
166 }
167
168 size_t out_pos = 0;
169 bool last_was_space = false;
170 bool in_quote = false;
171 char quote_char = '\0';
172
173 for (size_t i = 0; i < command_length; i++) {
174 char c = command_text[i];
175
176 /* Handle quotes */
177 if (!in_quote && (c == '"' || c == '\'')) {
178 in_quote = true;
179 quote_char = c;
180 output[out_pos++] = c;
181 last_was_space = false;
182 continue;
183 } else if (in_quote && c == quote_char) {
184 in_quote = false;
185 quote_char = '\0';
186 output[out_pos++] = c;
187 last_was_space = false;
188 continue;
189 }
190
191 /* Don't normalize inside quotes */
192 if (in_quote) {
193 output[out_pos++] = c;
194 last_was_space = false;
195 continue;
196 }
197
198 /* Normalize whitespace */
199 if (engine->options.normalize_spaces && isspace(c)) {
200 if (!last_was_space && c != '\n') {
201 output[out_pos++] = ' ';
202 last_was_space = true;
203 } else if (c == '\n') {
204 /* Handle trailing whitespace */
205 if (engine->options.trim_trailing && out_pos > 0 &&
206 output[out_pos - 1] == ' ') {
207 out_pos--; /* Remove trailing space */
208 }
209 output[out_pos++] = '\n';
210 last_was_space = false;
211 }
212 } else {
213 output[out_pos++] = c;
214 last_was_space = false;
215 }
216 }
217
218 /* Trim final trailing whitespace */
219 if (engine->options.trim_trailing && out_pos > 0) {
220 while (out_pos > 0 && isspace(output[out_pos - 1])) {
221 out_pos--;
222 }
223 }
224
225 output[out_pos] = '\0';
226
227 *normalized_text = output;
228 *normalized_length = out_pos;
229
230 return LLE_SUCCESS;
231}

233lle_result_t lle_formatting_engine_apply_style(lle_formatting_engine_t *engine,
234 const char *command_text,
235 size_t command_length,
236 lle_formatting_style_t style,
237 char **formatted_text,
238 size_t *formatted_length) {
239 if (!engine || !engine->active || !command_text || !formatted_text ||
240 !formatted_length) {
241 return LLE_ERROR_INVALID_PARAMETER;
242 }
243
244 lle_result_t result;
245
246 switch (style) {
247 case LLE_FORMAT_COMPACT:
248 result = format_compact(engine, command_text, command_length,
249 formatted_text, formatted_length);
250 break;
251
252 case LLE_FORMAT_READABLE:
253 result = format_readable(engine, command_text, command_length,
254 formatted_text, formatted_length);
255 break;
256
257 case LLE_FORMAT_EXPANDED:
258 result = format_expanded(engine, command_text, command_length,
259 formatted_text, formatted_length);
260 break;
261
262 case LLE_FORMAT_CUSTOM:
263 /* Use current options for formatting */
264 result = format_readable(engine, command_text, command_length,
265 formatted_text, formatted_length);
266 break;
267
268 default:
269 return LLE_ERROR_INVALID_PARAMETER;
270 }
271
272 return result;
273}

275lle_result_t lle_formatting_engine_format(lle_formatting_engine_t *engine,
276 const char *command_text,
277 size_t command_length,
278 lle_formatted_command_t **result) {
279 if (!engine || !engine->active || !command_text || !result) {
280 return LLE_ERROR_INVALID_PARAMETER;
281 }
282
283 /* Allocate result structure */
284 lle_formatted_command_t *formatted =
285 lle_pool_alloc(sizeof(lle_formatted_command_t));
286 if (!formatted) {
287 return LLE_ERROR_OUT_OF_MEMORY;
288 }
289
290 memset(formatted, 0, sizeof(lle_formatted_command_t));
291
292 /* Apply formatting according to configured style */
293 char *formatted_text = NULL;
294 size_t formatted_length = 0;
295
296 lle_result_t res = lle_formatting_engine_apply_style(
297 engine, command_text, command_length, engine->options.style,
298 &formatted_text, &formatted_length);
299
300 if (res != LLE_SUCCESS) {
301 return res;
302 }
303
304 formatted->text = formatted_text;
305 formatted->length = formatted_length;
306 formatted->style_applied = engine->options.style;
307 formatted->was_reformatted =
308 (formatted_length != command_length ||
309 memcmp(formatted_text, command_text, command_length) != 0);
310
311 *result = formatted;
312 return LLE_SUCCESS;
313}

329static lle_result_t format_compact(lle_formatting_engine_t *engine,
330 const char *text, size_t length,
331 char **output, size_t *output_len) {
332 (void)engine; /* Reserved for engine-specific formatting options */
333 /* Compact format: minimize whitespace */
334 char *result = lle_pool_alloc(length + 1);
335 if (!result) {
336 return LLE_ERROR_OUT_OF_MEMORY;
337 }
338
339 size_t out_pos = 0;
340 bool in_quote = false;
341 char quote_char = '\0';
342 bool last_was_space = false;
343
344 for (size_t i = 0; i < length; i++) {
345 char c = text[i];
346
347 /* Handle quotes */
348 if (!in_quote && (c == '"' || c == '\'')) {
349 in_quote = true;
350 quote_char = c;
351 } else if (in_quote && c == quote_char) {
352 in_quote = false;
353 quote_char = '\0';
354 }
355
356 if (in_quote) {
357 result[out_pos++] = c;
358 last_was_space = false;
359 } else if (c == '\n') {
360 /* Keep newlines but remove preceding spaces */
361 if (out_pos > 0 && result[out_pos - 1] == ' ') {
362 out_pos--;
363 }
364 result[out_pos++] = '\n';
365 last_was_space = false;
366 } else if (isspace(c)) {
367 /* Replace multiple spaces with single space */
368 if (!last_was_space && out_pos > 0) {
369 result[out_pos++] = ' ';
370 last_was_space = true;
371 }
372 } else {
373 result[out_pos++] = c;
374 last_was_space = false;
375 }
376 }
377
378 result[out_pos] = '\0';
379 *output = result;
380 *output_len = out_pos;
381
382 return LLE_SUCCESS;
383}

385static lle_result_t format_readable(lle_formatting_engine_t *engine,
386 const char *text, size_t length,
387 char **output, size_t *output_len) {
388 /* Readable format: balanced whitespace and indentation */
389 size_t estimated_size = length + (length / 10); /* 10% overhead estimate */
390 char *result = lle_pool_alloc(estimated_size);
391 if (!result) {
392 return LLE_ERROR_OUT_OF_MEMORY;
393 }
394
395 size_t out_pos = 0;
396 bool in_quote = false;
397 char quote_char = '\0';
398 bool last_was_space = false;
399
400 for (size_t i = 0; i < length; i++) {
401 char c = text[i];
402
403 /* Handle quotes */
404 if (!in_quote && (c == '"' || c == '\'')) {
405 in_quote = true;
406 quote_char = c;
407 } else if (in_quote && c == quote_char) {
408 in_quote = false;
409 quote_char = '\0';
410 }
411
412 if (in_quote) {
413 result[out_pos++] = c;
414 last_was_space = false;
415 } else if (isspace(c)) {
416 if (c == '\n' || !last_was_space) {
417 result[out_pos++] = c;
418 last_was_space = (c != '\n');
419 }
420 } else {
421 /* Add space around operators if configured */
422 if (engine->options.space_around_operators) {
423 if (should_add_space_before(c) && out_pos > 0 &&
424 !isspace(result[out_pos - 1])) {
425 result[out_pos++] = ' ';
426 }
427 }
428
429 result[out_pos++] = c;
430 last_was_space = false;
431
432 if (engine->options.space_around_operators &&
433 should_add_space_after(c)) {
434 result[out_pos++] = ' ';
435 last_was_space = true;
436 }
437 }
438 }
439
440 result[out_pos] = '\0';
441 *output = result;
442 *output_len = out_pos;
443
444 return LLE_SUCCESS;
445}

447static lle_result_t format_expanded(lle_formatting_engine_t *engine,
448 const char *text, size_t length,
449 char **output, size_t *output_len) {
450 /* Expanded format: maximum readability with line breaks */
451 size_t estimated_size =
452 length * 2; /* 100% overhead estimate for expansion */
453 char *result = lle_pool_alloc(estimated_size);
454 if (!result) {
455 return LLE_ERROR_OUT_OF_MEMORY;
456 }
457
458 size_t out_pos = 0;
459 bool in_quote = false;
460 char quote_char = '\0';
461
462 for (size_t i = 0; i < length; i++) {
463 char c = text[i];
464
465 /* Handle quotes */
466 if (!in_quote && (c == '"' || c == '\'')) {
467 in_quote = true;
468 quote_char = c;
469 } else if (in_quote && c == quote_char) {
470 in_quote = false;
471 quote_char = '\0';
472 }
473
474 if (in_quote) {
475 result[out_pos++] = c;
476 } else if (c == '|' && engine->options.break_pipes) {
477 /* Break pipelines into separate lines */
478 result[out_pos++] = ' ';
479 result[out_pos++] = '|';
480 result[out_pos++] = '\n';
481
482 /* Add indentation for next line */
483 for (uint8_t j = 0; j < engine->options.spaces_per_level; j++) {
484 result[out_pos++] = engine->options.indent_char;
485 }
486 } else {
487 result[out_pos++] = c;
488 }
489 }
490
491 result[out_pos] = '\0';
492 *output = result;
493 *output_len = out_pos;
494
495 return LLE_SUCCESS;
496}
