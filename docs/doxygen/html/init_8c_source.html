50/**
51 * Ensure safe bottom margin for interactive shell display.
52 * Creates scroll space at the bottom of the terminal.
53 */
54static void ensure_bottom_margin(void) {
55 static bool margin_created = false;
56 if (margin_created) {
57 return;
58 }
59
60 int rows = 0;
61 lle_get_terminal_size(NULL, &rows);
62 if (rows <= 0) {
63 rows = 24;
64 }
65
66 /* Save cursor, move to bottom, scroll, restore */
67 char cmd[64];
68 int len = snprintf(cmd, sizeof(cmd), "\x1b[s\x1b[%d;1H\n\x1b[u", rows);
69 if (len > 0 && (size_t)len < sizeof(cmd)) {
70 (void)write(STDOUT_FILENO, cmd, (size_t)len);
71 }
72
73 margin_created = true;
74}

76// Function to handle shebang processing
77static void process_shebang(FILE *file) {
78 if (!file) {
79 return;
80 }
81
82 // Check if the first line starts with #!
83 long pos = ftell(file);
84 char line[256];
85
86 if (fgets(line, sizeof(line), file) != NULL) {
87 if (strncmp(line, "#!", 2) == 0) {
88 // This is a shebang line - we just consume it and continue
89 // The shebang has already done its job by launching our shell
90 return;
91 }
92 }
93
94 // Not a shebang line, rewind to beginning
95 fseek(file, pos, SEEK_SET);
96}

555static int parse_opts(int argc, char **argv) {
556 // POSIX-compliant argument parsing: shell [options] script [script-args]
557 // Only parse options that come before the script name
558 // Everything after the script name should be treated as script arguments
559
560 int arg_index = 1; // Start from first argument (skip program name)
561
562 // Manual option parsing to properly handle POSIX argument separation
563 while (arg_index < argc) {
564 char *arg = argv[arg_index];
565
566 // If argument doesn't start with '-', it's the script name
567 // Stop parsing options here - everything after is script arguments
568 if (arg[0] != '-' || strlen(arg) == 1) {
569 break;
570 }
571
572 // Handle special case: "--" ends option parsing
573 if (strcmp(arg, "--") == 0) {
574 arg_index++; // Skip the "--"
575 break;
576 }
577
578 // Handle long options
579 if (arg[0] == '-' && arg[1] == '-') {
580 if (strcmp(arg, "--help") == 0) {
581 usage(EXIT_SUCCESS);
582 } else if (strcmp(arg, "--version") == 0) {
583 printf("%s %s\n", LUSUSH_NAME, LUSUSH_VERSION_STRING);
584 printf("%s\n", LUSUSH_DESCRIPTION);
585 printf("Copyright (C) 2021-2026 Michael Berry. Licensed under "
586 "GPL-3.0+.\n");
587 exit(EXIT_SUCCESS);
588 } else {
589 fprintf(stderr, "%s: invalid option -- '%s'\n", argv[0], arg);
590 usage(EXIT_FAILURE);
591 }
592 arg_index++;
593 continue;
594 }
595
596 // Handle short options (can be combined like -abc)
597 for (int i = 1; arg[i] != '\0'; i++) {
598 char opt = arg[i];
599
600 switch (opt) {
601 case 'h':
602 shell_opts.hash_commands = true;
603 break;
604 case 'V':
605 printf("%s %s\n", LUSUSH_NAME, LUSUSH_VERSION_STRING);
606 printf("%s\n", LUSUSH_DESCRIPTION);
607 printf("Copyright (C) 2021-2026 Michael Berry. Licensed under "
608 "GPL-3.0+.\n");
609 exit(EXIT_SUCCESS);
610 break;
611 case 'c':
612 // Execute command string - requires argument
613 if (arg[i + 1] != '\0') {
614 // Argument attached to option: -ccommand
615 shell_opts.command_mode = true;
616 shell_opts.command_string = strdup(&arg[i + 1]);
617 if (!shell_opts.command_string) {
618 error_abort(
619 "failed to allocate memory for command string");
620 }
621 goto next_arg; // Skip rest of this argument
622 } else if (arg_index + 1 < argc) {
623 // Argument in next parameter: -c command
624 arg_index++;
625 shell_opts.command_mode = true;
626 shell_opts.command_string = strdup(argv[arg_index]);
627 if (!shell_opts.command_string) {
628 error_abort(
629 "failed to allocate memory for command string");
630 }
631 goto next_arg;
632 } else {
633 fprintf(stderr, "%s: option requires an argument -- '%c'\n",
634 argv[0], opt);
635 usage(EXIT_FAILURE);
636 }
637 break;
638 case 's':
639 shell_opts.stdin_mode = true;
640 break;
641 case 'i':
642 shell_opts.interactive = true;
643 break;
644 case 'l':
645 shell_opts.login_shell = true;
646 break;
647 case 'e':
648 shell_opts.exit_on_error = true;
649 break;
650 case 'x':
651 shell_opts.trace_execution = true;
652 break;
653 case 'n':
654 shell_opts.syntax_check = true;
655 break;
656 case 'u':
657 shell_opts.unset_error = true;
658 break;
659 case 'v':
660 shell_opts.verbose = true;
661 break;
662 case 'f':
663 shell_opts.no_globbing = true;
664 break;
665 case 'm':
666 shell_opts.job_control = true;
667 break;
668 case 'b':
669 shell_opts.notify = true;
670 break;
671 case 't':
672 shell_opts.onecmd = true;
673 break;
674 default:
675 fprintf(stderr, "%s: invalid option -- '%c'\n", argv[0], opt);
676 usage(EXIT_FAILURE);
677 }
678 }
679
680 next_arg:
681 arg_index++;
682 }
683
684 // Return the index of the first non-option argument (script name)
685 return arg_index;
686}
