1/**
2 * @file custom_source.h
3 * @brief LLE Custom Completion Source API
4 *
5 * Public API for registering custom completion sources programmatically.
6 * Allows builtins, plugins, and other code to add custom completion providers.
7 *
8 * Example usage:
9 * @code
10 * static lle_result_t my_generate(void *user_data,
11 * const lle_completion_context_t *context,
12 * const char *prefix,
13 * lle_completion_result_t *result) {
14 * // Add completions matching prefix
15 * lle_completion_add_item(result, "option1", " ", "First option", 800);
16 * lle_completion_add_item(result, "option2", " ", "Second option", 700);
17 * return LLE_SUCCESS;
18 * }
19 *
20 * static bool my_applicable(void *user_data,
21 * const lle_completion_context_t *context) {
22 * return context->command_name &&
23 * strcmp(context->command_name, "mycommand") == 0;
24 * }
25 *
26 * lle_custom_completion_source_t my_source = {
27 * .name = "my-completions",
28 * .description = "Custom completions for mycommand",
29 * .priority = 800,
30 * .generate = my_generate,
31 * .is_applicable = my_applicable,
32 * .user_data = NULL
33 * };
34 *
35 * lle_completion_register_source(&my_source);
36 * @endcode
37 *
38 * Specification: Spec 12 (Completion System), Spec 07 (Extensibility)
39 * Version: 1.0.0
40 */

70/**
71 * @brief Custom completion source definition (public API)
72 *
73 * Defines a custom completion source that can be registered with the
74 * completion system. Sources provide completions for specific contexts
75 * based on command name, argument position, etc.
76 */
77typedef struct lle_custom_completion_source {
78 /** @brief Source identifier (must be unique) */
79 const char *name;
80
81 /** @brief Human-readable description (optional, can be NULL) */
82 const char *description;
83
84 /** @brief Priority (higher = queried earlier, default 500) */
85 int priority;
86
87 /**
88 * @brief Generate completions for given prefix (required)
89 *
90 * Called when the source is applicable for the current context.
91 * Should add matching completions to the result.
92 *
93 * @param user_data User data passed during registration
94 * @param context Completion context with command info
95 * @param prefix Current word prefix to match
96 * @param result Result structure to add completions to
97 * @return LLE_SUCCESS or error code
98 */
99 lle_result_t (*generate)(void *user_data,
100 const lle_context_analyzer_t *context,
101 const char *prefix,
102 lle_completion_result_t *result);
103
104 /**
105 * @brief Check if source applies to this context (optional)
106 *
107 * If NULL, source is always queried. Otherwise, source is only
108 * queried if this function returns true.
109 *
110 * @param user_data User data passed during registration
111 * @param context Completion context with command info
112 * @return true if source should be queried, false otherwise
113 */
114 bool (*is_applicable)(void *user_data,
115 const lle_context_analyzer_t *context);
116
117 /**
118 * @brief Cleanup function called on unregistration (optional)
119 *
120 * If not NULL, called when the source is unregistered to allow
121 * cleanup of user_data resources.
122 *
123 * @param user_data User data passed during registration
124 */
125 void (*cleanup)(void *user_data);
126
127 /** @brief User data passed to all callbacks */
128 void *user_data;
129
130} lle_custom_completion_source_t;

137/**
138 * @brief Register a custom completion source
139 *
140 * Registers a new completion source with the system. The source will be
141 * queried during completion generation if applicable.
142 *
143 * @param source Source definition (copied, caller retains ownership)
144 * @return LLE_SUCCESS on success
145 * @return LLE_ERROR_INVALID_PARAMETER if source or source->name is NULL
146 * @return LLE_ERROR_INVALID_PARAMETER if source->generate is NULL
147 * @return LLE_ERROR_ALREADY_EXISTS if source with same name exists
148 * @return LLE_ERROR_CAPACITY if max sources limit reached (16)
149 *
150 * @note The source struct is copied; caller can free it after registration.
151 * @note String fields (name, description) are duplicated internally.
152 */
153lle_result_t
154lle_completion_register_source(const lle_custom_completion_source_t *source);
155
156/**
157 * @brief Unregister a custom completion source
158 *
159 * Removes a previously registered source. If the source has a cleanup
160 * callback, it will be called with the user_data.
161 *
162 * @param name Source name to unregister
163 * @return LLE_SUCCESS on success
164 * @return LLE_ERROR_INVALID_PARAMETER if name is NULL
165 * @return LLE_ERROR_NOT_FOUND if source not found
166 */
167lle_result_t lle_completion_unregister_source(const char *name);

273typedef struct lle_command_source_config {
274 char *name; /**< Source identifier (from TOML section name) */
275 char *description; /**< Human-readable description */
276 char **applies_to; /**< Array of "cmd [subcommand]" patterns */
277 size_t applies_to_count; /**< Number of applies_to patterns */
278 int argument; /**< Argument position (0 = any position) */
279 char *command; /**< Shell command to execute for completions */
280 char *suffix; /**< Suffix to append after completion */
281 int cache_seconds; /**< Cache TTL in seconds (0 = no cache) */

283 /* Runtime state (managed internally) */
284 char **cached_results; /**< Cached completion results */
285 size_t cached_count; /**< Number of cached results */
286 time_t cache_time; /**< When cache was populated */
287} lle_command_source_config_t;
