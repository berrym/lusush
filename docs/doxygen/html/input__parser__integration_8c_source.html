43static uint64_t get_current_time_us(void) {
44 struct timespec ts;
45 clock_gettime(CLOCK_MONOTONIC, &ts);
46 return (uint64_t)ts.tv_sec * 1000000 + (uint64_t)ts.tv_nsec / 1000;

56/*
57 * Determine event priority based on input type
58 *
59 * Priority levels:
60 * - High (3): Control sequences, special keys (Ctrl+C, etc.)
61 * - Normal (2): Regular keys, mouse events
62 * - Low (1): Text input
63 */
64static int determine_event_priority(lle_parsed_input_type_t input_type,
65 const lle_parsed_input_t *parsed_input) {
66 if (!parsed_input) {
67 return 1; /* Default to low */
68 }
69
70 switch (input_type) {
71 case LLE_PARSED_INPUT_TYPE_TEXT:
72 return 1; /* Low priority - text input */
73
74 case LLE_PARSED_INPUT_TYPE_KEY:
75 /* Check if it's a control key */
76 if (parsed_input->data.key_info.modifiers & LLE_KEY_MOD_CTRL) {
77 return 3; /* High priority - control sequence */
78 }
79 return 2; /* Normal priority - regular key */
80
81 case LLE_PARSED_INPUT_TYPE_MOUSE:
82 return 2; /* Normal priority - mouse event */
83
84 case LLE_PARSED_INPUT_TYPE_SEQUENCE:
85 return 3; /* High priority - terminal sequence */
86
87 default:
88 return 1; /* Default to low */
89 }
90}

92/*
93 * Generate text input event
94 *
95 * Converts UTF-8 text input into event structure.
96 */
97lle_result_t
98lle_input_parser_generate_text_events(lle_input_parser_system_t *parser_sys,
99 lle_parsed_input_t *parsed_input) {
100 if (!parser_sys || !parsed_input) {
101 return LLE_ERROR_INVALID_PARAMETER;
102 }
103
104 if (parsed_input->type != LLE_PARSED_INPUT_TYPE_TEXT &&
105 parsed_input->type != LLE_PARSED_INPUT_TYPE_PASTE) {
106 return LLE_ERROR_INVALID_PARAMETER;
107 }
108
109 /* Get text info from parsed input */
110 lle_text_input_info_t *text_info = &parsed_input->data.text_info;
111
112 /* Create event with text data */
113 lle_event_t *event = NULL;
114 lle_result_t result = lle_event_create(
115 parser_sys->event_system, LLE_EVENT_KEY_PRESS,
116 (void *)text_info->utf8_bytes, text_info->utf8_length, &event);
117
118 if (result != LLE_SUCCESS) {
119 return result;
120 }
121
122 /* Dispatch event to event system */
123 result = lle_event_dispatch(parser_sys->event_system, event);
124
125 /* Clean up event after dispatch */
126 lle_event_destroy(parser_sys->event_system, event);
127
128 /* Mark input as handled */
129 parsed_input->handled = true;
130
131 return result;
132}

247/*
248 * Main event generation dispatcher
249 *
250 * Routes parsed input to appropriate event generator based on type.
251 */
252lle_result_t
253lle_input_parser_generate_events(lle_input_parser_system_t *parser_sys,
254 lle_parsed_input_t *parsed_input) {
255 if (!parser_sys || !parsed_input) {
256 return LLE_ERROR_INVALID_PARAMETER;
257 }
258
259 if (!parser_sys->event_system) {
260 return LLE_ERROR_NOT_INITIALIZED;
261 }
262
263 /* Dispatch to appropriate event generator based on input type */
264 switch (parsed_input->type) {
265 case LLE_PARSED_INPUT_TYPE_TEXT:
266 case LLE_PARSED_INPUT_TYPE_PASTE:
267 return lle_input_parser_generate_text_events(parser_sys, parsed_input);
268
269 case LLE_PARSED_INPUT_TYPE_KEY:
270 case LLE_PARSED_INPUT_TYPE_SEQUENCE:
271 return lle_input_parser_generate_key_events(parser_sys, parsed_input);
272
273 case LLE_PARSED_INPUT_TYPE_MOUSE:
274 return lle_input_parser_generate_mouse_events(parser_sys, parsed_input);
275
276 case LLE_PARSED_INPUT_TYPE_FOCUS:
277 case LLE_PARSED_INPUT_TYPE_UNKNOWN:
278 default:
279 /* Unsupported input type */
280 return LLE_ERROR_INVALID_PARAMETER;
281 }
