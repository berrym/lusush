138/**
139 * Handle LAYER_EVENT_REDRAW_NEEDED from command_layer
140 *
141 * This is the critical event handler that connects command_layer updates
142 * to actual terminal rendering. When LLE updates the command_layer content,
143 * command_layer publishes REDRAW_NEEDED, and this handler renders it.
144 *
145 * @param event The redraw event
146 * @param user_data Pointer to display_controller_t instance
147 * @return LAYER_EVENTS_SUCCESS on success, error code on failure
148 */
149/* SCREEN BUFFER SYSTEM - The proper way to handle terminal updates
150 *
151 * Instead of trying to move cursor back and redraw (which is unreliable),
152 * we maintain virtual screen buffers representing what's on the terminal.
153 * On each update, we diff old vs new and apply only the changes.
154 *
155 * This is the proven approach used by ZLE, Fish, and Replxx.
156 */
157static screen_buffer_t current_screen;
158static screen_buffer_t desired_screen;
159static bool screen_buffer_initialized = false;
160static bool prompt_rendered = false;
161static int last_terminal_end_row =
162 0; /* Actual terminal row after ghost text/menu */
163
164/**
165 * Reset display state - called when starting new input session
166 *
167 * Clears screen buffers so next render starts fresh.
168 */
169void dc_reset_prompt_display_state(void) {
170 prompt_rendered = false;
171 last_terminal_end_row = 0;
172 if (screen_buffer_initialized) {
173 screen_buffer_clear(&current_screen);
174 screen_buffer_clear(&desired_screen);
175 }
176
177 /* Reset command_layer's update_sequence_number so next render is treated
178 * as "first render" and publishes REDRAW_NEEDED even if buffer is empty.
179 * Without this, empty-to-empty transitions (like Ctrl+G on empty buffer)
180 * would skip the event and prompt wouldn't be drawn. */
181 display_controller_t *dc = display_integration_get_controller();
182 if (dc && dc->compositor && dc->compositor->command_layer) {
183 dc->compositor->command_layer->update_sequence_number = 0;
184 }
185}

201bool dc_apply_transient_prompt(const char *transient_prompt,
202 const char *command_text) {
203 if (!transient_prompt || !screen_buffer_initialized) {
204 return false;
205 }
206
207 /*
208 * Transient Prompt Replacement (Spec 25 Section 12)
209 *
210 * Replace the current (fancy) prompt with a minimal transient prompt
211 * while preserving the command text. This simplifies scrollback history.
212 *
213 * Current state (from current_screen):
214 * - command_start_row: row where command starts (0 = same line as prompt)
215 * - cursor_row: current cursor row (last row of display)
216 * - We need to go back and redraw everything from row 0
217 *
218 * Algorithm:
219 * 1. Move cursor up to row 0, column 1
220 * 2. Clear from there to end of screen
221 * 3. Write transient prompt + command text
222 * 4. Update screen buffer to reflect new state
223 */
224
225 int total_rows = current_screen.cursor_row + 1;
226 char seq_buf[32];
227 int seq_len;
228
229 /* Step 1: Move cursor up to the first row (where prompt started) */
230 if (current_screen.cursor_row > 0) {
231 seq_len = snprintf(seq_buf, sizeof(seq_buf), "\033[%dA",
232 current_screen.cursor_row);
233 if (seq_len > 0) {
234 write(STDOUT_FILENO, seq_buf, (size_t)seq_len);
235 }
236 }
237
238 /* Step 2: Move to column 1 */
239 write(STDOUT_FILENO, "\033[1G", 4);
240
241 /* Step 3: Clear from cursor to end of screen */
242 write(STDOUT_FILENO, "\033[J", 3);
243
244 /* Step 4: Write transient prompt */
245 if (transient_prompt[0] != '\0') {
246 write(STDOUT_FILENO, transient_prompt, strlen(transient_prompt));
247 }
248
249 /* Step 5: Write command text */
250 if (command_text && command_text[0] != '\0') {
251 write(STDOUT_FILENO, command_text, strlen(command_text));
252 }
253
254 /* Step 6: Update screen buffer to reflect new state
255 * Re-render with transient prompt so current_screen is accurate */
256 size_t cursor_offset = command_text ? strlen(command_text) : 0;
257 screen_buffer_render(&current_screen, transient_prompt,
258 command_text ? command_text : "", cursor_offset);
259
260 /* Mark that we handled this - prompt_rendered stays true since we
261 * wrote a prompt, but next dc_reset_prompt_display_state will clear it */
262
263 (void)total_rows; /* Used for documentation, may use later */
264
265 return true;
266}

268void dc_get_prompt_metrics(int *prompt_lines, int *total_lines,
269 int *command_col) {
270 if (!screen_buffer_initialized) {
271 if (prompt_lines)
272 *prompt_lines = 1;
273 if (total_lines)
274 *total_lines = 1;
275 if (command_col)
276 *command_col = 0;
277 return;
278 }
279
280 /*
281 * current_screen contains the last rendered state:
282 * - command_start_row: row where command text starts (0-based)
283 * - command_start_col: column where command text starts
284 * - cursor_row: current cursor row (0-based)
285 *
286 * For a single-line prompt "$ cmd":
287 * command_start_row = 0, command_start_col > 0
288 * prompt occupies part of row 0
289 *
290 * For a two-line prompt:
291 * Line 0: "[user@host] ~/path"
292 * Line 1: "$ cmd"
293 * command_start_row = 1, prompt occupies rows 0-1
294 *
295 * prompt_lines = command_start_row + 1 (since it's 0-indexed)
296 * But if command_start_row == 0, prompt shares line with command,
297 * so prompt_lines = 1.
298 */
299 if (prompt_lines) {
300 *prompt_lines = current_screen.command_start_row + 1;
301 }
302
303 if (total_lines) {
304 /* Total lines = cursor row + 1 (cursor is on the last used row) */
305 *total_lines = current_screen.cursor_row + 1;
306 }
307
308 if (command_col) {
309 *command_col = current_screen.command_start_col;
310 }
311}

313/**
314 * Callback for screen_buffer_render_with_continuation.
315 *
316 * Called when a newline is encountered during character-by-character rendering.
317 * Analyzes the line that just ended and returns the continuation prompt for
318 * the next line based on the current parsing state.
319 *
320 * @param line_text Plain text of the line that just ended (ANSI stripped)
321 * @param line_len Length of line_text
322 * @param line_number Logical line number (0-based)
323 * @param user_data Pointer to continuation_state_t
324 * @return Continuation prompt string, or NULL
325 */
326static const char *dc_continuation_prompt_callback(const char *line_text,
327 size_t line_len,
328 int line_number,
329 void *user_data) {
330 continuation_state_t *cont_state = (continuation_state_t *)user_data;
331 if (!cont_state) {
332 return "> "; /* Fallback */
333 }
334
335 /* Analyze this line to update continuation state */
336 continuation_analyze_line(line_text, cont_state);
337
338 /* Get the continuation prompt based on updated state */
339 const char *prompt = continuation_get_prompt(cont_state);
340
341 DC_DEBUG("Continuation callback: line=%d, text='%.40s%s', prompt='%s'",
342 line_number, line_text, line_len > 40 ? "..." : "",
343 prompt ? prompt : "(null)");
344
345 return prompt;
346}

348static layer_events_error_t dc_handle_redraw_needed(const layer_event_t *event,
349 void *user_data) {
350
351 (void)event;
352 display_controller_t *controller = (display_controller_t *)user_data;
353
354 if (!controller || !controller->is_initialized) {
355 return LAYER_EVENTS_ERROR_INVALID_PARAM;
356 }
357
358 command_layer_t *cmd_layer = controller->compositor->command_layer;
359 if (!cmd_layer) {
360 return LAYER_EVENTS_ERROR_INVALID_PARAM;
361 }
362
363 int term_width = 80;
364 if (controller->terminal_ctrl &&
365 controller->terminal_ctrl->capabilities.terminal_width > 0) {
366 term_width = controller->terminal_ctrl->capabilities.terminal_width;
367 }
368
369 if (!screen_buffer_initialized) {
370 screen_buffer_init(&current_screen, term_width);
371 screen_buffer_init(&desired_screen, term_width);
372 screen_buffer_initialized = true;
373 } else {
374 current_screen.terminal_width = term_width;
375 desired_screen.terminal_width = term_width;
376 }
377
378 prompt_layer_t *prompt_layer = controller->compositor->prompt_layer;
379 char prompt_buffer[PROMPT_LAYER_MAX_CONTENT_SIZE] = {0};
380 if (prompt_layer) {
381 prompt_layer_get_rendered_content(prompt_layer, prompt_buffer,
382 sizeof(prompt_buffer));
383 }
384
385 char command_buffer[COMMAND_LAYER_MAX_HIGHLIGHTED_SIZE];
386 command_layer_error_t cmd_result = command_layer_get_highlighted_text(
387 cmd_layer, command_buffer, sizeof(command_buffer));
388
389 if (cmd_result != COMMAND_LAYER_SUCCESS) {
390 return LAYER_EVENTS_ERROR_INVALID_PARAM;
391 }
392
393 /* Render completion menu if active (Proper Architecture - Spec 12)
394 * Menu is now composed at display time, not baked into command text */
395 char menu_buffer[8192] = {0};
396 char *menu_text = NULL;
397
398 if (controller->completion_menu_visible &&
399 controller->active_completion_menu) {
400 lle_menu_render_options_t options =
401 lle_menu_renderer_default_options(term_width);
402 options.max_rows = 20; /* Limit menu to 20 rows */
403
404 lle_menu_render_stats_t stats;
405 lle_result_t result = lle_completion_menu_render(
406 controller->active_completion_menu, &options, menu_buffer,
407 sizeof(menu_buffer), &stats);
408
409 if (result == LLE_SUCCESS && menu_buffer[0]) {
410 menu_text = menu_buffer;
411 }
412 }
413
414 /* CONTINUATION PROMPT SUPPORT:
415 *
416 * Use screen_buffer_render_with_continuation() which calls back on each
417 * newline to get the context-aware continuation prompt. This is the
418 * architecturally correct approach: prompts are set at the exact visual row
419 * where each newline lands during character-by-character rendering, not
420 * pre-calculated.
421 *
422 * The callback receives the plain text of each line (ANSI stripped) and
423 * updates the continuation state to determine the appropriate prompt.
424 */
425 int newline_count = count_newlines(command_buffer);
426 bool is_multiline = (newline_count > 0);
427
428 size_t cursor_byte_offset = cmd_layer->cursor_position;
429
430 if (is_multiline) {
431 /* Use callback-based rendering for proper visual row tracking */
432 continuation_state_t cont_state;
433 continuation_state_init(&cont_state);
434
435 screen_buffer_render_with_continuation(
436 &desired_screen, prompt_buffer, command_buffer, cursor_byte_offset,
437 dc_continuation_prompt_callback, &cont_state);
438
439 continuation_state_cleanup(&cont_state);
440 } else {
441 /* No newlines - use simple render */
442 screen_buffer_render(&desired_screen, prompt_buffer, command_buffer,
443 cursor_byte_offset);
444 }
445
446 /* DEBUG: Log what screen_buffer_render produced */
447 DC_DEBUG("After render: num_rows=%d, command_start_row=%d, cursor=(%d,%d), "
448 "term_width=%d",
449 desired_screen.num_rows, desired_screen.command_start_row,
450 desired_screen.cursor_row, desired_screen.cursor_col, term_width);
451
452 /* DEBUG: Log prefixes set on each row */
453 for (int r = 0; r < desired_screen.num_rows && r < 10; r++) {
454 const char *prefix = screen_buffer_get_line_prefix(&desired_screen, r);
455 if (prefix) {
456 DC_DEBUG(" Row %d has prefix: '%s'", r, prefix);
457 }
458 }
459
460 /* Add menu rows to screen_buffer per SCREEN_BUFFER_MENU_INTEGRATION_PLAN.md
461 *
462 * This is the key fix: by adding menu rows to screen_buffer AFTER rendering
463 * command text, the buffer knows the total display height. This allows
464 * screen_buffer_get_rows_below_cursor() to return the correct value for
465 * cursor positioning, fixing the "upward row consumption" bug.
466 *
467 * Cursor position (cursor_row, cursor_col) stays in the command area -
468 * menu rows are added AFTER and don't affect cursor tracking.
469 */
470 int menu_rows_added = 0;
471 if (menu_text && *menu_text) {
472 /* Add menu starting at row after command ends.
473 * Note: We add a newline before menu, so start at num_rows (which is
474 * one past the last command row). */
475 int menu_start_row = desired_screen.num_rows;
476 menu_rows_added = screen_buffer_add_text_rows(
477 &desired_screen, menu_start_row, menu_text);
478
479 DC_DEBUG("Added menu to screen_buffer: start_row=%d, rows_added=%d, "
480 "new_num_rows=%d",
481 menu_start_row, menu_rows_added, desired_screen.num_rows);
482 }
483
484 /* PROMPT-ONCE ARCHITECTURE per MODERN_EDITOR_WRAPPING_RESEARCH.md
485 *
486 * This implements the proven approach used by Replxx, Fish, and ZLE:
487 * 1. Prompt is drawn ONCE on first render and NEVER redrawn
488 * 2. Only the command text area is cleared and redrawn on updates
489 * 3. Absolute column positioning (\033[{n}G) is used instead of \r
490 *
491 * Key principle: Never move cursor to column 0 after first render,
492 * as this would allow \033[J to clear the prompt.
493 */
494
495 /* First render only: Draw prompt once */
496 if (!prompt_rendered) {
497 if (prompt_buffer[0]) {
498 write(STDOUT_FILENO, prompt_buffer, strlen(prompt_buffer));
499 }
500 prompt_rendered = true;
501 }
502
503 /* Every render (including first): Position to command start and redraw
504 * command */
505
506 /* Step 1: Move to absolute column where command starts (after prompt)
507 * For multi-line prompts, use command_start_col from screen buffer
508 * Use \033[{n}G for absolute positioning (1-based indexing) */
509 int command_start_col = desired_screen.command_start_col;
510 char move_to_col[32];
511 int col_len = snprintf(move_to_col, sizeof(move_to_col), "\033[%dG",
512 command_start_col + 1);
513 if (col_len > 0) {
514 write(STDOUT_FILENO, move_to_col, col_len);
515 }
516
517 /* Step 2: Handle ghost text/menu cleanup from previous render
518 *
519 * Problem: Ghost text (autosuggestions) may have wrapped to extra rows.
520 * After Step 5 of previous render, cursor was moved back to command cursor
521 * position. But the ghost text remains on screen below that.
522 *
523 * Solution: If previous render had extra rows (ghost text/menu), move DOWN
524 * to that row first, then clear, then move back up. This ensures we clear
525 * all the ghost text artifacts. */
526 int command_row = desired_screen.command_start_row;
527
528 if (last_terminal_end_row > current_screen.cursor_row) {
529 /* Previous render had ghost text/menu extending below cursor position.
530 * Move DOWN to that row to clear from there. */
531 int rows_down = last_terminal_end_row - current_screen.cursor_row;
532 DC_DEBUG("Step2: Moving DOWN %d rows to clear ghost text", rows_down);
533 char move_down[32];
534 int down_len =
535 snprintf(move_down, sizeof(move_down), "\033[%dB", rows_down);
536 if (down_len > 0) {
537 write(STDOUT_FILENO, move_down, down_len);
538 }
539
540 /* Clear from here to end of screen (clears ghost text) */
541 write(STDOUT_FILENO, "\033[J", 3);
542
543 /* Move back up to command start row */
544 int rows_up = last_terminal_end_row - command_row;
545 if (rows_up > 0) {
546 DC_DEBUG("Step2: Moving UP %d rows to command start", rows_up);
547 char move_up[32];
548 int up_len =
549 snprintf(move_up, sizeof(move_up), "\033[%dA", rows_up);
550 if (up_len > 0) {
551 write(STDOUT_FILENO, move_up, up_len);
552 }
553 }
554 } else if (current_screen.cursor_row > command_row) {
555 /* No ghost text overflow, but cursor is below command start - move up
556 */
557 int rows_up = current_screen.cursor_row - command_row;
558 DC_DEBUG("Step2: Moving up %d rows", rows_up);
559 char move_up[32];
560 int up_len = snprintf(move_up, sizeof(move_up), "\033[%dA", rows_up);
561 if (up_len > 0) {
562 write(STDOUT_FILENO, move_up, up_len);
563 }
564 }
565
566 /* Step 3: Clear from current position to end of screen
567 * This clears only the command area, never touches the prompt */
568 write(STDOUT_FILENO, "\033[J", 3);
569
570 /* Step 4: Write command text with continuation prompts
571 *
572 * For multiline input, we need to insert continuation prompts after each
573 * newline. The screen_buffer has prefixes set at visual rows (accounting
574 * for wrapping), so we must track visual rows during output to find the
575 * correct prefix.
576 *
577 * This uses character-by-character tracking to match how
578 * screen_buffer_render calculated visual rows, ensuring prefixes appear on
579 * the correct lines.
580 */
581 if (command_buffer[0]) {
582 if (is_multiline) {
583 /* Write command text character by character, tracking visual rows
584 * to output continuation prompts at the correct positions */
585 size_t i = 0;
586 size_t text_len = strlen(command_buffer);
587 int visual_row = desired_screen.command_start_row;
588 int visual_col = desired_screen.command_start_col;
589
590 while (i < text_len) {
591 unsigned char ch = (unsigned char)command_buffer[i];
592
593 /* Handle ANSI escape sequences - pass through without affecting
594 * position */
595 if (ch == '\033' || ch == '\x1b') {
596 size_t seq_start = i;
597 i++;
598 if (i < text_len && command_buffer[i] == '[') {
599 i++;
600 while (i < text_len) {
601 char c = command_buffer[i++];
602 if ((c >= 'A' && c <= 'Z') ||
603 (c >= 'a' && c <= 'z') || c == 'm' ||
604 c == 'H' || c == 'J' || c == 'K' || c == 'G' ||
605 c == 'f' || c == 's' || c == 'u') {
606 break;
607 }
608 }
609 }
610 /* Write the ANSI sequence */
611 write(STDOUT_FILENO, command_buffer + seq_start,
612 i - seq_start);
613 continue;
614 }
615
616 /* Handle newlines - move to next visual row and output
617 * continuation prompt */
618 if (ch == '\n') {
619 write(STDOUT_FILENO, "\n", 1);
620 visual_row++;
621
622 /* Get continuation prompt for this visual row */
623 const char *cont_prompt = screen_buffer_get_line_prefix(
624 &desired_screen, visual_row);
625 if (cont_prompt) {
626 /* Reset ANSI state before writing continuation prompt
627 */
628 write(STDOUT_FILENO, "\033[0m", 4);
629 write(STDOUT_FILENO, cont_prompt, strlen(cont_prompt));
630 visual_col =
631 (int)screen_buffer_get_line_prefix_visual_width(
632 &desired_screen, visual_row);
633 } else {
634 visual_col = 0;
635 }
636
637 i++;
638 continue;
639 }
640
641 /* Handle regular characters - track visual column for wrapping
642 */
643 uint32_t codepoint;
644 int char_bytes = lle_utf8_decode_codepoint(
645 command_buffer + i, text_len - i, &codepoint);
646
647 if (char_bytes > 0) {
648 int char_width = lle_utf8_codepoint_width(codepoint);
649
650 /* Write the character */
651 write(STDOUT_FILENO, command_buffer + i, char_bytes);
652
653 /* Update visual position */
654 visual_col += char_width;
655
656 /* Check for line wrap */
657 if (visual_col >= term_width) {
658 visual_row++;
659 visual_col = 0;
660 /* Note: wrapped lines don't get continuation prompts,
661 * only lines after explicit newlines do */
662 }
663
664 i += char_bytes;
665 } else {
666 /* Invalid UTF-8, write single byte */
667 write(STDOUT_FILENO, command_buffer + i, 1);
668 visual_col++;
669 if (visual_col >= term_width) {
670 visual_row++;
671 visual_col = 0;
672 }
673 i++;
674 }
675 }
676 } else {
677 /* Single-line input - write directly */
678 write(STDOUT_FILENO, command_buffer, strlen(command_buffer));
679 }
680 }

682 /* Step 4a: Write autosuggestion ghost text (Fish-style)
683 *
684 * Conditions for showing ghost text:
685 * 1. Autosuggestions enabled and layer available
686 * 2. No completion menu visible (menu takes precedence)
687 * 3. Not multiline input (simplifies initial implementation)
688 * 4. Cursor is at end of command (checked by autosuggestions_layer)
689 *
690 * The ghost text appears in BRIGHT_BLACK (gray/dimmed) after the command.
691 * Cursor positioning (Step 5) will move cursor back to correct position.
692 */
693 if (controller->autosuggestions_enabled &&
694 controller->autosuggestions_layer &&
695 !controller->completion_menu_visible && !is_multiline) {
696
697 const char *suggestion = autosuggestions_layer_get_current_suggestion(
698 controller->autosuggestions_layer);
699
700 if (suggestion && *suggestion) {
701 /* Write ghost text in BRIGHT_BLACK (dimmed gray) */
702 write(STDOUT_FILENO, "\033[90m",
703 5); /* Set bright black foreground */
704 write(STDOUT_FILENO, suggestion, strlen(suggestion));
705 write(STDOUT_FILENO, "\033[0m", 4); /* Reset all attributes */
706 }
707 }
708
709 /* Step 4b: Write completion menu WITHOUT continuation prompts */
710 if (menu_text && *menu_text) {
711 write(STDOUT_FILENO, "\n", 1);
712 write(STDOUT_FILENO, menu_text, strlen(menu_text));
713 }
714
715 /* Step 5: Position cursor at the correct location
716 *
717 * After drawing command text, ghost text, and menu, terminal cursor is at
718 * the end. We need to position it where the user's cursor actually is (in
719 * the command).
720 *
721 * With the new architecture (per SCREEN_BUFFER_MENU_INTEGRATION_PLAN.md):
722 * - Menu rows were added to desired_screen via
723 * screen_buffer_add_text_rows()
724 * - desired_screen.num_rows now includes menu rows
725 * - desired_screen.cursor_row is still in the command area (unaffected by
726 * menu)
727 * - screen_buffer_get_rows_below_cursor() gives us the exact count
728 *
729 * Ghost text is not yet tracked in screen_buffer, so we still calculate it
730 * separately.
731 */
732 int cursor_row = desired_screen.cursor_row;
733 int cursor_col = desired_screen.cursor_col;
734
735 /* Calculate extra rows added by ghost text (autosuggestion)
736 * Ghost text may wrap to additional lines beyond the command text.
737 * TODO: Add ghost text to screen_buffer for unified tracking. */
738 int ghost_text_extra_rows = 0;
739 if (controller->autosuggestions_enabled &&
740 controller->autosuggestions_layer &&
741 !controller->completion_menu_visible && !is_multiline) {
742
743 const char *suggestion = autosuggestions_layer_get_current_suggestion(
744 controller->autosuggestions_layer);
745
746 if (suggestion && *suggestion) {
747 size_t suggestion_width =
748 lle_utf8_string_width(suggestion, strlen(suggestion));
749 int command_end_col = desired_screen.command_end_col;
750 int total_cols_needed = command_end_col + (int)suggestion_width;
751
752 if (total_cols_needed > term_width) {
753 ghost_text_extra_rows = (total_cols_needed - 1) / term_width;
754 }
755 }
756 }
757
758 /* Calculate rows to move up using screen_buffer's tracked state.
759 *
760 * screen_buffer_get_rows_below_cursor() returns: (num_rows - 1) -
761 * cursor_row This accounts for menu rows that were added via
762 * screen_buffer_add_text_rows(). We still need to add ghost_text_extra_rows
763 * since those aren't in screen_buffer yet.
764 */
765 int rows_to_move_up = screen_buffer_get_rows_below_cursor(&desired_screen) +
766 ghost_text_extra_rows;
767
768 DC_DEBUG("Step5: cursor=(%d,%d), num_rows=%d, rows_below=%d, ghost=%d, "
769 "total_up=%d",
770 cursor_row, cursor_col, desired_screen.num_rows,
771 screen_buffer_get_rows_below_cursor(&desired_screen),
772 ghost_text_extra_rows, rows_to_move_up);
773
774 if (rows_to_move_up > 0) {
775 char up_seq[16];
776 int up_len =
777 snprintf(up_seq, sizeof(up_seq), "\033[%dA", rows_to_move_up);
778 if (up_len > 0) {
779 write(STDOUT_FILENO, up_seq, up_len);
780 }
781 }
782
783 /* Move to absolute column (never use \r - it goes to column 0!)
784 * Use \033[{n}G for absolute column positioning (1-based indexing) */
785 char col_seq[16];
786 int col_seq_len =
787 snprintf(col_seq, sizeof(col_seq), "\033[%dG", cursor_col + 1);
788 if (col_seq_len > 0) {
789 write(STDOUT_FILENO, col_seq, col_seq_len);
790 }
791
792 DC_DEBUG(
793 "Step5 done: copying desired_screen to current_screen (cursor_row=%d)",
794 desired_screen.cursor_row);
795 screen_buffer_copy(&current_screen, &desired_screen);
796 prompt_rendered = true;
797
798 /* Track where the terminal display actually ends (including ghost
799 * text/menu) This is needed by Step 2 on the next render to move up the
800 * correct amount.
801 *
802 * With menu rows now tracked in screen_buffer:
803 * - (num_rows - 1) gives the last row index in the buffer (includes menu)
804 * - ghost_text_extra_rows adds any additional wrapping from autosuggestions
805 */
806 last_terminal_end_row =
807 (desired_screen.num_rows - 1) + ghost_text_extra_rows;
808
809 /* NOTE: fsync() was causing input timeouts after cursor positioning -
810 * removed stdout is line-buffered by default and terminal I/O doesn't need
811 * fsync */
812
813 return LAYER_EVENTS_SUCCESS;
814}

850/**
851 * Extract semantic components from prompt for intelligent caching.
852 * Timestamp-aware normalization that ignores dynamic time elements.
853 */
854static uint32_t dc_hash_prompt_semantic(const char *prompt) {
855 if (!prompt || *prompt == '\0')
856 return DC_HASH_SEED;
857
858 // For maximum cache hit rates, normalize prompt by removing time-sensitive
859 // elements
860 char normalized[512] = {0};
861 const char *p = prompt;
862 size_t pos = 0;
863 bool in_color_seq = false;
864 bool skip_timestamp = false;
865
866 while (*p && pos < sizeof(normalized) - 1) {
867 if (*p == '\033' || *p == '\x1b') {
868 in_color_seq = true;
869 normalized[pos++] = 'C'; // Color marker only
870 p++;
871 continue;
872 }
873
874 if (in_color_seq) {
875 if (*p == 'm' || *p == 'K' || *p == 'J' || *p == 'H') {
876 in_color_seq = false;
877 }
878 p++;
879 continue;
880 }
881
882 // Detect and skip timestamp patterns (common formats)
883 // Skip sequences like "Mon Jan 27 15:30:42 2025" or "15:30:42" or "Jan
884 // 27"
885 if ((*p >= '0' && *p <= '9') &&
886 (*(p + 1) == ':' || (isdigit(*(p + 1)) && *(p + 2) == ':'))) {
887 // Skip time patterns like "15:30:42" or "3:45"
888 skip_timestamp = true;
889 } else if (skip_timestamp && (*p == ' ' || *p == '\t' || *p == '\n')) {
890 skip_timestamp = false;
891 p++;
892 continue;
893 } else if (skip_timestamp) {
894 p++;
895 continue;
896 }
897
898 // Skip common date patterns
899 if ((strncmp(p, "Mon ", 4) == 0) || (strncmp(p, "Tue ", 4) == 0) ||
900 (strncmp(p, "Wed ", 4) == 0) || (strncmp(p, "Thu ", 4) == 0) ||
901 (strncmp(p, "Fri ", 4) == 0) || (strncmp(p, "Sat ", 4) == 0) ||
902 (strncmp(p, "Sun ", 4) == 0) || (strncmp(p, "Jan ", 4) == 0) ||
903 (strncmp(p, "Feb ", 4) == 0) || (strncmp(p, "Mar ", 4) == 0) ||
904 (strncmp(p, "Apr ", 4) == 0) || (strncmp(p, "May ", 4) == 0) ||
905 (strncmp(p, "Jun ", 4) == 0) || (strncmp(p, "Jul ", 4) == 0) ||
906 (strncmp(p, "Aug ", 4) == 0) || (strncmp(p, "Sep ", 4) == 0) ||
907 (strncmp(p, "Oct ", 4) == 0) || (strncmp(p, "Nov ", 4) == 0) ||
908 (strncmp(p, "Dec ", 4) == 0)) {
909 // Skip to end of timestamp section
910 while (*p && *p != '\n' && *p != '$' && *p != '#' && *p != '>') {
911 p++;
912 }
913 continue;
914 }
915
916 // Normalize structural elements
917 if (*p == '$' || *p == '#' || *p == '>') {
918 normalized[pos++] = 'P'; // Prompt end marker
919 } else if (*p == '[' || *p == '(' || *p == '{') {
920 normalized[pos++] = 'B'; // Begin bracket
921 } else if (*p == ']' || *p == ')' || *p == '}') {
922 normalized[pos++] = 'E'; // End bracket
923 } else if (*p == '/' || *p == '~') {
924 normalized[pos++] = 'D'; // Directory marker
925 } else if (*p == '@') {
926 normalized[pos++] = 'A'; // User@host separator
927 } else if (isalpha(*p)) {
928 normalized[pos++] = tolower(*p); // Normalize case
929 } else if (*p != ' ' && *p != '\t' && *p != '\n') {
930 normalized[pos++] = *p; // Keep other significant chars
931 }
932
933 p++;
934 }
935
936 normalized[pos] = '\0';
937
938 // Hash the normalized prompt structure
939 uint32_t hash = DC_HASH_SEED;
940 const unsigned char *data = (const unsigned char *)normalized;
941 while (*data) {
942 hash ^= *data++;
943 hash *= DC_HASH_PRIME;
944 }
945
946 return hash;
947}

949/**
950 * Extract semantic components from command for intelligent caching.
951 * Deterministic command classification for consistent cache grouping.
952 */
953static uint32_t dc_hash_command_semantic(const char *command) {
954 if (!command || *command == '\0')
955 return DC_HASH_SEED;
956
957 // Simplified but more predictable command classification
958 uint32_t cmd_hash = DC_HASH_SEED;
959 const char *p = command;
960
961 // Skip leading whitespace
962 while (*p && isspace(*p))
963 p++;
964
965 if (!*p)
966 return DC_HASH_SEED;
967
968 // Extract base command for classification
969 char base_cmd[32] = {0};
970 size_t i = 0;
971 while (*p && !isspace(*p) && i < sizeof(base_cmd) - 1) {
972 base_cmd[i++] = (char)tolower(*p++);
973 }
974
975 // Hash the base command
976 for (i = 0; base_cmd[i]; i++) {
977 cmd_hash ^= (unsigned char)base_cmd[i];
978 cmd_hash *= DC_HASH_PRIME;
979 }
980
981 // Simple argument classification (not exact matching)
982 while (*p && isspace(*p))
983 p++;
984 int has_flags = 0, has_args = 0;
985
986 while (*p) {
987 if (*p == '-') {
988 has_flags = 1;
989 while (*p && !isspace(*p))
990 p++;
991 } else if (!isspace(*p)) {
992 has_args = 1;
993 while (*p && !isspace(*p))
994 p++;
995 } else {
996 p++;
997 }
998 }
999
1000 // Include argument pattern in hash for differentiation
1001 cmd_hash ^= (has_flags << 8) | (has_args << 4);
1002
1003 return cmd_hash;
1004}

1006/**
1007 * Generate intelligent state hash for caching and comparison.
1008 * Balanced approach for optimal cache hit rates with necessary differentiation.
1009 */
1010static void dc_generate_state_hash(const char *prompt, const char *command,
1011 const char *theme_name,
1012 symbol_compatibility_t symbol_mode,
1013 char *hash_buffer, size_t buffer_size) {
1014 if (!hash_buffer || buffer_size < DC_MAX_STATE_HASH_LENGTH)
1015 return;
1016
1017 // Get semantic hashes for grouping similar states
1018 uint32_t prompt_semantic = dc_hash_prompt_semantic(prompt);
1019 uint32_t command_semantic = dc_hash_command_semantic(command);
1020
1021 // Add theme context to hash calculation
1022 uint32_t theme_hash = dc_hash_string(theme_name ? theme_name : "default");
1023 uint32_t symbol_hash = (uint32_t)symbol_mode;
1024
1025 // Create primary hash from semantic components including theme context
1026 uint32_t primary_hash = prompt_semantic ^ (command_semantic << 1) ^
1027 (theme_hash << 2) ^ (symbol_hash << 3);
1028
1029 // For identical repeated commands, ensure exact cache hits
1030 uint32_t exact_prompt = dc_hash_string(prompt ? prompt : "");
1031 uint32_t exact_command = dc_hash_string(command ? command : "");
1032
1033 // Balance semantic grouping with exact matching
1034 // Use semantic hash as primary, exact hash for differentiation
1035 uint32_t secondary_hash =
1036 (exact_prompt & 0xFFFF) ^ ((exact_command & 0xFFFF) << 16);
1037
1038 // Final combined hash: semantic-driven but with exact differentiation and
1039 // theme context
1040 uint32_t combined_hash = primary_hash ^ secondary_hash;
1041
1042 snprintf(hash_buffer, buffer_size, "s%08x_e%08x_c%08x_t%08x_m%x",
1043 primary_hash, secondary_hash, combined_hash, theme_hash,
1044 symbol_hash);
1045}

1119/**
1120 * Clean expired cache entries.
1121 */
1122static void dc_cleanup_expired_cache_entries(display_controller_t *controller) {
1123 if (!controller || !controller->cache_entries)
1124 return;
1125
1126 struct timeval current_time;
1127 if (gettimeofday(&current_time, NULL) != 0)
1128 return;
1129
1130 for (size_t i = 0; i < controller->cache_count;) {
1131 display_cache_entry_t *entry = &controller->cache_entries[i];
1132
1133 if (!entry->is_valid) {
1134 // Remove invalid entry
1135 if (entry->display_content) {
1136 lusush_pool_free(entry->display_content);
1137 entry->display_content = NULL;
1138 }
1139 if (entry->state_hash) {
1140 lusush_pool_free(entry->state_hash);
1141 entry->state_hash = NULL;
1142 }
1143
1144 // Move last entry to this position
1145 if (i < controller->cache_count - 1) {
1146 *entry = controller->cache_entries[controller->cache_count - 1];
1147 }
1148 controller->cache_count--;
1149 continue;
1150 }
1151
1152 // Calculate age
1153 uint64_t age_ms = ((uint64_t)current_time.tv_sec -
1154 (uint64_t)entry->timestamp.tv_sec) *
1155 1000 +
1156 ((uint64_t)current_time.tv_usec -
1157 (uint64_t)entry->timestamp.tv_usec) /
1158 1000;
1159
1160 // Adaptive TTL based on access frequency
1161 uint32_t adaptive_ttl_ms = controller->config.cache_ttl_ms;
1162
1163 if (entry->access_count > 5) {
1164 // High-frequency entries get 4x longer TTL
1165 adaptive_ttl_ms *= 4;
1166 } else if (entry->access_count > 2) {
1167 // Medium-frequency entries get 2x longer TTL
1168 adaptive_ttl_ms *= 2;
1169 } else if (entry->access_count == 1) {
1170 // Single-use entries get much shorter TTL
1171 adaptive_ttl_ms = adaptive_ttl_ms / 3;
1172 }
1173
1174 if (age_ms > adaptive_ttl_ms) {
1175 // Entry has expired
1176 if (entry->display_content) {
1177 lusush_pool_free(entry->display_content);
1178 entry->display_content = NULL;
1179 }
1180 if (entry->state_hash) {
1181 lusush_pool_free(entry->state_hash);
1182 entry->state_hash = NULL;
1183 }
1184 entry->is_valid = false;
1185
1186 // Move last entry to this position
1187 if (i < controller->cache_count - 1) {
1188 *entry = controller->cache_entries[controller->cache_count - 1];
1189 }
1190 controller->cache_count--;
1191 continue;
1192 }
1193
1194 i++;
1195 }
1196}

1218/**
1219 * Add new cache entry.
1220 */
1221static display_controller_error_t
1222dc_add_cache_entry(display_controller_t *controller, const char *state_hash,
1223 const char *display_content, size_t content_length) {
1224 if (!controller || !state_hash || !display_content) {
1225 return DISPLAY_CONTROLLER_ERROR_INVALID_PARAM;
1226 }
1227
1228 // Check if cache is full
1229 if (controller->cache_count >= controller->cache_capacity) {
1230 // Find least recently used entry with improved protection for frequent
1231 // entries
1232 size_t lru_index = 0;
1233 uint32_t min_score = UINT32_MAX;
1234
1235 for (size_t i = 0; i < controller->cache_count; i++) {
1236 display_cache_entry_t *entry = &controller->cache_entries[i];
1237
1238 // Calculate eviction score: lower score = more likely to evict
1239 // Protect frequently accessed entries by heavily weighting access
1240 // count
1241 uint32_t score = entry->access_count * 1000;
1242
1243 // Also consider recency (entries accessed recently get protection)
1244 struct timeval current_time;
1245 gettimeofday(&current_time, NULL);
1246 uint64_t age_ms = ((uint64_t)current_time.tv_sec -
1247 (uint64_t)entry->timestamp.tv_sec) *
1248 1000 +
1249 ((uint64_t)current_time.tv_usec -
1250 (uint64_t)entry->timestamp.tv_usec) /
1251 1000;
1252
1253 // Reduce score based on age (older entries have lower scores)
1254 if (age_ms > 5000) {
1255 score =
1256 score / 2; // Halve score for entries older than 5 seconds
1257 }
1258
1259 if (score < min_score) {
1260 min_score = score;
1261 lru_index = i;
1262 }
1263 }
1264
1265 // Remove LRU entry
1266 display_cache_entry_t *lru_entry =
1267 &controller->cache_entries[lru_index];
1268 if (lru_entry->display_content)
1269 lusush_pool_free(lru_entry->display_content);
1270 if (lru_entry->state_hash)
1271 lusush_pool_free(lru_entry->state_hash);
1272
1273 // Use this slot for new entry
1274 display_cache_entry_t *new_entry = lru_entry;
1275 memset(new_entry, 0, sizeof(display_cache_entry_t));
1276
1277 new_entry->display_content = lusush_pool_alloc(content_length + 1);
1278 if (!new_entry->display_content) {
1279 return DISPLAY_CONTROLLER_ERROR_MEMORY_ALLOCATION;
1280 }
1281
1282 new_entry->state_hash = lusush_pool_alloc(strlen(state_hash) + 1);
1283 if (!new_entry->state_hash) {
1284 lusush_pool_free(new_entry->display_content);
1285 return DISPLAY_CONTROLLER_ERROR_MEMORY_ALLOCATION;
1286 }
1287
1288 memcpy(new_entry->display_content, display_content, content_length);
1289 new_entry->display_content[content_length] = '\0';
1290 strcpy(new_entry->state_hash, state_hash);
1291 new_entry->content_length = content_length;
1292 new_entry->access_count = 1;
1293 new_entry->is_valid = true;
1294 gettimeofday(&new_entry->timestamp, NULL);
1295 } else {
1296 // Add new entry
1297 display_cache_entry_t *new_entry =
1298 &controller->cache_entries[controller->cache_count];
1299 memset(new_entry, 0, sizeof(display_cache_entry_t));
1300
1301 new_entry->display_content = lusush_pool_alloc(content_length + 1);
1302 if (!new_entry->display_content) {
1303 return DISPLAY_CONTROLLER_ERROR_MEMORY_ALLOCATION;
1304 }
1305
1306 new_entry->state_hash = lusush_pool_alloc(strlen(state_hash) + 1);
1307 if (!new_entry->state_hash) {
1308 lusush_pool_free(new_entry->display_content);
1309 return DISPLAY_CONTROLLER_ERROR_MEMORY_ALLOCATION;
1310 }
1311
1312 memcpy(new_entry->display_content, display_content, content_length);
1313 new_entry->display_content[content_length] = '\0';
1314 strcpy(new_entry->state_hash, state_hash);
1315 new_entry->content_length = content_length;
1316 new_entry->access_count = 1;
1317 new_entry->is_valid = true;
1318 gettimeofday(&new_entry->timestamp, NULL);
1319
1320 controller->cache_count++;
1321 }
1322
1323 return DISPLAY_CONTROLLER_SUCCESS;
1324}

1678display_controller_error_t
1679display_controller_display(display_controller_t *controller,
1680 const char *prompt_text, const char *command_text,
1681 char *output, size_t output_size) {
1682
1683 if (!controller || !output) {
1684 return DISPLAY_CONTROLLER_ERROR_INVALID_PARAM;
1685 }
1686
1687 if (!controller->is_initialized) {
1688 return DISPLAY_CONTROLLER_ERROR_NOT_INITIALIZED;
1689 }
1690
1691 if (output_size == 0) {
1692 return DISPLAY_CONTROLLER_ERROR_BUFFER_TOO_SMALL;
1693 }
1694
1695 struct timeval start_time, end_time;
1696 gettimeofday(&start_time, NULL);
1697
1698 DC_DEBUG("display_controller_display called");
1699 DC_DEBUG("Controller: %p, is_initialized: %d", (void *)controller,
1700 controller->is_initialized);
1701 DC_DEBUG("Prompt text: %s", prompt_text ? prompt_text : "(null)");
1702 DC_DEBUG("Command text: %s", command_text ? command_text : "(null)");
1703
1704 DC_DEBUG("Starting display operation (seq: %u)",
1705 controller->operation_sequence_number++);
1706
1707 // Generate state hash for caching including theme context
1708 char state_hash[DC_MAX_STATE_HASH_LENGTH];
1709 dc_generate_state_hash(
1710 prompt_text, command_text, controller->current_theme_name,
1711 controller->current_symbol_mode, state_hash, sizeof(state_hash));
1712
1713 // Check cache if enabled
1714 if (controller->config.enable_caching) {
1715 display_cache_entry_t *cached_entry =
1716 dc_find_cache_entry(controller, state_hash);
1717
1718 if (cached_entry && cached_entry->content_length < output_size) {
1719 // Cache hit - return cached content immediately
1720 memcpy(output, cached_entry->display_content,
1721 cached_entry->content_length);
1722 output[cached_entry->content_length] = '\0';
1723
1724 controller->performance.cache_hits++;
1725 controller->performance.total_display_operations++;
1726
1727 gettimeofday(&end_time, NULL);
1728 uint64_t operation_time = dc_time_diff_ns(&start_time, &end_time);
1729 dc_update_performance_history(controller, operation_time);
1730
1731 // Performance Monitoring: Record cache hit and timing
1732 display_integration_record_layer_cache_operation(
1733 "display_controller", true);
1734 display_integration_record_display_timing(operation_time);
1735
1736 DC_DEBUG("Cache hit for state hash: %s (access_count: %u)",
1737 state_hash, cached_entry->access_count);
1738 return DISPLAY_CONTROLLER_SUCCESS;
1739 } else {
1740 controller->performance.cache_misses++;
1741
1742 // Performance Monitoring: Record cache miss
1743 display_integration_record_layer_cache_operation(
1744 "display_controller", false);
1745 DC_DEBUG("Cache miss for state hash: %s", state_hash);
1746 }
1747 }
1748
1749 // Initialize layers if needed
1750 bool compositor_initialized =
1751 composition_engine_is_initialized(controller->compositor);
1752 DC_DEBUG("Composition engine initialized check: %s",
1753 compositor_initialized ? "true" : "false");
1754 DC_DEBUG("Compositor pointer: %p", (void *)controller->compositor);
1755
1756 if (!compositor_initialized) {
1757 // Create and initialize prompt and command layers
1758 prompt_layer_t *prompt_layer = prompt_layer_create();
1759 command_layer_t *command_layer = command_layer_create();
1760
1761 if (!prompt_layer || !command_layer) {
1762 DC_ERROR("Failed to create display layers");
1763 if (prompt_layer)
1764 prompt_layer_destroy(prompt_layer);
1765 if (command_layer)
1766 command_layer_destroy(command_layer);
1767 return DISPLAY_CONTROLLER_ERROR_INITIALIZATION_FAILED;
1768 }
1769
1770 // Initialize individual layers with event system
1771 DC_DEBUG("About to initialize prompt layer");
1772 DC_DEBUG("Prompt layer pointer: %p", (void *)prompt_layer);
1773 DC_DEBUG("Event system pointer: %p", (void *)controller->event_system);
1774
1775 prompt_layer_error_t prompt_init_result =
1776 prompt_layer_init(prompt_layer, controller->event_system);
1777
1778 DC_DEBUG("Prompt layer init returned: %d", prompt_init_result);
1779
1780 if (prompt_init_result != PROMPT_LAYER_SUCCESS) {
1781 DC_ERROR("Failed to initialize prompt layer: error %d",
1782 prompt_init_result);
1783
1784 // Detailed error analysis for prompt layer
1785 switch (prompt_init_result) {
1786 case PROMPT_LAYER_ERROR_INVALID_PARAM:
1787 DC_ERROR("Prompt error cause: Invalid parameter");
1788 break;
1789 case PROMPT_LAYER_ERROR_EVENT_SYSTEM_FAILURE:
1790 DC_ERROR("Prompt error cause: Event system failure");
1791 break;
1792 case PROMPT_LAYER_ERROR_MEMORY_ALLOCATION:
1793 DC_ERROR("Prompt error cause: Memory allocation failure");
1794 break;
1795 default:
1796 DC_ERROR("Prompt error cause: Unknown error code %d",
1797 prompt_init_result);
1798 break;
1799 }
1800
1801 prompt_layer_destroy(prompt_layer);
1802 command_layer_destroy(command_layer);
1803 return DISPLAY_CONTROLLER_ERROR_INITIALIZATION_FAILED;
1804 }
1805
1806 DC_DEBUG("Prompt layer initialized successfully");
1807
1808 DC_DEBUG("About to initialize command layer");
1809 DC_DEBUG("Command layer pointer: %p", (void *)command_layer);
1810 DC_DEBUG("Event system pointer: %p", (void *)controller->event_system);
1811
1812 command_layer_error_t command_init_result =
1813 command_layer_init(command_layer, controller->event_system);
1814
1815 DC_DEBUG("Command layer init returned: %d", command_init_result);
1816
1817 if (command_init_result != COMMAND_LAYER_SUCCESS) {
1818 DC_ERROR("Failed to initialize command layer: error %d",
1819 command_init_result);
1820
1821 prompt_layer_cleanup(prompt_layer);
1822 prompt_layer_destroy(prompt_layer);
1823 command_layer_destroy(command_layer);
1824 return DISPLAY_CONTROLLER_ERROR_INITIALIZATION_FAILED;
1825 }
1826
1827 DC_DEBUG("Command layer initialized successfully");
1828
1829 // ============================================================================
1830 // CRITICAL FIX: Populate layers with provided content before
1831 // composition engine init
1832 // ============================================================================
1833
1834 DC_DEBUG("Layer content populated successfully");
1835
1836 // Initialize composition engine with populated layers
1837 composition_engine_error_t comp_result =
1838 composition_engine_init(controller->compositor, prompt_layer,
1839 command_layer, controller->event_system);
1840
1841 if (comp_result != COMPOSITION_ENGINE_SUCCESS) {
1842 DC_ERROR("Failed to initialize composition engine: %s",
1843 composition_engine_error_string(comp_result));
1844 prompt_layer_cleanup(prompt_layer);
1845 prompt_layer_destroy(prompt_layer);
1846 command_layer_cleanup(command_layer);
1847 command_layer_destroy(command_layer);
1848 return DISPLAY_CONTROLLER_ERROR_COMPOSITION_FAILED;
1849 }
1850 }
1851
1852 // ============================================================================
1853 // CRITICAL BUG FIX: Always update layer content with new input
1854 // ============================================================================
1855
1856 // Get layers from compositor (they are public fields)
1857 prompt_layer_t *prompt_layer = controller->compositor->prompt_layer;
1858 command_layer_t *command_layer = controller->compositor->command_layer;
1859
1860 if (!prompt_layer || !command_layer) {
1861
1862 return DISPLAY_CONTROLLER_ERROR_COMPOSITION_FAILED;
1863 }
1864
1865 // Always update layer content with new input (this was the missing piece!)
1866
1867 if (prompt_text && *prompt_text && prompt_layer) {
1868
1869 prompt_layer_error_t prompt_content_result =
1870 prompt_layer_set_content(prompt_layer, prompt_text);
1871 if (prompt_content_result != PROMPT_LAYER_SUCCESS) {
1872
1873 return DISPLAY_CONTROLLER_ERROR_COMPOSITION_FAILED;
1874 }
1875 }
1876
1877 // Always update command content with new input
1878 if (command_text && *command_text && command_layer) {
1879
1880 command_layer_error_t command_content_result =
1881 command_layer_set_command(command_layer, command_text, 0);
1882 if (command_content_result != COMMAND_LAYER_SUCCESS) {
1883
1884 return DISPLAY_CONTROLLER_ERROR_COMPOSITION_FAILED;
1885 }
1886 }
1887
1888 // Perform composition
1889 composition_engine_error_t comp_result =
1890 composition_engine_compose(controller->compositor);
1891 if (comp_result != COMPOSITION_ENGINE_SUCCESS) {
1892 DC_ERROR("Composition failed: %s",
1893 composition_engine_error_string(comp_result));
1894 return DISPLAY_CONTROLLER_ERROR_COMPOSITION_FAILED;
1895 }
1896
1897 // Get composed output
1898 comp_result = composition_engine_get_output(controller->compositor, output,
1899 output_size);
1900 if (comp_result != COMPOSITION_ENGINE_SUCCESS) {
1901 DC_ERROR("Failed to get composed output: %s",
1902 composition_engine_error_string(comp_result));
1903 return DISPLAY_CONTROLLER_ERROR_COMPOSITION_FAILED;
1904 }
1905
1906 // Fix cursor positioning: strip trailing newlines from composed output
1907 size_t output_length = strlen(output);
1908 while (output_length > 0 && output[output_length - 1] == '\n') {
1909 output[output_length - 1] = '\0';
1910 output_length--;
1911 }
1912
1913 // Update cache if enabled
1914 if (controller->config.enable_caching) {
1915 display_controller_error_t cache_result =
1916 dc_add_cache_entry(controller, state_hash, output, output_length);
1917 if (cache_result != DISPLAY_CONTROLLER_SUCCESS) {
1918 DC_DEBUG("Failed to add cache entry: %d", cache_result);
1919 // Non-fatal error, continue
1920 }
1921 }
1922
1923 // Update last display state
1924 if (controller->last_display_state) {
1925 free(controller->last_display_state);
1926 }
1927 controller->last_display_state = malloc(output_length + 1);
1928 if (controller->last_display_state) {
1929 memcpy(controller->last_display_state, output, output_length + 1);
1930 controller->last_display_length = output_length;
1931 strcpy(controller->current_state_hash, state_hash);
1932 controller->display_cache_valid = true;
1933 }
1934
1935 // Update performance metrics
1936 gettimeofday(&end_time, NULL);
1937 uint64_t operation_time = dc_time_diff_ns(&start_time, &end_time);
1938
1939 // Performance Monitoring: Record display timing for new
1940 // compositions
1941 display_integration_record_display_timing(operation_time);
1942
1943 controller->performance.total_display_operations++;
1944 if (controller->performance.total_display_operations == 1) {
1945 controller->performance.min_display_time_ns = operation_time;
1946 controller->performance.max_display_time_ns = operation_time;
1947 controller->performance.avg_display_time_ns = operation_time;
1948 } else {
1949 if (operation_time < controller->performance.min_display_time_ns) {
1950 controller->performance.min_display_time_ns = operation_time;
1951 }
1952 if (operation_time > controller->performance.max_display_time_ns) {
1953 controller->performance.max_display_time_ns = operation_time;
1954 }
1955 controller->performance.avg_display_time_ns =
1956 dc_calculate_average_performance(controller);
1957 }
1958
1959 dc_update_performance_history(controller, operation_time);
1960
1961 // Update cache hit rate
1962 if (controller->performance.total_display_operations > 0) {
1963 controller->performance.cache_hit_rate =
1964 (double)controller->performance.cache_hits /
1965 (double)controller->performance.total_display_operations;
1966 }
1967
1968 // Check if cache cleanup is needed
1969 struct timeval current_time;
1970 gettimeofday(&current_time, NULL);
1971 uint64_t cleanup_interval_ms =
1972 ((uint64_t)current_time.tv_sec -
1973 (uint64_t)controller->last_cache_cleanup.tv_sec) *
1974 1000 +
1975 ((uint64_t)current_time.tv_usec -
1976 (uint64_t)controller->last_cache_cleanup.tv_usec) /
1977 1000;
1978
1979 if (cleanup_interval_ms > DC_CACHE_CLEANUP_INTERVAL_MS) {
1980 dc_cleanup_expired_cache_entries(controller);
1981 controller->last_cache_cleanup = current_time;
1982 }
1983
1984 DC_DEBUG("Display operation completed (time: %" PRIu64 " ns)",
1985 operation_time);
1986 return DISPLAY_CONTROLLER_SUCCESS;

2097 }
2098 } else {
2099 DC_DEBUG("No prompt text provided");
2100 }
2101
2102 // Set command content if provided
2103 if (command_text && *command_text) {
2104 DC_DEBUG("Setting command content...");
2105 command_layer_error_t cmd_result = command_layer_set_command(

2178

2184 offset += clear_len;

2187 // Walk through composed output character-by-character and insert newlines
2188 // when wrapping
2189 const char *composed = comp_result.composed_output;
2190 size_t composed_len = strlen(composed);
2191 int visual_col = 0;
2192 bool in_escape = false;
2193
2194 for (size_t i = 0; i < composed_len; i++) {
2195 char ch = composed[i];
2196
2197 // Track ANSI escape sequences (they don't consume visual space)
2198 if (ch == '\033' || ch == '\x1b') {
2199 in_escape = true;
2200 }
2201
2202 // Check if we need to wrap BEFORE writing this character
2203 // (Only if not in escape sequence and not already a newline)
2204 if (!in_escape && ch != '\n' && ch != '\r' &&
2205 visual_col >= terminal_width) {
2206 // Insert newline to wrap to next line at column 0
2207 if (offset + 1 >= output_size) {
2208 return DISPLAY_CONTROLLER_ERROR_BUFFER_TOO_SMALL;
2209 }
2210 output[offset++] = '\n';
2211 visual_col = 0;
2212 }
2213
2214 // Write the character
2215 if (offset + 1 >= output_size) {
2216 return DISPLAY_CONTROLLER_ERROR_BUFFER_TOO_SMALL;
2217 }
2218 output[offset++] = ch;
2219
2220 // Update visual column position
2221 if (in_escape) {
2222 // Check for end of escape sequence
2223 if (ch == 'm' || ch == 'K' || ch == 'J' || ch == 'H' || ch == 'A' ||
2224 ch == 'B' || ch == 'C' || ch == 'D' || ch == 'G') {
2225 in_escape = false;
2226 }
2227 // Escape sequences don't advance visual column
2228 } else if (ch == '\n' || ch == '\r') {
2229 visual_col = 0;
2230 } else if (ch == '\t') {
2231 // Tab advances to next multiple of 8
2232 visual_col += 8 - (visual_col % 8);
2233 } else if ((unsigned char)ch >= 32) {
2234 // Printable character advances by 1
2235 // TODO: Handle wide characters (CJK) which advance by 2
2236 visual_col++;
2237 }
2238 }
2239
2240 // Step 3: Add cursor positioning
2241 // Convert from 0-based (composition) to 1-based (ANSI)
2242 int target_row = comp_result.cursor_screen_row + 1;
2243 int target_column = comp_result.cursor_screen_column + 1;
2244
2245 // Generate cursor positioning sequence using Layer 2 (terminal_control)

2248 tc, target_row, target_column, cursor_seq, sizeof(cursor_seq));
2249
2250 if (cursor_len < 0) {
2251 return DISPLAY_CONTROLLER_ERROR_COMPOSITION_FAILED;
2252 }
2253
2254 if (offset + cursor_len >= output_size) {
2255 return DISPLAY_CONTROLLER_ERROR_BUFFER_TOO_SMALL;
2256 }
2257
2258 memcpy(output + offset, cursor_seq, cursor_len);
2259 offset += cursor_len;
2260
2261 // Null-terminate
2262 output[offset] = '\0';
2263
2264 DC_DEBUG("Display with cursor completed: row=%d, col=%d, total output "
2265 "length=%zu",
2266 target_row, target_column, offset);
2267 DC_DEBUG("Output buffer first 100 chars: %.100s", output);
2268 return DISPLAY_CONTROLLER_SUCCESS;

2797 size_t expired_count = 0;
2798 bool corruption_found = false;

2800 if (controller->cache_entries) {

2817 (uint64_t)entry->timestamp.tv_usec) /

2821 expired_count++;

2828 corruption_found = true;
2829 }
2830 }
2831 }

2834 *valid_entries = valid_count;
2835 if (expired_entries)
