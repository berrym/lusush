16/* Get current time in microseconds */
17static uint64_t get_current_time_us(void) {
18 struct timespec ts;
19 clock_gettime(CLOCK_MONOTONIC, &ts);
20 return (uint64_t)ts.tv_sec * 1000000 + (uint64_t)ts.tv_nsec / 1000;
21}

158/*
159 * Initialize key sequence map
160 */
161static lle_result_t init_key_sequence_map(lle_key_sequence_map_t **map,
162 lle_memory_pool_t *memory_pool) {
163 if (!map || !memory_pool) {
164 return LLE_ERROR_INVALID_PARAMETER;
165 }
166
167 lle_key_sequence_map_t *new_map =
168 lle_pool_alloc(sizeof(lle_key_sequence_map_t));
169 if (!new_map) {
170 return LLE_ERROR_OUT_OF_MEMORY;
171 }
172
173 // Allocate entries array
174 new_map->capacity = KEY_MAPPING_COUNT;
175 new_map->entry_count = 0;
176 new_map->memory_pool = memory_pool;
177
178 new_map->entries = lle_pool_alloc(sizeof(lle_key_sequence_map_entry_t) *
179 new_map->capacity);
180 if (!new_map->entries) {
181 lle_pool_free(new_map);
182 return LLE_ERROR_OUT_OF_MEMORY;
183 }
184
185 // Populate map from static table
186 for (size_t i = 0; i < KEY_MAPPING_COUNT; i++) {
187 const key_mapping_entry_t *src = &key_mappings[i];
188 lle_key_sequence_map_entry_t *dst = &new_map->entries[i];
189
190 // Copy sequence
191 if (src->sequence_len > LLE_MAX_KEY_SEQUENCE_LENGTH) {
192 continue; // Skip if too long
193 }
194
195 memcpy(dst->sequence, src->sequence, src->sequence_len);
196 dst->sequence_length = src->sequence_len;
197
198 // Fill key info
199 dst->key_info.type = src->type;
200 dst->key_info.keycode = src->keycode;
201 dst->key_info.modifiers = src->modifiers;
202 strncpy(dst->key_info.key_name, src->key_name,
203 sizeof(dst->key_info.key_name) - 1);
204 dst->key_info.key_name[sizeof(dst->key_info.key_name) - 1] = '\0';
205 dst->key_info.is_repeat = false;
206 dst->key_info.timestamp = 0;
207
208 new_map->entry_count++;
209 }
210
211 *map = new_map;
212 return LLE_SUCCESS;
213}

215/*
216 * Match sequence to key
217 */
218static lle_key_sequence_match_type_t
219match_sequence(const lle_key_sequence_map_t *map, const char *sequence,
220 size_t sequence_len, lle_key_info_t *key_info) {
221 if (!map || !sequence || sequence_len == 0) {
222 return LLE_MATCH_NONE;
223 }
224
225 bool has_exact_match = false;
226 bool has_prefix_match = false;
227 lle_key_sequence_map_entry_t *exact_entry = NULL;
228
229 // Scan for matches
230 for (size_t i = 0; i < map->entry_count; i++) {
231 lle_key_sequence_map_entry_t *entry = &map->entries[i];
232
233 if (entry->sequence_length == sequence_len) {
234 // Check for exact match
235 if (memcmp(entry->sequence, sequence, sequence_len) == 0) {
236 has_exact_match = true;
237 exact_entry = entry;
238 // Don't break - check if there are longer sequences with this
239 // as prefix
240 }
241 } else if (entry->sequence_length > sequence_len) {
242 // Check if current sequence is a prefix
243 if (memcmp(entry->sequence, sequence, sequence_len) == 0) {
244 has_prefix_match = true;
245 }
246 }
247 }
248
249 if (has_exact_match) {
250 if (has_prefix_match) {
251 // Ambiguous - exact match but also prefix of longer sequences
252 if (key_info && exact_entry) {
253 memcpy(key_info, &exact_entry->key_info,
254 sizeof(lle_key_info_t));
255 }
256 return LLE_MATCH_AMBIGUOUS;
257 } else {
258 // Exact match, no ambiguity
259 if (key_info && exact_entry) {
260 memcpy(key_info, &exact_entry->key_info,
261 sizeof(lle_key_info_t));
262 }
263 return LLE_MATCH_EXACT;
264 }
265 } else if (has_prefix_match) {
266 // Prefix match - need more data
267 return LLE_MATCH_PREFIX;
268 }
269
270 return LLE_MATCH_NONE;
271}

303 }

328/*
329 * Check if sequence has timed out
330 */
331static bool has_timed_out(lle_key_detector_t *detector) {
332 if (detector->sequence_start_time == 0) {
333 return false;
334 }
335
336 uint64_t current_time = get_current_time_us();
337 uint64_t elapsed = current_time - detector->sequence_start_time;
338
339 return elapsed > LLE_KEY_SEQUENCE_TIMEOUT_US;
340}

342/*
343 * Process key sequence
344 */
345lle_result_t lle_key_detector_process_sequence(lle_key_detector_t *detector,
346 const char *sequence,
347 size_t sequence_len,
348 lle_key_info_t **key_info) {
349 if (!detector || !sequence || !key_info) {
350 return LLE_ERROR_INVALID_PARAMETER;
351 }
352
353 *key_info = NULL;
354
355 lle_key_sequence_match_result_t result;
356 memset(&result, 0, sizeof(lle_key_sequence_match_result_t));
357
358 // Add sequence to buffer
359 if (detector->sequence_pos + sequence_len > LLE_MAX_KEY_SEQUENCE_LENGTH) {
360 // Buffer overflow - reset
361 detector->sequence_pos = 0;
362 detector->sequence_start_time = 0;
363 detector->ambiguous_sequence = false;
364 return LLE_SUCCESS;
365 }
366
367 memcpy(detector->sequence_buffer + detector->sequence_pos, sequence,
368 sequence_len);
369 detector->sequence_pos += sequence_len;
370
371 // Start timing if first byte
372 if (detector->sequence_pos == sequence_len) {
373 detector->sequence_start_time = get_current_time_us();
374 }
375
376 // Try to match
377 lle_key_info_t matched_key;
378 lle_key_sequence_match_type_t match_type =
379 match_sequence(detector->sequence_map, detector->sequence_buffer,
380 detector->sequence_pos, &matched_key);
381
382 switch (match_type) {
383 case LLE_MATCH_EXACT:
384 // Exact match found - allocate and return key info
385 *key_info = lle_pool_alloc(sizeof(lle_key_info_t));
386 if (*key_info) {
387 memcpy(*key_info, &matched_key, sizeof(lle_key_info_t));
388 (*key_info)->timestamp = get_current_time_us();
389 memcpy((*key_info)->sequence, detector->sequence_buffer,
390 detector->sequence_pos);
391 (*key_info)->sequence_length = detector->sequence_pos;
392 }
393
394 detector->sequences_detected++;
395 detector->sequences_resolved++;
396
397 // Reset buffer
398 detector->sequence_pos = 0;
399 detector->sequence_start_time = 0;
400 detector->ambiguous_sequence = false;
401 break;
402
403 case LLE_MATCH_AMBIGUOUS:
404 // Ambiguous - could match or could be prefix
405 detector->ambiguous_sequence = true;
406 detector->sequences_detected++;
407
408 // Check for timeout
409 if (has_timed_out(detector)) {
410 // Timeout - accept the match
411 *key_info = lle_pool_alloc(sizeof(lle_key_info_t));
412 if (*key_info) {
413 memcpy(*key_info, &matched_key, sizeof(lle_key_info_t));
414 (*key_info)->timestamp = get_current_time_us();
415 memcpy((*key_info)->sequence, detector->sequence_buffer,
416 detector->sequence_pos);
417 (*key_info)->sequence_length = detector->sequence_pos;
418 }
419
420 detector->ambiguous_timeouts++;
421 detector->sequences_resolved++;
422
423 // Reset buffer
424 detector->sequence_pos = 0;
425 detector->sequence_start_time = 0;
426 detector->ambiguous_sequence = false;
427 }
428 // Otherwise keep waiting for more data or timeout
429 break;
430
431 case LLE_MATCH_PREFIX:
432 // Prefix match - need more data, keep buffering
433 break;
434
435 case LLE_MATCH_NONE:
436 // No match - reset
437 detector->sequence_pos = 0;
438 detector->sequence_start_time = 0;
439 detector->ambiguous_sequence = false;
440 break;
441 }
442
443 return LLE_SUCCESS;
444}
