64/*
65 * Validate input event data
66 */
67static bool validate_event(lle_input_event_t *event) {
68 if (!event) {
69 return false;
70 }
71
72 /* Validate based on event type */
73 switch (event->type) {
74 case LLE_INPUT_TYPE_CHARACTER:
75 /* Validate UTF-8 byte count */
76 if (event->data.character.byte_count == 0 ||
77 event->data.character.byte_count > 8) {
78 return false;
79 }
80 /* Validate codepoint range */
81 if (event->data.character.codepoint > 0x10FFFF) {
82 return false;
83 }
84 break;
85
86 case LLE_INPUT_TYPE_SPECIAL_KEY:
87 /* Validate key code - allow LLE_KEY_UNKNOWN if it has a valid keycode
88 * (for Ctrl+letter) */
89 if (event->data.special_key.key == LLE_KEY_UNKNOWN) {
90 /* LLE_KEY_UNKNOWN is valid if keycode is set (e.g., Ctrl+A has
91 * keycode='A') */
92 if (event->data.special_key.keycode == 0) {
93 return false; /* Invalid: UNKNOWN key with no keycode */
94 }
95 }
96 break;
97
98 case LLE_INPUT_TYPE_WINDOW_RESIZE:
99 /* Validate geometry */
100 if (event->data.resize.new_width == 0 ||
101 event->data.resize.new_height == 0) {
102 return false;
103 }
104 break;
105
106 case LLE_INPUT_TYPE_SIGNAL:
107 case LLE_INPUT_TYPE_TIMEOUT:
108 case LLE_INPUT_TYPE_ERROR:
109 case LLE_INPUT_TYPE_EOF:
110 /* These events are always valid */
111 break;
112
113 default:
114 return false;
115 }
116
117 return true;
118}

120/*
121 * Process input event (validate and track)
122 */
123lle_result_t lle_input_processor_process_event(lle_input_processor_t *processor,
124 lle_input_event_t *event) {
125 if (!processor || !event) {
126 return LLE_ERROR_INVALID_PARAMETER;
127 }
128
129 uint64_t processing_start = lle_get_current_time_microseconds();
130
131 /* Validate event */
132 if (!validate_event(event)) {
133 return LLE_ERROR_INVALID_INPUT_EVENT;
134 }
135
136 /* Assign sequence number */
137 event->sequence_number = processor->next_sequence_number++;
138
139 /* Update processing statistics */
140 processor->events_processed++;
141 uint64_t processing_time =
142 lle_get_current_time_microseconds() - processing_start;
143 processor->total_processing_time_us += processing_time;
144
145 return LLE_SUCCESS;
146}

148/*
149 * Read next input event from Unix interface
150 */
151lle_result_t
152lle_input_processor_read_next_event(lle_input_processor_t *processor,
153 lle_input_event_t **event,
154 uint32_t timeout_ms) {
155 if (!processor || !event) {
156 return LLE_ERROR_INVALID_PARAMETER;
157 }
158
159 /* Allocate event structure */
160 lle_input_event_t *new_event = calloc(1, sizeof(lle_input_event_t));
161 if (!new_event) {
162 return LLE_ERROR_OUT_OF_MEMORY;
163 }
164
165 /* Read event from Unix interface */
166 lle_result_t result = lle_unix_interface_read_event(
167 processor->unix_interface, new_event, timeout_ms);
168 if (result != LLE_SUCCESS) {
169 free(new_event);
170 return result;
171 }
172
173 /* Process (validate and track) the event */
174 result = lle_input_processor_process_event(processor, new_event);
175 if (result != LLE_SUCCESS) {
176 free(new_event);
177 return result;
178 }
179
180 *event = new_event;
181 return LLE_SUCCESS;
182}
