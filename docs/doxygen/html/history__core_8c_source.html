34 (void)memory_pool; /* Unused - we use global pool via lle_pool_alloc */
35
36 if (!config) {

38 }
39
40 /* Allocate configuration */

42 if (!cfg) {
43 return LLE_ERROR_OUT_OF_MEMORY;
44 }
45

68 cfg->ignore_space_prefix = true; /* Standard shell behavior */
69 cfg->save_timestamps = true;

105/**
106 * Create history entry
107 */
108lle_result_t lle_history_entry_create(lle_history_entry_t **entry,
109 const char *command,
110 lle_memory_pool_t *memory_pool) {
111 (void)memory_pool; /* Unused - we use global pool via lle_pool_alloc */
112
113 if (!entry || !command) {
114 return LLE_ERROR_INVALID_PARAMETER;
115 }
116
117 /* Validate command length */
118 size_t cmd_len = strlen(command);
119 if (cmd_len == 0) {
120 return LLE_ERROR_INVALID_PARAMETER;
121 }
122 if (cmd_len > LLE_HISTORY_MAX_COMMAND_LENGTH) {
123 return LLE_ERROR_BUFFER_OVERFLOW;
124 }
125
126 /* Allocate entry */
127 lle_history_entry_t *e = lle_pool_alloc(sizeof(lle_history_entry_t));
128 if (!e) {
129 return LLE_ERROR_OUT_OF_MEMORY;
130 }
131
132 /* Initialize entry */
133 memset(e, 0, sizeof(lle_history_entry_t));
134
135 /* Copy command */
136 e->command = lle_pool_alloc(cmd_len + 1);
137 if (!e->command) {
138 lle_pool_free(e);
139 return LLE_ERROR_OUT_OF_MEMORY;
140 }
141 memcpy(e->command, command, cmd_len + 1);
142 e->command_length = cmd_len;
143
144 /* Get current timestamp */
145 struct timeval tv;
146 gettimeofday(&tv, NULL);
147 e->timestamp = (uint64_t)tv.tv_sec;
148
149 /* Get current working directory */
150 char cwd_buffer[LLE_HISTORY_MAX_PATH_LENGTH];
151 if (lle_history_get_cwd(cwd_buffer, sizeof(cwd_buffer)) == LLE_SUCCESS) {
152 size_t cwd_len = strlen(cwd_buffer);
153 e->working_directory = lle_pool_alloc(cwd_len + 1);
154 if (e->working_directory) {
155 memcpy(e->working_directory, cwd_buffer, cwd_len + 1);
156 }
157 }
158
159 /* Initialize state */
160 e->state = LLE_HISTORY_STATE_ACTIVE;
161 e->exit_code = -1; /* Unknown */
162
163 /* Phase 4 fields - initialize to defaults */
164 e->is_multiline = false;
165 e->original_multiline = NULL;
166 e->duration_ms = 0;
167 e->edit_count = 0;
168
169 /* Phase 4 Day 11: Forensic fields - initialize to defaults */
170 e->process_id = 0;
171 e->session_id = 0;
172 e->user_id = 0;
173 e->group_id = 0;
174 e->terminal_name = NULL;
175 e->start_time_ns = 0;
176 e->end_time_ns = 0;
177 e->usage_count = 0;
178 e->last_access_time = 0;
179
180 /* List pointers */
181 e->next = NULL;
182 e->prev = NULL;
183
184 *entry = e;
185 return LLE_SUCCESS;
186}

260/**
261 * Create and initialize history core
262 */
263lle_result_t lle_history_core_create(lle_history_core_t **core,
264 lle_memory_pool_t *memory_pool,
265 const lle_history_config_t *config) {
266 if (!core) {
267 return LLE_ERROR_INVALID_PARAMETER;
268 }
269
270 lle_result_t result = LLE_SUCCESS;
271
272 /* Allocate core structure */
273 lle_history_core_t *c = lle_pool_alloc(sizeof(lle_history_core_t));
274 if (!c) {
275 return LLE_ERROR_OUT_OF_MEMORY;
276 }
277 memset(c, 0, sizeof(lle_history_core_t));
278
279 /* Store memory pool reference */
280 c->memory_pool = memory_pool;
281
282 /* Create or copy configuration */
283 if (config) {
284 /* Copy provided configuration */
285 c->config = lle_pool_alloc(sizeof(lle_history_config_t));
286 if (!c->config) {
287 lle_pool_free(c);
288 return LLE_ERROR_OUT_OF_MEMORY;
289 }
290 memcpy(c->config, config, sizeof(lle_history_config_t));
291
292 /* Deep copy history_file_path if present */
293 if (config->history_file_path) {
294 size_t path_len = strlen(config->history_file_path) + 1;
295 c->config->history_file_path = lle_pool_alloc(path_len);
296 if (c->config->history_file_path) {
297 memcpy(c->config->history_file_path, config->history_file_path,
298 path_len);
299 }
300 }
301 } else {
302 /* Create default configuration */
303 result = lle_history_config_create_default(&c->config, memory_pool);
304 if (result != LLE_SUCCESS) {
305 lle_pool_free(c);
306 return result;
307 }
308 }
309
310 /* Allocate initial entry array */
311 size_t initial_cap = c->config->initial_capacity;
312 c->entries = lle_pool_alloc(sizeof(lle_history_entry_t *) * initial_cap);
313 if (!c->entries) {
314 lle_history_config_destroy(c->config, memory_pool);
315 lle_pool_free(c);
316 return LLE_ERROR_OUT_OF_MEMORY;
317 }
318 memset(c->entries, 0, sizeof(lle_history_entry_t *) * initial_cap);
319
320 c->entry_capacity = initial_cap;
321 c->entry_count = 0;
322 c->next_entry_id = 1; /* Start IDs at 1 */
323
324 /* Initialize linked list pointers */
325 c->first_entry = NULL;
326 c->last_entry = NULL;
327
328 /* Phase 1 Day 2: Create hashtable index if enabled */
329 if (c->config->use_indexing) {
330 result = lle_history_index_create(&c->entry_lookup, initial_cap);
331 if (result != LLE_SUCCESS) {
332 lle_pool_free(c->entries);
333 lle_history_config_destroy(c->config, memory_pool);
334 lle_pool_free(c);
335 return result;
336 }
337 } else {
338 c->entry_lookup = NULL;
339 }
340
341 /* Phase 4 Day 12: Create deduplication engine if configured */
342 if (c->config->ignore_duplicates) {
343 /* Use configured strategy (default: KEEP_RECENT) */
344 result = lle_history_dedup_create(&c->dedup_engine, c,
345 c->config->dedup_strategy);
346 if (result != LLE_SUCCESS) {
347 if (c->entry_lookup) {
348 lle_history_index_destroy(c->entry_lookup);
349 }
350 lle_pool_free(c->entries);
351 lle_history_config_destroy(c->config, memory_pool);
352 lle_pool_free(c);
353 return result;
354 }
355 /* Configure Unicode normalization for dedup comparison */
356 lle_history_dedup_set_unicode_normalize(c->dedup_engine,
357 c->config->unicode_normalize);
358 } else {
359 c->dedup_engine = NULL;
360 }
361
362 /* Initialize statistics */
363 memset(&c->stats, 0, sizeof(lle_history_stats_t));
364
365 /* Initialize thread safety */
366 if (pthread_rwlock_init(&c->lock, NULL) != 0) {
367 lle_pool_free(c->entries);
368 lle_history_config_destroy(c->config, memory_pool);
369 lle_pool_free(c);
370 return LLE_ERROR_INITIALIZATION_FAILED;
371 }
372
373 c->initialized = true;
374
375 *core = c;
376 return LLE_SUCCESS;
377}

379/**
380 * Destroy history core
381 */
382lle_result_t lle_history_core_destroy(lle_history_core_t *core) {
383 if (!core) {
384 return LLE_ERROR_INVALID_PARAMETER;
385 }
386
387 /* Acquire write lock */
388 pthread_rwlock_wrlock(&core->lock);
389
390 /* Mark as not initialized */
391 core->initialized = false;
392
393 /* Free all entries */
394 for (size_t i = 0; i < core->entry_count; i++) {
395 if (core->entries[i]) {
396 lle_history_entry_destroy(core->entries[i], core->memory_pool);
397 core->entries[i] = NULL;
398 }
399 }
400
401 /* Free entries array */
402 if (core->entries) {
403 lle_pool_free(core->entries);
404 }
405
406 /* Phase 1 Day 2: Destroy hashtable index if present */
407 if (core->entry_lookup) {
408 lle_history_index_destroy(core->entry_lookup);
409 core->entry_lookup = NULL;
410 }
411
412 /* Phase 4 Day 12: Destroy deduplication engine if present */
413 if (core->dedup_engine) {
414 lle_history_dedup_destroy(core->dedup_engine);
415 core->dedup_engine = NULL;
416 }
417
418 /* Destroy configuration */
419 if (core->config) {
420 lle_history_config_destroy(core->config, core->memory_pool);
421 }
422
423 /* Release lock and destroy */
424 pthread_rwlock_unlock(&core->lock);
425 pthread_rwlock_destroy(&core->lock);
426
427 /* Free core structure */
428 lle_pool_free(core);
429
430 return LLE_SUCCESS;
431}

438/**
439 * Expand entry array capacity
440 */
441lle_result_t lle_history_expand_capacity(lle_history_core_t *core) {
442 if (!core) {
443 return LLE_ERROR_INVALID_PARAMETER;
444 }
445
446 /* Calculate new capacity (double it, or use initial minimum if currently 0)
447 */
448 size_t new_capacity;
449 if (core->entry_capacity == 0) {
450 /* Handle case where initial_capacity was set to 0 - use a reasonable
451 * minimum */
452 new_capacity = 100;
453 } else {
454 new_capacity = core->entry_capacity * 2;
455 }
456
457 /* Check maximum capacity */
458 if (new_capacity > core->config->max_entries) {
459 new_capacity = core->config->max_entries;
460 }
461
462 /* Check if already at max */
463 if (core->entry_capacity >= core->config->max_entries) {
464 return LLE_ERROR_BUFFER_OVERFLOW;
465 }
466
467 /* Allocate new array */
468 lle_history_entry_t **new_entries =
469 lle_pool_alloc(sizeof(lle_history_entry_t *) * new_capacity);
470 if (!new_entries) {
471 return LLE_ERROR_OUT_OF_MEMORY;
472 }
473
474 /* Copy existing entries */
475 memcpy(new_entries, core->entries,
476 sizeof(lle_history_entry_t *) * core->entry_count);
477
478 /* Zero new space */
479 memset(new_entries + core->entry_count, 0,
480 sizeof(lle_history_entry_t *) * (new_capacity - core->entry_count));
481
482 /* Free old array */
483 lle_pool_free(core->entries);
484
485 /* Update core */
486 core->entries = new_entries;
487 core->entry_capacity = new_capacity;
488
489 return LLE_SUCCESS;
490}

492/**
493 * Add entry to history
494 */
495lle_result_t lle_history_add_entry(lle_history_core_t *core,
496 const char *command, int exit_code,
497 uint64_t *entry_id) {
498 if (!core || !command) {
499 return LLE_ERROR_INVALID_PARAMETER;
500 }
501
502 if (!core->initialized) {
503 return LLE_ERROR_NOT_INITIALIZED;
504 }
505
506 /* Check for space prefix (ignore if configured) */
507 if (core->config->ignore_space_prefix && command[0] == ' ') {
508 return LLE_SUCCESS; /* Silently ignore */
509 }
510
511 /* Acquire write lock */
512 pthread_rwlock_wrlock(&core->lock);
513
514 lle_result_t result = LLE_SUCCESS;
515
516 /* Start performance measurement */
517 struct timeval start_time;
518 gettimeofday(&start_time, NULL);
519
520 /* Check if array is full */
521 if (core->entry_count >= core->entry_capacity) {
522 result = lle_history_expand_capacity(core);
523 if (result != LLE_SUCCESS) {
524 pthread_rwlock_unlock(&core->lock);
525 return result;
526 }
527 }
528
529 /* Create entry */
530 lle_history_entry_t *entry = NULL;
531 result = lle_history_entry_create(&entry, command, core->memory_pool);
532 if (result != LLE_SUCCESS) {
533 pthread_rwlock_unlock(&core->lock);
534 return result;
535 }
536
537 /* Assign entry ID */
538 entry->entry_id = core->next_entry_id++;
539 entry->exit_code = exit_code;
540
541 /* Phase 4 Day 11: Capture forensic context */
542 lle_forensic_context_t forensic_ctx;
543 if (lle_forensic_capture_context(&forensic_ctx) == LLE_SUCCESS) {
544 lle_forensic_apply_to_entry(entry, &forensic_ctx);
545 lle_forensic_free_context(&forensic_ctx);
546 }
547
548 /* Phase 4 Day 12: Check for duplicates if dedup engine is enabled */
549 if (core->dedup_engine) {
550 bool entry_rejected = false;
551 result =
552 lle_history_dedup_apply(core->dedup_engine, entry, &entry_rejected);
553
554 if (result != LLE_SUCCESS) {
555 lle_history_entry_destroy(entry, core->memory_pool);
556 pthread_rwlock_unlock(&core->lock);
557 return result;
558 }
559
560 if (entry_rejected) {
561 /* Duplicate was rejected - clean up and return success */
562 lle_history_entry_destroy(entry, core->memory_pool);
563 pthread_rwlock_unlock(&core->lock);
564
565 /* Optionally return the ID of the existing entry if requested */
566 if (entry_id) {
567 *entry_id = 0; /* Indicate entry was not added */
568 }
569
570 return LLE_SUCCESS;
571 }
572 }
573
574 /* Add to array */
575 core->entries[core->entry_count] = entry;
576
577 /* Update linked list */
578 if (core->last_entry) {
579 core->last_entry->next = entry;
580 entry->prev = core->last_entry;
581 } else {
582 core->first_entry = entry;
583 }
584 core->last_entry = entry;
585
586 core->entry_count++;
587
588 /* Phase 1 Day 2: Add to hashtable index if enabled */
589 if (core->entry_lookup) {
590 result = lle_history_index_insert(core->entry_lookup, entry->entry_id,
591 entry);
592 if (result != LLE_SUCCESS) {
593 /* Rollback: remove from array and linked list */
594 core->entry_count--;
595 core->entries[core->entry_count] = NULL;
596 if (entry->prev) {
597 entry->prev->next = NULL;
598 core->last_entry = entry->prev;
599 } else {
600 core->first_entry = NULL;
601 core->last_entry = NULL;
602 }
603 lle_history_entry_destroy(entry, core->memory_pool);
604 pthread_rwlock_unlock(&core->lock);
605 return result;
606 }
607 }
608
609 /* Update statistics */
610 core->stats.total_entries++;
611 core->stats.active_entries++;
612 core->stats.add_count++;
613
614 /* End performance measurement */
615 struct timeval end_time;
616 gettimeofday(&end_time, NULL);
617 uint64_t elapsed_us = (end_time.tv_sec - start_time.tv_sec) * 1000000 +
618 (end_time.tv_usec - start_time.tv_usec);
619 core->stats.total_add_time_us += elapsed_us;
620
621 /* Return entry ID if requested */
622 if (entry_id) {
623 *entry_id = entry->entry_id;
624 }
625
626 pthread_rwlock_unlock(&core->lock);
627
628 return LLE_SUCCESS;
629}

707/**
708 * Get entry by ID (Phase 1: Linear search, Phase 2: Hashtable)
709 */
710lle_result_t lle_history_get_entry_by_id(lle_history_core_t *core,
711 uint64_t entry_id,
712 lle_history_entry_t **entry) {
713 if (!core || !entry) {
714 return LLE_ERROR_INVALID_PARAMETER;
715 }
716
717 if (!core->initialized) {
718 return LLE_ERROR_NOT_INITIALIZED;
719 }
720
721 /* Acquire read lock */
722 pthread_rwlock_rdlock(&core->lock);
723
724 /* Start performance measurement */
725 struct timeval start_time;
726 gettimeofday(&start_time, NULL);
727
728 /* Phase 1 Day 2: Use hashtable lookup if available, otherwise linear search
729 */
730 lle_history_entry_t *found = NULL;
731 lle_result_t lookup_result;
732
733 if (core->entry_lookup) {
734 /* O(1) hashtable lookup */
735 lookup_result =
736 lle_history_index_lookup(core->entry_lookup, entry_id, &found);
737 (void)lookup_result; /* Lookup returns success even if not found (found
738 will be NULL) */
739 } else {
740 /* O(n) linear search fallback */
741 for (size_t i = 0; i < core->entry_count; i++) {
742 if (core->entries[i]->entry_id == entry_id) {
743 found = core->entries[i];
744 break;
745 }
746 }
747 }
748
749 /* Update statistics */
750 core->stats.retrieve_count++;
751
752 /* End performance measurement */
753 struct timeval end_time;
754 gettimeofday(&end_time, NULL);
755 uint64_t elapsed_us = (end_time.tv_sec - start_time.tv_sec) * 1000000 +
756 (end_time.tv_usec - start_time.tv_usec);
757 core->stats.total_retrieve_time_us += elapsed_us;
758
759 pthread_rwlock_unlock(&core->lock);
760
761 if (!found) {
762 return LLE_ERROR_NOT_FOUND;
763 }
764
765 *entry = found;
766 return LLE_SUCCESS;
767}
