105/**
106 * Calculate relevance score for a search result
107 *
108 * @param command Command string
109 * @param query Search query
110 * @param match_position Position of match in command
111 * @param entry_index Index in history (for recency)
112 * @param total_entries Total entries in history
113 * @param match_type Type of match
114 * @return Relevance score (higher = better)
115 */
116static int calculate_score(const char *command, const char *query,
117 size_t match_position, size_t entry_index,
118 size_t total_entries, lle_search_type_t match_type) {
119 int score = 0;
120
121 /* Base score by match type */
122 switch (match_type) {
123 case LLE_SEARCH_TYPE_EXACT:
124 score += SCORE_EXACT_MATCH;
125 break;
126 case LLE_SEARCH_TYPE_PREFIX:
127 score += SCORE_PREFIX_MATCH;
128 break;
129 case LLE_SEARCH_TYPE_SUBSTRING:
130 score += SCORE_SUBSTRING_MATCH;
131 break;
132 case LLE_SEARCH_TYPE_FUZZY:
133 score += SCORE_FUZZY_MATCH;
134 break;
135 }
136
137 /* Recency bonus: more recent commands score higher */
138 /* entry_index near total_entries = recent */
139 size_t recency = total_entries - entry_index;
140 score += (int)(recency * SCORE_RECENCY_WEIGHT / 100);
141
142 /* Position bonus: matches earlier in command score higher */
143 if (match_position == 0) {
144 score += SCORE_POSITION_WEIGHT;
145 }
146
147 /* Length bonus: prefer shorter commands (more specific) */
148 size_t cmd_len = strlen(command);
149 size_t query_len = strlen(query);
150 if (cmd_len > 0 && query_len > 0) {
151 int length_ratio = (int)((query_len * 100) / cmd_len);
152 score += length_ratio; /* Higher ratio = query is larger portion of
153 command */
154 }
155
156 return score;
157}

159/**
160 * Compare function for qsort - sort results by score (descending)
161 */
162static int compare_results_by_score(const void *a, const void *b) {
163 const lle_search_result_t *r1 = (const lle_search_result_t *)a;
164 const lle_search_result_t *r2 = (const lle_search_result_t *)b;
165
166 /* Sort by score (descending) */
167 if (r1->score > r2->score)
168 return -1;
169 if (r1->score < r2->score)
170 return 1;
171
172 /* Tie-breaker: more recent entry wins */
173 if (r1->entry_index > r2->entry_index)
174 return -1;
175 if (r1->entry_index < r2->entry_index)
176 return 1;
177
178 return 0;
179}

181/**
182 * Case-insensitive substring search
183 *
184 * @param haystack String to search in
185 * @param needle String to search for
186 * @return Position of match, or NULL if not found
187 */
188static const char *stristr(const char *haystack, const char *needle) {
189 if (!haystack || !needle)
190 return NULL;
191 if (*needle == '\0')
192 return haystack;
193
194 size_t needle_len = strlen(needle);
195
196 for (const char *p = haystack; *p != '\0'; p++) {
197 if (strncasecmp(p, needle, needle_len) == 0) {
198 return p;
199 }
200 }
201
202 return NULL;
203}

324/**
325 * Search history for exact command match
326 *
327 * @param history_core History core engine
328 * @param query Search query
329 * @param max_results Maximum results to return (0 = default)
330 * @return Search results or NULL on failure
331 */
332lle_history_search_results_t *
333lle_history_search_exact(lle_history_core_t *history_core, const char *query,
334 size_t max_results) {
335 if (!history_core || !query) {
336 return NULL;
337 }
338
339 struct timespec start_time;
340 clock_gettime(CLOCK_MONOTONIC, &start_time);
341
342 /* Create results container */
343 lle_history_search_results_t *results =
344 lle_history_search_results_create(max_results);
345 if (!results) {
346 return NULL;
347 }
348
349 /* Store query */
350 results->query = pool_strdup(query);
351 results->search_type = LLE_SEARCH_TYPE_EXACT;
352
353 /* Get total entry count */
354 size_t total_entries = 0;
355 if (lle_history_get_entry_count(history_core, &total_entries) !=
356 LLE_SUCCESS) {
357 lle_history_search_results_destroy(results);
358 return NULL;
359 }
360
361 /* Search backward through history (most recent first) */
362 for (size_t i = total_entries; i > 0; i--) {
363 size_t index = i - 1;
364
365 /* Get entry */
366 lle_history_entry_t *entry = NULL;
367 if (lle_history_get_entry_by_index(history_core, index, &entry) !=
368 LLE_SUCCESS ||
369 !entry) {
370 continue;
371 }
372 if (!entry || !entry->command) {
373 continue;
374 }
375
376 /* Check for exact match */
377 if (strcmp(entry->command, query) == 0) {
378 int score = calculate_score(
379 entry->command, query, 0, /* Exact match = position 0 */
380 index, total_entries, LLE_SEARCH_TYPE_EXACT);
381
382 add_search_result(results, entry->entry_id, index, entry->command,
383 entry->timestamp, score, 0,
384 LLE_SEARCH_TYPE_EXACT);
385
386 /* Check if we've hit max results */
387 if (results->count >= results->capacity) {
388 break;
389 }
390 }
391 }
392
393 /* Sort results by score */
394 lle_history_search_results_sort(results);
395
396 /* Record search time */
397 struct timespec end_time;
398 clock_gettime(CLOCK_MONOTONIC, &end_time);
399 results->search_time_us =
400 (uint64_t)((end_time.tv_sec - start_time.tv_sec) * 1000000 +
401 (end_time.tv_nsec - start_time.tv_nsec) / 1000);
402
403 return results;
404}
