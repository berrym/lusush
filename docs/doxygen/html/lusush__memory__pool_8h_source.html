25/**
26 * Memory Pool System - Enterprise-Grade Memory Management
27 *
28 * Designed for high-frequency allocation patterns in display operations,
29 * cache management, and composition engine. Provides automatic fallback
30 * to malloc for robustness and comprehensive memory usage tracking.
31 */
32
33// Memory pool size categories
34typedef enum {
35 LUSUSH_POOL_SMALL = 0, // 128B - state hashes, small strings, cache keys
36 LUSUSH_POOL_MEDIUM = 1, // 512B - prompts, short outputs, command strings
37 LUSUSH_POOL_LARGE =
38 2, // 4KB - display outputs, compositions, multiline inputs
39 LUSUSH_POOL_XLARGE =
40 3, // 16KB - tab completions, large buffers, complex outputs
41 LUSUSH_POOL_COUNT = 4
42} lusush_pool_size_t;
43
44// Pool block structure for memory management
45typedef struct lusush_pool_block {
46 void *memory; // Allocated memory block
47 size_t size; // Block size
48 bool in_use; // Allocation status
49 uint64_t allocation_time_us; // Allocation timestamp for debugging
50 uint32_t allocation_id; // Unique allocation ID for tracking
51 struct lusush_pool_block *next; // Next block in free list
52 struct lusush_pool_block *prev; // Previous block for efficient removal
53} lusush_pool_block_t;

55// Memory pool statistics for performance monitoring
56typedef struct {
57 uint64_t total_allocations; // Total allocation requests
58 uint64_t pool_hits; // Successful pool allocations
59 uint64_t pool_misses; // Fallback to malloc count
60 uint64_t current_pool_usage; // Current bytes allocated from pools
61 uint64_t peak_pool_usage; // Maximum pool usage recorded
62 uint64_t malloc_fallbacks; // Count of malloc fallback calls
63 uint64_t total_bytes_allocated; // Total bytes allocated (pool + malloc)
64 double pool_hit_rate; // Pool allocation success rate
65 uint64_t avg_allocation_time_ns; // Average allocation time
66 uint32_t active_allocations; // Current active allocation count
67} lusush_pool_stats_t;

69// Individual pool configuration and state
70typedef struct {
71 size_t block_size; // Size of each block in this pool
72 size_t initial_blocks; // Initial number of blocks to pre-allocate
73 size_t max_blocks; // Maximum blocks allowed (0 = unlimited)
74 size_t current_blocks; // Current number of allocated blocks
75 size_t free_blocks; // Number of available blocks
76 lusush_pool_block_t *free_list; // Linked list of free blocks

78 uint64_t pool_allocations; // Allocations from this specific pool
79 uint64_t pool_deallocations; // Deallocations to this specific pool
80} lusush_pool_t;
