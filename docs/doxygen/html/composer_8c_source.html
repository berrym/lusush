28/**
29 * @brief Cached terminal color capabilities
30 *
31 * Detected once at first use and cached for performance.
32 */
33static struct {
34 bool initialized;
35 bool has_256_color;
36 bool has_true_color;
37} g_terminal_color_caps = {false, false, false};
38
39/**
40 * @brief Get cached terminal color capabilities
41 */
42static void get_terminal_color_capabilities(bool *has_256, bool *has_true) {
43 if (!g_terminal_color_caps.initialized) {
44 /* Detect terminal capabilities.
45 * Note: The optimized detection returns a cached result that is
46 * managed by the detection system - do NOT destroy it. */
47 lle_terminal_detection_result_t *detection = NULL;
48 lle_result_t result =
49 lle_detect_terminal_capabilities_optimized(&detection);
50
51 if (result == LLE_SUCCESS && detection) {
52 g_terminal_color_caps.has_256_color =
53 detection->supports_256_colors;
54 g_terminal_color_caps.has_true_color =
55 detection->supports_truecolor;
56 } else {
57 /* Default to 256 colors if detection fails */
58 g_terminal_color_caps.has_256_color = true;
59 g_terminal_color_caps.has_true_color = false;
60 }
61 g_terminal_color_caps.initialized = true;
62 }
63
64 if (has_256)
65 *has_256 = g_terminal_color_caps.has_256_color;
66 if (has_true)
67 *has_true = g_terminal_color_caps.has_true_color;
68}

83/**
84 * @brief Get the semantic color for a segment from theme
85 *
86 * Maps segment names to their default semantic colors in the theme.
87 * This provides automatic coloring for built-in themes while still
88 * allowing users to override with explicit ${color:...} syntax.
89 *
90 * @param theme Active theme (may be NULL)
91 * @param segment_name Name of the segment
92 * @return Pointer to the color, or NULL if no mapping exists
93 */
94static const lle_color_t *get_segment_color(const lle_theme_t *theme,
95 const char *segment_name) {
96 if (!theme || !segment_name) {
97 return NULL;
98 }
99
100 /* Map segment names to semantic theme colors */
101 if (strcmp(segment_name, "user") == 0) {
102 return &theme->colors.primary;
103 } else if (strcmp(segment_name, "host") == 0) {
104 return &theme->colors.secondary;
105 } else if (strcmp(segment_name, "directory") == 0) {
106 return &theme->colors.path_normal;
107 } else if (strcmp(segment_name, "git") == 0) {
108 return &theme->colors.git_branch;
109 } else if (strcmp(segment_name, "status") == 0) {
110 return &theme->colors.status_error;
111 } else if (strcmp(segment_name, "jobs") == 0) {
112 return &theme->colors.warning;
113 } else if (strcmp(segment_name, "time") == 0) {
114 return &theme->colors.text_dim;
115 } else if (strcmp(segment_name, "symbol") == 0) {
116 return &theme->colors.primary;
117 }
118
119 return NULL;
120}

127/**
128 * @brief Get segment content for template rendering
129 *
130 * Renders the segment and automatically wraps with theme color if available.
131 * This provides colored prompts by default for built-in themes, while users
132 * can override by using explicit ${color:...} syntax in custom templates.
133 *
134 * @param segment_name Name of segment to render
135 * @param property Property name (NULL for full segment)
136 * @param user_data Composer callback context
137 * @return Rendered content (caller must free) or NULL
138 */
139static char *composer_get_segment(const char *segment_name,
140 const char *property, void *user_data) {
141 composer_callback_ctx_t *ctx = user_data;
142 if (!ctx || !ctx->composer || !segment_name) {
143 return NULL;
144 }
145
146 lle_prompt_composer_t *composer = ctx->composer;
147
148 /* Find segment in registry */
149 lle_prompt_segment_t *segment =
150 lle_segment_registry_find(composer->segments, segment_name);
151 if (!segment) {
152 return NULL;
153 }
154
155 /* Handle property access */
156 if (property && segment->get_property) {
157 const char *value = segment->get_property(segment, property);
158 if (value) {
159 return strdup(value);
160 }
161 return NULL;
162 }
163
164 /* Render full segment */
165 lle_segment_output_t output;
166 memset(&output, 0, sizeof(output));
167
168 if (segment->render) {
169 lle_result_t result =
170 segment->render(segment, &composer->context, ctx->theme, &output);
171 if (result != LLE_SUCCESS || output.is_empty) {
172 return NULL;
173 }
174
175 /* Check if theme provides a color for this segment */
176 const lle_color_t *color = get_segment_color(ctx->theme, segment_name);
177 if (color && color->mode != LLE_COLOR_MODE_NONE) {
178 /* Get terminal color capabilities for adaptive color support */
179 bool has_256_color = false;
180 bool has_true_color = false;
181 get_terminal_color_capabilities(&has_256_color, &has_true_color);
182
183 /* Downgrade color to match terminal capabilities */
184 lle_color_t adapted_color =
185 lle_color_downgrade(color, has_true_color, has_256_color);
186
187 /* Wrap content with color escape sequences */
188 char color_start[LLE_COLOR_CODE_MAX];
189 static const char *color_reset = "\033[0m";
190
191 lle_color_to_ansi(&adapted_color, true, color_start,
192 sizeof(color_start));
193
194 /* Allocate buffer for colored output */
195 size_t total_len = strlen(color_start) + output.content_len +
196 strlen(color_reset) + 1;
197 char *colored = malloc(total_len);
198 if (colored) {
199 snprintf(colored, total_len, "%s%s%s", color_start,
200 output.content, color_reset);
201 return colored;
202 }
203 }
204
205 return strdup(output.content);
206 }
207
208 return NULL;
209}

211/**
212 * @brief Check if segment is visible
213 *
214 * @param segment_name Name of segment to check
215 * @param property Property name (NULL for segment visibility)
216 * @param user_data Composer callback context
217 * @return true if visible, false otherwise
218 */
219static bool composer_is_visible(const char *segment_name, const char *property,
220 void *user_data) {
221 composer_callback_ctx_t *ctx = user_data;
222 if (!ctx || !ctx->composer || !segment_name) {
223 return false;
224 }
225
226 lle_prompt_composer_t *composer = ctx->composer;
227
228 /* Find segment in registry */
229 lle_prompt_segment_t *segment =
230 lle_segment_registry_find(composer->segments, segment_name);
231 if (!segment) {
232 return false;
233 }
234
235 /* Check if segment is enabled */
236 if (segment->is_enabled && !segment->is_enabled(segment)) {
237 return false;
238 }
239
240 /* Check if segment is visible in current context */
241 if (segment->is_visible) {
242 if (!segment->is_visible(segment, &composer->context)) {
243 return false;
244 }
245 }
246
247 /* For property check, verify property exists */
248 if (property && segment->get_property) {
249 const char *value = segment->get_property(segment, property);
250 return (value != NULL && strlen(value) > 0);
251 }
252
253 return true;
254}

420/**
421 * @brief Calculate visual width of rendered string (excluding ANSI codes)
422 *
423 * @param str String to measure
424 * @return Visual width in columns
425 */
426static size_t calculate_visual_width(const char *str) {
427 if (!str) {
428 return 0;
429 }
430
431 size_t width = 0;
432 const char *p = str;
433 bool in_escape = false;
434
435 while (*p) {
436 if (*p == '\033') {
437 in_escape = true;
438 p++;
439 continue;
440 }
441 if (in_escape) {
442 if (*p == 'm') {
443 in_escape = false;
444 }
445 p++;
446 continue;
447 }
448
449 /* Count UTF-8 character width */
450 unsigned char c = (unsigned char)*p;
451 if ((c & 0x80) == 0) {
452 /* ASCII */
453 width++;
454 p++;
455 } else if ((c & 0xE0) == 0xC0) {
456 /* 2-byte UTF-8 */
457 width++;
458 p += 2;
459 } else if ((c & 0xF0) == 0xE0) {
460 /* 3-byte UTF-8 (CJK characters are typically double-width) */
461 width += 2;
462 p += 3;
463 } else if ((c & 0xF8) == 0xF0) {
464 /* 4-byte UTF-8 */
465 width += 2;
466 p += 4;
467 } else {
468 /* Invalid or continuation byte, skip */
469 p++;
470 }
471 }
472
473 return width;
474}

476lle_result_t lle_composer_render(lle_prompt_composer_t *composer,
477 lle_prompt_output_t *output) {
478 if (!composer || !output || !composer->initialized) {
479 return LLE_ERROR_INVALID_PARAMETER;
480 }
481
482 memset(output, 0, sizeof(*output));
483
484 /* Get active theme */
485 lle_theme_t *theme = NULL;
486 if (composer->themes) {
487 theme = lle_theme_registry_get_active(composer->themes);
488 }
489
490 /* Use default template if no theme */
491 const char *left_format = "${directory} ${symbol} ";
492 const char *ps2_format = "> ";
493 const char *right_format = "";
494
495 if (theme) {
496 if (strlen(theme->layout.ps1_format) > 0) {
497 left_format = theme->layout.ps1_format;
498 }
499 if (strlen(theme->layout.ps2_format) > 0) {
500 ps2_format = theme->layout.ps2_format;
501 }
502 if (theme->layout.enable_right_prompt &&
503 strlen(theme->layout.rps1_format) > 0) {
504 right_format = theme->layout.rps1_format;
505 }
506 }
507
508 /* Create render context */
509 lle_template_render_ctx_t render_ctx =
510 lle_composer_create_render_ctx(composer);
511
512 /* Render PS1 */
513 lle_result_t result;
514
515 /* Prepend newline for visual separation if enabled */
516 if (composer->config.newline_before_prompt) {
517 output->ps1[0] = '\n';
518 result = lle_template_evaluate(
519 left_format, &render_ctx, output->ps1 + 1, sizeof(output->ps1) - 1);
520 if (result != LLE_SUCCESS) {
521 snprintf(output->ps1, sizeof(output->ps1), "\n$ ");
522 }
523 } else {
524 result = lle_template_evaluate(left_format, &render_ctx, output->ps1,
525 sizeof(output->ps1));
526 if (result != LLE_SUCCESS) {
527 snprintf(output->ps1, sizeof(output->ps1), "$ ");
528 }
529 }
530 output->ps1_len = strlen(output->ps1);
531
532 /* Append newlines after prompt if configured in theme */
533 if (theme && theme->layout.newline_after > 0) {
534 for (int i = 0; i < theme->layout.newline_after &&
535 output->ps1_len < sizeof(output->ps1) - 2;
536 i++) {
537 output->ps1[output->ps1_len++] = '\n';
538 }
539 output->ps1[output->ps1_len] = '\0';
540 }
541
542 output->ps1_visual_width = calculate_visual_width(output->ps1);
543
544 /* Check for multiline */
545 output->is_multiline = (strchr(output->ps1, '\n') != NULL);
546
547 /* Render PS2 */
548 result = lle_template_evaluate(ps2_format, &render_ctx, output->ps2,
549 sizeof(output->ps2));
550 if (result != LLE_SUCCESS) {
551 snprintf(output->ps2, sizeof(output->ps2), "> ");
552 }
553 output->ps2_len = strlen(output->ps2);
554 output->ps2_visual_width = calculate_visual_width(output->ps2);
555
556 /* Render RPROMPT if enabled */
557 if (composer->config.enable_right_prompt && strlen(right_format) > 0) {
558 result =
559 lle_template_evaluate(right_format, &render_ctx, output->rprompt,
560 sizeof(output->rprompt));
561 if (result == LLE_SUCCESS && strlen(output->rprompt) > 0) {
562 output->has_rprompt = true;
563 output->rprompt_len = strlen(output->rprompt);
564 output->rprompt_visual_width =
565 calculate_visual_width(output->rprompt);
566 }
567 }
568
569 composer->total_renders++;
570
571 return LLE_SUCCESS;
572}

687/* ============================================================================
688 * Shell Event Integration (Spec 26)
689 *
690 * This section implements the Issue #16 fix: when the directory changes,
691 * we invalidate all segment caches so stale git info doesn't persist.
692 * ============================================================================
693 */
694
695/**
696 * @brief Directory changed event handler (Issue #16 fix)
697 *
698 * When the working directory changes (cd, pushd, popd), this handler
699 * invalidates all segment caches. This ensures git status, directory
700 * info, and other path-dependent data is refreshed for the new location.
701 *
702 * @param event_data Pointer to lle_directory_changed_event_t
703 * @param user_data Pointer to lle_prompt_composer_t
704 */
705static void composer_on_directory_changed(void *event_data, void *user_data) {
706 lle_prompt_composer_t *composer = (lle_prompt_composer_t *)user_data;
707 (void)event_data; /* old_dir/new_dir available if needed */
708
709 if (!composer || !composer->initialized) {
710 return;
711 }
712
713 /* Refresh the context's directory info (cwd, cwd_display, cwd_is_git_repo).
714 * This also invalidates all segment caches. */
715 lle_composer_refresh_directory(composer);
716
717 /* Invalidate all segment caches - git status, directory display, etc. */
718 if (composer->segments) {
719 lle_segment_registry_invalidate_all(composer->segments);
720 }
721
722 /* Mark prompt for regeneration on next render */
723 composer->needs_regeneration = true;
724 composer->event_triggered_refreshes++;
725}

727/**
728 * @brief Handler for LLE_SHELL_EVENT_PRE_COMMAND
729 *
730 * Called just before command execution. Records command info for
731 * transient prompt support.
732 *
733 * @param event_data Pointer to lle_pre_command_event_t
734 * @param user_data Pointer to lle_prompt_composer_t
735 */
736static void composer_on_pre_command(void *event_data, void *user_data) {
737 lle_prompt_composer_t *composer = (lle_prompt_composer_t *)user_data;
738 lle_pre_command_event_t *event = (lle_pre_command_event_t *)event_data;
739
740 if (!composer || !composer->initialized || !event) {
741 return;
742 }
743
744 /* Save command info for post-command handling */
745 composer->current_command = event->command;
746 composer->current_command_is_bg = event->is_background;
747
748 /*
749 * Note: Transient prompt application (Spec 25 Section 12) is handled by
750 * the LINE_ACCEPTED widget hook in lle_readline.c, NOT here.
751 *
752 * The LINE_ACCEPTED hook fires earlier in the pipeline (before
753 * dc_finalize_input writes the newline), when cursor position and
754 * screen buffer state are still valid for relative cursor movement.
755 *
756 * By the time PRE_COMMAND fires here, the cursor has already moved
757 * to the output area and screen state has been reset.
758 */
759}
760
761/**
762 * @brief Handler for LLE_SHELL_EVENT_POST_COMMAND
763 *
764 * Called after command completes. Updates context with exit code and
765 * duration, then marks prompt for regeneration.
766 *
767 * @param event_data Pointer to lle_post_command_event_t
768 * @param user_data Pointer to lle_prompt_composer_t
769 */
770static void composer_on_post_command(void *event_data, void *user_data) {
771 lle_prompt_composer_t *composer = (lle_prompt_composer_t *)user_data;
772 lle_post_command_event_t *event = (lle_post_command_event_t *)event_data;
773
774 if (!composer || !composer->initialized || !event) {
775 return;
776 }
777
778 /* Update context with command results */
779 uint64_t duration_ms = event->duration_us / 1000;
780 lle_prompt_context_update(&composer->context, event->exit_code,
781 duration_ms);
782
783 /* Clear current command state */
784 composer->current_command = NULL;
785 composer->current_command_is_bg = false;
786
787 /* Invalidate all segment caches - commands like git push/pull/commit
788 * change repository state, so we must refetch git status */
789 if (composer->segments) {
790 lle_segment_registry_invalidate_all(composer->segments);
791 }
792
793 /* Mark prompt for regeneration - exit code/duration affects display */
794 composer->needs_regeneration = true;
795 composer->event_triggered_refreshes++;
796}

798lle_result_t
799lle_composer_register_shell_events(lle_prompt_composer_t *composer,
800 struct lle_shell_event_hub *event_hub) {
801 if (!composer) {
802 return LLE_ERROR_INVALID_PARAMETER;
803 }
804
805 if (!event_hub) {
806 /* No event hub provided - events won't be wired up.
807 * This is acceptable for unit testing or minimal configurations. */
808 return LLE_SUCCESS;
809 }
810
811 if (!composer->initialized) {
812 return LLE_ERROR_NOT_INITIALIZED;
813 }
814
815 if (composer->events_registered) {
816 /* Already registered */
817 return LLE_SUCCESS;
818 }
819
820 lle_result_t result;
821
822 /* Register directory change handler (Issue #16 fix) */
823 result = lle_shell_event_hub_register(
824 event_hub, LLE_SHELL_EVENT_DIRECTORY_CHANGED,
825 composer_on_directory_changed, composer, "prompt_composer_dir");
826
827 if (result != LLE_SUCCESS) {
828 return result;
829 }
830
831 /* Register pre-command handler (transient prompt support) */
832 result = lle_shell_event_hub_register(
833 event_hub, LLE_SHELL_EVENT_PRE_COMMAND, composer_on_pre_command,
834 composer, "prompt_composer_pre");
835
836 if (result != LLE_SUCCESS) {
837 /* Rollback directory handler */
838 lle_shell_event_hub_unregister(event_hub,
839 LLE_SHELL_EVENT_DIRECTORY_CHANGED,
840 "prompt_composer_dir");
841 return result;
842 }
843
844 /* Register post-command handler (exit code, duration) */
845 result = lle_shell_event_hub_register(
846 event_hub, LLE_SHELL_EVENT_POST_COMMAND, composer_on_post_command,
847 composer, "prompt_composer_post");
848
849 if (result != LLE_SUCCESS) {
850 /* Rollback previous handlers */
851 lle_shell_event_hub_unregister(event_hub,
852 LLE_SHELL_EVENT_DIRECTORY_CHANGED,
853 "prompt_composer_dir");
854 lle_shell_event_hub_unregister(event_hub, LLE_SHELL_EVENT_PRE_COMMAND,
855 "prompt_composer_pre");
856 return result;
857 }
858
859 /* Store reference and mark as registered */
860 composer->shell_event_hub = event_hub;
861 composer->events_registered = true;
862
863 return LLE_SUCCESS;
864}

866lle_result_t
867lle_composer_unregister_shell_events(lle_prompt_composer_t *composer) {
868 if (!composer) {
869 return LLE_ERROR_INVALID_PARAMETER;
870 }
871
872 if (!composer->events_registered || !composer->shell_event_hub) {
873 /* Nothing to unregister */
874 return LLE_SUCCESS;
875 }
876
877 lle_shell_event_hub_t *hub = composer->shell_event_hub;
878
879 /* Unregister all handlers */
880 lle_shell_event_hub_unregister(hub, LLE_SHELL_EVENT_DIRECTORY_CHANGED,
881 "prompt_composer_dir");
882 lle_shell_event_hub_unregister(hub, LLE_SHELL_EVENT_PRE_COMMAND,
883 "prompt_composer_pre");
884 lle_shell_event_hub_unregister(hub, LLE_SHELL_EVENT_POST_COMMAND,
885 "prompt_composer_post");
886
887 composer->shell_event_hub = NULL;
888 composer->events_registered = false;
889
890 return LLE_SUCCESS;
891}
