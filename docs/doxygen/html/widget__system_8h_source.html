1/**
2 * @file widget_system.h
3 * @brief LLE Widget Registry System
4 *
5 * ZSH-inspired widget system for LLE. Widgets are named editing operations
6 * that can be bound to keys, triggered by hooks, or invoked programmatically.
7 *
8 * Specification: docs/lle_specification/07_extensibility_framework_complete.md
9 * Section: 4 - Widget System Architecture
10 *
11 * Design Principles:
12 * - Simple, focused API following LLE patterns
13 * - Hash table based for O(1) widget lookup
14 * - Memory pool integration for all allocations
15 * - Thread-safe (single editing thread, no locks needed)
16 * - Error resilient (widget failures don't crash editor)
17 *
18 * Example Usage:
19 *
20 * // Initialize registry
21 * lle_widget_registry_t *registry;
22 * lle_widget_registry_init(&registry, memory_pool);
23 *
24 * // Register a widget
25 * lle_widget_register(registry, "my-widget", my_callback,
26 * LLE_WIDGET_USER, NULL);
27 *
28 * // Execute widget
29 * lle_widget_execute(registry, "my-widget", editor);
30 *
31 * // Cleanup
32 * lle_widget_registry_destroy(registry);
33 */
34

154/**
155 * Register a widget
156 *
157 * Registers a new widget with the given name and callback. Widget names
158 * must be unique within the registry. The name is copied internally.
159 *
160 * Widget names follow these conventions:
161 * - Built-in widgets: lowercase-with-hyphens (e.g., "forward-char")
162 * - User widgets: any valid identifier
163 * - Plugin widgets: plugin-name/widget-name (e.g., "myplugin/custom-action")
164 *
165 * @param registry Widget registry (must not be NULL)
166 * @param name Widget name (must not be NULL, must be unique)
167 * @param callback Widget callback function (must not be NULL)
168 * @param type Widget type
169 * @param user_data Optional user data (may be NULL)
170 * @return LLE_SUCCESS on success
171 * @return LLE_ERROR_INVALID_PARAMETER if registry, name, or callback is NULL
172 * @return LLE_ERROR_ALREADY_EXISTS if widget with same name exists
173 * @return LLE_ERROR_OUT_OF_MEMORY if allocation fails
174 */
175lle_result_t lle_widget_register(lle_widget_registry_t *registry,
176 const char *name,
177 lle_widget_callback_t callback,
178 lle_widget_type_t type, void *user_data);
179
180/**
181 * Unregister a widget

183 * Removes widget from registry and frees its resources. Any keybindings
184 * or hooks referencing this widget will fail if invoked after unregistration.

186 * @param registry Widget registry (must not be NULL)

188 * @return LLE_SUCCESS on success
189 * @return LLE_ERROR_INVALID_PARAMETER if registry or name is NULL
190 * @return LLE_ERROR_NOT_FOUND if widget does not exist

208/**
209 * Execute a widget by name
210 *
211 * Looks up widget by name and executes it with the given editor context.
212 * Tracks execution time and updates widget statistics.
213 *
214 * If widget execution fails, the error is returned but the editor remains
215 * in a valid state (error resilience).
216 *
217 * @param registry Widget registry (must not be NULL)
218 * @param name Widget name to execute (must not be NULL)
219 * @param editor Editor context (must not be NULL)
220 * @return LLE_SUCCESS if widget executed successfully
221 * @return LLE_ERROR_INVALID_PARAMETER if any parameter is NULL
222 * @return LLE_ERROR_NOT_FOUND if widget does not exist
223 * @return LLE_ERROR_DISABLED if widget is disabled
224 * @return Other error codes from widget callback
225 */
226lle_result_t lle_widget_execute(lle_widget_registry_t *registry,
227 const char *name, lle_editor_t *editor);

229/**
