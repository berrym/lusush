4 *
5 * Specification: Spec 03 - Buffer Management, Section 7

29/**
30 * @brief Get current timestamp in microseconds
31 */
32static uint64_t get_timestamp_us(void) {
33 struct timespec ts;
34 clock_gettime(CLOCK_MONOTONIC, &ts);
35 return (uint64_t)ts.tv_sec * 1000000ULL + (uint64_t)ts.tv_nsec / 1000ULL;
36}

38/**
39 * @brief Find last undoable sequence
40 */
41static lle_change_sequence_t *
42find_last_undoable_sequence(lle_change_tracker_t *tracker) {
43 if (!tracker || !tracker->current_position) {
44 return NULL;
45 }
46
47 /* Start from current position and search backwards */
48 lle_change_sequence_t *seq = tracker->current_position;
49 while (seq) {
50 if (seq->can_undo && seq->sequence_complete) {
51 return seq;
52 }
53 seq = seq->prev;
54 }
55
56 return NULL;
57}

73
74 while (seq) {

84/**
85 * @brief Free operation and its associated data
86 */
87static void free_operation(lle_change_operation_t *op,
88 lusush_memory_pool_t *pool) {
89 (void)pool; /* Reserved for future pool-based deallocation */
90 if (!op) {
91 return;
92 }
93
94 if (op->inserted_text) {
95 lle_pool_free(op->inserted_text);
96 }
97 if (op->deleted_text) {
98 lle_pool_free(op->deleted_text);
99 }
100
101 lle_pool_free(op);
102}

104/**
105 * @brief Free sequence and all its operations
106 */
107static void free_sequence(lle_change_sequence_t *seq,
108 lusush_memory_pool_t *pool) {
109 if (!seq) {
110 return;
111 }
112
113 /* Free all operations in sequence */
114 lle_change_operation_t *op = seq->first_op;
115 while (op) {
116 lle_change_operation_t *next = op->next;
117 free_operation(op, pool);
118 op = next;
119 }
120
121 lle_pool_free(seq);
122}

210lle_result_t
211lle_change_tracker_begin_sequence(lle_change_tracker_t *tracker,
212 const char *description,
213 lle_change_sequence_t **sequence) {
214 if (!tracker || !sequence) {
215 return LLE_ERROR_INVALID_PARAMETER;
216 }
217
218 /* Check if sequence already in progress */
219 if (tracker->sequence_in_progress) {
220 return LLE_ERROR_OPERATION_IN_PROGRESS;
221 }
222
223 /* Allocate new sequence */
224 lle_change_sequence_t *seq =
225 (lle_change_sequence_t *)lle_pool_alloc(sizeof(lle_change_sequence_t));
226 if (!seq) {
227 return LLE_ERROR_OUT_OF_MEMORY;
228 }
229
230 /* Initialize sequence */
231 memset(seq, 0, sizeof(lle_change_sequence_t));
232
233 seq->sequence_id = tracker->next_sequence_id++;
234 seq->start_time = get_timestamp_us();
235 seq->can_undo = true;
236 seq->can_redo = false;
237 seq->sequence_complete = false;
238
239 if (description) {
240 strncpy(seq->description, description, sizeof(seq->description) - 1);
241 seq->description[sizeof(seq->description) - 1] = '\0';
242 } else {
243 strcpy(seq->description, "Untitled operation");
244 }
245
246 /* Clear any redo history (branching timeline) */
247 if (tracker->current_position && tracker->current_position->next) {
248 lle_change_sequence_t *redo_seq = tracker->current_position->next;
249 while (redo_seq) {
250 lle_change_sequence_t *next = redo_seq->next;
251 free_sequence(redo_seq, tracker->memory_pool);
252 redo_seq = next;
253 }
254 tracker->current_position->next = NULL;
255 tracker->last_sequence = tracker->current_position;
256 }
257
258 /* Add to sequence chain */
259 if (!tracker->first_sequence) {
260 tracker->first_sequence = seq;
261 tracker->last_sequence = seq;
262 tracker->current_position = seq;
263 } else {
264 seq->prev = tracker->last_sequence;
265 tracker->last_sequence->next = seq;
266 tracker->last_sequence = seq;
267 tracker->current_position = seq;
268 }
269
270 tracker->active_sequence = seq;
271 tracker->sequence_in_progress = true;
272 tracker->sequence_count++;
273 tracker->memory_used += sizeof(lle_change_sequence_t);
274
275 /* Enforce undo limit */
276 while (tracker->sequence_count > tracker->max_undo_levels) {
277 lle_change_sequence_t *old = tracker->first_sequence;
278 if (old == tracker->current_position) {
279 break; /* Don't remove current position */
280 }
281
282 tracker->first_sequence = old->next;
283 if (tracker->first_sequence) {
284 tracker->first_sequence->prev = NULL;
285 }
286
287 free_sequence(old, tracker->memory_pool);
288 tracker->sequence_count--;
289 tracker->memory_used -= sizeof(lle_change_sequence_t);
290 }
291
292 *sequence = seq;
293 return LLE_SUCCESS;
294}

413lle_result_t lle_change_tracker_undo(lle_change_tracker_t *tracker,
414 lle_buffer_t *buffer) {
415 if (!tracker || !buffer) {
416 return LLE_ERROR_INVALID_PARAMETER;
417 }
418
419 /* Find the most recent sequence that can be undone */
420 lle_change_sequence_t *sequence = find_last_undoable_sequence(tracker);
421 if (!sequence) {
422 return LLE_ERROR_NO_UNDO_AVAILABLE;
423 }
424
425 lle_result_t result = LLE_SUCCESS;
426
427 /* Step 1: Disable change tracking during undo */
428 bool tracking_was_enabled = buffer->change_tracking_enabled;
429 buffer->change_tracking_enabled = false;
430
431 /* Step 2: Process operations in reverse order */
432 lle_change_operation_t *op = sequence->last_op;
433 while (op) {
434 switch (op->type) {
435 case LLE_CHANGE_TYPE_INSERT:
436 /* Undo insert by deleting the inserted text */
437 result = lle_buffer_delete_text(buffer, op->start_position,
438 op->inserted_length);
439 break;
440
441 case LLE_CHANGE_TYPE_DELETE:
442 /* Undo delete by reinserting the deleted text */
443 result =
444 lle_buffer_insert_text(buffer, op->start_position,
445 op->deleted_text, op->deleted_length);
446 break;
447
448 case LLE_CHANGE_TYPE_REPLACE:
449 /* Undo replace by deleting new text and inserting old text */
450 result = lle_buffer_delete_text(buffer, op->start_position,
451 op->inserted_length);
452 if (result == LLE_SUCCESS) {
453 result = lle_buffer_insert_text(buffer, op->start_position,
454 op->deleted_text,
455 op->deleted_length);
456 }
457 break;
458
459 default:
460 /* Other operation types don't modify buffer content */
461 break;
462 }
463
464 if (result != LLE_SUCCESS) {
465 break;
466 }
467
468 op = op->prev;
469 }
470
471 /* Step 3: Restore cursor position */
472 if (result == LLE_SUCCESS && sequence->first_op) {
473 buffer->cursor = sequence->first_op->cursor_before;
474 }
475
476 /* Step 4: Mark sequence as undone */
477 if (result == LLE_SUCCESS) {
478 sequence->can_undo = false;
479 sequence->can_redo = true;
480 tracker->undo_count++;
481
482 /* Move current position back */
483 if (tracker->current_position == sequence) {
484 tracker->current_position =
485 sequence->prev; /* Can be NULL for first sequence */
486 }
487 }
488
489 /* Step 5: Re-enable change tracking */
490 buffer->change_tracking_enabled = tracking_was_enabled;
491
492 return result;
493}

495lle_result_t lle_change_tracker_redo(lle_change_tracker_t *tracker,
496 lle_buffer_t *buffer) {
497 if (!tracker || !buffer) {
498 return LLE_ERROR_INVALID_PARAMETER;
499 }
500
501 /* Find the most recent sequence that can be redone */
502 lle_change_sequence_t *sequence = find_last_redoable_sequence(tracker);
503 if (!sequence) {
504 return LLE_ERROR_NO_REDO_AVAILABLE;
505 }
506
507 lle_result_t result = LLE_SUCCESS;
508
509 /* Step 1: Disable change tracking during redo */
510 bool tracking_was_enabled = buffer->change_tracking_enabled;
511 buffer->change_tracking_enabled = false;
512
513 /* Step 2: Process operations in forward order */
514 lle_change_operation_t *op = sequence->first_op;
515 while (op) {
516 switch (op->type) {
517 case LLE_CHANGE_TYPE_INSERT:
518 /* Redo insert */
519 result =
520 lle_buffer_insert_text(buffer, op->start_position,
521 op->inserted_text, op->inserted_length);
522 break;
523
524 case LLE_CHANGE_TYPE_DELETE:
525 /* Redo delete */
526 result = lle_buffer_delete_text(buffer, op->start_position,
527 op->deleted_length);
528 break;
529
530 case LLE_CHANGE_TYPE_REPLACE:
531 /* Redo replace */
532 result = lle_buffer_delete_text(buffer, op->start_position,
533 op->deleted_length);
534 if (result == LLE_SUCCESS) {
535 result = lle_buffer_insert_text(buffer, op->start_position,
536 op->inserted_text,
537 op->inserted_length);
538 }
539 break;
540
541 default:
542 /* Other operation types don't modify buffer content */
543 break;
544 }
545
546 if (result != LLE_SUCCESS) {
547 break;
548 }
549
550 op = op->next;
551 }
552
553 /* Step 3: Restore cursor position */
554 if (result == LLE_SUCCESS && sequence->last_op) {
555 buffer->cursor = sequence->last_op->cursor_after;
556 }
557
558 /* Step 4: Mark sequence as redone */
559 if (result == LLE_SUCCESS) {
560 sequence->can_undo = true;
561 sequence->can_redo = false;
562 tracker->redo_count++;
563
564 /* Move current position forward */
565 tracker->current_position = sequence;
566 }
567
568 /* Step 5: Re-enable change tracking */
569 buffer->change_tracking_enabled = tracking_was_enabled;
570
571 return result;
572}
