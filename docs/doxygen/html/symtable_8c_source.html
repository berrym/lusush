10 * - FNV1A hash algorithm for superior distribution vs djb2

61/**
62 * @brief Serialize variable metadata into a string format
63 *
64 * Creates a serialized representation of variable data in the format:
65 * "value|type|flags|scope_level" for storage in the hash table.
66 *
67 * @param value Variable value (NULL treated as empty string)
68 * @param type Variable type (string, integer, array, etc.)
69 * @param flags Variable flags (exported, readonly, local, etc.)
70 * @param scope_level Scope level where variable is defined
71 * @return Allocated serialized string, or NULL on allocation failure
72 */
73static char *serialize_variable(const char *value, symvar_type_t type,
74 symvar_flags_t flags, size_t scope_level) {
75 if (!value) {
76 value = "";
77 }
78
79 // Calculate needed size
80 size_t value_len = strlen(value);
81 size_t total_size = value_len + METADATA_BUFFER_SIZE;
82
83 char *serialized = malloc(total_size);
84 if (!serialized) {
85 return NULL;
86 }
87
88 snprintf(serialized, total_size, "%s%s%d%s%d%s%zu", value,
89 METADATA_SEPARATOR, (int)type, METADATA_SEPARATOR, (int)flags,
90 METADATA_SEPARATOR, scope_level);
91
92 return serialized;
93}

95/**
96 * @brief Deserialize variable metadata from string format
97 *
98 * Parses a serialized variable string and creates a symvar_t structure.
99 * Handles empty fields correctly by parsing separators manually.
100 *
101 * @param name Variable name to associate with the result
102 * @param serialized Serialized string in "value|type|flags|scope_level" format
103 * @return Allocated symvar_t structure, or NULL on failure
104 */
105static symvar_t *deserialize_variable(const char *name,
106 const char *serialized) {
107 if (!serialized || !name) {
108 return NULL;
109 }
110
111 symvar_t *var = malloc(sizeof(symvar_t));
112 if (!var) {
113 return NULL;
114 }
115
116 // Initialize defaults
117 var->name = strdup(name);
118 var->value = NULL;
119 var->type = SYMVAR_STRING;
120 var->flags = SYMVAR_NONE;
121 var->scope_level = 0;
122 var->next = NULL;
123
124 if (!var->name) {
125 free(var);
126 return NULL;
127 }
128
129 // Parse serialized string manually to handle empty values correctly
130 char *serialized_copy = strdup(serialized);
131 if (!serialized_copy) {
132 free(var->name);
133 free(var);
134 return NULL;
135 }
136
137 // Find the separators manually to handle empty fields
138 char *pos = serialized_copy;
139 char *sep1 = strstr(pos, METADATA_SEPARATOR);
140 if (sep1) {
141 *sep1 = '\0';
142 var->value = strdup(pos);
143
144 pos = sep1 + strlen(METADATA_SEPARATOR);
145 char *sep2 = strstr(pos, METADATA_SEPARATOR);
146 if (sep2) {
147 *sep2 = '\0';
148 var->type = (symvar_type_t)atoi(pos);
149
150 pos = sep2 + strlen(METADATA_SEPARATOR);
151 char *sep3 = strstr(pos, METADATA_SEPARATOR);
152 if (sep3) {
153 *sep3 = '\0';
154 var->flags = (symvar_flags_t)atoi(pos);
155
156 pos = sep3 + strlen(METADATA_SEPARATOR);
157 var->scope_level = (size_t)atoi(pos);
158 }
159 }
160 }
161
162 free(serialized_copy);
163
164 // Ensure we have a value
165 if (!var->value) {
166 var->value = strdup("");
167 }
168
169 return var;
170}

193/**
194 * @brief Create a new symbol table manager
195 *
196 * Allocates and initializes a symbol table manager with a global scope.
197 * The global scope uses libhashtable with FNV1A hashing for performance.
198 *
199 * @return Newly allocated manager, or NULL on allocation failure
200 */
201symtable_manager_t *symtable_manager_new(void) {
202 symtable_manager_t *manager = calloc(1, sizeof(symtable_manager_t));
203 if (!manager) {
204 return NULL;
205 }
206
207 // Create global scope
208 symtable_scope_t *global = calloc(1, sizeof(symtable_scope_t));
209 if (!global) {
210 free(manager);
211 return NULL;
212 }
213
214 global->scope_type = SCOPE_GLOBAL;
215 global->level = 0;
216 global->vars_ht = ht_strstr_create(DEFAULT_HT_FLAGS);
217 global->parent = NULL;
218 global->scope_name = strdup("global");
219
220 if (!global->vars_ht || !global->scope_name) {
221 if (global->vars_ht) {
222 ht_strstr_destroy(global->vars_ht);
223 }
224 free(global->scope_name);
225 free(global);
226 free(manager);
227 return NULL;
228 }
229
230 manager->current_scope = global;
231 manager->global_scope = global;
232 manager->max_scope_level = 0;
233 manager->debug_mode = false;
234
235 return manager;
236}

238/**
239 * @brief Free a symbol table manager and all its scopes
240 *
241 * Pops all scopes from the scope chain and frees all associated memory
242 * including hash tables and scope names.
243 *
244 * @param manager Manager to free (NULL is safe)
245 */
246void symtable_manager_free(symtable_manager_t *manager) {
247 if (!manager) {
248 return;
249 }
250
251 // Pop all scopes to free memory
252 while (manager->current_scope &&
253 manager->current_scope != manager->global_scope) {
254 symtable_scope_t *old_scope = manager->current_scope;
255 manager->current_scope = old_scope->parent;
256
257 if (old_scope->vars_ht) {
258 ht_strstr_destroy(old_scope->vars_ht);
259 }
260 free(old_scope->scope_name);
261 free(old_scope);
262 }
263
264 // Free global scope
265 if (manager->global_scope) {
266 if (manager->global_scope->vars_ht) {
267 ht_strstr_destroy(manager->global_scope->vars_ht);
268 }
269 free(manager->global_scope->scope_name);
270 free(manager->global_scope);
271 }
272
273 free(manager);
274}

291/**
292 * @brief Find a variable in the scope chain
293 *
294 * Searches from the given scope up through parent scopes until a variable
295 * with the given name is found. Skips variables marked as unset.
296 *
297 * @param scope Starting scope for the search
298 * @param name Variable name to find
299 * @return Allocated symvar_t if found, NULL otherwise
300 */
301static symvar_t *find_var(symtable_scope_t *scope, const char *name) {
302 if (!scope || !name) {
303 return NULL;
304 }
305
306 while (scope) {
307 const char *serialized = ht_strstr_get(scope->vars_ht, name);
308 if (serialized) {
309 symvar_t *var = deserialize_variable(name, serialized);
310 if (var && !(var->flags & SYMVAR_UNSET)) {
311 return var;
312 }
313 free_symvar(var);
314 }
315 scope = scope->parent;
316 }
317
318 return NULL;
319}

321/**
322 * @brief Push a new scope onto the scope stack
323 *
324 * Creates a new scope with its own hash table and links it as the
325 * current scope. The new scope inherits from parent scopes for variable
326 * lookups but has its own storage for local variables.
327 *
328 * @param manager Symbol table manager
329 * @param type Type of scope (global, function, loop, subshell, conditional)
330 * @param name Human-readable name for debugging
331 * @return 0 on success, -1 on failure or max depth exceeded
332 */
333int symtable_push_scope(symtable_manager_t *manager, scope_type_t type,
334 const char *name) {
335 if (!manager || !name) {
336 return -1;
337 }
338
339 if (manager->current_scope->level >= MAX_SCOPE_DEPTH) {
340 if (manager->debug_mode) {
341 fprintf(stderr, "ERROR: Maximum scope depth exceeded\n");
342 }
343 return -1;
344 }
345
346 symtable_scope_t *new_scope = calloc(1, sizeof(symtable_scope_t));
347 if (!new_scope) {
348 return -1;
349 }
350
351 new_scope->scope_type = type;
352 new_scope->level = manager->current_scope->level + 1;
353 new_scope->vars_ht = ht_strstr_create(DEFAULT_HT_FLAGS);
354 new_scope->parent = manager->current_scope;
355 new_scope->scope_name = strdup(name);
356
357 if (!new_scope->vars_ht || !new_scope->scope_name) {
358 if (new_scope->vars_ht) {
359 ht_strstr_destroy(new_scope->vars_ht);
360 }
361 free(new_scope->scope_name);
362 free(new_scope);
363 return -1;
364 }
365
366 manager->current_scope = new_scope;
367 if (new_scope->level > manager->max_scope_level) {
368 manager->max_scope_level = new_scope->level;
369 }
370
371 if (manager->debug_mode) {
372 printf("DEBUG: Pushed scope '%s' (level %zu)\n", name,
373 new_scope->level);
374 }
375
376 return 0;
377}

379/**
380 * @brief Pop the current scope from the scope stack
381 *
382 * Removes the current scope, freeing its hash table and memory,
383 * and restores the parent scope as current. Cannot pop the global scope.
384 *
385 * @param manager Symbol table manager
386 * @return 0 on success, -1 if at global scope or invalid manager
387 */
388int symtable_pop_scope(symtable_manager_t *manager) {
389 if (!manager || !manager->current_scope ||
390 manager->current_scope == manager->global_scope) {
391 return -1; // Can't pop global scope
392 }
393
394 symtable_scope_t *old_scope = manager->current_scope;
395 manager->current_scope = old_scope->parent;
396
397 if (manager->debug_mode) {
398 printf("DEBUG: Popped scope '%s' (level %zu)\n", old_scope->scope_name,
399 old_scope->level);
400 }
401
402 ht_strstr_destroy(old_scope->vars_ht);
403 free(old_scope->scope_name);
404 free(old_scope);
405
406 return 0;
407}

443 * Serializes the variable data and stores it in the current scope's

446 * @param manager Symbol table manager

661/**
662 * @brief Dump variables in a specific scope type for debugging
663 *
664 * Finds the first scope of the given type in the scope chain and
665 * prints all variables in that scope to stdout.
666 *
667 * @param manager Symbol table manager
668 * @param scope Scope type to dump (global, function, loop, etc.)
669 */
670void symtable_dump_scope(symtable_manager_t *manager, scope_type_t scope) {
671 if (!manager) {
672 printf("DEBUG: No manager provided\n");
673 return;
674 }
675
676 // Find the requested scope in the scope chain
677 symtable_scope_t *target_scope = NULL;
678 symtable_scope_t *current = manager->current_scope;
679
680 while (current) {
681 if (current->scope_type == scope) {
682 target_scope = current;
683 break;
684 }
685 current = current->parent;
686 }
687
688 if (!target_scope) {
689 printf("DEBUG: Scope type %d not found\n", scope);
690 return;
691 }
692
693 const char *scope_name = "unknown";
694 switch (scope) {
695 case SCOPE_GLOBAL:
696 scope_name = "global";
697 break;
698 case SCOPE_FUNCTION:
699 scope_name = "function";
700 break;
701 case SCOPE_LOOP:
702 scope_name = "loop";
703 break;
704 case SCOPE_SUBSHELL:
705 scope_name = "subshell";
706 break;
707 case SCOPE_CONDITIONAL:
708 scope_name = "conditional";
709 break;
710 }
711
712 printf("=== %s scope (level %zu) ===\n", scope_name, target_scope->level);
713
714 if (!target_scope->vars_ht) {
715 printf(" (no variables)\n");
716 return;
717 }
718
719 // Enumerate all variables in this scope's hashtable
720 ht_enum_t *enum_iter = ht_strstr_enum_create(target_scope->vars_ht);
721 if (!enum_iter) {
722 printf(" (enumeration failed)\n");
723 return;
724 }
725
726 const char *key, *value;
727 bool has_vars = false;
728
729 while (ht_strstr_enum_next(enum_iter, &key, &value)) {
730 has_vars = true;
731 printf(" %-12s = '%s'\n", key ? key : "(null)",
732 value ? value : "(null)");
733 }
734
735 if (!has_vars) {
736 printf(" (no variables in scope)\n");
737 }
738
739 ht_strstr_enum_destroy(enum_iter);
740}

743 * @brief Dump all scopes in the scope chain for debugging
744 *
745 * Walks the scope chain from current to global, printing each scope's
746 * name, type, level, and all variables it contains.
747 *
748 * @param manager Symbol table manager

764 int scope_count = 0;
765
766 while (current) {

769 const char *type_name = "unknown";
