141/**
142 * @brief Trigger widget hooks for input
143 *
144 * Evaluates which widget hooks should be triggered based on the parsed input
145 * and executes registered widgets through the hooks manager.
146 *
147 * @param parser Parser system with widget hook integration
148 * @param input Parsed input to evaluate for hook triggers
149 * @return LLE_SUCCESS on success, error code on failure
150 */
151lle_result_t lle_input_trigger_widget_hooks(lle_input_parser_system_t *parser,
152 lle_parsed_input_t *input) {
153 if (!parser || !input) {
154 return LLE_ERROR_INVALID_PARAMETER;
155 }
156
157 if (!parser->widget_hook_triggers) {
158 /* No widget hook triggers configured */
159 return LLE_SUCCESS;
160 }
161
162 lle_widget_hook_triggers_t *wh = parser->widget_hook_triggers;
163
164 /* Check if hook execution is enabled */
165 if (!wh->hook_execution_enabled) {
166 return LLE_SUCCESS;
167 }
168
169 /* Check if hooks manager is available */
170 if (!wh->hooks_manager) {
171 /* Hooks manager not yet initialized - skip triggering */
172 return LLE_SUCCESS;
173 }
174
175 /* Record start time for performance tracking */
176 uint64_t start_time = lle_event_get_timestamp_us();
177
178 /* Determine which hook to trigger based on input type */
179 bool should_trigger = false;
180
181 /* Map input types to hook types */
182 switch (input->type) {
183 case LLE_PARSED_INPUT_TYPE_TEXT:
184 /* Text input could trigger buffer-modified hook */
185 should_trigger = true;
186 break;
187
188 case LLE_PARSED_INPUT_TYPE_KEY:
189 /* Some key events might trigger hooks */
190 /* For now, don't trigger on every key */
191 should_trigger = false;
192 break;
193
194 default:
195 /* Other input types don't trigger hooks yet */
196 should_trigger = false;
197 break;
198 }
199
200 lle_result_t result = LLE_SUCCESS;
201
202 /* Hook triggering is tracked. Actual hook execution happens at a higher
203 * level where editor context is available. The input parser detects when
204 * hooks should trigger and sets appropriate flags in the parsed input
205 * structure. The caller (typically lle_readline or command processing) then
206 * triggers the hooks with the full editor context via
207 * lle_widget_hook_trigger(). */
208 if (should_trigger) {
209 __atomic_fetch_add(&wh->hooks_triggered, 1, __ATOMIC_SEQ_CST);
210 }
211
212 /* Track execution time */
213 uint64_t execution_time = lle_event_get_timestamp_us() - start_time;
214 __atomic_fetch_add(&wh->total_execution_time_us, execution_time,
215 __ATOMIC_SEQ_CST);
216
217 if (execution_time > wh->max_execution_time_us) {
218 wh->max_execution_time_us = execution_time;
219 }
220
221 return result;
222}

279/**
280 * @brief Get widget hook trigger statistics
281 *
282 * Retrieves performance statistics for widget hook triggering.
283 *
284 * @param triggers Trigger system
285 * @param triggered Output for total hooks triggered
286 * @param executed Output for total hooks executed
287 * @param avg_time_us Output for average execution time
288 * @return LLE_SUCCESS on success, error code on failure
289 */
290lle_result_t lle_widget_hooks_get_stats(lle_widget_hook_triggers_t *triggers,
291 uint64_t *triggered, uint64_t *executed,
292 uint64_t *avg_time_us) {
293 if (!triggers) {
294 return LLE_ERROR_INVALID_PARAMETER;
295 }
296
297 if (triggered) {
298 *triggered = triggers->hooks_triggered;
299 }
300
301 if (executed) {
302 *executed = triggers->hooks_executed;
303 }
304
305 if (avg_time_us) {
306 if (triggers->hooks_executed > 0) {
307 *avg_time_us =
308 triggers->total_execution_time_us / triggers->hooks_executed;
309 } else {
310 *avg_time_us = 0;
311 }
312 }
313
314 return LLE_SUCCESS;
315}
