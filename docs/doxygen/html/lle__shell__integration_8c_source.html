63/**
64 * @brief Get current timestamp in microseconds
65 */
66static uint64_t get_timestamp_us(void) {
67 struct timespec ts;
68 if (clock_gettime(CLOCK_MONOTONIC, &ts) == 0) {
69 return (uint64_t)ts.tv_sec * 1000000ULL + (uint64_t)ts.tv_nsec / 1000;
70 }
71 return 0;
72}

78 if (!hist_config) {
79 return;
80 }
81
82 memset(hist_config, 0, sizeof(lle_history_config_t));
83
84 /* Capacity settings */
85 hist_config->max_entries =
86 config.history_size > 0 ? config.history_size : 5000;
87 hist_config->max_command_length = 8192;
88
89 /* File settings */
90 if (config.lle_history_file && config.lle_history_file[0] != '\0') {
91 hist_config->history_file_path = config.lle_history_file;
92 } else {
93 hist_config->history_file_path = NULL;
94 }
95 hist_config->auto_save = true;
96 hist_config->load_on_init = true;
97
98 /* Deduplication behavior */
99 hist_config->ignore_duplicates =
100 config.lle_enable_deduplication &&
101 (config.lle_dedup_scope != LLE_DEDUP_SCOPE_NONE);
102
103 /* Map dedup strategy */
104 switch (config.lle_dedup_strategy) {
105 case LLE_DEDUP_STRATEGY_IGNORE:
106 hist_config->dedup_strategy = LLE_DEDUP_IGNORE;
107 break;
108 case LLE_DEDUP_STRATEGY_KEEP_FREQUENT:
109 hist_config->dedup_strategy = LLE_DEDUP_KEEP_FREQUENT;
110 break;
111 case LLE_DEDUP_STRATEGY_MERGE:
112 hist_config->dedup_strategy = LLE_DEDUP_MERGE_METADATA;
113 break;
114 case LLE_DEDUP_STRATEGY_KEEP_ALL:
115 hist_config->dedup_strategy = LLE_DEDUP_KEEP_ALL;
116 break;
117 case LLE_DEDUP_STRATEGY_KEEP_RECENT:

119 hist_config->dedup_strategy = LLE_DEDUP_KEEP_RECENT;
120 break;

122

124 hist_config->ignore_space_prefix = false;

144lle_result_t lle_shell_integration_init(void) {
145 /* Already initialized? */
146 if (g_lle_integration) {
147 return LLE_SUCCESS;
148 }
149
150 /* Allocate integration structure */
151 lle_shell_integration_t *integ = calloc(1, sizeof(lle_shell_integration_t));
152 if (!integ) {
153 return LLE_ERROR_OUT_OF_MEMORY;
154 }
155
156 integ->init_time_us = get_timestamp_us();
157
158 /* Step 1: Verify global memory pool exists */
159 if (!global_memory_pool) {
160 free(integ);
161 return LLE_ERROR_NOT_INITIALIZED;
162 }
163 integ->init_state.memory_pool_verified = true;
164
165 /* Step 2: Verify terminal detection is complete
166 * Terminal detection is handled by Lusush's display system,
167 * so we just verify it has been initialized */
168 integ->init_state.terminal_detected = true;
169
170 /* Step 3: Create shell event hub */
171 lle_result_t result = lle_shell_event_hub_create(&integ->event_hub);
172 if (result != LLE_SUCCESS) {
173 free(integ);
174 return result;
175 }
176 integ->init_state.event_hub_initialized = true;
177
178 /* Step 4: Create and configure LLE editor */
179 result = create_and_configure_editor(integ);
180 if (result != LLE_SUCCESS) {
181 lle_shell_event_hub_destroy(integ->event_hub);
182 free(integ);
183 return result;
184 }
185 integ->init_state.editor_initialized = true;
186
187 /* Step 5: Initialize history (already done in create_and_configure_editor)
188 */
189 integ->init_state.history_initialized = true;
190
191 /* Step 6: Create and configure prompt composer (Spec 25) */
192 result = create_and_configure_prompt_composer(integ);
193 if (result != LLE_SUCCESS) {
194 /* Prompt composer is optional - log warning but continue */
195 /* The shell can still function without the fancy prompt system */
196 } else {
197 integ->init_state.prompt_initialized = true;
198 }
199
200 /* Step 7: Register atexit handler for cleanup */
201 if (!atexit_registered) {
202 if (atexit(lle_shell_integration_atexit_handler) == 0) {
203 atexit_registered = true;
204 integ->init_state.atexit_registered = true;
205 }
206 }
207
208 /* Step 8: Initialize watchdog subsystem for deadlock detection */
209 result = lle_watchdog_init();
210 if (result != LLE_SUCCESS) {
211 /* Watchdog is optional - log warning but continue */
212 /* Shell can still function without watchdog protection */
213 }
214
215 /* Mark shell hooks as installed */
216 integ->init_state.shell_hooks_installed = true;
217
218 /* Set global pointer */
219 g_lle_integration = integ;
220
221 return LLE_SUCCESS;
222}

224void lle_shell_integration_shutdown(void) {
225 if (!g_lle_integration) {
226 return;
227 }
228
229 lle_shell_integration_t *integ = g_lle_integration;
230
231 /* Save history before shutdown */
232 if (integ->editor && integ->editor->history_system) {
233 const char *home = getenv("HOME");
234 if (home) {
235 char history_path[1024];
236 snprintf(history_path, sizeof(history_path), "%s/.lusush_history",
237 home);
238 lle_history_save_to_file(integ->editor->history_system,
239 history_path);
240 }
241 }
242
243 /* Destroy prompt composer (unregisters from event hub) */
244 destroy_prompt_composer(integ);
245
246 /* Destroy editor */
247 destroy_editor(integ);
248
249 /* Destroy event hub */
250 if (integ->event_hub) {
251 lle_shell_event_hub_destroy(integ->event_hub);
252 integ->event_hub = NULL;
253 }
254
255 /* Cleanup watchdog subsystem */
256 lle_watchdog_cleanup();
257
258 /* Clear global pointer */
259 g_lle_integration = NULL;
260
261 /* Free integration structure */
262 free(integ);
263}

294/**
295 * @brief Create and configure the LLE editor instance
296 */
297static lle_result_t
298create_and_configure_editor(lle_shell_integration_t *integ) {
299 if (!integ) {
300 return LLE_ERROR_INVALID_PARAMETER;
301 }
302
303 /* Create editor */
304 lle_result_t result = lle_editor_create(&integ->editor, global_memory_pool);
305 if (result != LLE_SUCCESS || !integ->editor) {
306 return result != LLE_SUCCESS ? result : LLE_ERROR_OUT_OF_MEMORY;
307 }
308
309 /* Initialize history subsystem */
310 lle_history_config_t hist_config;
311 populate_history_config(&hist_config);
312
313 result = lle_history_core_create(&integ->editor->history_system,
314 integ->editor->lle_pool, &hist_config);
315
316 if (result == LLE_SUCCESS && integ->editor->history_system) {
317 /* Load existing history from file */
318 const char *home = getenv("HOME");
319 if (home) {
320 char history_path[1024];
321 snprintf(history_path, sizeof(history_path), "%s/.lusush_history",
322 home);
323 lle_history_load_from_file(integ->editor->history_system,
324 history_path);
325 }
326
327 /* Initialize the history bridge for builtin commands
328 * This connects the LLE history core to the shell's history builtin */
329 lle_result_t bridge_result =
330 lle_history_bridge_init(integ->editor->history_system,
331 NULL, /* No POSIX manager - LLE-only now */
332 integ->editor->lle_pool);
333 if (bridge_result != LLE_SUCCESS) {
334 /* Non-fatal - history builtin won't work but shell continues */
335 }
336 }
337
338 return LLE_SUCCESS;
339}

365/**
366 * @brief Create and configure the prompt composer
367 *
368 * Initializes the prompt composer and registers it with the shell event hub
369 * for automatic cache invalidation on directory changes and command events.
370 */
371static lle_result_t
372create_and_configure_prompt_composer(lle_shell_integration_t *integ) {
373 if (!integ || !integ->event_hub) {
374 return LLE_ERROR_INVALID_PARAMETER;
375 }
376
377 /* Initialize registries only if not already initialized
378 * This prevents double-init issues during hard reset */
379 if (!g_registries_initialized) {
380 /* Initialize segment registry and register built-in segments */
381 lle_result_t result = lle_segment_registry_init(&g_segment_registry);
382 if (result != LLE_SUCCESS) {
383 return result;
384 }
385 lle_segment_register_builtins(&g_segment_registry);
386
387 /* Initialize theme registry and register built-in themes */
388 result = lle_theme_registry_init(&g_theme_registry);
389 if (result != LLE_SUCCESS) {
390 lle_segment_registry_cleanup(&g_segment_registry);
391 return result;
392 }
393 lle_theme_register_builtins(&g_theme_registry);
394
395 /* Load user themes from standard locations (Issue #21)
396 * This loads themes from:
397 * - $XDG_CONFIG_HOME/lusush/themes/ (~/.config/lusush/themes/)
398 * - /etc/lusush/themes/ (system-wide)
399 */
400 lle_theme_load_user_themes(&g_theme_registry);
401
402 g_registries_initialized = true;
403 }
404
405 lle_result_t result;
406
407 /* Set default theme as active */
408 lle_theme_registry_set_active(&g_theme_registry, "default");
409
410 /* Allocate prompt composer */
411 integ->prompt_composer = calloc(1, sizeof(lle_prompt_composer_t));
412 if (!integ->prompt_composer) {
413 return LLE_ERROR_OUT_OF_MEMORY;
414 }
415
416 /* Initialize the composer with segment and theme registries */
417 result = lle_composer_init(integ->prompt_composer, &g_segment_registry,
418 &g_theme_registry);
419 if (result != LLE_SUCCESS) {
420 free(integ->prompt_composer);
421 integ->prompt_composer = NULL;
422 return result;
423 }
424
425 /* Sync composer config with global config settings */
426 integ->prompt_composer->config.enable_transient =
427 config.display_transient_prompt;
428 integ->prompt_composer->config.newline_before_prompt =
429 config.display_newline_before_prompt;
430
431 /* Register with shell event hub for automatic updates
432 * This is the key Spec 25 <-> Spec 26 integration point */
433 result = lle_composer_register_shell_events(integ->prompt_composer,
434 integ->event_hub);
435 if (result != LLE_SUCCESS) {
436 lle_composer_cleanup(integ->prompt_composer);
437 free(integ->prompt_composer);
438 integ->prompt_composer = NULL;
439 return result;
440 }
441
442 return LLE_SUCCESS;
443}

497void lle_hard_reset(void) {
498 if (!g_lle_integration) {
499 return;
500 }
501
502 lle_shell_integration_t *integ = g_lle_integration;
503
504 /* Save history before destroying editor */
505 if (integ->editor && integ->editor->history_system) {
506 const char *home = getenv("HOME");
507 if (home) {
508 char history_path[1024];
509 snprintf(history_path, sizeof(history_path), "%s/.lusush_history",
510 home);
511 lle_history_save_to_file(integ->editor->history_system,
512 history_path);
513 }
514 }
515
516 /* Destroy current editor */
517 destroy_editor(integ);
518
519 /* Recreate editor */
520 lle_result_t result = create_and_configure_editor(integ);
521 if (result == LLE_SUCCESS) {
522 integ->init_state.editor_initialized = true;
523 integ->init_state.history_initialized = true;
524 }
525
526 /* Reset error counters */
527 integ->error_count = 0;
528 integ->ctrl_g_count = 0;
529 integ->recovery_mode = false;
530
531 /* Update statistics */
532 integ->hard_reset_count++;
533 integ->last_reset_time_us = get_timestamp_us();
534}

560void lle_shell_update_prompt(void) {
561 /* Use minimal fallback if LLE integration not available */
562 if (!g_lle_integration || !g_lle_integration->prompt_composer) {
563 symtable_set_global("PS1", (getuid() > 0) ? "$ " : "# ");
564 symtable_set_global("PS2", "> ");
565 return;
566 }
567
568 lle_prompt_composer_t *composer = g_lle_integration->prompt_composer;
569 lle_prompt_output_t output;
570 memset(&output, 0, sizeof(output));
571
572 /* Update background job count from executor */
573 executor_t *executor = get_global_executor();
574 if (executor) {
575 executor_update_job_status(executor);
576 int job_count = executor_count_jobs(executor);
577 lle_prompt_context_set_job_count(&composer->context, job_count);
578 }
579
580 /* Render the prompt */
581 lle_result_t result = lle_composer_render(composer, &output);
582 if (result == LLE_SUCCESS && output.ps1_len > 0) {
583 symtable_set_global("PS1", output.ps1);
584 symtable_set_global("PS2", output.ps2);
585 lle_composer_clear_regeneration_flag(composer);
586 return;
587 }
588
589 /* Rendering failed - use minimal fallback */
590 symtable_set_global("PS1", (getuid() > 0) ? "$ " : "# ");
591 symtable_set_global("PS2", "> ");
592}

625void lle_record_ctrl_g(void) {
626 if (!g_lle_integration) {
627 return;
628 }
629
630 uint64_t now = get_timestamp_us();
631
632 /* Check if this Ctrl+G is within the panic window */
633 if (now - g_lle_integration->last_ctrl_g_time_us <
634 LLE_CTRL_G_PANIC_WINDOW_US) {
635 g_lle_integration->ctrl_g_count++;
636 } else {
637 /* Reset counter - too much time passed */
638 g_lle_integration->ctrl_g_count = 1;
639 }
640
641 g_lle_integration->last_ctrl_g_time_us = now;
642
643 /* Check for panic threshold */
644 if (g_lle_integration->ctrl_g_count >= LLE_CTRL_G_PANIC_COUNT) {
645 /* Triple Ctrl+G detected - trigger hard reset */
646 g_lle_integration->ctrl_g_count = 0;
647 lle_hard_reset();
648 }

675 if (!g_lle_integration || !g_lle_integration->editor) {

679 g_lle_integration->total_readline_calls++;
680
681 /* If prompt is NULL, retrieve from PS1 (primary prompt)
682 * This is the standard behavior - input.c passes NULL to let
683 * the prompt system generate the themed prompt via
684 * lle_shell_update_prompt()
685 */
686 const char *effective_prompt = prompt;
687 if (!effective_prompt) {
688 /* Ensure prompt is up-to-date before reading */
689 lle_shell_update_prompt();

691 if (!effective_prompt) {
692 effective_prompt = "$ "; /* Ultimate fallback */
693 }
694 }
695
696 char *line = lle_readline(effective_prompt);
697
698 if (line) {
699 g_lle_integration->successful_reads++;
