1/**
2 * @file history_interactive_search.c
3 * @brief LLE History System - Interactive Search Implementation (Spec 09 Phase
4 * 3 Day 9)
5 *
6 * Implements Ctrl+R reverse incremental search for the LLE history system:
7 * - Real-time search as user types
8 * - Reverse chronological search (most recent first)
9 * - Navigation through search results (Ctrl+R for next, Ctrl+S for previous)
10 * - Accept/cancel search operations
11 * - State management for search sessions
12 *
13 * Behavior (matches bash/readline):
14 * - Ctrl+R enters search mode with empty query
15 * - Each keypress updates query and triggers new search
16 * - Ctrl+R during search moves to next (older) match
17 * - Ctrl+S during search moves to previous (newer) match
18 * - Enter accepts current match and exits search
19 * - Ctrl+C/Ctrl+G cancels search and returns to original line
20 * - Other keys exit search and process the key
21 *
22 * @date 2025-11-01
23 * @author LLE Implementation Team
24 */

53 size_t len = strlen(str);
54 char *dup = lle_pool_alloc(len + 1);

123/**
124 * Update prompt string based on current search state
125 */
126static void update_prompt_string(void) {
127 lle_interactive_search_session_t *session = &g_search_session;
128
129 if (!session->active) {
130 session->prompt_string[0] = '\0';
131 return;
132 }
133
134 switch (session->state) {
135 case LLE_SEARCH_STATE_ACTIVE:
136 if (session->results &&
137 lle_history_search_results_get_count(session->results) > 0) {
138 snprintf(session->prompt_string, SEARCH_PROMPT_MAX_LEN,
139 "(reverse-i-search)`%.255s': ", session->query);
140 } else {
141 snprintf(session->prompt_string, SEARCH_PROMPT_MAX_LEN,
142 "(reverse-i-search)`%.255s': ", session->query);
143 }
144 break;
145
146 case LLE_SEARCH_STATE_NO_RESULTS:
147 snprintf(session->prompt_string, SEARCH_PROMPT_MAX_LEN,
148 "(failed reverse-i-search)`%.255s': ", session->query);
149 break;
150
151 case LLE_SEARCH_STATE_FAILED:
152 snprintf(session->prompt_string, SEARCH_PROMPT_MAX_LEN,
153 "(failed reverse-i-search)`%.255s': ", session->query);
154 break;
155
156 default:
157 session->prompt_string[0] = '\0';
158 break;
159 }
160}

162/**
163 * Perform search with current query
164 *
165 * @return true if search succeeded and found results, false otherwise
166 */
167static bool perform_search(void) {
168 lle_interactive_search_session_t *session = &g_search_session;
169
170 if (!session->history_core) {
171 session->state = LLE_SEARCH_STATE_FAILED;
172 return false;
173 }
174
175 /* Clean up previous results */
176 if (session->results) {
177 lle_history_search_results_destroy(session->results);
178 session->results = NULL;
179 }
180
181 /* Empty query = no search */
182 if (session->query_len == 0) {
183 session->state = LLE_SEARCH_STATE_NO_RESULTS;
184 session->current_result_index = 0;
185 update_prompt_string();
186 return false;
187 }
188
189 /* Perform substring search (most useful for interactive search) */
190 session->results = lle_history_search_substring(
191 session->history_core, session->query, 100 /* max results */
192 );
193
194 if (!session->results) {
195 session->state = LLE_SEARCH_STATE_FAILED;
196 update_prompt_string();
197 return false;
198 }
199
200 /* Update statistics */
201 session->searches_performed++;
202 session->total_search_time_us +=
203 lle_history_search_results_get_time_us(session->results);
204
205 /* Check if we found any results */
206 size_t result_count =
207 lle_history_search_results_get_count(session->results);
208 if (result_count == 0) {
209 session->state = LLE_SEARCH_STATE_NO_RESULTS;
210 session->current_result_index = 0;
211 update_prompt_string();
212 return false;
213 }
214
215 /* Success - we have results */
216 session->state = LLE_SEARCH_STATE_ACTIVE;
217 session->current_result_index =
218 0; /* Start with most recent (highest score) */
219 update_prompt_string();
220
221 return true;
222}

229/**
230 * Initialize interactive search session
231 *
232 * Starts a new Ctrl+R search session. Saves the current line and cursor
233 * position so they can be restored on cancel.
234 *
235 * @param history_core History core engine
236 * @param current_line Current line buffer (will be saved for cancel)
237 * @param cursor_pos Current cursor position
238 * @return LLE_SUCCESS or error code
239 */
240lle_result_t
241lle_history_interactive_search_init(lle_history_core_t *history_core,
242 const char *current_line,
243 size_t cursor_pos) {
244 if (!history_core) {
245 return LLE_ERROR_INVALID_PARAMETER;
246 }
247
248 lle_interactive_search_session_t *session = &g_search_session;
249
250 /* If already active, cancel the previous search first */
251 if (session->active) {
252 lle_history_interactive_search_cancel();
253 }
254
255 /* CRITICAL FIX: Clean up any leftover results from previous session
256 * If the previous search ended with accept() rather than cancel(),
257 * the results were kept alive for the caller to read. We must free
258 * them now to prevent memory leak.
259 */
260 if (session->results) {
261 lle_history_search_results_destroy(session->results);
262 session->results = NULL;
263 }
264
265 /* Initialize session */
266 session->state = LLE_SEARCH_STATE_ACTIVE;
267 session->active = true;
268 session->history_core = history_core;
269 session->query[0] = '\0';
270 session->query_len = 0;
271 session->results = NULL;
272 session->current_result_index = 0;
273
274 /* Save original state for cancel */
275 if (session->original_line) {
276 lle_pool_free(session->original_line);
277 }
278 session->original_line = current_line ? pool_strdup(current_line) : NULL;
279 session->original_cursor_pos = cursor_pos;
280
281 /* Update prompt */
282 update_prompt_string();
283
284 return LLE_SUCCESS;
285}

287/**
288 * Update search query with new character
289 *
290 * Appends a character to the search query and re-runs the search.
291 * This is called for each keypress during interactive search.
292 *
293 * @param c Character to append to query
294 * @return LLE_SUCCESS or error code
295 */
296lle_result_t lle_history_interactive_search_update_query(char c) {
297 lle_interactive_search_session_t *session = &g_search_session;
298
299 if (!session->active) {
300 return LLE_ERROR_INVALID_STATE;
301 }
302
303 /* Check if we have room for another character */
304 if (session->query_len >= SEARCH_QUERY_MAX_LEN - 1) {
305 return LLE_ERROR_BUFFER_OVERFLOW;
306 }
307
308 /* Append character to query */
309 session->query[session->query_len] = c;
310 session->query_len++;
311 session->query[session->query_len] = '\0';
312
313 /* Re-run search with updated query */
314 perform_search();
315
316 return LLE_SUCCESS;
317}

347/**
348 * Move to next (older) search result
349 *
350 * Called when user presses Ctrl+R during an active search.
351 * Moves to the next match in the result list (older command).
352 *
353 * @return LLE_SUCCESS or error code
354 */
355lle_result_t lle_history_interactive_search_next(void) {
356 lle_interactive_search_session_t *session = &g_search_session;
357
358 if (!session->active) {
359 return LLE_ERROR_INVALID_STATE;
360 }
361
362 if (!session->results || session->state != LLE_SEARCH_STATE_ACTIVE) {
363 /* No results to navigate */
364 return LLE_SUCCESS;
365 }
366
367 size_t result_count =
368 lle_history_search_results_get_count(session->results);
369 if (result_count == 0) {
370 return LLE_SUCCESS;
371 }
372
373 /* Move to next result (wrap around) */
374 session->current_result_index++;
375 if (session->current_result_index >= result_count) {
376 session->current_result_index = 0; /* Wrap to first result */
377 }
378
379 return LLE_SUCCESS;

419/**
420 * Accept current search result and exit search mode
421 *
422 * Returns the selected command for execution. The caller is responsible
423 * for placing this command in the input buffer.
424 *
425 * @return Command string (owned by search session, must be copied) or NULL
426 */
427const char *lle_history_interactive_search_accept(void) {
428 lle_interactive_search_session_t *session = &g_search_session;
429
430 if (!session->active) {
431 return NULL;
432 }
433
434 const char *selected_command = NULL;
435
436 /* Get the currently selected result */
437 if (session->results && session->state == LLE_SEARCH_STATE_ACTIVE) {
438 size_t result_count =
439 lle_history_search_results_get_count(session->results);
440 if (result_count > 0 && session->current_result_index < result_count) {
441 const lle_search_result_t *result = lle_history_search_results_get(
442 session->results, session->current_result_index);
443 if (result) {
444 selected_command = result->command;
445 }
446 }
447 }
448
449 /* Clean up session but don't free the command yet (caller needs it) */
450 if (session->results) {
451 /* Note: We return a pointer to command in results, so we can't destroy
452 * yet. Caller must copy the string before we destroy results. For now,
453 * we'll keep results alive and destroy on next init. */
454 }
455
456 /* Mark session as inactive */
457 session->active = false;
458 session->state = LLE_SEARCH_STATE_INACTIVE;
459
460 return selected_command;
461}

463/**
464 * Cancel search and restore original line
465 *
466 * Returns the original line that was active when search started.
467 *
468 * @return Original line (owned by session, must be copied) or NULL
469 */
470const char *lle_history_interactive_search_cancel(void) {
471 lle_interactive_search_session_t *session = &g_search_session;
472
473 if (!session->active) {
474 return NULL;
475 }
476
477 const char *original = session->original_line;
478
479 /* Clean up search results */
480 if (session->results) {
481 lle_history_search_results_destroy(session->results);
482 session->results = NULL;
483 }
484
485 /* Mark session as inactive */
486 session->active = false;
487 session->state = LLE_SEARCH_STATE_INACTIVE;
488
489 /* Note: We don't free original_line here because caller needs it.
490 * It will be freed on next init or when session is destroyed. */
491
492 return original;
493}
