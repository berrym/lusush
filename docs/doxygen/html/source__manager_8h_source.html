32/**
33 * Completion source types
34 */
35typedef enum {
36 LLE_SOURCE_BUILTINS, /* Shell builtin commands */
37 LLE_SOURCE_EXTERNAL_COMMANDS, /* External commands in PATH */
38 LLE_SOURCE_FILES, /* File/directory paths */
39 LLE_SOURCE_VARIABLES, /* Environment variables */
40 LLE_SOURCE_HISTORY, /* Command history */
41 LLE_SOURCE_ALIASES, /* Shell aliases (future) */
42 LLE_SOURCE_FUNCTIONS, /* Shell functions (future) */
43 LLE_SOURCE_CUSTOM, /* User-registered custom sources */
44} lle_source_type_t;

50/**

53 * @param pool Memory pool for allocations

55 * @param prefix Prefix to match
56 * @param result Result structure to append completions to

64 * Source applicability function signature

69typedef bool (*lle_source_applicable_fn)(const lle_context_analyzer_t *context);

71/**
72 * Single completion source
73 */
74struct lle_completion_source {
75 lle_source_type_t type;
76 const char *name;
77
78 /* Source function - generates completions for given prefix */
79 lle_source_generate_fn generate;

81 /* Optional: Check if source is applicable for context */
82 lle_source_applicable_fn is_applicable;

84 void *user_data; /* Source-specific data */
85};
