1/*
2 * parser_state_machine.c - Input Parser State Machine
3 *
4 * Manages parser state transitions for coordinating input processing across
5 * multiple parsing components (UTF-8 processor, sequence parser, key detector,
6 * mouse parser).
7 *
8 * States:
9 * - NORMAL: Regular text input
10 * - ESCAPE: Escape sequence started (ESC received)
11 * - CSI: CSI sequence in progress (ESC[ received)
12 * - OSC: OSC sequence in progress (ESC] received)
13 * - DCS: DCS sequence in progress (ESC P received)
14 * - UTF8_MULTIBYTE: UTF-8 multibyte sequence in progress
15 * - KEY_SEQUENCE: Key sequence detection in progress
16 * - MOUSE: Mouse event parsing in progress
17 * - ERROR_RECOVERY: Error recovery state
18 *
19 * The state machine tracks state transitions and provides query functions.
20 * Actual parser invocations are handled by the integration layer.
21 *
22 * Spec 06: Input Parsing - Phase 6
23 */

30/* Get current time in microseconds */
31static uint64_t get_current_time_us(void) {
32 struct timespec ts;
33 clock_gettime(CLOCK_MONOTONIC, &ts);
34 return (uint64_t)ts.tv_sec * 1000000 + (uint64_t)ts.tv_nsec / 1000;
35}

42 lle_error_context_t *error_ctx,
43 lle_memory_pool_t *memory_pool) {
44 if (!state_machine) {
45 return LLE_ERROR_INVALID_PARAMETER;

48 lle_parser_state_machine_t *sm =
49 lle_pool_alloc(sizeof(lle_parser_state_machine_t));
50 if (!sm) {
51 return LLE_ERROR_OUT_OF_MEMORY;
52 }
53
54 memset(sm, 0, sizeof(lle_parser_state_machine_t));
55
56 sm->current_state = LLE_PARSER_STATE_NORMAL;
57 sm->previous_state = LLE_PARSER_STATE_NORMAL;
58 sm->state_transitions = 0;
59 sm->state_change_time = get_current_time_us();
60 sm->error_ctx = error_ctx;
61 sm->error_recoveries = 0;
62 sm->memory_pool = memory_pool;
63
64 *state_machine = sm;
65 return LLE_SUCCESS;
66}

81/*
82 * Transition to new state
83 */
84lle_result_t
85lle_parser_state_machine_transition(lle_parser_state_machine_t *state_machine,
86 lle_parser_state_t new_state) {
87 if (!state_machine) {
88 return LLE_ERROR_INVALID_PARAMETER;
89 }
90
91 if (state_machine->current_state != new_state) {
92 state_machine->previous_state = state_machine->current_state;
93 state_machine->current_state = new_state;
94 state_machine->state_transitions++;
95 state_machine->state_change_time = get_current_time_us();
96 }
97
98 return LLE_SUCCESS;
99}

101/*
102 * Main processing function - analyzes input and determines state transitions
103 *
104 * This is a lightweight function that examines the input data and determines
105 * which parser state should handle it. The actual parsing is done by the
106 * integration layer.
107 */
108lle_result_t
109lle_parser_state_machine_process(lle_parser_state_machine_t *state_machine,
110 lle_input_parser_system_t *parser_sys,
111 const char *data, size_t data_len) {
112 (void)parser_sys; /* Reserved for parser-system-aware processing */
113 if (!state_machine || !data) {
114 return LLE_ERROR_INVALID_PARAMETER;
115 }
116
117 if (data_len == 0) {
118 return LLE_SUCCESS;
119 }
120
121 /* The process function provides basic state routing hints based on input.
122 * The integration layer will do the actual parsing. */
123
124 /* Check current state and input to determine transitions */
125 switch (state_machine->current_state) {
126 case LLE_PARSER_STATE_NORMAL:
127 /* Check if escape sequence is starting */
128 if (data[0] == '\x1B') {
129 lle_parser_state_machine_transition(state_machine,
130 LLE_PARSER_STATE_ESCAPE);
131 }
132 /* Otherwise stay in NORMAL state for text processing */
133 break;
134
135 case LLE_PARSER_STATE_ESCAPE:
136 /* Analyze second byte to determine sequence type */
137 if (data_len >= 2) {
138 if (data[1] == '[') {
139 /* Could be CSI, mouse, or key sequence */
140 if (data_len >= 3) {
141 if (data[2] == 'M' || data[2] == '<') {
142 /* Mouse sequence */
143 lle_parser_state_machine_transition(
144 state_machine, LLE_PARSER_STATE_MOUSE);
145 } else {
146 /* CSI sequence */
147 lle_parser_state_machine_transition(
148 state_machine, LLE_PARSER_STATE_CSI);
149 }
150 }
151 } else if (data[1] == ']') {
152 /* OSC sequence */
153 lle_parser_state_machine_transition(state_machine,
154 LLE_PARSER_STATE_OSC);
155 } else if (data[1] == 'P') {
156 /* DCS sequence */
157 lle_parser_state_machine_transition(state_machine,
158 LLE_PARSER_STATE_DCS);
159 } else if (data[1] == 'O' || data[1] == 'N') {
160 /* SS3 or SS2 - key sequence */
161 lle_parser_state_machine_transition(
162 state_machine, LLE_PARSER_STATE_KEY_SEQUENCE);
163 } else {
164 /* Other escape sequence - treat as key */
165 lle_parser_state_machine_transition(
166 state_machine, LLE_PARSER_STATE_KEY_SEQUENCE);
167 }
168 }
169 break;
170
171 case LLE_PARSER_STATE_CSI:
172 case LLE_PARSER_STATE_OSC:
173 case LLE_PARSER_STATE_DCS:
174 case LLE_PARSER_STATE_KEY_SEQUENCE:
175 case LLE_PARSER_STATE_MOUSE:
176 case LLE_PARSER_STATE_UTF8_MULTIBYTE:
177 /* These states are handled by the integration layer.
178 * The state machine just tracks that we're in these states.
179 * Integration layer will call transition() when parsing completes. */
180 break;
181
182 case LLE_PARSER_STATE_ERROR_RECOVERY:
183 /* Error recovery - transition back to normal */
184 lle_parser_state_machine_transition(state_machine,
185 LLE_PARSER_STATE_NORMAL);
186 state_machine->error_recoveries++;
187 break;
188
189 default:
190 /* Unknown state - recover */
191 lle_parser_state_machine_transition(state_machine,
192 LLE_PARSER_STATE_ERROR_RECOVERY);
193 state_machine->error_recoveries++;
194 break;
195 }
196
197 return LLE_SUCCESS;
198}
