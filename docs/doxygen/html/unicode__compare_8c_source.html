3 *
4 * Implements NFC normalization and Unicode-aware string comparison.
5 * Uses lookup tables for common Latin characters with diacritics.

7 * For full Unicode coverage, a complete implementation would need the
8 * Unicode Character Database (UCD), but for shell command comparison
9 * we focus on Latin-1 Supplement and Latin Extended-A/B which cover
10 * the vast majority of real-world cases.

308/*
309 * Binary search for decomposition entry
310 */
311static const decomposition_entry_t *find_decomposition(uint32_t codepoint) {
312 size_t left = 0;
313 size_t right = decomposition_table_size;
314
315 while (left < right) {
316 size_t mid = left + (right - left) / 2;
317 if (decomposition_table[mid].composed < codepoint) {
318 left = mid + 1;
319 } else if (decomposition_table[mid].composed > codepoint) {
320 right = mid;
321 } else {
322 return &decomposition_table[mid];
323 }
324 }
325
326 return NULL;
327}

329/*
330 * Find composition (reverse lookup)
331 */
332static bool find_composition(uint32_t base, uint32_t combining,
333 uint32_t *composed) {
334 /* Linear search for composition (could optimize with hash table) */
335 for (size_t i = 0; i < decomposition_table_size; i++) {
336 if (decomposition_table[i].base == base &&
337 decomposition_table[i].combining == combining) {
338 *composed = decomposition_table[i].composed;
339 return true;
340 }
341 }
342 return false;
343}

345/*
346 * Convert codepoint to lowercase (simple case folding)
347 */
348static uint32_t to_lowercase(uint32_t cp) {
349 /* ASCII range */
350 if (cp >= 'A' && cp <= 'Z') {
351 return cp + ('a' - 'A');
352 }
353
354 /* Latin-1 Supplement uppercase */
355 if (cp >= 0x00C0 && cp <= 0x00D6) {
356 return cp + 0x20;
357 }
358 if (cp >= 0x00D8 && cp <= 0x00DE) {
359 return cp + 0x20;
360 }
361
362 /* Latin Extended-A pairs (most are sequential uppercase/lowercase) */
363 if (cp >= 0x0100 && cp <= 0x0137 && (cp & 1) == 0) {
364 return cp + 1;
365 }
366 if (cp >= 0x0139 && cp <= 0x0148 && (cp & 1) == 1) {
367 return cp + 1;
368 }
369 if (cp >= 0x014A && cp <= 0x0177 && (cp & 1) == 0) {
370 return cp + 1;
371 }
372 if (cp >= 0x0179 && cp <= 0x017E && (cp & 1) == 1) {
373 return cp + 1;
374 }
375
376 return cp;
377}

384bool lle_unicode_is_combining(uint32_t codepoint) {
385 /* Combining Diacritical Marks (0300-036F) */
386 if (codepoint >= 0x0300 && codepoint <= 0x036F) {
387 return true;
388 }
389 /* Combining Diacritical Marks Extended (1AB0-1AFF) */
390 if (codepoint >= 0x1AB0 && codepoint <= 0x1AFF) {
391 return true;
392 }
393 /* Combining Diacritical Marks Supplement (1DC0-1DFF) */
394 if (codepoint >= 0x1DC0 && codepoint <= 0x1DFF) {
395 return true;
396 }
397 /* Combining Diacritical Marks for Symbols (20D0-20FF) */
398 if (codepoint >= 0x20D0 && codepoint <= 0x20FF) {
399 return true;
400 }
401 /* Combining Half Marks (FE20-FE2F) */
402 if (codepoint >= 0xFE20 && codepoint <= 0xFE2F) {
403 return true;
404 }
405 return false;
406}

408uint8_t lle_unicode_combining_class(uint32_t codepoint) {
409 /* Binary search in combining class table */
410 size_t left = 0;
411 size_t right = combining_class_table_size;
412
413 while (left < right) {
414 size_t mid = left + (right - left) / 2;
415 if (combining_class_table[mid].codepoint < codepoint) {
416 left = mid + 1;
417 } else if (combining_class_table[mid].codepoint > codepoint) {
418 right = mid;
419 } else {
420 return combining_class_table[mid].ccc;
421 }
422 }
423
424 /* Default: 0 for base characters, 230 for combining marks */
425 if (lle_unicode_is_combining(codepoint)) {
426 return 230; /* Default above class */
427 }
428 return 0; /* Starter */
429}

431int lle_unicode_decompose(uint32_t codepoint, uint32_t *decomposed,
432 int max_length) {
433 if (!decomposed || max_length < 1) {
434 return 0;
435 }
436
437 const decomposition_entry_t *entry = find_decomposition(codepoint);
438 if (!entry) {
439 return 0; /* No decomposition */
440 }
441
442 decomposed[0] = entry->base;
443 if (entry->combining != 0 && max_length >= 2) {
444 decomposed[1] = entry->combining;
445 return 2;
446 }
447
448 return 1;
449}

466#define MAX_DECOMPOSED 256
467 uint32_t decomposed[MAX_DECOMPOSED];
468 size_t decomposed_count = 0;
469
470 const char *ptr = input;
471 const char *end = input + input_len;
472
473 while (ptr < end && decomposed_count < MAX_DECOMPOSED - 4) {
474 uint32_t codepoint;
475 int seq_len = lle_utf8_decode_codepoint(ptr, end - ptr, &codepoint);
476 if (seq_len <= 0) {
477 return -3; /* Invalid UTF-8 */
478 }
479
480 /* Try to decompose */
481 uint32_t parts[4];
482 int parts_count = lle_unicode_decompose(codepoint, parts, 4);
483
484 if (parts_count > 0) {
485 /* Has decomposition */
486 for (int i = 0;
487 i < parts_count && decomposed_count < MAX_DECOMPOSED; i++) {
488 decomposed[decomposed_count++] = parts[i];
489 }
490 } else {
491 /* No decomposition, keep original */
492 decomposed[decomposed_count++] = codepoint;
493 }
494
495 ptr += seq_len;
496 }
497
498 /* Step 2: Canonical ordering of combining marks */
499 for (size_t i = 1; i < decomposed_count; i++) {
500 uint8_t ccc = lle_unicode_combining_class(decomposed[i]);
501 if (ccc == 0)
502 continue; /* Starter, don't reorder */
503
504 size_t j = i;
505 while (j > 0) {
506 uint8_t prev_ccc = lle_unicode_combining_class(decomposed[j - 1]);
507 if (prev_ccc == 0 || prev_ccc <= ccc) {
508 break; /* Correct order */
509 }
510 /* Swap */
511 uint32_t tmp = decomposed[j];
512 decomposed[j] = decomposed[j - 1];
513 decomposed[j - 1] = tmp;
514 j--;
515 }
516 }
517
518 /* Step 3: Compose (NFC) */
519 size_t composed_count = 0;
520 uint32_t composed[MAX_DECOMPOSED];
521
522 for (size_t i = 0; i < decomposed_count; i++) {
523 uint32_t cp = decomposed[i];
524
525 if (composed_count == 0) {
526 composed[composed_count++] = cp;
527 continue;
528 }
529
530 /* Try to compose with last starter */
531 size_t last_starter = composed_count - 1;
532 while (last_starter > 0 &&
533 lle_unicode_combining_class(composed[last_starter]) != 0) {
534 last_starter--;
535 }
536
537 uint32_t result;
538 if (lle_unicode_compose(composed[last_starter], cp, &result)) {
539 composed[last_starter] = result;
540 } else {
541 composed[composed_count++] = cp;
542 }
543 }
544
545 /* Step 4: Encode back to UTF-8 */
546 char *out_ptr = output;
547 size_t remaining = output_size - 1; /* Reserve for null terminator */
548
549 for (size_t i = 0; i < composed_count; i++) {
550 char buf[4];
551 int len = lle_utf8_encode_codepoint(composed[i], buf);
552 if (len <= 0 || (size_t)len > remaining) {
553 return -2; /* Buffer too small */
554 }
555 memcpy(out_ptr, buf, len);
556 out_ptr += len;
557 remaining -= len;
558 }
559
560 *out_ptr = '\0';
561 if (output_len) {
562 *output_len = out_ptr - output;
563 }
564
565 return 0;
566}

579bool lle_unicode_is_prefix(const char *prefix, size_t prefix_len,
580 const char *str, size_t str_len,
581 const lle_unicode_compare_options_t *options) {
582 /* Handle edge cases */
583 if (!prefix || prefix_len == 0) {
584 return true; /* Empty prefix matches everything */
585 }
586 if (!str || str_len == 0) {
587 return false; /* Non-empty prefix can't match empty string */
588 }
589 if (prefix_len > str_len) {
590 return false; /* Prefix longer than string - quick rejection */
591 }
592
593 /* Use default options if none provided */
594 lle_unicode_compare_options_t opts =
595 options ? *options : LLE_UNICODE_COMPARE_DEFAULT;
596
597 /*
598 * Fast path: If no normalization needed, use simple byte comparison.
599 * This handles the common case of ASCII-only input efficiently.
600 */
601 if (!opts.normalize && !opts.case_insensitive) {
602 return memcmp(prefix, str, prefix_len) == 0;
603 }
604
605 /*
606 * Unicode-aware path: Normalize both strings and compare.
607 * We need to be careful about grapheme boundaries - the prefix
608 * should end at a grapheme boundary in the normalized form.
609 */
610#define PREFIX_NORM_BUF_SIZE 4096
611 char norm_prefix[PREFIX_NORM_BUF_SIZE];
612 char norm_str[PREFIX_NORM_BUF_SIZE];
613 size_t norm_prefix_len, norm_str_len;
614
615 /* Normalize prefix */
616 if (opts.normalize) {
617 if (lle_unicode_normalize_nfc(prefix, prefix_len, norm_prefix,
618 PREFIX_NORM_BUF_SIZE,
619 &norm_prefix_len) != 0) {
620 /* Normalization failed, fall back to byte comparison */
621 return memcmp(prefix, str, prefix_len) == 0;
622 }
623
624 /* Normalize string */
625 if (lle_unicode_normalize_nfc(str, str_len, norm_str,
626 PREFIX_NORM_BUF_SIZE,
627 &norm_str_len) != 0) {
628 /* Normalization failed, fall back to byte comparison */
629 return memcmp(prefix, str, prefix_len) == 0;
630 }
631 } else {
632 /* No normalization - copy to buffers for uniform handling */
633 if (prefix_len >= PREFIX_NORM_BUF_SIZE ||
634 str_len >= PREFIX_NORM_BUF_SIZE) {
635 /* Strings too long for buffers, fall back */
636 return memcmp(prefix, str, prefix_len) == 0;
637 }
638 memcpy(norm_prefix, prefix, prefix_len);
639 norm_prefix[prefix_len] = '\0';
640 norm_prefix_len = prefix_len;
641 memcpy(norm_str, str, str_len);
642 norm_str[str_len] = '\0';
643 norm_str_len = str_len;
644 }
645
646 /* Quick length check after normalization */
647 if (norm_prefix_len > norm_str_len) {
648 return false;
649 }
650
651 /*
652 * Compare codepoint by codepoint, optionally with case folding.
653 * Track position in both strings to ensure we match exactly
654 * norm_prefix_len worth of normalized prefix.
655 */
656 const char *pp = norm_prefix;
657 const char *pp_end = norm_prefix + norm_prefix_len;
658 const char *sp = norm_str;
659 const char *sp_end = norm_str + norm_str_len;
660
661 if (opts.case_insensitive) {
662 while (pp < pp_end && sp < sp_end) {
663 uint32_t cp_prefix, cp_str;
664 int len_prefix =
665 lle_utf8_decode_codepoint(pp, pp_end - pp, &cp_prefix);
666 int len_str = lle_utf8_decode_codepoint(sp, sp_end - sp, &cp_str);
667
668 if (len_prefix <= 0 || len_str <= 0) {
669 return false; /* Invalid UTF-8 */
670 }
671
672 if (to_lowercase(cp_prefix) != to_lowercase(cp_str)) {
673 return false;
674 }
675
676 pp += len_prefix;
677 sp += len_str;
678 }
679 } else {
680 /* Direct comparison without case folding */
681 if (memcmp(norm_prefix, norm_str, norm_prefix_len) != 0) {
682 return false;
683 }
684 pp = pp_end; /* Mark as fully consumed */
685 }
686
687 /* Prefix matched if we consumed all of it */
688 return (pp >= pp_end);
689}

704bool lle_unicode_strings_equal_n(const char *str1, size_t len1,
705 const char *str2, size_t len2,
706 const lle_unicode_compare_options_t *options) {
707 if (!str1 && !str2)
708 return true;
709 if (!str1 || !str2)
710 return false;
711
712 /* Use default options if none provided */
713 lle_unicode_compare_options_t opts =
714 options ? *options : LLE_UNICODE_COMPARE_DEFAULT;
715
716 /* Fast path: if no normalization needed and same length, use memcmp */
717 if (!opts.normalize && !opts.case_insensitive && !opts.trim_whitespace) {
718 if (len1 != len2)
719 return false;
720 return memcmp(str1, str2, len1) == 0;
721 }
722
723 /* Handle whitespace trimming */
724 const char *s1 = str1;
725 const char *e1 = str1 + len1;
726 const char *s2 = str2;
727 const char *e2 = str2 + len2;
728
729 if (opts.trim_whitespace) {
730 while (s1 < e1 && (*s1 == ' ' || *s1 == '\t'))
731 s1++;
732 while (e1 > s1 && (e1[-1] == ' ' || e1[-1] == '\t'))
733 e1--;
734 while (s2 < e2 && (*s2 == ' ' || *s2 == '\t'))
735 s2++;
736 while (e2 > s2 && (e2[-1] == ' ' || e2[-1] == '\t'))
737 e2--;
738 }
739
740 size_t trimmed_len1 = e1 - s1;
741 size_t trimmed_len2 = e2 - s2;
742
743 /* Normalize if requested */
744 if (opts.normalize) {
745#define NORM_BUF_SIZE 4096
746 char norm1[NORM_BUF_SIZE];
747 char norm2[NORM_BUF_SIZE];
748 size_t norm1_len, norm2_len;
749
750 if (lle_unicode_normalize_nfc(s1, trimmed_len1, norm1, NORM_BUF_SIZE,
751 &norm1_len) != 0) {
752 /* Normalization failed, fall back to byte comparison */
753 goto byte_compare;
754 }
755 if (lle_unicode_normalize_nfc(s2, trimmed_len2, norm2, NORM_BUF_SIZE,
756 &norm2_len) != 0) {
757 goto byte_compare;
758 }
759
760 if (opts.case_insensitive) {
761 /* Compare codepoint by codepoint with case folding */
762 const char *p1 = norm1;
763 const char *end1 = norm1 + norm1_len;
764 const char *p2 = norm2;
765 const char *end2 = norm2 + norm2_len;
766
767 while (p1 < end1 && p2 < end2) {
768 uint32_t cp1, cp2;
769 int len1 = lle_utf8_decode_codepoint(p1, end1 - p1, &cp1);
770 int len2 = lle_utf8_decode_codepoint(p2, end2 - p2, &cp2);
771
772 if (len1 <= 0 || len2 <= 0) {
773 return false;
774 }
775
776 if (to_lowercase(cp1) != to_lowercase(cp2)) {
777 return false;
778 }
779
780 p1 += len1;
781 p2 += len2;
782 }
783
784 return (p1 == end1 && p2 == end2);
785 } else {
786 /* Direct comparison of normalized strings */
787 if (norm1_len != norm2_len)
788 return false;
789 return memcmp(norm1, norm2, norm1_len) == 0;
790 }
791 }
792
793byte_compare:
794 /* Case-insensitive without normalization */
795 if (opts.case_insensitive) {
796 const char *p1 = s1;
797 const char *p2 = s2;
798
799 while (p1 < e1 && p2 < e2) {
800 uint32_t cp1, cp2;
801 int len1 = lle_utf8_decode_codepoint(p1, e1 - p1, &cp1);
802 int len2 = lle_utf8_decode_codepoint(p2, e2 - p2, &cp2);
803
804 if (len1 <= 0 || len2 <= 0) {
805 return false;
806 }
807
808 if (to_lowercase(cp1) != to_lowercase(cp2)) {
809 return false;
810 }
811
812 p1 += len1;
813 p2 += len2;
814 }
815
816 return (p1 == e1 && p2 == e2);
817 }
818
819 /* Simple byte comparison */
820 if (trimmed_len1 != trimmed_len2)
821 return false;
822 return memcmp(s1, s2, trimmed_len1) == 0;
823}
