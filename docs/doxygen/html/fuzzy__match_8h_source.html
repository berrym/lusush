1/**
2 * @file fuzzy_match.h
3 * @brief Unicode-aware fuzzy string matching library
4 *
5 * Provides fuzzy string matching algorithms with full Unicode support.
6 * Used by both the shell's autocorrect system and LLE's completion/history
7 * search.
8 *
9 * Features:
10 * - Levenshtein (edit) distance
11 * - Damerau-Levenshtein distance (with transpositions)
12 * - Jaro-Winkler similarity
13 * - Subsequence matching
14 * - Combined weighted scoring
15 * - Unicode NFC normalization support
16 * - Case-insensitive matching with Unicode case folding
17 *
18 * @author Michael Berry <trismegustis@gmail.com>
19 * @copyright Copyright (C) 2021-2026 Michael Berry
20 */

64/**
65 * Calculate combined fuzzy similarity score between two strings
66 *
67 * Uses a weighted combination of multiple algorithms:
68 * - Levenshtein/Damerau-Levenshtein distance (40%)
69 * - Jaro-Winkler similarity (30%)
70 * - Common prefix bonus (20%)
71 * - Subsequence score (10%)
72 *
73 * @param s1 First string (UTF-8)
74 * @param s2 Second string (UTF-8)
75 * @param options Matching options (NULL for defaults)
76 * @return Similarity score 0-100 (100 = identical)
77 */
78int fuzzy_match_score(const char *s1, const char *s2,
79 const fuzzy_match_options_t *options);

91int fuzzy_match_score_n(const char *s1, size_t len1, const char *s2,

111/**
112 * Calculate Levenshtein (edit) distance
113 *
114 * Counts minimum single-character edits (insertions, deletions, substitutions)
115 * needed to transform s1 into s2.
116 *
117 * @param s1 First string (UTF-8)
118 * @param s2 Second string (UTF-8)
119 * @param options Matching options (NULL for defaults)
120 * @return Edit distance (0 = identical, higher = more different)
121 */
122int fuzzy_levenshtein_distance(const char *s1, const char *s2,
123 const fuzzy_match_options_t *options);
124
125/**
126 * Calculate Damerau-Levenshtein distance
127 *
128 * Like Levenshtein but also considers transpositions (swapping adjacent chars)
129 * as a single edit. Better for detecting typos like "teh" -> "the".
130 *
131 * @param s1 First string (UTF-8)
132 * @param s2 Second string (UTF-8)
133 * @param options Matching options (NULL for defaults)
134 * @return Edit distance with transpositions
135 */
136int fuzzy_damerau_levenshtein_distance(const char *s1, const char *s2,
137 const fuzzy_match_options_t *options);
138
139/**
140 * Calculate Jaro similarity
141 *
142 * Measures character matches within a sliding window, accounting for
143 * transpositions. Good for short strings and typo detection.
144 *
145 * @param s1 First string (UTF-8)
146 * @param s2 Second string (UTF-8)
147 * @param options Matching options (NULL for defaults)
148 * @return Similarity score 0-100
149 */
150int fuzzy_jaro_score(const char *s1, const char *s2,
151 const fuzzy_match_options_t *options);
152
153/**
154 * Calculate Jaro-Winkler similarity
155 *
156 * Jaro similarity with a bonus for matching prefixes.
157 * Especially good for command names that often share prefixes.
158 *
159 * @param s1 First string (UTF-8)
160 * @param s2 Second string (UTF-8)
161 * @param options Matching options (NULL for defaults)
162 * @return Similarity score 0-100
163 */
164int fuzzy_jaro_winkler_score(const char *s1, const char *s2,
165 const fuzzy_match_options_t *options);

178/**
