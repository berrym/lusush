21/*
22 * Context types for the context stack
23 * Used to track nested shell constructs for proper prompt display
24 */
25typedef enum {
26 CONTEXT_NONE = 0,
27 CONTEXT_IF,
28 CONTEXT_WHILE,
29 CONTEXT_FOR,
30 CONTEXT_UNTIL,
31 CONTEXT_CASE,
32 CONTEXT_FUNCTION,

41/*
42 * Continuation state structure
43 *
44 * Tracks the parsing state for multiline input constructs including:
45 * - Quote tracking (single, double, backtick)
46 * - Bracket/brace/parenthesis counting
47 * - Control structure detection (if/then/fi, case, loops)
48 * - Here document handling
49 * - Function definitions
50 * - Context stack for proper nested construct tracking
51 */
52typedef struct {
53 // Quote tracking
54 int quote_count;
55 int double_quote_count;
56 int backtick_count;
57 bool in_single_quote;
58 bool in_double_quote;
59 bool in_backtick;
60
61 // Bracket/brace/parenthesis tracking
62 int paren_count;
63 int brace_count;
64 int bracket_count;
65
66 // Escape and continuation
67 bool escaped;
68 bool has_continuation;
69
70 // Here document handling
71 bool in_here_doc;
72 char *here_doc_delimiter;
73
74 // Command substitution and arithmetic
75 bool in_command_substitution;
76 bool in_arithmetic;
77
78 // Control structure tracking (legacy flags - kept for compatibility)
79 bool in_function_definition;
80 bool in_case_statement;
81 bool in_if_statement;
82 bool in_while_loop;
83 bool in_for_loop;
84 bool in_until_loop;
85 int compound_command_depth;
86
87 // Context stack for nested construct tracking
88 // This allows proper prompt switching when entering/exiting nested
89 // constructs
90 continuation_context_type_t context_stack[CONTINUATION_MAX_CONTEXT_DEPTH];
91 int context_stack_depth;
92} continuation_state_t;

94/*
95 * Initialize a continuation state structure
96 *
97 * Parameters:
98 * state - Pointer to state structure to initialize

100void continuation_state_init(continuation_state_t *state);

102/*
103 * Clean up a continuation state structure
104 *
105 * Frees any dynamically allocated memory (e.g., here_doc_delimiter)
106 * and resets the structure to initial state.
107 *
108 * Parameters:
109 * state - Pointer to state structure to clean up
110 */
111void continuation_state_cleanup(continuation_state_t *state);

113/*
114 * Analyze a line and update the continuation state
115 *
116 * Parses the line for shell constructs, quotes, brackets, control
117 * keywords, and updates the state accordingly. This is the core
118 * parsing function.
119 *
120 * Parameters:
121 * line - The line to analyze (null-terminated string)
122 * state - Pointer to state structure to update
123 */
124void continuation_analyze_line(const char *line, continuation_state_t *state);
125
126/*
127 * Check if the current input is complete
128 *
129 * Returns true if all constructs are closed and the input does not
130 * require continuation. Returns false if the input needs more lines.
131 *
132 * Parameters:
133 * state - Pointer to state structure to check
134 *
135 * Returns:
136 * true if input is complete, false if continuation is needed
137 */
138bool continuation_is_complete(const continuation_state_t *state);
139
140/*
141 * Check if the current state needs continuation
142 *
143 * Similar to continuation_is_complete but returns the opposite value.
144 * This is a convenience function.
145 *
146 * Parameters:
147 * state - Pointer to state structure to check
148 *
149 * Returns:
150 * true if continuation is needed, false if input is complete
151 */
152bool continuation_needs_continuation(const continuation_state_t *state);
153
154/*
155 * Get an appropriate continuation prompt string
156 *
157 * Returns a prompt string based on the current parsing state.
158 * The returned string is a static or symbol table string and should
159 * not be freed by the caller.
160 *
161 * Parameters:
162 * state - Pointer to state structure
163 *
164 * Returns:
165 * Prompt string (e.g., "> ", "quote> ", "if> ", etc.)
166 */
167const char *continuation_get_prompt(const continuation_state_t *state);
168
169/*
170 * Check if a word is a shell control keyword
171 *
172 * Returns true for keywords like "if", "then", "while", "do", etc.
173 *
174 * Parameters:
175 * word - The word to check (null-terminated string)
176 *
177 * Returns:
178 * true if word is a control keyword, false otherwise
179 */
180bool continuation_is_control_keyword(const char *word);
181
182/*
183 * Check if a line is a control structure terminator
184 *
185 * Returns true for terminators like "fi", "done", "esac", "}".
186 *
187 * Parameters:
188 * line - The line to check (null-terminated string)
189 *
190 * Returns:
191 * true if line is a terminator, false otherwise
192 */
193bool continuation_is_terminator(const char *line);
