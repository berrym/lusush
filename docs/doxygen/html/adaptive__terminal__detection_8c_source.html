48/**
49 * Simple wildcard pattern matching.
50 * Supports * wildcard only (matches any sequence of characters).
51 */
52static bool pattern_match(const char *pattern, const char *string) {
53 if (!pattern || !string) {
54 return false;
55 }
56
57 /* Simple wildcard matching implementation */
58 const char *p = pattern;
59 const char *s = string;
60 const char *star = NULL;
61 const char *ss = NULL;
62
63 while (*s) {
64 if (*p == '*') {
65 /* Remember position of * for backtracking */
66 star = p++;
67 ss = s;
68 } else if (*p == *s || (*p >= 'A' && *p <= 'Z' && *p + 32 == *s) ||
69 (*p >= 'a' && *p <= 'z' && *p - 32 == *s)) {
70 /* Match (case-insensitive) */
71 p++;
72 s++;
73 } else if (star) {
74 /* Backtrack to * */
75 p = star + 1;
76 s = ++ss;
77 } else {
78 return false;
79 }
80 }
81
82 /* Skip trailing * in pattern */
83 while (*p == '*') {
84 p++;
85 }
86
87 return *p == '\0';
88}

112/**
113 * Analyze environment variables for terminal information.
114 */
115static lle_result_t
116analyze_environment_variables(lle_terminal_detection_result_t *detection) {
117 const char *term = getenv("TERM");
118 const char *term_program = getenv("TERM_PROGRAM");
119 const char *colorterm = getenv("COLORTERM");
120
121 /* Copy environment values */
122 safe_strncpy(detection->term_name, term ? term : "",
123 sizeof(detection->term_name));
124 safe_strncpy(detection->term_program, term_program ? term_program : "",
125 sizeof(detection->term_program));
126 safe_strncpy(detection->colorterm, colorterm ? colorterm : "",
127 sizeof(detection->colorterm));
128
129 /* Basic capability inference from environment */
130 detection->supports_colors =
131 (term && (strstr(term, "color") || strstr(term, "256")));
132 detection->supports_256_colors = (term && strstr(term, "256"));
133 detection->supports_truecolor =
134 (colorterm && (strcmp(colorterm, "truecolor") == 0 ||
135 strcmp(colorterm, "24bit") == 0));
136 detection->supports_unicode =
137 (colorterm != NULL); /* COLORTERM usually implies UTF-8 */
138
139 /* Detect terminal multiplexer */
140 const char *tmux = getenv("TMUX");
141 const char *sty = getenv("STY");
142
143 if (tmux && tmux[0] != '\0') {
144 detection->multiplexer_type = LLE_MUX_TYPE_TMUX;
145 } else if ((term && strstr(term, "screen")) || (sty && sty[0] != '\0')) {
146 detection->multiplexer_type = LLE_MUX_TYPE_SCREEN;
147 } else {
148 detection->multiplexer_type = LLE_MUX_TYPE_NONE;
149 }
150
151 return LLE_SUCCESS;
152}

159/**
160 * Match terminal signature from database.
161 */
162const lle_terminal_signature_t *
163lle_match_terminal_signature(const lle_terminal_detection_result_t *detection) {
164
165 if (!detection) {
166 return NULL;
167 }
168
169 size_t count = 0;
170 const lle_terminal_signature_t *signatures =
171 lle_get_terminal_signature_database(&count);
172
173 /* Iterate through signatures in priority order */
174 for (size_t i = 0; i < count; i++) {
175 const lle_terminal_signature_t *sig = &signatures[i];
176 bool match = true;
177
178 /* Check TERM_PROGRAM pattern */
179 if (sig->term_program_pattern) {
180 if (detection->term_program[0] == '\0' ||
181 !pattern_match(sig->term_program_pattern,
182 detection->term_program)) {
183 match = false;
184 }
185 }
186
187 /* Check TERM pattern */
188 if (match && sig->term_pattern) {
189 if (detection->term_name[0] == '\0' ||
190 !pattern_match(sig->term_pattern, detection->term_name)) {
191 match = false;
192 }
193 }
194
195 /* Check additional environment variable */
196 if (match && sig->env_var_check) {
197 const char *env_val = getenv(sig->env_var_check);
198 if (!env_val || env_val[0] == '\0') {
199 match = false;
200 }
201 }
202
203 if (match) {
204 return sig;
205 }
206 }
207
208 return NULL; /* No match found */
209}

216/**
217 * Probe single capability with timeout.
218 */
219static bool probe_capability_with_timeout(const char *query, int timeout_ms) {
220 /* Only probe if stdout is a TTY */
221 if (!isatty(STDOUT_FILENO)) {
222 return false;
223 }
224
225 /* Write query sequence */
226 if (write(STDOUT_FILENO, query, strlen(query)) < 0) {
227 return false;
228 }
229
230 /* Wait for response with timeout */
231 fd_set readfds;
232 struct timeval timeout;
233
234 FD_ZERO(&readfds);
235 FD_SET(STDIN_FILENO, &readfds);
236
237 timeout.tv_sec = timeout_ms / 1000;
238 timeout.tv_usec = (timeout_ms % 1000) * 1000;
239
240 int result = select(STDIN_FILENO + 1, &readfds, NULL, NULL, &timeout);
241
242 if (result > 0) {
243 /* Data available - read and discard response */
244 char buffer[256];
245 ssize_t bytes_read = read(STDIN_FILENO, buffer, sizeof(buffer) - 1);
246 return bytes_read > 0;
247 }
248
249 return false; /* Timeout or error */
250}

252/**
253 * Safe terminal capability probing with timeout protection.
254 */
255lle_result_t lle_probe_terminal_capabilities_safe(
256 lle_terminal_detection_result_t *detection) {
257
258 /* Cannot probe without stdout TTY - mark as unsuccessful but not an error
259 */
260 if (!detection->stdout_is_tty) {
261 detection->probing_successful = false;
262 /* Set all probe flags to false - this is complete behavior for non-TTY
263 */
264 detection->supports_cursor_positioning = false;
265 detection->supports_cursor_queries = false;
266 detection->supports_bracketed_paste = false;
267 detection->supports_mouse = false;
268 return LLE_SUCCESS; /* Successfully determined we cannot probe */
269 }
270
271 /* Save terminal state */
272 struct termios saved_termios;
273 if (tcgetattr(STDIN_FILENO, &saved_termios) != 0) {
274 detection->probing_successful = false;
275 return LLE_ERROR_TERMINAL_ABSTRACTION;
276 }
277
278 /* Set raw mode for accurate probing */
279 struct termios raw_termios = saved_termios;
280 raw_termios.c_iflag &= ~(BRKINT | ICRNL | INPCK | ISTRIP | IXON);
281 raw_termios.c_oflag &= ~(OPOST);
282 raw_termios.c_cflag |= (CS8);
283 raw_termios.c_lflag &= ~(ECHO | ICANON | IEXTEN | ISIG);
284 raw_termios.c_cc[VMIN] = 0;
285 raw_termios.c_cc[VTIME] = 0;
286
287 if (tcsetattr(STDIN_FILENO, TCSANOW, &raw_termios) != 0) {
288 detection->probing_successful = false;
289 return LLE_ERROR_TERMINAL_ABSTRACTION;
290 }
291
292 /* Progressive capability probing with timeout protection */
293
294 /* Test cursor positioning (DSR - Device Status Report) */
295 detection->supports_cursor_positioning =
296 probe_capability_with_timeout("\x1b[6n", 100);
297
298 /* If basic cursor support works, we likely have ANSI capability */
299 if (detection->supports_cursor_positioning) {
300 detection->supports_cursor_queries = true;
301 }
302
303 /* Test bracketed paste mode */
304 detection->supports_bracketed_paste =
305 probe_capability_with_timeout("\x1b[?2004h", 25);
306
307 /* Test mouse support */
308 detection->supports_mouse =
309 probe_capability_with_timeout("\x1b[?1000h", 50);
310
311 /* Restore terminal state */
312 tcsetattr(STDIN_FILENO, TCSANOW, &saved_termios);
313
314 detection->probing_successful = true;
315 return LLE_SUCCESS;
316}

323/**
324 * Determine fallback mode based on detection results.
325 */
326static lle_adaptive_mode_t
327determine_fallback_mode(const lle_terminal_detection_result_t *detection) {
328
329 /* No TTY at all - non-interactive */
330 if (!detection->stdin_is_tty && !detection->stdout_is_tty) {
331 return LLE_ADAPTIVE_MODE_NONE;
332 }
333
334 /* Both stdin and stdout are TTY - native mode */
335 if (detection->stdin_is_tty && detection->stdout_is_tty) {
336 return LLE_ADAPTIVE_MODE_NATIVE;
337 }
338
339 /* Only stdout is TTY - enhanced mode (editor terminal pattern) */
340 if (!detection->stdin_is_tty && detection->stdout_is_tty) {
341 return LLE_ADAPTIVE_MODE_ENHANCED;
342 }
343
344 /* Capable output but no TTY - minimal mode */
345 return LLE_ADAPTIVE_MODE_MINIMAL;
346}

348/**
349 * Validate and adjust recommended mode.
350 */
351static lle_adaptive_mode_t
352validate_and_adjust_mode(const lle_terminal_detection_result_t *detection) {
353
354 lle_adaptive_mode_t mode = detection->recommended_mode;
355
356 /* Validate mode is compatible with TTY status */
357 switch (mode) {
358 case LLE_ADAPTIVE_MODE_NATIVE:
359 /* Native mode requires stdin TTY */
360 if (!detection->stdin_is_tty) {
361 mode = LLE_ADAPTIVE_MODE_ENHANCED;
362 }
363 break;
364
365 case LLE_ADAPTIVE_MODE_ENHANCED:
366 /* Enhanced mode requires stdout TTY */
367 if (!detection->stdout_is_tty) {
368 mode = LLE_ADAPTIVE_MODE_MINIMAL;
369 }
370 break;
371
372 case LLE_ADAPTIVE_MODE_MULTIPLEXED:
373 /* Multiplexed mode requires both TTYs */
374 if (!detection->stdin_is_tty || !detection->stdout_is_tty) {
375 mode = LLE_ADAPTIVE_MODE_ENHANCED;
376 }
377 break;
378
379 case LLE_ADAPTIVE_MODE_MINIMAL:
380 case LLE_ADAPTIVE_MODE_NONE:
381 /* No validation needed */
382 break;
383 }
384
385 return mode;
386}

393/**
394 * Perform comprehensive terminal capability detection.
395 */
396lle_result_t lle_detect_terminal_capabilities_comprehensive(
397 lle_terminal_detection_result_t **result) {
398
399 if (!result) {
400 return LLE_ERROR_INVALID_PARAMETER;
401 }
402
403 /* Allocate detection result using standard malloc, not the pool.
404 * This is intentional: the detection result is cached as a singleton
405 * and may outlive pool shutdown. Using the pool would cause double-free
406 * or use-after-free when the pool is cleaned up while cached_result
407 * still holds a reference. */
408 lle_terminal_detection_result_t *detection =
409 calloc(1, sizeof(lle_terminal_detection_result_t));
410 if (!detection) {
411 return LLE_ERROR_OUT_OF_MEMORY;
412 }
413
414 uint64_t start_time = get_current_time_us();
415
416 /* Step 1: Basic TTY status detection */
417 detection->stdin_is_tty = isatty(STDIN_FILENO);
418 detection->stdout_is_tty = isatty(STDOUT_FILENO);
419 detection->stderr_is_tty = isatty(STDERR_FILENO);
420
421 /* Get terminal dimensions */
422 lle_get_terminal_size(&detection->terminal_cols, &detection->terminal_rows);
423
424 /* Step 2: Environment variable analysis */
425 lle_result_t env_result = analyze_environment_variables(detection);
426 if (env_result != LLE_SUCCESS) {
427 /* Continue with conservative defaults */
428 detection->capability_level = LLE_CAPABILITY_BASIC;
429 }
430
431 /* Step 3: Terminal signature matching */
432 detection->matched_signature = lle_match_terminal_signature(detection);
433 if (detection->matched_signature) {
434 detection->capability_level =
435 detection->matched_signature->capability_level;
436 detection->recommended_mode =
437 detection->matched_signature->preferred_mode;
438 detection->detection_confidence_high = true;
439 } else {
440 /* Step 4: Runtime capability probing (for unknown terminals) */
441 lle_result_t probe_result =
442 lle_probe_terminal_capabilities_safe(detection);
443 detection->probing_successful = (probe_result == LLE_SUCCESS);
444 detection->detection_confidence_high = detection->probing_successful;
445
446 /* Step 5: Fallback mode determination */
447 detection->recommended_mode = determine_fallback_mode(detection);
448
449 /* Infer capability level from probing results */
450 if (detection->supports_truecolor) {
451 detection->capability_level = LLE_CAPABILITY_PREMIUM;
452 } else if (detection->supports_256_colors) {
453 detection->capability_level = LLE_CAPABILITY_FULL;
454 } else if (detection->supports_colors) {
455 detection->capability_level = LLE_CAPABILITY_STANDARD;
456 } else {
457 detection->capability_level = LLE_CAPABILITY_BASIC;
458 }
459 }
460
461 /* Step 6: Final mode validation and adjustment */
462 detection->recommended_mode = validate_and_adjust_mode(detection);
463
464 detection->detection_time_us = get_current_time_us() - start_time;
465
466 /* Update statistics */
467 detection_stats.total_detections++;
468 detection_stats.avg_detection_time_us =
469 ((detection_stats.avg_detection_time_us *
470 (detection_stats.total_detections - 1)) +
471 detection->detection_time_us) /
472 detection_stats.total_detections;
473 if (detection->detection_time_us > detection_stats.max_detection_time_us) {
474 detection_stats.max_detection_time_us = detection->detection_time_us;
475 }
476
477 *result = detection;
478 return LLE_SUCCESS;
479}

481/**
482 * Optimized detection with caching.
483 */
484lle_result_t lle_detect_terminal_capabilities_optimized(
485 lle_terminal_detection_result_t **result) {
486
487 if (!result) {
488 return LLE_ERROR_INVALID_PARAMETER;
489 }
490
491 uint64_t current_time = get_current_time_us();
492
493 /* Check cache validity */
494 if (cached_result && (current_time - cache_timestamp_us) < CACHE_TTL_US) {
495 /* Return cached result */
496 *result = cached_result;
497 detection_stats.cache_hits++;
498 return LLE_SUCCESS;
499 }
500
501 /* Cache miss - perform full detection */
502 detection_stats.cache_misses++;
503
504 lle_result_t res = lle_detect_terminal_capabilities_comprehensive(result);
505 if (res == LLE_SUCCESS) {
506 /* Update cache */
507 if (cached_result) {
508 free(cached_result);
509 }
510 cached_result = *result;
511 cache_timestamp_us = current_time;
512 }
513
514 return res;
515}

670/**
671 * Get current terminal dimensions.
672 */
673lle_result_t lle_get_terminal_size(int *cols, int *rows) {
674 struct winsize ws;
675
676 if (ioctl(STDOUT_FILENO, TIOCGWINSZ, &ws) == 0) {
677 if (cols) {
678 *cols = ws.ws_col > 0 ? ws.ws_col : 80;
679 }
680 if (rows) {
681 *rows = ws.ws_row > 0 ? ws.ws_row : 24;
682 }
683 return LLE_SUCCESS;
684 }
685
686 /* Fallback to environment variables */
687 const char *columns = getenv("COLUMNS");
688 const char *lines = getenv("LINES");
689
690 if (cols) {
691 *cols = columns ? atoi(columns) : 80;
692 if (*cols <= 0) {
693 *cols = 80;
694 }
695 }
696 if (rows) {
697 *rows = lines ? atoi(lines) : 24;
698 if (*rows <= 0) {
699 *rows = 24;
700 }
701 }
702
703 return LLE_SUCCESS;
704}
