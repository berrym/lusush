36/**
37 * Represents a single character cell in the virtual screen
38 *
39 * Stores full UTF-8 grapheme clusters (1-4 bytes) to properly support:
40 * - ASCII characters (1 byte)
41 * - Extended Latin, Cyrillic, etc. (2 bytes)
42 * - CJK characters, box drawing (3 bytes)
43 * - Emoji and other SMP characters (4 bytes)
44 *
45 * The visual_width field indicates display columns (0, 1, or 2):
46 * - 0 for zero-width characters (combining marks)
47 * - 1 for normal width (ASCII, most Unicode)
48 * - 2 for wide characters (CJK, emoji)
49 */
50typedef struct {
51 char utf8_bytes[4]; // Full UTF-8 sequence (1-4 bytes)
52 uint8_t byte_len; // Actual bytes used (1-4)
53 uint8_t visual_width; // Display width in columns (0, 1, or 2)
54 bool is_prompt; // True if this cell is part of the prompt
55} screen_cell_t;
56
57/**
58 * Represents a line prefix (e.g., continuation prompt)
59 *
60 * Prefixes are rendered before line content and tracked separately for
61 * efficient updates. Used for continuation prompts and future features
62 * like autosuggestions.
63 */
64typedef struct {
65 char *text; // Prefix text (e.g., "> ", "loop> ")
66 size_t length; // Length in bytes
67 size_t visual_width; // Visual width in columns (excluding ANSI codes)
68 bool contains_ansi; // True if prefix contains ANSI escape codes
69 bool dirty; // True if prefix changed since last render
70} screen_line_prefix_t;
71
72/**
73 * Represents one line in the virtual screen
74 */
75typedef struct {
76 screen_cell_t cells[SCREEN_BUFFER_MAX_COLS];
77 int length; // Number of characters in this line
78 bool dirty; // True if line content changed since last render
79
80 screen_line_prefix_t *prefix; // Optional prefix (NULL if none)
81 bool prefix_dirty; // True if prefix changed since last render
82} screen_line_t;
83
84/**
85 * Virtual screen buffer
86 *
87 * Tracks the complete display state including command text and any menu/overlay
88 * content. By tracking the menu as part of the buffer, cursor positioning
89 * calculations work correctly regardless of terminal scrolling.
90 */
91typedef struct {
92 screen_line_t lines[SCREEN_BUFFER_MAX_ROWS];
93 int num_rows; // Number of rows currently used (command only)
94 int terminal_width; // Terminal width in columns
95 int cursor_row; // Cursor row position (0-based, within command)
96 int cursor_col; // Cursor column position (0-based)
97 int command_start_row; // Row where command text starts (after prompt)
98 int command_start_col; // Column where command text starts (after prompt)
99
100 // Menu/overlay tracking - tracks content displayed after command text
101 // This allows cursor positioning to account for all displayed content
102 int menu_lines; // Number of lines the menu occupies (0 if none)
103 int ghost_text_lines; // Extra lines from autosuggestion wrapping (0 if
104 // none)
105 int total_display_rows; // Total rows: num_rows + ghost_text_lines +
106 // menu_lines
107 int command_end_row; // Row where command text ends (before ghost/menu)
108 int command_end_col; // Column where command text ends
109} screen_buffer_t;

157/**
158 * Callback type for continuation prompt generation during rendering.
159 *
160 * Called when screen_buffer_render_with_continuation encounters a newline.
161 * The callback receives the plain text content of the line that just ended
162 * (with ANSI codes stripped) and should return the continuation prompt
163 * to display on the next line.
164 *
165 * @param line_text Plain text content of the line that ended (no ANSI codes)
166 * @param line_len Length of line_text in bytes
167 * @param line_number The logical line number (0-based, line that just ended)
168 * @param user_data Opaque pointer passed to
169 * screen_buffer_render_with_continuation
170 * @return Continuation prompt string for next line, or NULL for no prefix.
171 * String must remain valid until next callback or render completes.
172 */
173typedef const char *(*screen_buffer_continuation_cb)(const char *line_text,
174 size_t line_len,
175 int line_number,
176 void *user_data);

178/**
179 * Render prompt and command into screen buffer with continuation prompt
180 * support.
181 *
182 * Like screen_buffer_render but calls a callback on each newline to get
183 * context-aware continuation prompts. This enables proper
184 * character-by-character tracking where prompts are set at the exact visual row
185 * determined during rendering, not pre-calculated.
186 *
187 * @param buffer Buffer to render into
188 * @param prompt_text Prompt string (may contain ANSI codes)
189 * @param command_text Command string (may contain ANSI codes)
190 * @param cursor_byte_offset Cursor position in command_text (byte offset)
191 * @param continuation_cb Callback to get continuation prompt after each newline
192 * @param user_data Opaque pointer passed to continuation_cb
193 */
194void screen_buffer_render_with_continuation(
195 screen_buffer_t *buffer, const char *prompt_text, const char *command_text,
196 size_t cursor_byte_offset, screen_buffer_continuation_cb continuation_cb,
197 void *user_data);
