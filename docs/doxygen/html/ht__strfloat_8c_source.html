19/**
20 * __floatdup:
21 * Create a pointer duplicating val.
22 */
23static float *__floatdup(const float *val) {
24 float *f = calloc(1, sizeof(float));
25 if (!f) {
26 perror("__floatdup");
27 return NULL;
28 }
29
30 return memcpy(f, val, sizeof(float));
31}

49 return (ht_strfloat_t *)ht_create(hash, keyeq, &callbacks, flags);
