117/**
118 * @brief Validate line structure
119 *
120 * @param buffer Buffer to validate
121 * @param validator Buffer validator
122 * @return LLE_SUCCESS or error code
123 */
124lle_result_t
125lle_buffer_validate_line_structure(lle_buffer_t *buffer,
126 lle_buffer_validator_t *validator) {
127 if (!buffer || !validator) {
128 return LLE_ERROR_INVALID_PARAMETER;
129 }
130
131 /* Skip if line structure validation is disabled */
132 if (!validator->line_structure_validation) {
133 return LLE_SUCCESS;
134 }
135
136 /* Check that line count doesn't exceed capacity */
137 if (buffer->line_count > buffer->line_capacity) {
138 validator->validation_failures++;
139 validator->corruption_detections++;
140 return LLE_ERROR_INVALID_STATE;
141 }
142
143 /* Validate each line's offsets */
144 for (size_t i = 0; i < buffer->line_count; i++) {
145 lle_line_info_t *line = &buffer->lines[i];
146
147 /* Check that start offset is within buffer */
148 if (line->start_offset > buffer->length) {
149 validator->validation_failures++;
150 validator->bounds_violations++;
151 return LLE_ERROR_INVALID_RANGE;
152 }
153
154 /* Check that line doesn't extend past buffer end */
155 if (line->start_offset + line->length > buffer->length) {
156 validator->validation_failures++;
157 validator->bounds_violations++;
158 return LLE_ERROR_INVALID_RANGE;
159 }
160
161 /* Check line ordering (each line should start after previous) */
162 if (i > 0) {
163 lle_line_info_t *prev_line = &buffer->lines[i - 1];
164 if (line->start_offset <
165 prev_line->start_offset + prev_line->length) {
166 validator->validation_failures++;
167 validator->corruption_detections++;
168 return LLE_ERROR_MEMORY_CORRUPTION;
169 }
170 }
171 }
172
173 return LLE_SUCCESS;
174}

177 * @brief Validate cursor position
178 *
179 * @param buffer Buffer to validate
180 * @param validator Buffer validator
181 * @return LLE_SUCCESS or error code
182 */
183lle_result_t
184lle_buffer_validate_cursor_position(lle_buffer_t *buffer,
185 lle_buffer_validator_t *validator) {
186 if (!buffer || !validator) {
187 return LLE_ERROR_INVALID_PARAMETER;
188 }
189
190 /* Skip if cursor validation is disabled */
191 if (!validator->cursor_validation_enabled) {
192 return LLE_SUCCESS;
193 }
194
195 /* Check that cursor byte offset is within buffer bounds */
196 if (buffer->cursor.byte_offset > buffer->length) {
197 validator->validation_failures++;
198 validator->bounds_violations++;
199 return LLE_ERROR_INVALID_RANGE;
200 }
201
202 /* Check that cursor line number is within line count */
203 if (buffer->cursor.line_number >= buffer->line_count &&
204 buffer->line_count > 0) {
205 validator->validation_failures++;
206 validator->bounds_violations++;
207 return LLE_ERROR_INVALID_RANGE;
208 }
209
210 /* If cursor position is marked as valid, check version consistency */
211 if (buffer->cursor.position_valid) {
212 /* Cursor version should not be ahead of buffer modification count */
213 if (buffer->cursor.buffer_version > buffer->modification_count) {
214 validator->validation_failures++;
215 validator->corruption_detections++;
216 return LLE_ERROR_STATE_CORRUPTION;
217 }
218 }
219
220 return LLE_SUCCESS;
221}

243 validator->validation_failures++;

247
248 /* Check that used space doesn't exceed capacity */
249 if (buffer->used > buffer->capacity) {
250 validator->validation_failures++;
251 validator->bounds_violations++;
252 return LLE_ERROR_BUFFER_OVERFLOW;

254

256 if (buffer->length > buffer->used) {
257 validator->validation_failures++;
258 validator->corruption_detections++;
259 return LLE_ERROR_INVALID_STATE;
260 }
261
262 /* Check minimum capacity requirement */
263 if (buffer->capacity < LLE_BUFFER_MIN_CAPACITY) {
264 validator->validation_failures++;
265 validator->bounds_violations++;
266 return LLE_ERROR_INVALID_STATE;
267 }
268
269 /* Check maximum capacity limit */
270 if (buffer->capacity > LLE_BUFFER_MAX_CAPACITY) {
271 validator->validation_failures++;
272 validator->bounds_violations++;
273 return LLE_ERROR_BUFFER_OVERFLOW;
274 }
275
276 /* Check that data pointer is not NULL if capacity > 0 */
277 if (buffer->capacity > 0 && buffer->data == NULL) {
278 validator->validation_failures++;
279 validator->corruption_detections++;

286/**
287 * @brief Validate complete buffer
288 *
289 * Spec Reference: Spec 03, Section 8.1
290 *
291 * @param buffer Buffer to validate
292 * @param validator Buffer validator
293 * @return LLE_SUCCESS or error code
294 */
295lle_result_t lle_buffer_validate_complete(lle_buffer_t *buffer,
296 lle_buffer_validator_t *validator) {
297 if (!buffer || !validator) {
298 return LLE_ERROR_INVALID_PARAMETER;
299 }
300
301 lle_result_t result = LLE_SUCCESS;
302
303 /* Update validation statistics */
304 validator->validation_count++;
305 uint64_t start_time = lle_get_current_time_us();
306
307 /* Step 1: Validate buffer bounds first (most critical) */
308 result = lle_buffer_validate_bounds(buffer, validator);
309 if (result != LLE_SUCCESS) {
310 validator->last_validation_result = result;
311 validator->last_validation_time =
312 lle_get_current_time_us() - start_time;
313 return result;
314 }
315
316 /* Step 2: Validate UTF-8 encoding */
317 result = lle_buffer_validate_utf8(buffer, validator);
318 if (result != LLE_SUCCESS) {
319 validator->last_validation_result = result;
320 validator->last_validation_time =
321 lle_get_current_time_us() - start_time;
322 return result;
323 }
324
325 /* Step 3: Validate null termination */
326 if (buffer->data && buffer->data[buffer->length] != '\0') {
327 validator->validation_failures++;
328 validator->corruption_detections++;
329 validator->last_validation_result = LLE_ERROR_MEMORY_CORRUPTION;
330 validator->last_validation_time =
331 lle_get_current_time_us() - start_time;
332 return LLE_ERROR_MEMORY_CORRUPTION;
333 }
334
335 /* Step 4: Validate line structure */
336 result = lle_buffer_validate_line_structure(buffer, validator);
337 if (result != LLE_SUCCESS) {
338 validator->last_validation_result = result;
339 validator->last_validation_time =
340 lle_get_current_time_us() - start_time;
341 return result;
342 }
343
344 /* Step 5: Validate cursor position */
345 result = lle_buffer_validate_cursor_position(buffer, validator);
346 if (result != LLE_SUCCESS) {
347 validator->last_validation_result = result;
348 validator->last_validation_time =
349 lle_get_current_time_us() - start_time;
350 return result;
351 }
352
353 /* All validations passed */
354 validator->last_validation_result = LLE_SUCCESS;
355 validator->last_validation_time = lle_get_current_time_us() - start_time;
356
357 return LLE_SUCCESS;
358}
