66 ssh_host_cache_t *cache = calloc(1, sizeof(ssh_host_cache_t));

98int ssh_host_cache_add(ssh_host_cache_t *cache, const ssh_host_t *host) {
99 if (!cache || !host) {
100 return -1;
101 }
102
103 /* Grow if needed */
104 if (cache->count >= cache->capacity) {
105 size_t new_capacity = cache->capacity * 2;
106 ssh_host_t *new_hosts =
107 realloc(cache->hosts, new_capacity * sizeof(ssh_host_t));
108 if (!new_hosts) {
109 return -1;
110 }
111 cache->hosts = new_hosts;
112 cache->capacity = new_capacity;
113 }
114
115 /* Copy host entry */
116 cache->hosts[cache->count] = *host;
117 cache->count++;
118
119 return 0;
120}

122ssh_host_t *ssh_host_cache_find(ssh_host_cache_t *cache, const char *hostname) {
123 if (!cache || !hostname) {
124 return NULL;
125 }
126
127 for (size_t i = 0; i < cache->count; i++) {
128 if (strcmp(cache->hosts[i].hostname, hostname) == 0 ||
129 strcmp(cache->hosts[i].alias, hostname) == 0) {
130 return &cache->hosts[i];
131 }
132 }
133
134 return NULL;
135}

142int ssh_parse_config(const char *config_path, ssh_host_cache_t *cache) {
143 if (!config_path || !cache) {
144 return -1;
145 }
146
147 FILE *file = fopen(config_path, "r");
148 if (!file) {
149 return -1;
150 }
151
152 char line[MAX_CONFIG_LINE_LEN];
153 ssh_host_t current_host = {0};
154 bool in_host_block = false;
155 int hosts_added = 0;
156
157 while (fgets(line, sizeof(line), file)) {
158 line[strcspn(line, "\n")] = '\0';
159 trim_whitespace(line);
160
161 if (line[0] == '\0' || line[0] == '#') {
162 continue;
163 }
164
165 if (is_ssh_host_line(line)) {
166 /* Save previous host if valid */
167 if (in_host_block && current_host.hostname[0]) {
168 current_host.from_config = true;
169 current_host.priority = 80;
170 if (ssh_host_cache_add(cache, &current_host) == 0) {
171 hosts_added++;
172 }
173 }
174
175 /* Start new host block */
176 memset(&current_host, 0, sizeof(current_host));
177 extract_ssh_config_value(line, current_host.alias,
178 MAX_HOSTNAME_LEN);
179
180 /* If alias doesn't contain wildcards, use as hostname too */
181 if (!strchr(current_host.alias, '*') &&
182 !strchr(current_host.alias, '?')) {
183 strncpy(current_host.hostname, current_host.alias,
184 MAX_HOSTNAME_LEN - 1);
185 current_host.hostname[MAX_HOSTNAME_LEN - 1] = '\0';
186 }
187
188 in_host_block = true;
189 } else if (in_host_block) {
190 if (is_ssh_hostname_line(line)) {
191 extract_ssh_config_value(line, current_host.hostname,
192 MAX_HOSTNAME_LEN);
193 } else if (is_ssh_user_line(line)) {
194 extract_ssh_config_value(line, current_host.user,
195 MAX_USERNAME_LEN);
196 } else if (is_ssh_port_line(line)) {
197 extract_ssh_config_value(line, current_host.port, MAX_PORT_LEN);
198 }
199 }
200 }
201
202 /* Save last host */
203 if (in_host_block && current_host.hostname[0]) {
204 current_host.from_config = true;
205 current_host.priority = 80;
206 if (ssh_host_cache_add(cache, &current_host) == 0) {
207 hosts_added++;
208 }
209 }
210
211 fclose(file);
212 return hosts_added;
213}

215int ssh_parse_known_hosts(const char *known_hosts_path,
216 ssh_host_cache_t *cache) {
217 if (!known_hosts_path || !cache) {
218 return -1;
219 }
220
221 FILE *file = fopen(known_hosts_path, "r");
222 if (!file) {
223 return -1;
224 }
225
226 char line[MAX_CONFIG_LINE_LEN];
227 int hosts_added = 0;
228
229 while (fgets(line, sizeof(line), file)) {
230 line[strcspn(line, "\n")] = '\0';
231
232 /* Skip empty lines, comments, and hashed entries */
233 if (line[0] == '\0' || line[0] == '#' || line[0] == '|') {
234 continue;
235 }
236
237 /* Extract hostname (first field before space) */
238 char *space = strchr(line, ' ');
239 if (!space) {
240 continue;
241 }
242
243 size_t hostname_len = space - line;
244 if (hostname_len >= MAX_HOSTNAME_LEN) {
245 hostname_len = MAX_HOSTNAME_LEN - 1;
246 }
247
248 ssh_host_t host = {0};
249 strncpy(host.hostname, line, hostname_len);
250 host.hostname[hostname_len] = '\0';
251
252 /* Handle [host]:port format */
253 if (host.hostname[0] == '[') {
254 char *bracket = strchr(host.hostname, ']');
255 if (bracket) {
256 memmove(host.hostname, host.hostname + 1,
257 bracket - host.hostname - 1);
258 host.hostname[bracket - host.hostname - 1] = '\0';
259 }
260 }
261
262 /* Remove any comma-separated aliases, keep first */
263 char *comma = strchr(host.hostname, ',');
264 if (comma) {
265 *comma = '\0';
266 }
267
268 /* Skip if already in cache */
269 if (ssh_host_cache_find(cache, host.hostname)) {
270 continue;
271 }
272
273 host.from_known_hosts = true;
274 host.priority = 50;
275
276 if (ssh_host_cache_add(cache, &host) == 0) {
277 hosts_added++;
278 }
279 }
280
281 fclose(file);
282 return hosts_added;
283}

311void ssh_hosts_refresh(void) {
312 if (!g_ssh_host_cache) {
313 return;
314 }
315
316 /* Clear existing entries */
317 g_ssh_host_cache->count = 0;
318
319 /* Get home directory */
320 const char *home = getenv("HOME");
321 if (!home) {
322 return;
323 }
324
325 char path[1024];
326
327 /* Parse user SSH config */
328 snprintf(path, sizeof(path), "%s/.ssh/config", home);
329 ssh_parse_config(path, g_ssh_host_cache);
330
331 /* Parse system SSH config */
332 ssh_parse_config("/etc/ssh/ssh_config", g_ssh_host_cache);
333
334 /* Parse known_hosts */
335 snprintf(path, sizeof(path), "%s/.ssh/known_hosts", home);
336 ssh_parse_known_hosts(path, g_ssh_host_cache);
337
338 g_ssh_host_cache->last_updated = time(NULL);
339 g_ssh_host_cache->needs_refresh = false;
340}

342ssh_host_cache_t *get_ssh_host_cache(void) {
343 if (!g_ssh_host_cache) {
344 ssh_hosts_init();
345 }
346
347 /* Check if refresh needed */
348 if (g_ssh_host_cache && g_ssh_host_cache->needs_refresh) {
349 ssh_hosts_refresh();
350 }
351
352 /* Check cache timeout */
353 if (g_ssh_host_cache) {
354 time_t now = time(NULL);
355 if (now - g_ssh_host_cache->last_updated > SSH_CONFIG_CACHE_TIMEOUT) {
356 ssh_hosts_refresh();
357 }
358 }
359
360 return g_ssh_host_cache;
361}

368static void trim_whitespace(char *str) {
369 if (!str) {
370 return;
371 }
372
373 char *start = str;
374 while (*start && isspace((unsigned char)*start)) {
375 start++;
376 }
377
378 if (start != str) {
379 memmove(str, start, strlen(start) + 1);
380 }
381
382 char *end = str + strlen(str) - 1;
383 while (end > str && isspace((unsigned char)*end)) {
384 *end-- = '\0';
385 }
386}

408static void extract_ssh_config_value(const char *line, char *value,
409 size_t max_len) {
410 if (!line || !value || max_len == 0) {
411 return;
412 }
413
414 /* Skip keyword */
415 const char *p = line;
416 while (*p && !isspace((unsigned char)*p)) {
417 p++;
418 }
419
420 /* Skip whitespace */
421 while (*p && isspace((unsigned char)*p)) {
422 p++;
423 }
424
425 /* Copy value */
426 size_t len = strlen(p);
427 if (len >= max_len) {
428 len = max_len - 1;
429 }
430 strncpy(value, p, len);
431 value[len] = '\0';
432
433 /* Remove trailing whitespace */
434 trim_whitespace(value);
435}
