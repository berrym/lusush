101 new_analyzer->active = true;

129lle_result_t lle_structure_analyzer_detect_keywords(
130 lle_structure_analyzer_t *analyzer, const char *command_text,
131 size_t command_length, lle_keyword_match_t **keywords,
132 size_t *keyword_count) {
133 if (!analyzer || !analyzer->active || !command_text || !keywords ||
134 !keyword_count) {
135 return LLE_ERROR_INVALID_PARAMETER;
136 }
137
138 if (command_length > analyzer->config.max_command_length) {
139 return LLE_ERROR_INVALID_PARAMETER;
140 }
141
142 /* Reset context for new analysis */
143 memset(&analyzer->context, 0, sizeof(lle_analysis_context_t));
144
145 /* Count keywords first pass */
146 size_t count = 0;
147 for (size_t i = 0; i < command_length; i++) {
148 update_context_for_char(&analyzer->context, command_text[i]);
149
150 /* Skip if we're inside quotes or comments */
151 if (analyzer->context.in_quoted_string ||
152 analyzer->context.in_comment) {
153 continue;
154 }
155
156 /* Check each keyword definition */
157 for (const keyword_definition_t *kw = SHELL_KEYWORDS; kw->text != NULL;
158 kw++) {
159 bool at_start = false;
160 if (is_keyword_at_position(command_text, i, command_length, kw,
161 &at_start)) {
162 if (!kw->requires_command_start || at_start) {
163 count++;
164 i += kw->length - 1; /* Skip past keyword */
165 break;
166 }
167 }
168 }
169 }
170
171 if (count == 0) {
172 *keywords = NULL;
173 *keyword_count = 0;
174 return LLE_SUCCESS;
175 }
176
177 /* Allocate keyword array */
178 lle_keyword_match_t *kw_array =
179 lle_pool_alloc(count * sizeof(lle_keyword_match_t));
180 if (!kw_array) {
181 return LLE_ERROR_OUT_OF_MEMORY;
182 }
183
184 /* Second pass: extract keywords */
185 memset(&analyzer->context, 0, sizeof(lle_analysis_context_t));
186 size_t kw_idx = 0;
187
188 for (size_t i = 0; i < command_length; i++) {
189 char c = command_text[i];
190
191 /* Track line numbers */
192 if (c == '\n') {
193 analyzer->context.current_line++;
194 }
195
196 update_context_for_char(&analyzer->context, c);
197
198 /* Skip if we're inside quotes or comments */
199 if (analyzer->context.in_quoted_string ||
200 analyzer->context.in_comment) {
201 continue;
202 }
203
204 /* Check each keyword definition */
205 for (const keyword_definition_t *kw = SHELL_KEYWORDS; kw->text != NULL;
206 kw++) {
207 bool at_start = false;
208 if (is_keyword_at_position(command_text, i, command_length, kw,
209 &at_start)) {
210 if (!kw->requires_command_start || at_start) {
211 kw_array[kw_idx].type = kw->type;
212 kw_array[kw_idx].start_offset = i;
213 kw_array[kw_idx].length = kw->length;
214 kw_array[kw_idx].line_number =
215 analyzer->context.current_line;
216 kw_array[kw_idx].is_command_start = at_start;
217 memset(kw_array[kw_idx].reserved, 0,
218 sizeof(kw_array[kw_idx].reserved));
219
220 kw_idx++;
221 i += kw->length - 1; /* Skip past keyword */
222 break;
223 }
224 }
225 }
226 }
227
228 *keywords = kw_array;
229 *keyword_count = count;
230
231 return LLE_SUCCESS;
232}

289 /* Stack-based matching of opening/closing keywords */
290 typedef struct {
291 lle_keyword_type_t opening;
292 lle_keyword_type_t closing;
293 } keyword_pair_t;
294
295 static const keyword_pair_t pairs[] = {
296 {LLE_KEYWORD_FOR, LLE_KEYWORD_DONE},
297 {LLE_KEYWORD_WHILE, LLE_KEYWORD_DONE},
298 {LLE_KEYWORD_UNTIL, LLE_KEYWORD_DONE},
299 {LLE_KEYWORD_SELECT, LLE_KEYWORD_DONE},
300 {LLE_KEYWORD_IF, LLE_KEYWORD_FI},
301 {LLE_KEYWORD_CASE, LLE_KEYWORD_ESAC},
302 {LLE_KEYWORD_NONE, LLE_KEYWORD_NONE}};
303
304 /* Simple stack to track unclosed keywords */
305 lle_keyword_type_t stack[DEFAULT_MAX_NESTING_DEPTH];
306 size_t stack_size = 0;
307
308 for (size_t i = 0; i < keyword_count; i++) {
309 lle_keyword_type_t kw_type = keywords[i].type;
310
311 /* Check if it's an opening keyword */
312 bool is_opening = false;
313 lle_keyword_type_t expected_closing = LLE_KEYWORD_NONE;
314
315 for (const keyword_pair_t *pair = pairs;
316 pair->opening != LLE_KEYWORD_NONE; pair++) {
317 if (kw_type == pair->opening) {
318 is_opening = true;
319 expected_closing = pair->closing;
320 break;
321 }
322 }
323
324 if (is_opening) {
325 if (stack_size >= DEFAULT_MAX_NESTING_DEPTH) {
326 return LLE_ERROR_INVALID_STATE;
327 }
328 stack[stack_size++] = expected_closing;
329 } else {
330 /* Check if it's a closing keyword */
331 bool is_closing = false;
332 for (const keyword_pair_t *pair = pairs;
333 pair->opening != LLE_KEYWORD_NONE; pair++) {
334 if (kw_type == pair->closing) {
335 is_closing = true;
336 break;
337 }
338 }
339
340 if (is_closing) {
341 if (stack_size == 0) {
342 /* Closing without opening */
343 *is_complete = false;
344 if (missing_keyword) {
345 *missing_keyword = LLE_KEYWORD_NONE;
346 }
347 return LLE_SUCCESS;
348 }
349
350 if (stack[stack_size - 1] != kw_type) {
351 /* Mismatched closing */
352 *is_complete = false;
353 if (missing_keyword) {
354 *missing_keyword = stack[stack_size - 1];
355 }
356 return LLE_SUCCESS;
357 }
358
359 stack_size--;
360 }
361 }
362 }
363
364 *is_complete = (stack_size == 0);
365 if (missing_keyword) {
366 *missing_keyword =
367 (stack_size > 0) ? stack[stack_size - 1] : LLE_KEYWORD_NONE;
368 }
369
370 return LLE_SUCCESS;
371}

373lle_result_t lle_structure_analyzer_calculate_indentation(
374 lle_structure_analyzer_t *analyzer, const char *command_text,
375 size_t command_length, lle_indentation_info_t **indent_info) {
376 if (!analyzer || !analyzer->active || !command_text || !indent_info) {
377 return LLE_ERROR_INVALID_PARAMETER;
378 }
379
380 if (!analyzer->config.track_indentation) {
381 return LLE_ERROR_INVALID_STATE;
382 }
383
384 lle_indentation_info_t *info =
385 lle_pool_alloc(sizeof(lle_indentation_info_t));
386 if (!info) {
387 return LLE_ERROR_OUT_OF_MEMORY;
388 }
389
390 memset(info, 0, sizeof(lle_indentation_info_t));
391
392 /* Count lines */
393 size_t line_count = 1;
394 for (size_t i = 0; i < command_length; i++) {
395 if (command_text[i] == '\n') {
396 line_count++;
397 }
398 }
399
400 info->line_count = line_count;
401 info->spaces_per_level = 2; /* Default to 2 spaces */
402 info->use_tabs = false;
403 info->indent_char = ' ';
404
405 /* Allocate line indentation array */
406 info->level_per_line = lle_pool_alloc(line_count * sizeof(size_t));
407 if (!info->level_per_line) {
408 return LLE_ERROR_OUT_OF_MEMORY;
409 }
410
411 /* Calculate indentation for each line */
412 size_t current_line = 0;
413 size_t line_start = 0;
414
415 for (size_t i = 0; i <= command_length; i++) {
416 if (i == command_length || command_text[i] == '\n') {
417 /* Calculate indentation for this line */
418 size_t indent = 0;
419 for (size_t j = line_start; j < i; j++) {
420 if (command_text[j] == ' ') {
421 indent++;
422 } else if (command_text[j] == '\t') {
423 indent += 4; /* Tab = 4 spaces */
424 info->use_tabs = true;
425 } else {
426 break; /* Non-whitespace found */
427 }
428 }
429
430 info->level_per_line[current_line] = indent;
431 current_line++;
432 line_start = i + 1;
433 }
434 }
435
436 *indent_info = info;
437 return LLE_SUCCESS;
438}

440lle_result_t
441lle_structure_analyzer_analyze(lle_structure_analyzer_t *analyzer,
442 const char *command_text, size_t command_length,
443 lle_command_structure_t **structure) {
444 if (!analyzer || !analyzer->active || !command_text || !structure) {
445 return LLE_ERROR_INVALID_PARAMETER;
446 }
447
448 /* Create command structure */
449 lle_result_t result =
450 lle_command_structure_create(structure, analyzer->memory_pool);
451 if (result != LLE_SUCCESS) {
452 return result;
453 }
454
455 /* Detect keywords */
456 lle_keyword_match_t *keywords = NULL;
457 size_t keyword_count = 0;
458
459 result = lle_structure_analyzer_detect_keywords(
460 analyzer, command_text, command_length, &keywords, &keyword_count);
461
462 if (result != LLE_SUCCESS) {
463 lle_command_structure_destroy(*structure);
464 *structure = NULL;
465 return result;
466 }
467
468 /* Add keywords to structure */
469 for (size_t i = 0; i < keyword_count; i++) {
470 result = lle_command_structure_add_keyword(
471 *structure, keywords[i].type, keywords[i].start_offset,
472 keywords[i].line_number, 0 /* Indent level calculated separately */
473 );
474
475 if (result != LLE_SUCCESS) {
476 lle_command_structure_destroy(*structure);
477 *structure = NULL;
478 return result;
479 }
480 }
481
482 /* Determine primary construct type */
483 if (keyword_count > 0) {
484 (*structure)->primary_type = determine_construct_type(keywords[0].type);
485 } else {
486 (*structure)->primary_type = LLE_CONSTRUCT_SIMPLE;
487 }
488
489 /* Check completeness */
490 bool is_complete = false;
491 result = lle_structure_analyzer_check_complete(
492 analyzer, command_text, command_length, &is_complete, NULL);
493
494 if (result == LLE_SUCCESS) {
495 (*structure)->is_complete = is_complete;
496 }
497
498 /* Calculate indentation if enabled */
499 if (analyzer->config.track_indentation) {
500 lle_indentation_info_t *indent_info = NULL;
501 result = lle_structure_analyzer_calculate_indentation(
502 analyzer, command_text, command_length, &indent_info);
503
504 if (result == LLE_SUCCESS) {
505 (*structure)->indentation = indent_info;
506 (*structure)->total_lines = indent_info->line_count;
507 }
508 }
509
510 /* Build construct tree */
511 result = build_construct_tree(analyzer, *structure);
512 if (result != LLE_SUCCESS) {
513 lle_command_structure_destroy(*structure);
514 *structure = NULL;
515 return result;
516 }
517
518 return LLE_SUCCESS;
519}

524 return isspace(c) || c == ';' || c == '|' || c == '&' || c == '(' ||
525 c == ')' || c == '{' || c == '}' || c == '<' || c == '>' ||
526 c == '\0';

529static bool is_keyword_at_position(const char *text, size_t pos,
530 size_t text_len,
531 const keyword_definition_t *keyword,
532 bool *at_start) {
533 /* Check if we have enough characters */
534 if (pos + keyword->length > text_len) {
535 return false;
536 }
537
538 /* Check if keyword matches */
539 if (strncmp(&text[pos], keyword->text, keyword->length) != 0) {
540 return false;
541 }
542
543 /* Check word boundaries */
544 if (pos > 0 && !is_word_boundary(text[pos - 1])) {
545 return false;
546 }
547
548 if (pos + keyword->length < text_len &&
549 !is_word_boundary(text[pos + keyword->length])) {
550 return false;
551 }
552
553 /* Check if at command/line start */
554 if (at_start) {
555 *at_start = (pos == 0);
556 if (!*at_start) {
557 /* Check if we're at the start of a line */
558 for (size_t i = pos; i > 0; i--) {
559 if (text[i - 1] == '\n') {
560 *at_start = true;
561 break;
562 }
563 if (!isspace(text[i - 1])) {
564 break;
565 }
566 }
567 }
568 }
569
570 return true;
571}

573static void update_context_for_char(lle_analysis_context_t *ctx, char c) {
574 if (ctx->last_was_escape) {
575 ctx->last_was_escape = false;
576 return;
577 }
578
579 if (c == '\\') {
580 ctx->last_was_escape = true;
581 return;
582 }
583
584 if (ctx->in_comment) {
585 if (c == '\n') {
586 ctx->in_comment = false;
587 }
588 return;
589 }
590
591 if (!ctx->in_quoted_string && c == '#') {
592 ctx->in_comment = true;
593 return;
594 }
595
596 if (c == '"' || c == '\'' || c == '`') {
597 if (!ctx->in_quoted_string) {
598 ctx->in_quoted_string = true;
599 ctx->quote_char = c;
600 } else if (c == ctx->quote_char) {
601 ctx->in_quoted_string = false;
602 ctx->quote_char = '\0';
603 }
604 }
605}
