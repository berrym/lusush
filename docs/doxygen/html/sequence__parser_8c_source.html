16/* Helper macros */
17#define IS_CONTROL_CHAR(c) ((c) < 0x20 || (c) == 0x7F)
18#define IS_CSI_PARAMETER(c) (((c) >= '0' && (c) <= '9') || (c) == ';')
19#define IS_CSI_INTERMEDIATE(c) ((c) >= 0x20 && (c) <= 0x2F)
20#define IS_CSI_FINAL(c) ((c) >= 0x40 && (c) <= 0x7E)

22/* Get current time in microseconds */
23static uint64_t get_current_time_us(void) {
24 struct timespec ts;
25 clock_gettime(CLOCK_MONOTONIC, &ts);
26 return (uint64_t)ts.tv_sec * 1000000 + (uint64_t)ts.tv_nsec / 1000;
27}

119/*
120 * Parse CSI parameters from buffer
121 *
122 * CSI format: ESC [ [parameters] [intermediate bytes] final_byte
123 * Parameters are semicolon-separated decimal numbers
124 */
125static lle_result_t parse_csi_parameters(lle_sequence_parser_t *parser,
126 const char *params_start,
127 size_t params_len) {
128 if (!parser || !params_start || params_len == 0) {
129 return LLE_SUCCESS; // No parameters is valid
130 }
131
132 parser->parameter_count = 0;
133 uint32_t current_param = 0;
134 bool has_current = false;
135
136 for (size_t i = 0;
137 i < params_len && parser->parameter_count < LLE_MAX_CSI_PARAMETERS;
138 i++) {
139 char c = params_start[i];
140
141 if (c >= '0' && c <= '9') {
142 current_param = current_param * 10 + (c - '0');
143 has_current = true;
144 } else if (c == ';') {
145 // Semicolon separates parameters
146 parser->parameters[parser->parameter_count++] =
147 has_current ? current_param : 0;
148 current_param = 0;
149 has_current = false;
150 } else {
151 // Invalid character in parameters
152 break;
153 }
154 }
155
156 // Add final parameter
157 if (has_current && parser->parameter_count < LLE_MAX_CSI_PARAMETERS) {
158 parser->parameters[parser->parameter_count++] = current_param;
159 }
160
161 return LLE_SUCCESS;
162}

164/*
165 * Process a complete CSI sequence
166 *
167 * CSI sequences have the format: ESC [ [params] [intermediates] final
168 * Examples:
169 * ESC[2J - Clear screen
170 * ESC[1;5H - Move cursor to row 1, col 5
171 * ESC[31m - Set foreground color to red
172 */
173static lle_result_t process_csi_sequence(lle_sequence_parser_t *parser,
174 lle_parsed_input_t **parsed_input) {
175 if (!parser || !parsed_input) {
176 return LLE_ERROR_INVALID_PARAMETER;
177 }
178
179 // CSI sequence is complete - parse it
180 // Buffer contains: ESC [ [parameters] [intermediates] final
181
182 // Find where parameters end and intermediates begin
183 size_t param_start = 2; // Skip ESC [
184 size_t param_end = param_start;
185 size_t intermediate_start = param_start;
186
187 // Scan for parameters (digits and semicolons)
188 while (param_end < parser->buffer_pos) {
189 char c = parser->buffer[param_end];
190 if (IS_CSI_PARAMETER(c)) {
191 param_end++;
192 } else {
193 break;
194 }
195 }
196
197 // Parse parameters
198 if (param_end > param_start) {
199 parse_csi_parameters(parser, parser->buffer + param_start,
200 param_end - param_start);
201 }
202
203 // Scan for intermediate bytes
204 intermediate_start = param_end;
205 while (intermediate_start < parser->buffer_pos - 1) { // -1 for final byte
206 char c = parser->buffer[intermediate_start];
207 if (IS_CSI_INTERMEDIATE(c)) {
208 if (parser->intermediate_count <
209 sizeof(parser->intermediate_chars)) {
210 parser->intermediate_chars[parser->intermediate_count++] = c;
211 }
212 intermediate_start++;
213 } else {
214 break;
215 }
216 }
217
218 // Final character
219 parser->final_char = parser->buffer[parser->buffer_pos - 1];
220
221 // Create parsed input result
222 lle_parsed_input_t *result = lle_pool_alloc(sizeof(lle_parsed_input_t));
223 if (!result) {
224 return LLE_ERROR_OUT_OF_MEMORY;
225 }
226
227 memset(result, 0, sizeof(lle_parsed_input_t));
228 result->type = LLE_PARSED_INPUT_TYPE_SEQUENCE;
229 result->handled = false;
230
231 *parsed_input = result;
232 return LLE_SUCCESS;
233}

263/*
264 * Process data through the sequence parser
265 *
266 * This is the main parsing loop that implements the state machine.
267 */
268lle_result_t
269lle_sequence_parser_process_data(lle_sequence_parser_t *parser,
270 const char *data, size_t data_len,
271 lle_parsed_input_t **parsed_input) {
272 if (!parser || !data || !parsed_input) {
273 return LLE_ERROR_INVALID_PARAMETER;
274 }
275
276 *parsed_input = NULL;
277
278 for (size_t i = 0; i < data_len; i++) {
279 unsigned char c = data[i];
280
281 // Check for sequence timeout
282 if (parser->state != LLE_PARSER_STATE_NORMAL &&
283 has_sequence_timed_out(parser)) {
284 parser->timeout_sequences++;
285 parser->last_error_time = get_current_time_us();
286 lle_sequence_parser_reset_state(parser);
287 }
288
289 switch (parser->state) {
290 case LLE_PARSER_STATE_NORMAL:
291 if (c == 0x1B) { // ESC
292 parser->state = LLE_PARSER_STATE_ESCAPE;
293 parser->buffer[0] = c;
294 parser->buffer_pos = 1;
295 parser->sequence_start_time = get_current_time_us();
296 } else if (IS_CONTROL_CHAR(c)) {
297 // Control character in normal state
298 return process_control_char(parser, c, parsed_input);
299 } else {
300 // Regular character - not our responsibility (UTF-8 processor
301 // handles this)
302 continue;
303 }
304 break;
305
306 case LLE_PARSER_STATE_ESCAPE:
307 // Store character
308 if (parser->buffer_pos >= parser->buffer_capacity) {
309 // Buffer overflow - treat as malformed
310 parser->malformed_sequences++;
311 parser->last_error_time = get_current_time_us();
312 lle_sequence_parser_reset_state(parser);
313 break;
314 }
315
316 parser->buffer[parser->buffer_pos++] = c;
317
318 // Determine sequence type based on second character
319 if (c == '[') {
320 // CSI sequence
321 parser->state = LLE_PARSER_STATE_CSI;
322 parser->type = LLE_SEQ_TYPE_CSI;
323 } else if (c == ']') {
324 // OSC sequence
325 parser->state = LLE_PARSER_STATE_OSC;
326 parser->type = LLE_SEQ_TYPE_OSC;
327 } else if (c == 'P') {
328 // DCS sequence
329 parser->state = LLE_PARSER_STATE_DCS;
330 parser->type = LLE_SEQ_TYPE_DCS;
331 } else if (c == 'O') {
332 // SS3 sequence (function keys)
333 parser->type = LLE_SEQ_TYPE_SS3;
334 // SS3 sequences are typically followed by one more character
335 parser->state = LLE_PARSER_STATE_KEY_SEQUENCE;
336 } else if (c == 'N') {
337 // SS2 sequence
338 parser->type = LLE_SEQ_TYPE_SS2;
339 parser->state = LLE_PARSER_STATE_KEY_SEQUENCE;
340 } else if ((c >= 0x20 && c < 0x7F) || c == 0x7F) {
341 // ESC + printable ASCII or DEL = Meta/Alt + character
342 // This is how macOS Terminal sends Alt+key when Option is Meta,
343 // or when user physically presses ESC then a letter (e.g., ESC
344 // f for M-f). 0x7F (DEL/Backspace) is included for
345 // Alt+Backspace.
346 lle_parsed_input_t *result =
347 lle_pool_alloc(sizeof(lle_parsed_input_t));
348 if (!result) {
349 lle_sequence_parser_reset_state(parser);
350 return LLE_ERROR_OUT_OF_MEMORY;
351 }
352
353 memset(result, 0, sizeof(lle_parsed_input_t));
354 result->type = LLE_PARSED_INPUT_TYPE_KEY;
355 // 0x7F (DEL/Backspace) needs SPECIAL type to be recognized
356 result->data.key_info.type =
357 (c == 0x7F) ? LLE_KEY_TYPE_SPECIAL : LLE_KEY_TYPE_REGULAR;
358 result->data.key_info.keycode = c;
359 result->data.key_info.modifiers = LLE_KEY_MOD_ALT;
360 result->handled = false;
361
362 *parsed_input = result;
363 lle_sequence_parser_reset_state(parser);
364 return LLE_SUCCESS;
365 } else {
366 // Other two-character escape sequence
367 lle_result_t result =
368 process_control_char(parser, c, parsed_input);
369 lle_sequence_parser_reset_state(parser);
370 return result;
371 }
372 break;
373
374 case LLE_PARSER_STATE_CSI:
375 // Store character
376 if (parser->buffer_pos >= parser->buffer_capacity) {
377 parser->malformed_sequences++;
378 parser->last_error_time = get_current_time_us();
379 lle_sequence_parser_reset_state(parser);
380 break;
381 }
382
383 parser->buffer[parser->buffer_pos++] = c;
384
385 // Check if this is the final character
386 if (IS_CSI_FINAL(c)) {
387 // Sequence is complete
388 lle_result_t result =
389 process_csi_sequence(parser, parsed_input);
390 lle_sequence_parser_reset_state(parser);
391 return result;
392 }
393 break;
394
395 case LLE_PARSER_STATE_OSC:
396 case LLE_PARSER_STATE_DCS:
397 // OSC and DCS sequences are terminated by ST (ESC \‍) or BEL (0x07)
398 if (parser->buffer_pos >= parser->buffer_capacity) {
399 parser->malformed_sequences++;
400 parser->last_error_time = get_current_time_us();
401 lle_sequence_parser_reset_state(parser);
402 break;
403 }
404
405 parser->buffer[parser->buffer_pos++] = c;
406
407 // Check for terminator
408 if (c == 0x07) { // BEL
409 // Sequence complete
410 lle_parsed_input_t *result =
411 lle_pool_alloc(sizeof(lle_parsed_input_t));
412 if (result) {
413 memset(result, 0, sizeof(lle_parsed_input_t));
414 result->type = LLE_PARSED_INPUT_TYPE_SEQUENCE;
415 *parsed_input = result;
416 }
417 lle_sequence_parser_reset_state(parser);
418 return result ? LLE_SUCCESS : LLE_ERROR_OUT_OF_MEMORY;
419 } else if (c == '\\' && parser->buffer_pos >= 2 &&
420 parser->buffer[parser->buffer_pos - 2] == 0x1B) {
421 // ESC \ (ST - String Terminator)
422 lle_parsed_input_t *result =
423 lle_pool_alloc(sizeof(lle_parsed_input_t));
424 if (result) {
425 memset(result, 0, sizeof(lle_parsed_input_t));
426 result->type = LLE_PARSED_INPUT_TYPE_SEQUENCE;
427 *parsed_input = result;
428 }
429 lle_sequence_parser_reset_state(parser);
430 return result ? LLE_SUCCESS : LLE_ERROR_OUT_OF_MEMORY;
431 }
432 break;
433
434 case LLE_PARSER_STATE_KEY_SEQUENCE:
435 // SS2/SS3 sequences followed by one character
436 if (parser->buffer_pos >= parser->buffer_capacity) {
437 parser->malformed_sequences++;
438 parser->last_error_time = get_current_time_us();
439 lle_sequence_parser_reset_state(parser);
440 break;
441 }
442
443 parser->buffer[parser->buffer_pos++] = c;
444
445 // Key sequence is complete
446 lle_parsed_input_t *result =
447 lle_pool_alloc(sizeof(lle_parsed_input_t));
448 if (result) {
449 memset(result, 0, sizeof(lle_parsed_input_t));
450 result->type = LLE_PARSED_INPUT_TYPE_KEY;
451 *parsed_input = result;
452 }
453 lle_sequence_parser_reset_state(parser);
454 return result ? LLE_SUCCESS : LLE_ERROR_OUT_OF_MEMORY;
455
456 case LLE_PARSER_STATE_ERROR_RECOVERY:
457 // In error recovery, skip characters until we see ESC or normal
458 // text
459 if (c == 0x1B) {
460 parser->state = LLE_PARSER_STATE_ESCAPE;
461 parser->buffer[0] = c;
462 parser->buffer_pos = 1;
463 parser->sequence_start_time = get_current_time_us();
464 } else if (!IS_CONTROL_CHAR(c)) {
465 lle_sequence_parser_reset_state(parser);
466 }
467 break;
468
469 default:
470 // Unknown state - reset
471 lle_sequence_parser_reset_state(parser);
472 break;
473 }
474 }
475
476 return LLE_SUCCESS;
477}

552/*
553 * Check if parser has timed out waiting for sequence completion.
554 * If in ESCAPE state and timeout exceeded, returns the ESC as a standalone key.
555 *
556 * @param parser The sequence parser
557 * @param timeout_us Timeout in microseconds (typically 50000-100000 for ESC)
558 * @param parsed_input Output: If timeout occurred, contains ESC key event
559 * @return LLE_SUCCESS if timeout handled, LLE_ERROR_NOT_FOUND if no timeout
560 */
561lle_result_t
562lle_sequence_parser_check_timeout(lle_sequence_parser_t *parser,
563 uint64_t timeout_us,
564 lle_parsed_input_t **parsed_input) {
565 if (!parser || !parsed_input) {
566 return LLE_ERROR_INVALID_PARAMETER;
567 }
568
569 *parsed_input = NULL;
570
571 /* Only check timeout if parser is accumulating a sequence */
572 if (parser->state == LLE_PARSER_STATE_NORMAL) {
573 return LLE_ERROR_NOT_FOUND; /* Not accumulating, no timeout to check */
574 }
575
576 /* Check if we have a sequence start time */
577 if (parser->sequence_start_time == 0) {
578 return LLE_ERROR_NOT_FOUND;
579 }
580
581 uint64_t current_time = get_current_time_us();
582 uint64_t elapsed = current_time - parser->sequence_start_time;
583
584 if (elapsed < timeout_us) {
585 return LLE_ERROR_NOT_FOUND; /* Timeout not yet exceeded */
586 }
587
588 /* Timeout exceeded - if in ESCAPE state with just ESC buffered, return ESC
589 * key */
590 if (parser->state == LLE_PARSER_STATE_ESCAPE && parser->buffer_pos == 1 &&
591 parser->buffer[0] == 0x1B) {
592
593 /* Create ESC key event */
594 lle_parsed_input_t *result = lle_pool_alloc(sizeof(lle_parsed_input_t));
595 if (!result) {
596 return LLE_ERROR_OUT_OF_MEMORY;
597 }
598
599 memset(result, 0, sizeof(lle_parsed_input_t));
600 result->type = LLE_PARSED_INPUT_TYPE_KEY;
601 result->data.key_info.type = LLE_KEY_TYPE_SPECIAL;
602 result->data.key_info.keycode = 27; /* ESC */
603 result->data.key_info.modifiers = 0;
604 result->data.key_info.timestamp = current_time;
605 result->handled = false;
606 result->parse_time_us = 0;
607
608 /* Reset parser state */
609 parser->timeout_sequences++;
610 lle_sequence_parser_reset_state(parser);
611
612 *parsed_input = result;
613 return LLE_SUCCESS;
614 }
615
616 /* Timeout in other state - reset parser and discard partial sequence */
617 parser->timeout_sequences++;
618 lle_sequence_parser_reset_state(parser);
619
620 return LLE_ERROR_NOT_FOUND;
