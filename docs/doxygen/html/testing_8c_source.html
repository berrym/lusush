100 if (!framework) {

103
104 lle_testing_framework_t *fw = calloc(1, sizeof(lle_testing_framework_t));
105 if (!fw) {
106 return LLE_ERROR_OUT_OF_MEMORY;
107 }
108
109 /* Initialize suite registry */

135

190/**
191 * @brief Run all registered tests
192 */
193lle_result_t
194lle_testing_framework_run_all_tests(lle_testing_framework_t *framework,
195 lle_test_run_config_t *config,
196 lle_test_results_t *results) {
197 if (!framework || !results) {
198 return LLE_ERROR_INVALID_PARAMETER;
199 }
200
201 /* Initialize results */
202 memset(results, 0, sizeof(lle_test_results_t));
203 results->start_timestamp = (uint64_t)time(NULL);
204
205 if (!framework->test_runner || !framework->suite_registry) {
206 return LLE_ERROR_INVALID_STATE;
207 }
208
209 lle_test_runner_t *runner = framework->test_runner;
210 runner->tests_run = 0;
211 runner->tests_passed = 0;
212 runner->tests_failed = 0;
213
214 /* Run all test suites */
215 for (size_t i = 0; i < framework->suite_registry->suite_count; i++) {
216 lle_test_suite_t *suite = framework->suite_registry->suites[i];
217
218 /* Run all tests in suite */
219 for (size_t j = 0; j < suite->test_count; j++) {
220 lle_test_case_t *test_case = suite->test_cases[j];
221
222 /* Check config for test filtering */
223 if (config && !config->include_nightly_tests &&
224 test_case->priority == LLE_TEST_PRIORITY_NIGHTLY) {
225 continue;
226 }
227
228 /* Create test context */
229 lle_test_context_t *ctx = calloc(1, sizeof(lle_test_context_t));
230 if (!ctx) {
231 continue;
232 }
233
234 ctx->current_test = test_case;
235 runner->current_context = ctx;
236
237 /* Execute test */
238 lle_test_execution_result_t exec_result;
239 lle_result_t run_result =
240 lle_test_runner_execute_test(runner, ctx, &exec_result);
241
242 results->total_tests++;
243
244 if (run_result == LLE_SUCCESS &&
245 exec_result.result == LLE_TEST_RESULT_SUCCESS) {
246 runner->tests_passed++;
247 results->passed_tests++;
248 } else {
249 runner->tests_failed++;
250 results->failed_tests++;
251 }
252
253 results->total_execution_time_us += exec_result.execution_time_us;
254 if (exec_result.peak_memory_usage > results->peak_memory_usage) {
255 results->peak_memory_usage = exec_result.peak_memory_usage;
256 }
257
258 runner->tests_run++;
259
260 lle_test_context_destroy(ctx);
261 runner->current_context = NULL;
262
263 /* Check fail-fast */
264 if (config && config->fail_fast &&
265 exec_result.result != LLE_TEST_RESULT_SUCCESS) {
266 break;
267 }
268 }
269
270 /* Check fail-fast for outer loop */
271 if (config && config->fail_fast && runner->tests_failed > 0) {
272 break;
273 }
274 }
275
276 results->end_timestamp = (uint64_t)time(NULL);
277 results->total_duration_us =
278 (results->end_timestamp - results->start_timestamp) * 1000000ULL;
279
280 return LLE_SUCCESS;
281}

501/**
502 * @brief Execute a single test
503 */
504lle_result_t lle_test_runner_execute_test(lle_test_runner_t *runner,
505 lle_test_context_t *context,
506 lle_test_execution_result_t *result) {
507 if (!runner || !context || !result) {
508 return LLE_ERROR_INVALID_PARAMETER;
509 }
510
511 lle_test_case_t *test_case = context->current_test;
512 if (!test_case || !test_case->test_function) {
513 return LLE_ERROR_INVALID_STATE;
514 }
515
516 /* Initialize result */
517 memset(result, 0, sizeof(lle_test_execution_result_t));
518
519 /* Record test start time */
520 struct timespec start_time;
521 clock_gettime(CLOCK_MONOTONIC, &start_time);
522
523 /* Run setup if provided */
524 if (test_case->setup_function) {
525 lle_result_t setup_result = test_case->setup_function(context);
526 if (setup_result != LLE_SUCCESS) {
527 result->result = LLE_TEST_RESULT_ERROR;
528 result->failure_reason = strdup("Setup failed");
529 return LLE_ERROR_INVALID_STATE;
530 }
531 }
532
533 /* Execute test */
534 lle_test_result_t test_result = test_case->test_function(context);
535
536 /* Run teardown if provided */
537 if (test_case->teardown_function) {
538 test_case->teardown_function(context);
539 }
540
541 /* Record test end time */
542 struct timespec end_time;
543 clock_gettime(CLOCK_MONOTONIC, &end_time);
544
545 /* Calculate duration in microseconds */
546 uint64_t duration_us =
547 ((end_time.tv_sec - start_time.tv_sec) * 1000000ULL) +
548 ((end_time.tv_nsec - start_time.tv_nsec) / 1000ULL);
549
550 /* Fill result */
551 result->result = test_result;
552 result->execution_time_us = duration_us;
553 result->peak_memory_usage =
554 0; /* Would need memory tracking for real value */
555
556 if (test_result != LLE_TEST_RESULT_SUCCESS) {
557 result->failure_reason = strdup(context->last_error);
558 }
559
560 return LLE_SUCCESS;
561}
