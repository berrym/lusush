78/*
79 * Parse X10 compatible mouse sequence
80 *
81 * Format: ESC[M<btn><x><y>
82 * - btn: button byte (32 + button_code + modifiers)
83 * - x, y: coordinate bytes (32 + coordinate, limited to 223)
84 */
85static lle_result_t parse_x10_sequence(lle_mouse_parser_t *parser,
86 const char *data, size_t data_len,
87 lle_mouse_event_info_t *event) {
88 if (!parser || !data || data_len < 6 || !event) {
89 return LLE_ERROR_INVALID_PARAMETER;
90 }
91
92 // Verify prefix: ESC[M
93 if (data[0] != '\x1B' || data[1] != '[' || data[2] != 'M') {
94 return LLE_ERROR_INVALID_FORMAT;
95 }
96
97 // Extract button byte
98 unsigned char btn = (unsigned char)data[3];
99 unsigned char x_byte = (unsigned char)data[4];
100 unsigned char y_byte = (unsigned char)data[5];
101
102 // Decode button
103 int button_code = (btn - 32) & 0x03;
104 bool is_motion = (btn - 32) & 0x20;
105 bool is_wheel = (btn - 32) & 0x40;
106
107 // Decode modifiers
108 event->modifiers = LLE_KEY_MOD_NONE;
109 if ((btn - 32) & 0x04)
110 event->modifiers |= LLE_KEY_MOD_SHIFT;
111 if ((btn - 32) & 0x08)
112 event->modifiers |= LLE_KEY_MOD_ALT;
113 if ((btn - 32) & 0x10)
114 event->modifiers |= LLE_KEY_MOD_CTRL;
115
116 // Decode coordinates (1-based, subtract 32 offset and convert to 0-based)
117 event->x = (x_byte - 32 - 1);
118 event->y = (y_byte - 32 - 1);
119
120 // Determine event type and button
121 if (is_wheel) {
122 event->type = LLE_MOUSE_EVENT_WHEEL;
123 event->button = (button_code == 0) ? LLE_MOUSE_BUTTON_WHEEL_UP
124 : LLE_MOUSE_BUTTON_WHEEL_DOWN;
125 event->wheel_delta = (button_code == 0) ? 1 : -1;
126 } else if (is_motion) {
127 event->type = LLE_MOUSE_EVENT_MOVE;
128 event->button = LLE_MOUSE_BUTTON_NONE;
129
130 // Check if dragging
131 if (parser->pressed_buttons != LLE_MOUSE_BUTTON_NONE) {
132 event->type = LLE_MOUSE_EVENT_DRAG;
133 event->button = parser->pressed_buttons;
134 }
135 } else if (button_code == 3) {
136 // Button release
137 event->type = LLE_MOUSE_EVENT_RELEASE;
138 event->button = parser->pressed_buttons;
139 parser->pressed_buttons = LLE_MOUSE_BUTTON_NONE;
140 } else {
141 // Button press
142 event->type = LLE_MOUSE_EVENT_PRESS;
143 event->button =
144 (lle_mouse_button_t)(button_code + 1); // Convert 0-2 to 1-3
145 parser->pressed_buttons = event->button;
146 }
147
148 event->timestamp = get_current_time_us();
149
150 // Update last position
151 parser->last_x = event->x;
152 parser->last_y = event->y;
153
154 return LLE_SUCCESS;
155}

157/*
158 * Parse SGR extended mouse sequence
159 *
160 * Format: ESC[<btn;x;y>M (press) or ESC[<btn;x;y>m (release)
161 * - btn: button code with modifiers
162 * - x, y: decimal coordinates (no limit)
163 * - M: press, m: release
164 */
165static lle_result_t parse_sgr_sequence(lle_mouse_parser_t *parser,
166 const char *data, size_t data_len,
167 lle_mouse_event_info_t *event) {
168 if (!parser || !data || data_len < 9 || !event) {
169 return LLE_ERROR_INVALID_PARAMETER;
170 }
171
172 // Verify prefix: ESC[<
173 if (data[0] != '\x1B' || data[1] != '[' || data[2] != '<') {
174 return LLE_ERROR_INVALID_FORMAT;
175 }
176
177 // Parse button;x;y
178 int btn = 0, x = 0, y = 0;
179 const char *ptr = data + 3;
180 const char *end = data + data_len;
181
182 // Parse button
183 while (ptr < end && *ptr >= '0' && *ptr <= '9') {
184 btn = btn * 10 + (*ptr - '0');
185 ptr++;
186 }
187
188 if (ptr >= end || *ptr != ';') {
189 return LLE_ERROR_INVALID_FORMAT;
190 }
191 ptr++; // Skip semicolon
192
193 // Parse x
194 while (ptr < end && *ptr >= '0' && *ptr <= '9') {
195 x = x * 10 + (*ptr - '0');
196 ptr++;
197 }
198
199 if (ptr >= end || *ptr != ';') {
200 return LLE_ERROR_INVALID_FORMAT;
201 }
202 ptr++; // Skip semicolon
203
204 // Parse y
205 while (ptr < end && *ptr >= '0' && *ptr <= '9') {
206 y = y * 10 + (*ptr - '0');
207 ptr++;
208 }
209
210 if (ptr >= end) {
211 return LLE_ERROR_INVALID_FORMAT;
212 }
213
214 // Check terminator (M for press, m for release)
215 bool is_release = (*ptr == 'm');
216 if (*ptr != 'M' && *ptr != 'm') {
217 return LLE_ERROR_INVALID_FORMAT;
218 }
219
220 // Decode button
221 int button_code = btn & 0x03;
222 bool is_motion = btn & 0x20;
223 bool is_wheel = btn & 0x40;
224
225 // Decode modifiers
226 event->modifiers = LLE_KEY_MOD_NONE;
227 if (btn & 0x04)
228 event->modifiers |= LLE_KEY_MOD_SHIFT;
229 if (btn & 0x08)
230 event->modifiers |= LLE_KEY_MOD_ALT;
231 if (btn & 0x10)
232 event->modifiers |= LLE_KEY_MOD_CTRL;
233
234 // Set coordinates (SGR uses 1-based, convert to 0-based)
235 event->x = x - 1;
236 event->y = y - 1;
237
238 // Determine event type and button
239 if (is_wheel) {
240 event->type = LLE_MOUSE_EVENT_WHEEL;
241 event->button = (button_code == 0) ? LLE_MOUSE_BUTTON_WHEEL_UP
242 : LLE_MOUSE_BUTTON_WHEEL_DOWN;
243 event->wheel_delta = (button_code == 0) ? 1 : -1;
244 } else if (is_motion) {
245 event->type = LLE_MOUSE_EVENT_MOVE;
246 event->button = LLE_MOUSE_BUTTON_NONE;
247
248 // Check if dragging
249 if (parser->pressed_buttons != LLE_MOUSE_BUTTON_NONE) {
250 event->type = LLE_MOUSE_EVENT_DRAG;
251 event->button = parser->pressed_buttons;
252 }
253 } else if (is_release) {
254 event->type = LLE_MOUSE_EVENT_RELEASE;
255 event->button = (parser->pressed_buttons != LLE_MOUSE_BUTTON_NONE)
256 ? parser->pressed_buttons
257 : (lle_mouse_button_t)(button_code + 1);
258 parser->pressed_buttons = LLE_MOUSE_BUTTON_NONE;
259 } else {
260 event->type = LLE_MOUSE_EVENT_PRESS;
261 event->button = (lle_mouse_button_t)(button_code + 1);
262 parser->pressed_buttons = event->button;
263 }
264
265 event->timestamp = get_current_time_us();
266
267 // Update last position
268 parser->last_x = event->x;
269 parser->last_y = event->y;
270
271 return LLE_SUCCESS;
272}

274/*
275 * Detect multi-click events (double/triple click)
276 */
277static void detect_multi_click(lle_mouse_parser_t *parser,
278 lle_mouse_event_info_t *event) {
279 if (!parser || !event || event->type != LLE_MOUSE_EVENT_PRESS) {
280 return;
281 }
282
283 uint64_t current_time = event->timestamp;
284 uint64_t time_diff = current_time - parser->last_click_time;
285
286 // Check if within double-click time window (500ms)
287 if (time_diff < (LLE_MOUSE_DOUBLE_CLICK_TIME_MS * 1000)) {
288 parser->click_count++;
289 } else {
290 parser->click_count = 1;
291 }
292
293 parser->last_click_time = current_time;
294
295 // Set multi-click flags
296 if (parser->click_count == 2) {
297 event->double_click = true;
298 event->triple_click = false;
299 } else if (parser->click_count >= 3) {
300 event->double_click = false;
301 event->triple_click = true;
302 } else {
303 event->double_click = false;
304 event->triple_click = false;
305 }
306}

308/*
309 * Parse mouse sequence
310 *
311 * Detects format and dispatches to appropriate parser
312 */
313lle_result_t
314lle_mouse_parser_parse_sequence(lle_mouse_parser_t *parser,
315 const char *sequence, size_t sequence_len,
316 lle_mouse_event_info_t **event_info) {
317 if (!parser || !sequence || !event_info) {
318 return LLE_ERROR_INVALID_PARAMETER;
319 }
320
321 if (sequence_len < 6) {
322 parser->invalid_mouse_sequences++;
323 return LLE_ERROR_INVALID_FORMAT;
324 }
325
326 *event_info = NULL;
327
328 // Allocate event structure
329 lle_mouse_event_info_t *event =
330 lle_pool_alloc(sizeof(lle_mouse_event_info_t));
331 if (!event) {
332 return LLE_ERROR_OUT_OF_MEMORY;
333 }
334
335 memset(event, 0, sizeof(lle_mouse_event_info_t));
336
337 lle_result_t result;
338
339 // Detect format and parse
340 if (sequence_len >= 6 && sequence[0] == '\x1B' && sequence[1] == '[' &&
341 sequence[2] == 'M') {
342 // X10 format
343 result = parse_x10_sequence(parser, sequence, sequence_len, event);
344 } else if (sequence_len >= 9 && sequence[0] == '\x1B' &&
345 sequence[1] == '[' && sequence[2] == '<') {
346 // SGR format
347 result = parse_sgr_sequence(parser, sequence, sequence_len, event);
348 } else {
349 // Unknown format
350 lle_pool_free(event);
351 parser->invalid_mouse_sequences++;
352 return LLE_ERROR_INVALID_FORMAT;
353 }
354
355 if (result != LLE_SUCCESS) {
356 lle_pool_free(event);
357 parser->invalid_mouse_sequences++;
358 return result;
359 }
360
361 // Detect multi-click
362 detect_multi_click(parser, event);
363
364 parser->mouse_events_parsed++;
365 *event_info = event;
366
367 return LLE_SUCCESS;
368}
