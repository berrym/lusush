1/**
2 * @file keybinding_config.h
3 * @brief LLE User Keybinding Configuration System
4 *
5 * Provides user-customizable keybindings via configuration files.
6 * Users can override default Emacs keybindings by creating:
7 * ~/.config/lusush/keybindings.toml
8 *
9 * Example config:
10 * [bindings]
11 * "C-a" = "end-of-line" # Swap C-a and C-e
12 * "C-e" = "beginning-of-line"
13 * "M-p" = "history-search-backward"
14 * "C-s" = "none" # Unbind
15 *
16 * Specification: Spec 13 (User Customization), Spec 22 (User Interface)
17 * Version: 1.0.0
18 */
19

52/**
53 * @brief Action registry entry mapping name to function pointer
54 *
55 * Maps GNU Readline action names (e.g., "forward-char") to LLE
56 * function pointers (e.g., lle_forward_char).
57 */
58typedef struct lle_action_registry_entry {
59 const char *name; /**< GNU Readline action name */
60 lle_action_type_t type; /**< Action type (SIMPLE or CONTEXT) */
61 union {
62 lle_action_simple_t simple; /**< Simple action function */
63 lle_action_context_t context; /**< Context-aware action function */
64 } func;
65 const char *description; /**< Human-readable description */
66} lle_action_registry_entry_t;
