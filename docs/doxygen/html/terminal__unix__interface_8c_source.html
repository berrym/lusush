2 * terminal_unix_interface.c - Unix Terminal Interface (Spec 02 Subsystem 6)
3 *
4 * Minimal Unix terminal interface abstraction providing:
5 * - Raw mode setup and teardown
6 * - Terminal attribute saving and restoration

11 * - EOF and error detection

13 * Critical Principles:
14 * - Always restore terminal state on exit
15 * - Thread-safe state transitions
16 * - Async-signal-safe signal handlers
17 * - Idempotent operations (safe to call multiple times)

36/* ============================================================================
37 * GLOBAL STATE FOR SIGNAL HANDLERS
38 * ============================================================================
39 *
40 * Unfortunately, POSIX signal handlers require global state. We use a single
41 * global pointer protected by mutex where possible.
42 */
43
44static lle_unix_interface_t *g_signal_interface = NULL;
45
46/* ============================================================================
47 * SIGNAL HANDLERS
48 * ============================================================================
49 *
50 * Signal handlers must be async-signal-safe. Only a limited set of functions
51 * are allowed: tcsetattr, signal, raise, write, etc. NO malloc, printf, etc.
52 */
53
54/*
55 * SIGWINCH handler - window size changed
56 *
57 * This is called when the terminal window is resized. We don't do much here
58 * because we can't safely call complex functions in signal context.
59 */
60static void handle_sigwinch(int sig) {
61 (void)sig;
62
63 /* Set flag to be checked in event loop (async-signal-safe) */
64 if (g_signal_interface) {
65 g_signal_interface->sigwinch_received = true;
66 }
67}
68
69/*
70 * SIGTSTP handler - suspend (Ctrl-Z)
71 *
72 * Before suspending, we must restore the terminal to its original state
73 * so the user gets a normal shell prompt when backgrounded.
74 */
75static void handle_sigtstp(int sig) {
76 if (!g_signal_interface)
77 return;
78
79 /* Exit raw mode before suspending (async-signal-safe) */
80 if (g_signal_interface->raw_mode_active) {
81 tcsetattr(g_signal_interface->terminal_fd, TCSAFLUSH,
82 &g_signal_interface->original_termios);
83 }
84
85 /* Re-raise signal with default handler to actually suspend */
86 signal(sig, SIG_DFL);
87 raise(sig);
88}
89
90/*
91 * SIGCONT handler - resume after suspend
92 *
93 * When resumed from background, re-enter raw mode if we were in it.
94 */
95static void handle_sigcont(int sig) {
96 (void)sig;
97
98 if (!g_signal_interface)
99 return;
100
101 /* Re-enter raw mode if we were in it (async-signal-safe) */
102 if (g_signal_interface->raw_mode_active) {
103 tcsetattr(g_signal_interface->terminal_fd, TCSAFLUSH,
104 &g_signal_interface->raw_termios);
105 }
106
107 /* Re-install SIGTSTP handler (it was reset to default) */
108 signal(SIGTSTP, handle_sigtstp);
109}

122/*
123 * atexit handler - ensure terminal is restored even on abnormal exit
124 */
125static void cleanup_on_exit(void) {
126 if (g_signal_interface && g_signal_interface->raw_mode_active) {
127 /* This is safe in atexit context */
128 tcsetattr(g_signal_interface->terminal_fd, TCSAFLUSH,
129 &g_signal_interface->original_termios);
130 }
131}

133/*
134 * Register atexit cleanup handler (called once)
135 */
136static void register_cleanup(void) {
137 static bool registered = false;
138
139 if (!registered) {
140 atexit(cleanup_on_exit);
141 registered = true;
142 }

145/* ============================================================================
146 * SIGNAL HANDLER INSTALLATION
147 * ============================================================================
148 */
149
150/* Storage for original signal handlers (static lifetime) */
151static struct sigaction original_sigwinch;
152static struct sigaction original_sigtstp;
153static struct sigaction original_sigcont;
154/* Note: We don't install SIGINT/SIGTERM handlers anymore (lusush handles them)
155 */
156static bool signals_installed = false;
157
158/*
159 * Install all signal handlers
160 */
161static lle_result_t install_signal_handlers(lle_unix_interface_t *interface) {
162 if (signals_installed) {
163 return LLE_SUCCESS; /* Already installed */
164 }
165
166 struct sigaction sa;
167
168 /* SIGWINCH - window resize */
169 memset(&sa, 0, sizeof(sa));
170 sa.sa_handler = handle_sigwinch;
171 sigemptyset(&sa.sa_mask);
172 sa.sa_flags = SA_RESTART; /* Restart interrupted system calls */
173
174 if (sigaction(SIGWINCH, &sa, &original_sigwinch) != 0) {
175 return LLE_ERROR_SYSTEM_CALL;
176 }
177
178 /* SIGTSTP - suspend (Ctrl-Z) */
179 sa.sa_handler = handle_sigtstp;
180 if (sigaction(SIGTSTP, &sa, &original_sigtstp) != 0) {
181 sigaction(SIGWINCH, &original_sigwinch, NULL);
182 return LLE_ERROR_SYSTEM_CALL;
183 }
184
185 /* SIGCONT - resume */
186 sa.sa_handler = handle_sigcont;
187 if (sigaction(SIGCONT, &sa, &original_sigcont) != 0) {
188 sigaction(SIGWINCH, &original_sigwinch, NULL);
189 sigaction(SIGTSTP, &original_sigtstp, NULL);
190 return LLE_ERROR_SYSTEM_CALL;
191 }
192
193 /* NOTE: We do NOT install SIGINT/SIGTERM handlers here.
194 * Lusush's signal handlers (src/signals.c) manage these properly:
195 * - SIGINT: kills child process OR clears line (but never exits shell)
196 * - SIGTERM: handles graceful shutdown
197 *
198 * LLE previously installed handlers that would exit the shell on Ctrl+C,
199 * which is incorrect shell behavior. Now that ISIG is enabled in raw mode,
200 * Ctrl+C generates SIGINT which lusush's handler will catch and handle
201 * correctly.
202 */
203
204 /* Set global pointer for handlers */
205 g_signal_interface = interface;
206 signals_installed = true;
207
208 return LLE_SUCCESS;
209}
210
211/*
212 * Restore original signal handlers
213 */
214static void restore_signal_handlers(lle_unix_interface_t *interface) {
215 if (!signals_installed) {
216 return;
217 }
218
219 /* Only restore if this was the interface that installed them */
220 if (g_signal_interface != interface) {
221 return;
222 }
223
224 /* Restore signal handlers that we installed */
225 sigaction(SIGWINCH, &original_sigwinch, NULL);
226 sigaction(SIGTSTP, &original_sigtstp, NULL);
227 sigaction(SIGCONT, &original_sigcont, NULL);
228 /* Note: We don't restore SIGINT/SIGTERM because we never installed them */
229
230 /* Clear global pointer */
231 g_signal_interface = NULL;
232 signals_installed = false;
233}

240/*
241 * Initialize Unix terminal interface
242 */
243lle_result_t lle_unix_interface_init(lle_unix_interface_t **interface) {
244 if (!interface) {
245 return LLE_ERROR_INVALID_PARAMETER;
246 }
247
248 /* Allocate interface structure */
249 lle_unix_interface_t *iface = calloc(1, sizeof(lle_unix_interface_t));
250 if (!iface) {
251 return LLE_ERROR_OUT_OF_MEMORY;
252 }
253
254 /* Set file descriptor (using STDIN for terminal operations) */
255 iface->terminal_fd = STDIN_FILENO;
256
257 /* Save original terminal state (if this is a TTY) */
258 /* In non-TTY environments (tests, pipes), tcgetattr will fail - that's OK
259 */
260 if (tcgetattr(iface->terminal_fd, &iface->original_termios) != 0) {
261 /* Not a TTY - initialize with empty termios */
262 memset(&iface->original_termios, 0, sizeof(struct termios));
263 }
264
265 /* Initialize state */
266 iface->raw_mode_active = false;
267 iface->size_changed = false;
268 iface->sigwinch_received = false;
269 iface->last_error = LLE_SUCCESS;
270
271 /* Get initial window size */
272 struct winsize ws;
273 if (ioctl(STDOUT_FILENO, TIOCGWINSZ, &ws) == 0) {
274 iface->current_width = ws.ws_col;
275 iface->current_height = ws.ws_row;
276 } else {
277 iface->current_width = 80;
278 iface->current_height = 24;
279 }
280
281 /* Install signal handlers */
282 lle_result_t result = install_signal_handlers(iface);
283 if (result != LLE_SUCCESS) {
284 free(iface);
285 return result;
286 }
287
288 /* Register atexit cleanup */
289 register_cleanup();
290
291 /* Initialize parser-related fields to NULL (will be set up later) */
292 iface->sequence_parser = NULL;
293 iface->key_detector = NULL;
294 iface->capabilities = NULL;
295 iface->memory_pool = NULL;
296
297 *interface = iface;
298 return LLE_SUCCESS;
299}

301/*
302 * Initialize sequence parser for escape sequence processing
303 *
304 * This must be called after lle_unix_interface_init() and after
305 * terminal capabilities have been detected.
306 */
307lle_result_t lle_unix_interface_init_sequence_parser(
308 lle_unix_interface_t *interface, lle_terminal_capabilities_t *capabilities,
309 lle_memory_pool_t *memory_pool) {
310 if (!interface || !capabilities || !memory_pool) {
311 return LLE_ERROR_INVALID_PARAMETER;
312 }
313
314 /* Store references for later use */
315 interface->capabilities = capabilities;
316 interface->memory_pool = memory_pool;
317
318 /* Initialize sequence parser */
319 lle_result_t result = lle_sequence_parser_init(&interface->sequence_parser,
320 capabilities, memory_pool);
321
322 if (result != LLE_SUCCESS) {
323 interface->capabilities = NULL;
324 interface->memory_pool = NULL;
325 return result;
326 }
327
328 /* Initialize key detector */
329 result = lle_key_detector_init(&interface->key_detector, capabilities,

331

333 lle_sequence_parser_destroy(interface->sequence_parser);
334 interface->sequence_parser = NULL;
335 interface->capabilities = NULL;
336 interface->memory_pool = NULL;
337 return result;
338 }
339
340 return LLE_SUCCESS;
341}

343/*
344 * Destroy Unix terminal interface
345 */
346void lle_unix_interface_destroy(lle_unix_interface_t *interface) {
347 if (!interface) {
348 return;
349 }
350
351 /* Ensure we exit raw mode before cleanup */
352 if (interface->raw_mode_active) {
353 lle_unix_interface_exit_raw_mode(interface);
354 }
355
356 /* Clean up sequence parser if initialized */
357 if (interface->sequence_parser) {
358 lle_sequence_parser_destroy(interface->sequence_parser);
359 interface->sequence_parser = NULL;
360 }
361
362 /* Clean up key detector if initialized */
363 if (interface->key_detector) {
364 lle_key_detector_destroy(interface->key_detector);
365 interface->key_detector = NULL;
366 }
367
368 /* Clear references (we don't own capabilities or memory_pool) */
369 interface->capabilities = NULL;
370 interface->memory_pool = NULL;
371
372 /* Restore original signal handlers */
373 restore_signal_handlers(interface);
374
375 /* Free structure */
376 free(interface);
377}

379/*
380 * Enter raw (non-canonical) mode
381 */
382lle_result_t
383lle_unix_interface_enter_raw_mode(lle_unix_interface_t *interface) {
384 if (!interface) {
385 return LLE_ERROR_INVALID_PARAMETER;
386 }
387
388 /* Already in raw mode? (idempotent) */
389 if (interface->raw_mode_active) {
390 return LLE_SUCCESS;
391 }
392
393 /* Copy original settings */
394 interface->raw_termios = interface->original_termios;
395
396 /* Modify for raw mode */
397 struct termios *raw = &interface->raw_termios;
398
399 /* Input flags - disable special processing */
400 raw->c_iflag &= ~(BRKINT | /* No break signal */
401 ICRNL | /* Don't translate CR to NL */
402 INPCK | /* Disable parity checking */
403 ISTRIP | /* Don't strip 8th bit */
404 IXON); /* Disable XON/XOFF flow control */
405
406 /* Output flags - KEEP output processing for proper display */
407 /* NOTE: Disabling OPOST causes display corruption - \n won't return to
408 * column 0 */
409 /* We need raw INPUT mode, but output should remain processed for display */
410
411 /* Control flags - 8-bit characters */
412 raw->c_cflag |= (CS8); /* 8 bits per byte */
413
414 /* Local flags - disable canonical mode and echo, but KEEP signals enabled
415 */
416 raw->c_lflag &= ~(ECHO | /* No echo */
417 ICANON | /* Non-canonical mode */
418 IEXTEN); /* Disable extended input processing */
419 /* KEEP ISIG ENABLED - allow Ctrl-C to generate SIGINT for proper shell
420 * behavior */
421 /* This ensures lusush's signal handler (src/signals.c) can manage child
422 * processes */
423
424 /* Control characters - non-blocking read */
425 raw->c_cc[VMIN] = 0; /* Non-blocking: return immediately */
426 raw->c_cc[VTIME] = 0; /* No timeout */
427
428 /* Apply settings - TCSAFLUSH discards unread input */
429 if (tcsetattr(interface->terminal_fd, TCSAFLUSH, raw) != 0) {
430 interface->last_error = LLE_ERROR_SYSTEM_CALL;
431 return LLE_ERROR_SYSTEM_CALL;
432 }
433
434 interface->raw_mode_active = true;
435 return LLE_SUCCESS;
436}

462/*
463 * Get current window size
464 */
465lle_result_t lle_unix_interface_get_window_size(lle_unix_interface_t *interface,
466 size_t *width, size_t *height) {
467 if (!interface || !width || !height) {
468 return LLE_ERROR_INVALID_PARAMETER;
469 }
470
471 struct winsize ws;
472
473 /* Try ioctl first */
474 if (ioctl(STDOUT_FILENO, TIOCGWINSZ, &ws) == 0 && ws.ws_col > 0 &&
475 ws.ws_row > 0) {
476 *width = ws.ws_col;
477 *height = ws.ws_row;
478
479 /* Update cached size */
480 interface->current_width = ws.ws_col;
481 interface->current_height = ws.ws_row;
482
483 return LLE_SUCCESS;
484 }
485
486 /* Fallback to environment variables */
487 const char *env_cols = getenv("COLUMNS");
488 const char *env_lines = getenv("LINES");
489
490 *width = (env_cols && *env_cols) ? (size_t)atoi(env_cols) : 80;
491 *height = (env_lines && *env_lines) ? (size_t)atoi(env_lines) : 24;
492
493 /* Update cached size */
494 interface->current_width = *width;
495 interface->current_height = *height;
496
497 return LLE_SUCCESS;
498}

505/*
506 * Convert lle_key_info_t keycode to lle_special_key_t
507 */
508static lle_special_key_t convert_key_code(uint32_t keycode,
509 lle_key_type_t key_type) {
510 /* For cursor keys and editing keys, keycode typically maps directly */
511 switch (key_type) {
512 case LLE_KEY_TYPE_CURSOR:
513 /* Cursor keys: handle both numeric (1-4) and ASCII ('A'-'D') formats */
514 if (keycode >= 1 && keycode <= 4) {
515 return (lle_special_key_t)(LLE_KEY_UP + keycode - 1);
516 }
517 /* CSI cursor keys use ASCII: A=Up, B=Down, C=Right, D=Left */
518 switch (keycode) {
519 case 'A':
520 return LLE_KEY_UP;
521 case 'B':
522 return LLE_KEY_DOWN;
523 case 'C':
524 return LLE_KEY_RIGHT;
525 case 'D':
526 return LLE_KEY_LEFT;
527 /* Home/End also sometimes reported as cursor type */
528 case 'H':
529 return LLE_KEY_HOME;
530 case 'F':
531 return LLE_KEY_END;
532 /* PageUp/PageDown with ASCII digit keycodes */
533 case '5':
534 return LLE_KEY_PAGE_UP;
535 case '6':
536 return LLE_KEY_PAGE_DOWN;
537 default:
538 break;
539 }
540 break;
541 case LLE_KEY_TYPE_FUNCTION:
542 /* Function keys: F1-F12 */
543 if (keycode >= 1 && keycode <= 12) {
544 return (lle_special_key_t)(LLE_KEY_F1 + keycode - 1);
545 }
546 break;
547 case LLE_KEY_TYPE_EDITING:
548 /* Editing keys: map common codes (both numeric and ASCII) */
549 switch (keycode) {
550 case 1:
551 return LLE_KEY_HOME;
552 case 2:
553 case '2':
554 return LLE_KEY_INSERT;
555 case 3:
556 case '3':
557 return LLE_KEY_DELETE;
558 case 4:
559 return LLE_KEY_END;
560 case 5:
561 case '5':
562 return LLE_KEY_PAGE_UP;
563 case 6:
564 case '6':
565 return LLE_KEY_PAGE_DOWN;
566 default:
567 break;
568 }
569 break;
570 case LLE_KEY_TYPE_CONTROL:
571 /* Control characters - treat as regular characters for now */
572 /* Ctrl+C (0x03) should be handled by the application layer */
573 /* We'll return UNKNOWN here and let the character handler deal with it
574 */
575 break;
576 case LLE_KEY_TYPE_SPECIAL:
577 /* Special keys */
578 switch (keycode) {
579 case 9:
580 return LLE_KEY_TAB;
581 case 13:
582 case 10:
583 return LLE_KEY_ENTER;
584 case 27:
585 return LLE_KEY_ESCAPE;
586 case 127:
587 return LLE_KEY_BACKSPACE;
588 default:
589 break;
590 }
591 break;
592 default:
593 break;
594 }
595 return LLE_KEY_UNKNOWN;
596}

598/*
599 * Convert lle_key_modifiers_t to lle_key_modifier_t
600 */
601static lle_key_modifier_t convert_modifiers(lle_key_modifiers_t parser_mods) {
602 lle_key_modifier_t result = LLE_MOD_NONE;
603
604 if (parser_mods & LLE_KEY_MOD_SHIFT)
605 result |= LLE_MOD_SHIFT;
606 if (parser_mods & LLE_KEY_MOD_ALT)
607 result |= LLE_MOD_ALT;
608 if (parser_mods & LLE_KEY_MOD_CTRL)
609 result |= LLE_MOD_CTRL;
610 if (parser_mods & LLE_KEY_MOD_META)
611 result |= LLE_MOD_META;
612
613 return result;
614}

616/*
617 * Convert lle_parsed_input_t to lle_input_event_t
618 *
619 * This function bridges the comprehensive sequence parser output with the
620 * terminal abstraction input event format.
621 */
622static lle_result_t
623convert_parsed_input_to_event(const lle_parsed_input_t *parsed,
624 lle_input_event_t *event) {
625 if (!parsed || !event) {
626 return LLE_ERROR_INVALID_PARAMETER;
627 }
628
629 memset(event, 0, sizeof(lle_input_event_t));
630 event->timestamp = parsed->data.text_info.timestamp;
631 event->sequence_number = 0; /* Will be set by caller if needed */
632
633 switch (parsed->type) {
634 case LLE_PARSED_INPUT_TYPE_TEXT:
635 /* Regular text input */
636 event->type = LLE_INPUT_TYPE_CHARACTER;
637 event->data.character.codepoint = parsed->data.text_info.codepoint;
638 memcpy(event->data.character.utf8_bytes,
639 parsed->data.text_info.utf8_bytes,
640 parsed->data.text_info.utf8_length);
641 event->data.character.byte_count =
642 (uint8_t)parsed->data.text_info.utf8_length;
643 break;
644
645 case LLE_PARSED_INPUT_TYPE_KEY: {
646 /* Key press or combination */
647 /* First try to convert to special key */
648 lle_special_key_t special_key = convert_key_code(
649 parsed->data.key_info.keycode, parsed->data.key_info.type);
650
651 /* If it's a control character that didn't map to a special key,
652 * treat it as a regular character event (e.g., Ctrl+C) */
653 if (special_key == LLE_KEY_UNKNOWN &&
654 parsed->data.key_info.type == LLE_KEY_TYPE_CONTROL) {
655 /* Control character - return as CHARACTER event */
656 /* For control chars, keycode might be ASCII letter, so use raw
657 * value */
658 uint32_t ctrl_code = parsed->data.key_info.keycode;
659 /* If keycode is uppercase letter, convert to control code (Ctrl+C =
660 * 'C'-64 = 3) */
661 if (ctrl_code >= 'A' && ctrl_code <= 'Z') {
662 ctrl_code = ctrl_code - 64;
663 } else if (ctrl_code >= 'a' && ctrl_code <= 'z') {
664 ctrl_code = ctrl_code - 96;
665 }
666 event->type = LLE_INPUT_TYPE_CHARACTER;
667 event->data.character.codepoint = ctrl_code;
668 event->data.character.utf8_bytes[0] = (char)ctrl_code;
669 event->data.character.byte_count = 1;
670 } else {
671 /* Regular special key */
672 event->type = LLE_INPUT_TYPE_SPECIAL_KEY;
673 event->data.special_key.key = special_key;
674 event->data.special_key.keycode = parsed->data.key_info.keycode;
675 event->data.special_key.modifiers =
676 convert_modifiers(parsed->data.key_info.modifiers);
677 }
678 event->timestamp = parsed->data.key_info.timestamp;
679 break;
680 }
681
682 case LLE_PARSED_INPUT_TYPE_MOUSE:
683 /* Mouse events are not directly supported by lle_input_event_t */
684 /* We'll need to handle these through a different mechanism later */
685 /* For now, treat as unknown/error */
686 event->type = LLE_INPUT_TYPE_ERROR;
687 event->data.error.error_code = LLE_ERROR_FEATURE_NOT_AVAILABLE;
688 snprintf(event->data.error.error_message,
689 sizeof(event->data.error.error_message),
690 "Mouse events not yet supported");
691 break;
692
693 case LLE_PARSED_INPUT_TYPE_SEQUENCE:
694 /* Terminal sequence - treat as error for now */
695 event->type = LLE_INPUT_TYPE_ERROR;
696 event->data.error.error_code = LLE_ERROR_FEATURE_NOT_AVAILABLE;
697 snprintf(event->data.error.error_message,
698 sizeof(event->data.error.error_message),
699 "Terminal sequence events not yet supported");
700 break;
701
702 case LLE_PARSED_INPUT_TYPE_PASTE:
703 /* Bracketed paste - treat as error for now */
704 event->type = LLE_INPUT_TYPE_ERROR;
705 event->data.error.error_code = LLE_ERROR_FEATURE_NOT_AVAILABLE;
706 snprintf(event->data.error.error_message,
707 sizeof(event->data.error.error_message),
708 "Paste events not yet supported");
709 break;
710
711 case LLE_PARSED_INPUT_TYPE_FOCUS:
712 /* Focus events - treat as error for now */
713 event->type = LLE_INPUT_TYPE_ERROR;
714 event->data.error.error_code = LLE_ERROR_FEATURE_NOT_AVAILABLE;
715 snprintf(event->data.error.error_message,
716 sizeof(event->data.error.error_message),
717 "Focus events not yet supported");
718 break;
719
720 case LLE_PARSED_INPUT_TYPE_UNKNOWN:
721 default:
722 /* Unknown input type */
723 event->type = LLE_INPUT_TYPE_ERROR;
724 event->data.error.error_code = LLE_ERROR_INPUT_PARSING;
725 snprintf(event->data.error.error_message,
726 sizeof(event->data.error.error_message),
727 "Unknown input type from parser");
728 break;
729 }
730
731 return LLE_SUCCESS;
732}

739/*
740 * Determine UTF-8 sequence length from first byte
741 */
742static int get_utf8_length(unsigned char first_byte) {
743 if ((first_byte & 0x80) == 0x00)
744 return 1; /* 0xxxxxxx - ASCII */
745 if ((first_byte & 0xE0) == 0xC0)
746 return 2; /* 110xxxxx - 2 bytes */
747 if ((first_byte & 0xF0) == 0xE0)
748 return 3; /* 1110xxxx - 3 bytes */
749 if ((first_byte & 0xF8) == 0xF0)
750 return 4; /* 11110xxx - 4 bytes */
751 return -1; /* Invalid first byte */
752}

754/*
755 * Decode UTF-8 sequence to Unicode codepoint
756 *
757 * This decoder:
758 * - Validates basic structure (continuation bytes)
759 * - Returns replacement character (U+FFFD) for invalid sequences
760 * - Does not detect overlong sequences (acceptable tradeoff)
761 */
762static lle_result_t decode_utf8(lle_unix_interface_t *interface,
763 unsigned char first_byte,
764 uint32_t *codepoint_out, char *utf8_bytes,
765 uint8_t *byte_count_out) {
766 int expected_bytes = get_utf8_length(first_byte);
767
768 if (expected_bytes < 0) {
769 /* Invalid first byte - use replacement character */
770 *codepoint_out = 0xFFFD;
771 utf8_bytes[0] = (char)first_byte;
772 *byte_count_out = 1;
773 return LLE_SUCCESS;
774 }
775
776 utf8_bytes[0] = (char)first_byte;
777 *byte_count_out = (uint8_t)expected_bytes;
778
779 if (expected_bytes == 1) {
780 /* ASCII - fast path */
781 *codepoint_out = first_byte;
782 return LLE_SUCCESS;
783 }
784
785 /* Read additional bytes for multi-byte sequence */
786 for (int i = 1; i < expected_bytes; i++) {
787 unsigned char byte;
788 ssize_t n = read(interface->terminal_fd, &byte, 1);
789
790 if (n <= 0) {
791 /* Incomplete sequence - use replacement character */
792 *codepoint_out = 0xFFFD;
793 return LLE_SUCCESS;
794 }
795
796 /* Validate continuation byte (10xxxxxx) */
797 if ((byte & 0xC0) != 0x80) {
798 /* Invalid continuation - use replacement character */
799 *codepoint_out = 0xFFFD;
800 return LLE_SUCCESS;
801 }
802
803 utf8_bytes[i] = (char)byte;
804 }
805
806 /* Decode to Unicode codepoint */
807 switch (expected_bytes) {
808 case 2:
809 *codepoint_out =
810 ((first_byte & 0x1F) << 6) | ((unsigned char)utf8_bytes[1] & 0x3F);
811 break;
812 case 3:
813 *codepoint_out = ((first_byte & 0x0F) << 12) |
814 (((unsigned char)utf8_bytes[1] & 0x3F) << 6) |
815 ((unsigned char)utf8_bytes[2] & 0x3F);
816 break;
817 case 4:
818 *codepoint_out = ((first_byte & 0x07) << 18) |
819 (((unsigned char)utf8_bytes[1] & 0x3F) << 12) |
820 (((unsigned char)utf8_bytes[2] & 0x3F) << 6) |
821 ((unsigned char)utf8_bytes[3] & 0x3F);
822 break;
823 default:
824 *codepoint_out = 0xFFFD;
825 break;
826 }
827
828 return LLE_SUCCESS;
829}

836/*
837 * Read input event from terminal with timeout support
838 *
839 * This implementation provides:
840 * - Non-blocking input with configurable timeout
841 * - UTF-8 character decoding
842 * - Window resize event generation (from SIGWINCH)
843 * - EOF detection
844 * - Timeout events
845 *
846 * Higher-level parsing (escape sequences, special keys) is handled by
847 * Spec 06 Input Parsing, which wraps this primitive interface.
848 */
849lle_result_t lle_unix_interface_read_event(lle_unix_interface_t *interface,
850 lle_input_event_t *event,
851 uint32_t timeout_ms) {
852 if (!interface || !event) {
853 return LLE_ERROR_INVALID_PARAMETER;
854 }
855
856 /* Clear event structure */
857 memset(event, 0, sizeof(lle_input_event_t));
858
859 /* Check for pending SIGWINCH (resize event has priority) */
860 if (interface->sigwinch_received) {
861 interface->sigwinch_received = false;
862
863 size_t width, height;
864 lle_result_t result =
865 lle_unix_interface_get_window_size(interface, &width, &height);
866 if (result != LLE_SUCCESS) {
867 event->type = LLE_INPUT_TYPE_ERROR;
868 event->timestamp = lle_get_current_time_microseconds();
869 event->data.error.error_code = result;
870 snprintf(event->data.error.error_message,
871 sizeof(event->data.error.error_message),
872 "Failed to get window size after SIGWINCH");
873 return result;
874 }
875
876 event->type = LLE_INPUT_TYPE_WINDOW_RESIZE;
877 event->timestamp = lle_get_current_time_microseconds();
878 event->data.resize.new_width = width;
879 event->data.resize.new_height = height;
880 interface->size_changed = true;
881
882 return LLE_SUCCESS;
883 }
884
885 /* Use select() for timeout support */
886 fd_set readfds;
887 FD_ZERO(&readfds);
888 FD_SET(interface->terminal_fd, &readfds);
889
890 struct timeval tv;
891 struct timeval *tv_ptr;
892
893 /* Determine effective timeout */
894 uint32_t effective_timeout_ms = timeout_ms;
895
896 /* If parser is accumulating an escape sequence, use a shorter timeout
897 * to detect standalone ESC key (50ms is typical escape sequence timeout) */
898 if (interface->sequence_parser) {
899 lle_parser_state_t parser_state =
900 lle_sequence_parser_get_state(interface->sequence_parser);
901 if (parser_state != LLE_PARSER_STATE_NORMAL) {
902 /* Parser is waiting for more sequence bytes - use 60ms timeout
903 * (slightly longer than the 50ms sequence timeout to ensure we
904 * detect it) */
905 if (effective_timeout_ms == UINT32_MAX ||
906 effective_timeout_ms > 60) {
907 effective_timeout_ms = 60;
908 }
909 }
910 }
911
912 if (effective_timeout_ms == UINT32_MAX) {
913 /* Infinite timeout - pass NULL to select() */
914 tv_ptr = NULL;
915 } else {
916 tv.tv_sec = (time_t)(effective_timeout_ms / 1000);
917 tv.tv_usec = (suseconds_t)((effective_timeout_ms % 1000) * 1000);
918 tv_ptr = &tv;
919 }
920
921 int ready =
922 select(interface->terminal_fd + 1, &readfds, NULL, NULL, tv_ptr);
923
924 if (ready == -1) {
925 if (errno == EINTR) {
926 /* Interrupted by signal - check for resize */
927 if (interface->sigwinch_received) {
928 /* Recursively handle resize event */
929 return lle_unix_interface_read_event(interface, event,
930 timeout_ms);
931 }
932 /* Other signal - return timeout */
933 event->type = LLE_INPUT_TYPE_TIMEOUT;
934 event->timestamp = lle_get_current_time_microseconds();
935 return LLE_SUCCESS;
936 }
937 /* System call error */
938 event->type = LLE_INPUT_TYPE_ERROR;
939 event->timestamp = lle_get_current_time_microseconds();
940 event->data.error.error_code = LLE_ERROR_SYSTEM_CALL;
941 snprintf(event->data.error.error_message,
942 sizeof(event->data.error.error_message), "select() failed: %s",
943 strerror(errno));
944 return LLE_ERROR_SYSTEM_CALL;
945 }
946
947 if (ready == 0) {
948 /* Timeout - no data available */
949 /* Check if parser is accumulating a sequence that has timed out */
950 if (interface->sequence_parser) {
951 lle_parsed_input_t *timeout_input = NULL;
952 lle_result_t timeout_result = lle_sequence_parser_check_timeout(
953 interface->sequence_parser,
954 300000, /* 300ms timeout for ESC+key (Meta) sequences */
955 &timeout_input);
956
957 if (timeout_result == LLE_SUCCESS && timeout_input) {
958 /* Timeout occurred - return the ESC key event */
959 lle_result_t convert_result =
960 convert_parsed_input_to_event(timeout_input, event);
961 lle_pool_free(timeout_input);
962 return convert_result;
963 }
964 }
965
966 event->type = LLE_INPUT_TYPE_TIMEOUT;
967 event->timestamp = lle_get_current_time_microseconds();
968 return LLE_SUCCESS;
969 }
970
971 /* Data available - read first byte */
972 unsigned char first_byte = 0;
973 ssize_t bytes_read = read(interface->terminal_fd, &first_byte, 1);
974
975 if (bytes_read == -1) {
976 if (errno == EINTR || errno == EAGAIN || errno == EWOULDBLOCK) {
977 /* Non-blocking read interrupted - treat as timeout */
978 event->type = LLE_INPUT_TYPE_TIMEOUT;
979 event->timestamp = lle_get_current_time_microseconds();
980 return LLE_SUCCESS;
981 }
982 /* Read error */
983 event->type = LLE_INPUT_TYPE_ERROR;
984 event->timestamp = lle_get_current_time_microseconds();
985 event->data.error.error_code = LLE_ERROR_SYSTEM_CALL;
986 snprintf(event->data.error.error_message,
987 sizeof(event->data.error.error_message), "read() failed: %s",
988 strerror(errno));
989 return LLE_ERROR_SYSTEM_CALL;
990 }
991
992 if (bytes_read == 0) {
993 /* EOF - stdin closed */
994 event->type = LLE_INPUT_TYPE_EOF;
995 event->timestamp = lle_get_current_time_microseconds();
996 return LLE_SUCCESS;
997 }
998
999 /* Use comprehensive sequence parser if available */
1000 if (interface->sequence_parser) {
1001 /* Check if parser is accumulating a sequence or if this is ESC/control
1002 * char IMPORTANT: Don't send standalone control chars (Ctrl-A through
1003 * Ctrl-Z, Enter, etc.) to parser unless we're already accumulating an
1004 * escape sequence. Only ESC (0x1B) should initiate parsing. */
1005 lle_parser_state_t parser_state =
1006 lle_sequence_parser_get_state(interface->sequence_parser);
1007 bool parser_accumulating = (parser_state != LLE_PARSER_STATE_NORMAL);
1008 bool should_parse = parser_accumulating || (first_byte == 0x1B);
1009
1010 if (should_parse) {
1011 /* Feed byte to comprehensive parser */
1012 lle_parsed_input_t *parsed_input = NULL;
1013 char byte_buffer[1] = {(char)first_byte};
1014
1015 /* Save parser buffer BEFORE process_data (in case it needs to be
1016 * retrieved) */
1017 const char *pre_buffer = NULL;
1018 size_t pre_buffer_len = 0;
1019 lle_sequence_parser_get_buffer(interface->sequence_parser,
1020 &pre_buffer, &pre_buffer_len);
1021
1022 /* Make a copy since parser will reset after returning a result */
1023 char saved_buffer[256];
1024 size_t saved_len = (pre_buffer_len < sizeof(saved_buffer))
1025 ? pre_buffer_len
1026 : sizeof(saved_buffer);
1027 if (pre_buffer && saved_len > 0) {
1028 memcpy(saved_buffer, pre_buffer, saved_len);
1029 }
1030
1031 lle_result_t parse_result = lle_sequence_parser_process_data(
1032 interface->sequence_parser, byte_buffer, 1, &parsed_input);
1033
1034 if (parse_result != LLE_SUCCESS) {
1035 event->type = LLE_INPUT_TYPE_ERROR;
1036 event->timestamp = lle_get_current_time_microseconds();
1037 event->data.error.error_code = parse_result;
1038 snprintf(event->data.error.error_message,
1039 sizeof(event->data.error.error_message),
1040 "Parser failed to process input");
1041 return parse_result;
1042 }
1043
1044 if (parsed_input) {
1045 /* Parser returned a complete sequence */
1046 /* Note: Parser has already reset its buffer, but we saved it
1047 * beforehand */
1048
1049 /* Add the last byte we just processed to the saved buffer */
1050 if (saved_len < sizeof(saved_buffer)) {
1051 saved_buffer[saved_len++] = first_byte;
1052 }
1053
1054 /* Try key_detector if:
1055 * 1. It's a generic SEQUENCE type, OR
1056 * 2. It's a KEY type but with unknown/zero keycode (parser
1057 * didn't identify it)
1058 */
1059 bool should_try_detector = false;
1060 if (parsed_input->type == LLE_PARSED_INPUT_TYPE_SEQUENCE) {
1061 should_try_detector = true;
1062 } else if (parsed_input->type == LLE_PARSED_INPUT_TYPE_KEY &&
1063 parsed_input->data.key_info.keycode == 0) {
1064 should_try_detector = true;
1065 }
1066
1067 if (should_try_detector && interface->key_detector &&
1068 saved_len > 0) {
1069
1070 /* Try to identify the key */
1071 lle_key_info_t *key_info = NULL;
1072 lle_result_t detect_result =
1073 lle_key_detector_process_sequence(
1074 interface->key_detector, saved_buffer, saved_len,
1075 &key_info);
1076
1077 if (detect_result == LLE_SUCCESS && key_info) {
1078 /* Successfully identified the key - update parsed_input
1079 */
1080 parsed_input->type = LLE_PARSED_INPUT_TYPE_KEY;
1081 parsed_input->data.key_info = *key_info;
1082 lle_pool_free(key_info);
1083 }
1084 }
1085
1086 /* Convert to event */
1087 lle_result_t convert_result =
1088 convert_parsed_input_to_event(parsed_input, event);
1089 lle_pool_free(parsed_input);
1090 return convert_result;
1091 }
1092
1093 /* Parser is accumulating a sequence - check for timeout first */
1094 /* If ESC key was pressed and enough time has passed, return ESC as
1095 * standalone key */
1096 lle_parsed_input_t *timeout_input = NULL;
1097 lle_result_t timeout_result = lle_sequence_parser_check_timeout(
1098 interface->sequence_parser,
1099 300000, /* 300ms timeout for ESC+key (Meta) sequences */
1100 &timeout_input);
1101
1102 if (timeout_result == LLE_SUCCESS && timeout_input) {
1103 /* Timeout occurred - return the ESC key event */
1104 lle_result_t convert_result =
1105 convert_parsed_input_to_event(timeout_input, event);
1106 lle_pool_free(timeout_input);
1107 return convert_result;
1108 }
1109
1110 /* No timeout yet - return timeout so caller will call again */
1111 event->type = LLE_INPUT_TYPE_TIMEOUT;
1112 event->timestamp = lle_get_current_time_microseconds();
1113 return LLE_SUCCESS;
1114 }
1115 /* else: regular character, parser not accumulating - fall through to
1116 * UTF-8 handling */
1117 }
1118
1119 /* Check for escape sequences (ESC = 0x1B = 27) */
1120 if (first_byte == 0x1B) {
1121 /* Read next byte with short timeout to detect escape sequences */
1122 unsigned char second_byte;
1123 fd_set read_fds;
1124 struct timeval escape_timeout;
1125
1126 FD_ZERO(&read_fds);
1127 FD_SET(interface->terminal_fd, &read_fds);
1128 escape_timeout.tv_sec = 0;
1129 escape_timeout.tv_usec =
1130 100000; /* 100ms timeout for ESC+key (Meta) sequences */
1131
1132 int ready = select(interface->terminal_fd + 1, &read_fds, NULL, NULL,
1133 &escape_timeout);
1134
1135 if (ready > 0) {
1136 ssize_t read2 = read(interface->terminal_fd, &second_byte, 1);
1137
1138 if (read2 == 1 && second_byte == '[') {
1139 /* CSI sequence - read the final byte */
1140 unsigned char final_byte;
1141 ssize_t read3 = read(interface->terminal_fd, &final_byte, 1);
1142
1143 if (read3 == 1) {
1144 /* Detect common arrow key sequences: ESC [ A/B/C/D */
1145 event->type = LLE_INPUT_TYPE_SPECIAL_KEY;
1146 event->timestamp = lle_get_current_time_microseconds();
1147 event->data.special_key.modifiers = 0;
1148
1149 switch (final_byte) {
1150 case 'A':
1151 event->data.special_key.key = LLE_KEY_UP;
1152 return LLE_SUCCESS;
1153 case 'B':
1154 event->data.special_key.key = LLE_KEY_DOWN;
1155 return LLE_SUCCESS;
1156 case 'C':
1157 event->data.special_key.key = LLE_KEY_RIGHT;
1158 return LLE_SUCCESS;
1159 case 'D':
1160 event->data.special_key.key = LLE_KEY_LEFT;
1161 return LLE_SUCCESS;
1162 case 'H':
1163 event->data.special_key.key = LLE_KEY_HOME;
1164 return LLE_SUCCESS;
1165 case 'F':
1166 event->data.special_key.key = LLE_KEY_END;
1167 return LLE_SUCCESS;
1168 case '3':
1169 /* Delete key: ESC [ 3 ~ - need to read the ~ */
1170 {
1171 unsigned char tilde;
1172 ssize_t read4 =
1173 read(interface->terminal_fd, &tilde, 1);
1174 if (read4 == 1 && tilde == '~') {
1175 event->data.special_key.key = LLE_KEY_DELETE;
1176 return LLE_SUCCESS;
1177 }
1178 }
1179 break;
1180 default:
1181 /* Unknown CSI sequence - fall through to return as
1182 * character */
1183 break;
1184 }
1185 }
1186 } else if (read2 == 1 && second_byte == 'O') {
1187 /* SS3 sequence - alternate function keys */
1188 unsigned char final_byte;
1189 ssize_t read3 = read(interface->terminal_fd, &final_byte, 1);
1190
1191 if (read3 == 1) {
1192 event->type = LLE_INPUT_TYPE_SPECIAL_KEY;
1193 event->timestamp = lle_get_current_time_microseconds();
1194 event->data.special_key.modifiers = 0;
1195
1196 switch (final_byte) {
1197 case 'H':
1198 event->data.special_key.key = LLE_KEY_HOME;
1199 return LLE_SUCCESS;
1200 case 'F':
1201 event->data.special_key.key = LLE_KEY_END;
1202 return LLE_SUCCESS;
1203 default:
1204 /* Unknown SS3 sequence */
1205 break;
1206 }
1207 }
1208 } else if (read2 == 1 && second_byte >= 0x20 &&
1209 second_byte < 0x7F) {
1210 /* ESC + printable ASCII character = Meta/Alt + character
1211 * This is how macOS Terminal and other terminals send Alt+key
1212 * when the Option key is configured as Meta, or when user
1213 * physically presses ESC then a letter (e.g., ESC f for Alt-f).
1214 *
1215 * Range 0x20-0x7E covers printable ASCII (space through tilde).
1216 * This enables M-f (forward-word), M-b (backward-word), etc.
1217 */
1218 event->type = LLE_INPUT_TYPE_SPECIAL_KEY;
1219 event->timestamp = lle_get_current_time_microseconds();
1220 event->data.special_key.key = LLE_KEY_UNKNOWN;
1221 event->data.special_key.modifiers = LLE_MOD_ALT;
1222 event->data.special_key.keycode = second_byte;
1223 return LLE_SUCCESS;
1224 }
1225 }
1226
1227 /* If we get here, it's just a plain ESC key (no second byte within
1228 * timeout) or an unrecognized sequence - return ESC as a regular
1229 * character */
1230 }
1231
1232 /* Decode UTF-8 character */
1233 uint32_t codepoint;
1234 char utf8_bytes[8] = {0};
1235 uint8_t byte_count;
1236
1237 lle_result_t decode_result =
1238 decode_utf8(interface, first_byte, &codepoint, utf8_bytes, &byte_count);
1239 if (decode_result != LLE_SUCCESS) {
1240 event->type = LLE_INPUT_TYPE_ERROR;
1241 event->timestamp = lle_get_current_time_microseconds();
1242 event->data.error.error_code = decode_result;
1243 snprintf(event->data.error.error_message,
1244 sizeof(event->data.error.error_message),
1245 "UTF-8 decoding failed");
1246 return decode_result;
1247 }
1248
1249 /* Check if this is a Ctrl+letter combination (0x01-0x1A = Ctrl-A through
1250 * Ctrl-Z) According to spec, these should be SPECIAL_KEY events with
1251 * keycode and modifiers EXCEPT for special control characters that have
1252 * their own meaning:
1253 * - 0x09 (Tab / Ctrl-I)
1254 * - 0x0A (Newline / Ctrl-J)
1255 * - 0x0D (Enter / Ctrl-M)
1256 */
1257 if (codepoint >= 0x01 && codepoint <= 0x1A && codepoint != 0x09 &&
1258 codepoint != 0x0A && codepoint != 0x0D) {
1259 /* Ctrl+letter: Convert to SPECIAL_KEY event with keycode and
1260 * LLE_MOD_CTRL */
1261 event->type = LLE_INPUT_TYPE_SPECIAL_KEY;
1262 event->timestamp = lle_get_current_time_microseconds();
1263 event->data.special_key.key =
1264 LLE_KEY_UNKNOWN; /* Not a special key like arrow/F-key */
1265 event->data.special_key.modifiers = LLE_MOD_CTRL;
1266 event->data.special_key.keycode =
1267 codepoint + 0x40; /* 0x01->0x41='A', 0x02->0x42='B', etc. */
1268 return LLE_SUCCESS;
1269 }
1270
1271 /* Populate character event */
1272 event->type = LLE_INPUT_TYPE_CHARACTER;
1273 event->timestamp = lle_get_current_time_microseconds();
1274 event->data.character.codepoint = codepoint;
1275 memcpy(event->data.character.utf8_bytes, utf8_bytes, byte_count);
1276 event->data.character.byte_count = byte_count;
1277
1278 return LLE_SUCCESS;
1279}
