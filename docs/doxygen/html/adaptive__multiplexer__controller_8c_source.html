93/**
94 * Detect multiplexer type from environment.
95 */
96static lle_multiplexer_type_t lle_detect_multiplexer_type(void) {
97 /* Check for tmux */
98 const char *tmux_env = getenv("TMUX");
99 if (tmux_env && strlen(tmux_env) > 0) {
100 return LLE_MUX_TYPE_TMUX;
101 }
102
103 /* Check for screen */
104 const char *sty_env = getenv("STY");
105 if (sty_env && strlen(sty_env) > 0) {
106 return LLE_MUX_TYPE_SCREEN;
107 }
108
109 /* Check TERM variable patterns */
110 const char *term = getenv("TERM");
111 if (term) {
112 if (strstr(term, "tmux") != NULL) {
113 return LLE_MUX_TYPE_TMUX;
114 }
115 if (strstr(term, "screen") != NULL) {
116 return LLE_MUX_TYPE_SCREEN;
117 }
118 }
119
120 return LLE_MUX_TYPE_NONE;
121}

123/**
124 * Configure multiplexer-specific capabilities.
125 */
126static void lle_configure_multiplexer_capabilities(
127 lle_multiplexer_controller_t *mux,
128 const lle_terminal_detection_result_t *detection) {
129 (void)detection; /* Reserved for detection-based capability tuning */
130
131 switch (mux->multiplexer_type) {
132 case LLE_MUX_TYPE_TMUX:
133 /* tmux supports DCS passthrough */
134 mux->supports_passthrough = true;
135 mux->supports_focus_events = true;
136 mux->requires_escape_doubling = false;
137 break;
138
139 case LLE_MUX_TYPE_SCREEN:
140 /* screen has limited passthrough */
141 mux->supports_passthrough = false;
142 mux->supports_focus_events = false;
143 mux->requires_escape_doubling = true;
144 break;
145
146 case LLE_MUX_TYPE_OTHER:
147 /* Unknown multiplexer - conservative settings */
148 mux->supports_passthrough = false;
149 mux->supports_focus_events = false;
150 mux->requires_escape_doubling = false;
151 break;
152
153 default:
154 mux->supports_passthrough = false;
155 mux->supports_focus_events = false;
156 mux->requires_escape_doubling = false;
157 break;
158 }
159}

166/**
167 * Create multiplexer adapter.
168 */
169lle_result_t lle_multiplexer_adapter_create(lle_multiplexer_adapter_t **adapter,
170 lle_multiplexer_type_t type) {
171
172 lle_multiplexer_adapter_t *adapt =
173 calloc(1, sizeof(lle_multiplexer_adapter_t));
174 if (!adapt) {
175 return LLE_ERROR_OUT_OF_MEMORY;
176 }
177
178 adapt->type = type;
179
180 /* Configure passthrough based on multiplexer type */
181 switch (type) {
182 case LLE_MUX_TYPE_TMUX:
183 /* tmux DCS passthrough: \ePtmux;\e<sequence>\e\\ */
184 adapt->passthrough_prefix = "\x1bPtmux;\x1b";
185 adapt->passthrough_suffix = "\x1b\\";
186 adapt->needs_escape_doubling = true;
187 adapt->supports_focus_events = true;
188 adapt->supports_true_mouse = true;
189 adapt->supports_clipboard = true;
190 break;
191
192 case LLE_MUX_TYPE_SCREEN:
193 /* screen DCS passthrough: \eP\e<sequence>\e\\ */
194 adapt->passthrough_prefix = "\x1bP\x1b";
195 adapt->passthrough_suffix = "\x1b\\";
196 adapt->needs_escape_doubling = true;
197 adapt->supports_focus_events = false;
198 adapt->supports_true_mouse = false;
199 adapt->supports_clipboard = false;
200 break;
201
202 default:
203 adapt->passthrough_prefix = NULL;
204 adapt->passthrough_suffix = NULL;
205 adapt->needs_escape_doubling = false;
206 adapt->supports_focus_events = false;
207 adapt->supports_true_mouse = false;
208 adapt->supports_clipboard = false;
209 break;
210 }
211
212 *adapter = adapt;
213 return LLE_SUCCESS;
214}

224/**
225 * Wrap escape sequence for multiplexer passthrough.
226 */
227static lle_result_t
228lle_multiplexer_adapter_wrap_sequence(lle_multiplexer_adapter_t *adapter,
229 const char *sequence, size_t seq_len,
230 char **wrapped, size_t *wrapped_len) {
231
232 if (!adapter->passthrough_prefix || !adapter->passthrough_suffix) {
233 /* No wrapping needed */
234 *wrapped = strdup(sequence);
235 if (!*wrapped) {
236 return LLE_ERROR_OUT_OF_MEMORY;
237 }
238 *wrapped_len = seq_len;
239 return LLE_SUCCESS;
240 }
241
242 size_t prefix_len = strlen(adapter->passthrough_prefix);
243 size_t suffix_len = strlen(adapter->passthrough_suffix);
244
245 /* Calculate wrapped size (account for escape doubling) */
246 size_t doubled_len = seq_len;
247 if (adapter->needs_escape_doubling) {
248 /* Count escapes in sequence */
249 size_t escape_count = 0;
250 for (size_t i = 0; i < seq_len; i++) {
251 if (sequence[i] == '\x1b') {
252 escape_count++;
253 }
254 }
255 doubled_len += escape_count;
256 }
257
258 size_t total_len = prefix_len + doubled_len + suffix_len;
259 char *result = malloc(total_len + 1);
260 if (!result) {
261 return LLE_ERROR_OUT_OF_MEMORY;
262 }
263
264 /* Build wrapped sequence */
265 size_t pos = 0;
266
267 /* Add prefix */
268 memcpy(result + pos, adapter->passthrough_prefix, prefix_len);
269 pos += prefix_len;
270
271 /* Add sequence (with escape doubling if needed) */
272 if (adapter->needs_escape_doubling) {
273 for (size_t i = 0; i < seq_len; i++) {
274 result[pos++] = sequence[i];
275 if (sequence[i] == '\x1b') {
276 result[pos++] = '\x1b'; /* Double the escape */
277 }
278 }
279 } else {
280 memcpy(result + pos, sequence, seq_len);
281 pos += seq_len;
282 }
283
284 /* Add suffix */
285 memcpy(result + pos, adapter->passthrough_suffix, suffix_len);
286 pos += suffix_len;
287
288 result[pos] = '\0';
289
290 *wrapped = result;
291 *wrapped_len = pos;
292
293 adapter->sequences_wrapped++;
294 adapter->passthrough_operations++;
295
296 return LLE_SUCCESS;
297}

299/**
300 * Check if sequence needs wrapping.
301 */
302static bool
303lle_multiplexer_adapter_needs_wrapping(lle_multiplexer_adapter_t *adapter,
304 const char *sequence) {
305
306 if (!adapter->passthrough_prefix) {
307 return false;
308 }
309
310 /* Check for sequences that need passthrough */
311 /* OSC sequences (clipboard, window title, etc.) */
312 if (strstr(sequence, "\x1b]") != NULL) {
313 return true;
314 }
315
316 /* Device control strings */
317 if (strstr(sequence, "\x1bP") != NULL) {
318 return true;
319 }
320
321 return false;
322}

329/**
330 * Initialize multiplexer controller.
331 */
332lle_result_t
333lle_initialize_multiplexer_controller(lle_adaptive_context_t *context,
334 lusush_memory_pool_t *memory_pool) {
335
336 lle_multiplexer_controller_t *mux =
337 calloc(1, sizeof(lle_multiplexer_controller_t));
338 if (!mux) {
339 return LLE_ERROR_OUT_OF_MEMORY;
340 }
341
342 /* Detect multiplexer type */
343 mux->multiplexer_type = lle_detect_multiplexer_type();
344
345 /* Configure multiplexer-specific capabilities */
346 lle_configure_multiplexer_capabilities(mux, context->detection_result);
347
348 /* Initialize base native controller */
349 lle_result_t result =
350 lle_initialize_native_controller(context, memory_pool);
351 if (result != LLE_SUCCESS) {
352 free(mux);
353 return result;
354 }
355
356 /* Save base controller reference */
357 mux->base_controller = context->controller.native;
358
359 /* Create multiplexer adapter */
360 result =
361 lle_multiplexer_adapter_create(&mux->adapter, mux->multiplexer_type);
362 if (result != LLE_SUCCESS) {
363 lle_cleanup_native_controller(mux->base_controller);
364 free(mux);
365 return result;
366 }
367
368 mux->memory_pool = memory_pool;
369
370 /* Update context to use multiplexer controller */
371 context->controller.mux = mux;
372
373 return LLE_SUCCESS;
374}

409/**
410 * Send adapted sequence through multiplexer.
411 */
412lle_result_t lle_multiplexer_send_sequence(lle_multiplexer_controller_t *mux,
413 const char *sequence,
414 size_t length) {
415
416 if (!mux || !sequence) {
417 return LLE_ERROR_INVALID_PARAMETER;
418 }
419
420 /* Check if sequence needs wrapping */
421 if (!lle_multiplexer_adapter_needs_wrapping(mux->adapter, sequence)) {
422 /* Send directly */
423 ssize_t written = write(STDOUT_FILENO, sequence, length);
424 if (written < 0 || (size_t)written != length) {
425 return LLE_ERROR_TERMINAL_ABSTRACTION;
426 }
427 return LLE_SUCCESS;
428 }
429
430 /* Wrap sequence for multiplexer passthrough */
431 char *wrapped = NULL;
432 size_t wrapped_len = 0;
433 lle_result_t result = lle_multiplexer_adapter_wrap_sequence(
434 mux->adapter, sequence, length, &wrapped, &wrapped_len);
435
436 if (result != LLE_SUCCESS) {
437 return result;
438 }
439
440 /* Send wrapped sequence */
441 ssize_t written = write(STDOUT_FILENO, wrapped, wrapped_len);
442 free(wrapped);
443
444 if (written < 0 || (size_t)written != wrapped_len) {
445 return LLE_ERROR_TERMINAL_ABSTRACTION;
446 }
447
448 mux->adapted_sequences++;
449 return LLE_SUCCESS;
450}
