26 * LAYER EVENTS SYSTEM (Layer Communication Protocol)
27 *
28 * This implementation provides event-driven communication between display
29 * layers without creating direct dependencies. It uses a publisher/subscriber
30 * pattern with priority-based event processing and performance optimization.
31 *
32 * Key Responsibilities:
33 * - Event queue management and processing
34 * - Publisher/subscriber registration and notification
35 * - Priority-based event ordering and delivery
36 * - Memory-safe event data management
37 * - Performance monitoring and optimization
38 * - Event debugging and diagnostics
39 *
40 * Design Principles:
41 * - Zero direct layer dependencies
42 * - High-performance event processing
43 * - Type-safe event handling
44 * - Comprehensive error handling
45 * - Memory-efficient operations
46 * - Thread-safe design (future consideration)
47 */

164/**
165 * Create a new layer events system
166 */
167layer_event_system_t *layer_events_create(const layer_events_config_t *config) {
168 layer_event_system_t *events = calloc(1, sizeof(layer_event_system_t));
169 if (!events) {
170 return NULL;
171 }
172
173 events->magic = EVENT_SYSTEM_MAGIC;
174 events->initialized = false;
175
176 // Set configuration (use defaults if config is NULL)
177 if (config) {
178 events->config = *config;
179 } else {
180 events->config = layer_events_create_default_config();
181 }
182
183 // Initialize queues
184 for (int i = 0; i < 4; i++) {
185 events->queues[i].head = NULL;
186 events->queues[i].tail = NULL;
187 events->queues[i].size = 0;
188 events->queues[i].max_size = events->config.max_queue_size;
189 }
190
191 // Initialize subscriber list
192 events->subscribers = NULL;
193 events->subscriber_count = 0;
194
195 // Initialize counters
196 events->next_sequence_number = 1;
197 events->next_event_id = 1;
198
199 // Initialize statistics
200 memset(&events->stats, 0, sizeof(layer_event_stats_t));
201
202 events->debug_enabled = events->config.enable_debugging;
203 events->last_error = LAYER_EVENTS_SUCCESS;
204
205 return events;
206}

208/**
209 * Initialize the layer events system

211layer_events_error_t layer_events_init(layer_event_system_t *events) {
212 if (!validate_event_system(events)) {
213 return LAYER_EVENTS_ERROR_INVALID_PARAM;
214 }
215
216 if (events->initialized) {
217 return LAYER_EVENTS_SUCCESS; // Already initialized

220 debug_log(events, "Initializing layer events system");
221
222 // Validate configuration
223 if (events->config.max_queue_size == 0 ||
224 events->config.max_subscribers == 0) {
225 events->last_error = LAYER_EVENTS_ERROR_INVALID_PARAM;
226 return LAYER_EVENTS_ERROR_INVALID_PARAM;
227 }
228
229 events->initialized = true;
230 layer_events_global_initialized = true;
231
232 debug_log(events, "Layer events system initialized successfully");
233
234 return LAYER_EVENTS_SUCCESS;
235}

237/**
238 * Clean up event system resources
239 */
240layer_events_error_t layer_events_cleanup(layer_event_system_t *events) {
241 if (!validate_event_system(events)) {
242 return LAYER_EVENTS_ERROR_INVALID_PARAM;
243 }
244
245 debug_log(events, "Cleaning up layer events system");
246
247 // Clear all event queues
248 for (int priority = 0; priority < 4; priority++) {
249 event_queue_t *queue = &events->queues[priority];
250 event_queue_entry_t *entry = queue->head;
251
252 while (entry) {
253 event_queue_entry_t *next = entry->next;
254 cleanup_event_data(&entry->event);
255 destroy_queue_entry(entry);
256 entry = next;
257 }
258
259 queue->head = NULL;
260 queue->tail = NULL;
261 queue->size = 0;
262 }
263
264 // Clear subscriber list
265 subscriber_entry_t *subscriber = events->subscribers;
266 while (subscriber) {
267 subscriber_entry_t *next = subscriber->next;
268 free(subscriber);
269 subscriber = next;
270 }
271 events->subscribers = NULL;
272 events->subscriber_count = 0;
273
274 events->initialized = false;
275
276 debug_log(events, "Layer events system cleanup completed");
277
278 return LAYER_EVENTS_SUCCESS;
279}

301/**
302 * Publish an event to the system
303 */
304layer_events_error_t layer_events_publish(layer_event_system_t *events,
305 const layer_event_t *event) {
306 if (!validate_event_system(events) || !event) {
307 if (events)
308 events->last_error = LAYER_EVENTS_ERROR_INVALID_PARAM;
309 return LAYER_EVENTS_ERROR_INVALID_PARAM;
310 }
311
312 if (!events->initialized) {
313 events->last_error = LAYER_EVENTS_ERROR_NOT_INITIALIZED;
314 return LAYER_EVENTS_ERROR_NOT_INITIALIZED;
315 }
316
317 if (!layer_events_validate_event(event)) {
318 events->last_error = LAYER_EVENTS_ERROR_INVALID_PARAM;
319 return LAYER_EVENTS_ERROR_INVALID_PARAM;
320 }
321
322 debug_log(events, "Publishing event type %d from layer %d", event->type,
323 event->source_layer);
324
325 events->stats.events_published++;
326
327 // Create a copy of the event with system-generated fields
328 layer_event_t event_copy = *event;
329 event_copy.timestamp = get_timestamp_ns();
330 event_copy.sequence_number = events->next_sequence_number++;
331 event_copy.event_id = events->next_event_id++;
332 event_copy.processing_attempts = 0;
333 event_copy.last_processing_time = 0;
334 event_copy.processed = false;
335
336 // Handle immediate processing for NO_QUEUE flag
337 if (event->flags & LAYER_EVENT_FLAG_NO_QUEUE) {
338 layer_events_error_t result = notify_subscribers(events, &event_copy);
339 if (result == LAYER_EVENTS_SUCCESS) {
340 events->stats.events_processed++;
341 } else {
342 events->stats.events_failed++;
343 }
344 return result;
345 }
346
347 // Queue the event for later processing
348 layer_events_error_t result = enqueue_event(events, &event_copy);
349 if (result != LAYER_EVENTS_SUCCESS) {
350 events->last_error = result;
351 if (result == LAYER_EVENTS_ERROR_QUEUE_FULL) {
352 events->stats.events_dropped++;
353 events->stats.queue_overflows++;
354 }
355 }
356
357 return result;
358}

378/**
379 * Publish a content changed event
380 */
381layer_events_error_t layer_events_publish_content_changed(
382 layer_event_system_t *events, layer_id_t source_layer, const char *content,
383 size_t content_length, bool needs_reflow) {
384 layer_event_t event = {0};
385 event.type = LAYER_EVENT_CONTENT_CHANGED;
386 event.source_layer = source_layer;
387 event.target_layer = 0; // Broadcast
388 event.priority = LAYER_EVENT_PRIORITY_NORMAL;
389 event.flags = LAYER_EVENT_FLAG_BROADCAST;
390
391 // Create a copy of the content string to avoid dangling pointers
392 char *content_copy = NULL;
393 if (content && content_length > 0) {
394 content_copy = malloc(content_length + 1);
395 if (content_copy) {
396 memcpy(content_copy, content, content_length);
397 content_copy[content_length] = '\0';
398 }
399 }
400
401 event.data.content_changed.new_content = content_copy;
402 event.data.content_changed.content_length = content_length;
403 event.data.content_changed.needs_reflow = needs_reflow;
404
405 return layer_events_publish(events, &event);
406}

433/**
434 * Subscribe to events of a specific type
435 */
436layer_events_error_t
437layer_events_subscribe(layer_event_system_t *events,
438 layer_event_type_t event_type, layer_id_t subscriber_id,
439 layer_event_callback_t callback, void *user_data,
440 layer_event_priority_t min_priority) {
441 if (!validate_event_system(events) || !callback) {
442 if (events)
443 events->last_error = LAYER_EVENTS_ERROR_INVALID_PARAM;
444 return LAYER_EVENTS_ERROR_INVALID_PARAM;
445 }
446
447 if (!events->initialized) {
448 events->last_error = LAYER_EVENTS_ERROR_NOT_INITIALIZED;
449 return LAYER_EVENTS_ERROR_NOT_INITIALIZED;
450 }
451
452 if (events->subscriber_count >= events->config.max_subscribers) {
453 events->last_error = LAYER_EVENTS_ERROR_MAX_SUBSCRIBERS;
454 return LAYER_EVENTS_ERROR_MAX_SUBSCRIBERS;
455 }
456
457 // Check if already subscribed
458 if (find_subscriber(events, event_type, subscriber_id)) {
459 return LAYER_EVENTS_SUCCESS; // Already subscribed
460 }
461
462 debug_log(events, "Subscribing layer %d to event type %d", subscriber_id,
463 event_type);
464
465 // Create new subscriber entry
466 subscriber_entry_t *entry = malloc(sizeof(subscriber_entry_t));
467 if (!entry) {
468 events->last_error = LAYER_EVENTS_ERROR_MEMORY_ALLOCATION;
469 return LAYER_EVENTS_ERROR_MEMORY_ALLOCATION;
470 }
471
472 entry->subscriber.event_type = event_type;
473 entry->subscriber.subscriber_id = subscriber_id;
474 entry->subscriber.callback = callback;
475 entry->subscriber.user_data = user_data;
476 entry->subscriber.min_priority = min_priority;
477 entry->subscriber.active = true;
478 entry->subscriber.events_received = 0;
479 entry->subscriber.events_processed = 0;
480 entry->subscriber.last_event_time = 0;
481
482 // Add to subscriber list
483 entry->next = events->subscribers;
484 events->subscribers = entry;
485 events->subscriber_count++;
486 events->stats.active_subscribers = events->subscriber_count;
487
488 return LAYER_EVENTS_SUCCESS;
489}

491/**
492 * Unsubscribe from events
493 */
494layer_events_error_t layer_events_unsubscribe(layer_event_system_t *events,
495 layer_event_type_t event_type,
496 layer_id_t subscriber_id) {
497 if (!validate_event_system(events)) {
498 return LAYER_EVENTS_ERROR_INVALID_PARAM;
499 }
500
501 if (!events->initialized) {
502 events->last_error = LAYER_EVENTS_ERROR_NOT_INITIALIZED;
503 return LAYER_EVENTS_ERROR_NOT_INITIALIZED;
504 }
505
506 debug_log(events, "Unsubscribing layer %d from event type %d",
507 subscriber_id, event_type);
508
509 subscriber_entry_t **current = &events->subscribers;
510 while (*current) {
511 if ((*current)->subscriber.event_type == event_type &&
512 (*current)->subscriber.subscriber_id == subscriber_id) {
513
514 subscriber_entry_t *to_remove = *current;
515 *current = (*current)->next;
516 free(to_remove);
517 events->subscriber_count--;
518 events->stats.active_subscribers = events->subscriber_count;
519 return LAYER_EVENTS_SUCCESS;
520 }
521 current = &(*current)->next;
522 }
523
524 events->last_error = LAYER_EVENTS_ERROR_SUBSCRIBER_NOT_FOUND;
525 return LAYER_EVENTS_ERROR_SUBSCRIBER_NOT_FOUND;
526}

568/**
569 * Process pending events
570 */
571int layer_events_process_pending(layer_event_system_t *events,
572 uint32_t max_events, uint32_t timeout_ms) {
573 if (!validate_event_system(events)) {
574 return -1;
575 }
576
577 if (!events->initialized) {
578 events->last_error = LAYER_EVENTS_ERROR_NOT_INITIALIZED;
579 return -1;
580 }
581
582 uint64_t start_time = get_timestamp_ns();
583 uint64_t timeout_ns = timeout_ms * 1000000ULL;
584 int processed_count = 0;
585
586 // Process events in priority order: Critical, High, Normal, Low
587 for (int priority = LAYER_EVENT_PRIORITY_CRITICAL;
588 priority >= LAYER_EVENT_PRIORITY_LOW; priority--) {
589
590 while (true) {
591 // Check timeout
592 if (timeout_ms > 0) {
593 uint64_t elapsed = get_timestamp_ns() - start_time;
594 if (elapsed >= timeout_ns) {
595 return processed_count;
596 }
597 }
598
599 // Check max events limit
600 if (max_events > 0 && processed_count >= (int)max_events) {
601 return processed_count;
602 }
603
604 // Get next event from this priority queue
605 layer_event_t *event =
606 dequeue_event(events, (layer_event_priority_t)priority);
607 if (!event) {
608 break; // No more events at this priority
609 }
610
611 // Process the event
612 uint64_t processing_start = get_timestamp_ns();
613 layer_events_error_t result = notify_subscribers(events, event);
614 uint64_t processing_time = get_timestamp_ns() - processing_start;
615
616 events->stats.total_processing_time_ns += processing_time;
617
618 if (result == LAYER_EVENTS_SUCCESS) {
619 events->stats.events_processed++;
620 processed_count++;
621 } else {
622 events->stats.events_failed++;
623 }
624
625 // Check for performance warnings
626 if (events->config.enable_performance_monitoring) {
627 uint64_t processing_time_ms = processing_time / 1000000;
628 if (processing_time_ms >
629 PERFORMANCE_WARNING_PROCESSING_TIME_MS) {
630 debug_log(
631 events,
632 "Performance warning: Event processing took %lu ms",
633 processing_time_ms);
634 }
635 }
636
637 cleanup_event_data(event);
638 free(event);
639 }
640 }
641
642 return processed_count;
643}

645/**
646 * Process events of specific priority
647 */
648int layer_events_process_priority(layer_event_system_t *events,
649 layer_event_priority_t priority,
650 uint32_t max_events) {
651 if (!validate_event_system(events)) {
652 return -1;
653 }
654
655 if (!events->initialized) {
656 events->last_error = LAYER_EVENTS_ERROR_NOT_INITIALIZED;
657 return -1;
658 }
659
660 int processed_count = 0;
661
662 while (max_events == 0 || processed_count < (int)max_events) {
663 layer_event_t *event = dequeue_event(events, priority);
664 if (!event) {
665 break; // No more events at this priority
666 }
667
668 layer_events_error_t result = notify_subscribers(events, event);
669
670 if (result == LAYER_EVENTS_SUCCESS) {
671 events->stats.events_processed++;
672 processed_count++;
673 } else {
674 events->stats.events_failed++;
675 }
676
677 cleanup_event_data(event);
678 free(event);
679 }
680
681 return processed_count;
682}

948/**
949 * Enqueue event
950 */
951static layer_events_error_t enqueue_event(layer_event_system_t *events,
952 const layer_event_t *event) {
953 event_queue_t *queue = &events->queues[event->priority];
954
955 if (queue->size >= queue->max_size) {
956 return LAYER_EVENTS_ERROR_QUEUE_FULL;
957 }
958
959 event_queue_entry_t *entry = create_queue_entry(event);
960 if (!entry) {
961 return LAYER_EVENTS_ERROR_MEMORY_ALLOCATION;
962 }
963
964 // Add to tail of queue
965 if (queue->tail) {
966 queue->tail->next = entry;
967 entry->prev = queue->tail;
968 queue->tail = entry;
969 } else {
970 queue->head = queue->tail = entry;
971 }
972
973 queue->size++;
974 events->stats.current_queue_size++;
975
976 if (queue->size > events->stats.max_queue_size_reached) {
977 events->stats.max_queue_size_reached = queue->size;
978 }
979
980 return LAYER_EVENTS_SUCCESS;
981}

983/**
984 * Dequeue event
985 */
986static layer_event_t *dequeue_event(layer_event_system_t *events,
987 layer_event_priority_t priority) {
988 event_queue_t *queue = &events->queues[priority];
989
990 if (queue->size == 0) {
991 return NULL;
992 }
993
994 event_queue_entry_t *entry = queue->head;
995 layer_event_t *event = malloc(sizeof(layer_event_t));
996 if (!event) {
997 return NULL;
998 }
999
1000 *event = entry->event;
1001
1002 // Remove from head of queue
1003 queue->head = entry->next;
1004 if (queue->head) {
1005 queue->head->prev = NULL;
1006 } else {
1007 queue->tail = NULL;
1008 }
1009
1010 queue->size--;
1011 events->stats.current_queue_size--;
1012
1013 free(entry);
1014 return event;
1015}

1017/**
1018 * Notify subscribers
1019 */
1020static layer_events_error_t notify_subscribers(layer_event_system_t *events,
1021 const layer_event_t *event) {
1022 subscriber_entry_t *subscriber = events->subscribers;
1023 bool found_subscriber = false;
1024 layer_events_error_t last_error = LAYER_EVENTS_SUCCESS;
1025
1026 while (subscriber) {
1027 if (subscriber->subscriber.event_type == event->type &&
1028 subscriber->subscriber.active &&
1029 event->priority >= subscriber->subscriber.min_priority) {
1030
1031 // Check if event is targeted to this subscriber
1032 if (event->target_layer != 0 &&
1033 event->target_layer != subscriber->subscriber.subscriber_id) {
1034 subscriber = subscriber->next;
1035 continue;
1036 }
1037
1038 found_subscriber = true;
1039 subscriber->subscriber.events_received++;
1040 subscriber->subscriber.last_event_time = event->timestamp;
1041
1042 debug_log(events, "Notifying subscriber %d for event %d",
1043 subscriber->subscriber.subscriber_id, event->type);
1044
1045 layer_events_error_t result = subscriber->subscriber.callback(
1046 event, subscriber->subscriber.user_data);
1047
1048 if (result == LAYER_EVENTS_SUCCESS) {
1049 subscriber->subscriber.events_processed++;
1050 } else {
1051 last_error = result;
1052 debug_log(events, "Subscriber callback failed with error: %s",
1053 layer_events_error_string(result));
1054 }
1055 }
1056
1057 subscriber = subscriber->next;
1058 }
1059
1060 return found_subscriber ? last_error : LAYER_EVENTS_SUCCESS;
1061}
