105typedef struct cmd_cache_entry {

115static unsigned int hash_string(const char *str) {
116 unsigned int hash = 5381;
117 int c;
118 while ((c = *str++)) {
119 hash = ((hash << 5) + hash) + c;
120 }
121 return hash % CMD_CACHE_SIZE;
122}

129 "if", "then", "else", "elif", "fi", "for", "in", "do",

137static bool is_shell_keyword(const char *word, size_t len) {
138 for (int i = 0; shell_keywords[i]; i++) {
139 if (strlen(shell_keywords[i]) == len &&
140 strncmp(word, shell_keywords[i], len) == 0) {
141 return true;
142 }
143 }
144 return false;
145}

148 for (int i = 0; block_ending_keywords[i]; i++) {
149 if (strlen(block_ending_keywords[i]) == len &&
150 strncmp(word, block_ending_keywords[i], len) == 0) {
151 return true;
152 }
153 }
154 return false;
155}

161static bool is_word_char(char c) {
162 unsigned char uc = (unsigned char)c;
163
164 /* UTF-8 continuation bytes (10xxxxxx) and lead bytes (11xxxxxx) are part of
165 * words. This ensures multi-byte UTF-8 characters like 'Ã©' (0xC3 0xA9) are
166 * not split. */
167 if (uc >= 0x80) {
168 return true; /* Any non-ASCII byte is part of a word */
169 }
170
171 return isalnum(uc) || c == '_' || c == '-' || c == '.' || c == '/' ||
172 c == '~' || c == '+' || c == '@' || c == ':' || c == '=';
173}

175static bool is_option_start(const char *s, size_t remaining) {
176 return remaining >= 2 && s[0] == '-' &&
177 (isalnum((unsigned char)s[1]) || s[1] == '-');
178}

189/**
190 * Check if a word is a variable assignment (VAR=value pattern)
191 * Valid variable names start with letter or underscore, followed by
192 * alphanumeric or underscore, then '='
193 */
194static bool is_assignment(const char *word, size_t len) {
195 if (len < 2)
196 return false; /* Minimum: "x=" */
197
198 /* Find the '=' */
199 const char *eq = memchr(word, '=', len);
200 if (!eq)
201 return false;
202
203 /* Variable name must be before the '=' */
204 size_t name_len = eq - word;
205 if (name_len == 0)
206 return false;
207
208 /* First char must be letter or underscore */
209 if (!isalpha((unsigned char)word[0]) && word[0] != '_') {
210 return false;
211 }
212
213 /* Rest must be alphanumeric or underscore */
214 for (size_t i = 1; i < name_len; i++) {
215 if (!isalnum((unsigned char)word[i]) && word[i] != '_') {
216 return false;
217 }
218 }
219
220 return true;
221}

223static size_t skip_whitespace(const char *input, size_t pos, size_t len) {
224 while (pos < len && isspace((unsigned char)input[pos])) {
225 pos++;
226 }

237static bool command_exists_in_path(const char *command) {
238 /* Check if command contains a path separator */
239 if (strchr(command, '/')) {
240 return access(command, X_OK) == 0;
241 }
242
243 const char *path_env = getenv("PATH");
244 if (!path_env)
245 return false;
246
247 char *path_copy = strdup(path_env);
248 if (!path_copy)
249 return false;
250
251 bool found = false;
252 char *dir = strtok(path_copy, ":");
253
254 while (dir && !found) {
255 char full_path[4096];
256 snprintf(full_path, sizeof(full_path), "%s/%s", dir, command);
257 if (access(full_path, X_OK) == 0) {
258 found = true;
259 }
260 dir = strtok(NULL, ":");
261 }
262
263 free(path_copy);
264 return found;
265}

267lle_syntax_token_type_t
268lle_syntax_check_command(lle_syntax_highlighter_t *highlighter,
269 const char *command) {
270 if (!command || !*command) {
271 return LLE_TOKEN_UNKNOWN;
272 }
273
274 /* Check cache first */
275 if (highlighter && highlighter->command_cache) {
276 cmd_cache_t *cache = (cmd_cache_t *)highlighter->command_cache;
277 unsigned int idx = hash_string(command);
278 cmd_cache_entry_t *entry = &cache->entries[idx];
279
280 if (entry->command && strcmp(entry->command, command) == 0) {
281 time_t now = time(NULL);
282 if (now - entry->timestamp < CMD_CACHE_TTL) {
283 return entry->type;
284 }
285 }
286 }
287
288 /* Check command type */
289 lle_syntax_token_type_t type;
290
291 if (is_builtin(command)) {
292 type = LLE_TOKEN_COMMAND_BUILTIN;
293 } else if (lookup_alias(command) != NULL) {
294 type = LLE_TOKEN_COMMAND_ALIAS;
295 } else if (command_exists_in_path(command)) {
296 type = LLE_TOKEN_COMMAND_VALID;
297 } else {
298 type = LLE_TOKEN_COMMAND_INVALID;
299 }
300
301 /* Update cache */
302 if (highlighter && highlighter->command_cache) {
303 cmd_cache_t *cache = (cmd_cache_t *)highlighter->command_cache;
304 unsigned int idx = hash_string(command);
305 cmd_cache_entry_t *entry = &cache->entries[idx];
306
307 free(entry->command);
308 entry->command = strdup(command);
309 entry->type = type;
310 entry->timestamp = time(NULL);
311 }
312
313 return type;
314}

318 return stat(path, &st) == 0;
319}

325static int ensure_token_capacity(lle_syntax_highlighter_t *h, size_t needed) {
326 if (needed <= h->token_capacity)
327 return 0;
328
329 size_t new_cap = h->token_capacity ? h->token_capacity * 2 : 32;
330 while (new_cap < needed)
331 new_cap *= 2;
332
333 lle_syntax_token_t *new_tokens =
334 realloc(h->tokens, new_cap * sizeof(lle_syntax_token_t));
335 if (!new_tokens)
336 return -1;
337
338 h->tokens = new_tokens;
339 h->token_capacity = new_cap;
340 return 0;
341}

343static int add_token(lle_syntax_highlighter_t *h, lle_syntax_token_type_t type,
344 size_t start, size_t end) {
345 if (ensure_token_capacity(h, h->token_count + 1) < 0)
346 return -1;
347
348 lle_syntax_token_t *tok = &h->tokens[h->token_count++];
349 tok->type = type;
350 tok->start = start;
351 tok->end = end;
352 tok->color = 0;
353 tok->attributes = LLE_ATTR_NONE;
354
355 return 0;
356}

358int lle_syntax_highlight(lle_syntax_highlighter_t *highlighter,
359 const char *input, size_t input_len) {
360 if (!highlighter || !input)
361 return -1;
362
363 highlighter->token_count = 0;
364
365 size_t pos = 0;
366 bool expect_command = true; /* Next word is a command */
367
368 while (pos < input_len) {
369 /* Skip whitespace */
370 size_t ws_start = pos;
371 pos = skip_whitespace(input, pos, input_len);
372 if (pos > ws_start) {
373 add_token(highlighter, LLE_TOKEN_WHITESPACE, ws_start, pos);
374 /* Check if whitespace contained a newline - new line = new command
375 */
376 for (size_t i = ws_start; i < pos; i++) {
377 if (input[i] == '\n') {
378 expect_command = true;
379 break;
380 }
381 }
382 }
383 if (pos >= input_len)
384 break;
385
386 char c = input[pos];
387 size_t token_start = pos;
388
389 /* Comment */
390 if (c == '#') {
391 while (pos < input_len && input[pos] != '\n')
392 pos++;
393 add_token(highlighter, LLE_TOKEN_COMMENT, token_start, pos);
394 continue;
395 }
396
397 /* Single-quoted string */
398 if (c == '\'') {
399 pos++;
400 while (pos < input_len && input[pos] != '\'')
401 pos++;
402 if (pos < input_len)
403 pos++; /* Skip closing quote */
404 add_token(highlighter,
405 pos <= input_len && input[pos - 1] == '\''
406 ? LLE_TOKEN_STRING_SINGLE
407 : LLE_TOKEN_UNCLOSED_STRING,
408 token_start, pos);
409 expect_command = false;
410 continue;
411 }
412
413 /* Double-quoted string */
414 if (c == '"') {
415 pos++;
416 while (pos < input_len && input[pos] != '"') {
417 if (input[pos] == '\\' && pos + 1 < input_len)
418 pos++;
419 pos++;
420 }
421 if (pos < input_len)
422 pos++; /* Skip closing quote */
423 add_token(highlighter,
424 pos <= input_len && input[pos - 1] == '"'
425 ? LLE_TOKEN_STRING_DOUBLE
426 : LLE_TOKEN_UNCLOSED_STRING,
427 token_start, pos);
428 expect_command = false;
429 continue;
430 }
431
432 /* Backtick command substitution */
433 if (c == '`') {
434 pos++;
435 while (pos < input_len && input[pos] != '`') {
436 if (input[pos] == '\\' && pos + 1 < input_len)
437 pos++;
438 pos++;
439 }
440 if (pos < input_len)
441 pos++;
442 add_token(highlighter, LLE_TOKEN_STRING_BACKTICK, token_start, pos);
443 expect_command = false;
444 continue;
445 }
446
447 /* Variable and $-prefixed constructs */
448 if (c == '$') {
449 pos++;
450 if (pos < input_len) {
451 char next = input[pos];
452 lle_syntax_token_type_t vtype = LLE_TOKEN_VARIABLE;
453
454 /* ANSI-C quoting: $'...' */
455 if (next == '\'') {
456 pos++; /* Skip opening quote */
457 while (pos < input_len && input[pos] != '\'') {
458 /* Handle escape sequences in ANSI-C strings */
459 if (input[pos] == '\\' && pos + 1 < input_len) {
460 pos++;
461 }
462 pos++;
463 }
464 if (pos < input_len)
465 pos++; /* Skip closing quote */
466 add_token(highlighter,
467 pos <= input_len && input[pos - 1] == '\''
468 ? LLE_TOKEN_STRING_ANSIC
469 : LLE_TOKEN_UNCLOSED_STRING,
470 token_start, pos);
471 expect_command = false;
472 continue;
473 }
474 /* Arithmetic expansion: $((...)) */
475 else if (next == '(' && pos + 1 < input_len &&
476 input[pos + 1] == '(') {
477 pos += 2; /* Skip (( */
478 int depth = 1;
479 while (pos < input_len && depth > 0) {
480 if (pos + 1 < input_len && input[pos] == '(' &&
481 input[pos + 1] == '(') {
482 depth++;
483 pos++;
484 } else if (pos + 1 < input_len && input[pos] == ')' &&
485 input[pos + 1] == ')') {
486 depth--;
487 pos++;
488 }
489 pos++;
490 }
491 add_token(highlighter, LLE_TOKEN_ARITHMETIC, token_start,
492 pos);
493 expect_command = false;
494 continue;
495 }
496 /* Special variables */
497 else if (next == '?' || next == '#' || next == '@' ||
498 next == '*' || next == '$' || next == '!' ||
499 next == '-' || next == '_' ||
500 (next >= '0' && next <= '9')) {
501 pos++;
502 vtype = LLE_TOKEN_VARIABLE_SPECIAL;
503 }
504 /* ${...} or $(...) command substitution */
505 else if (next == '{' || next == '(') {
506 char close = (next == '{') ? '}' : ')';
507 int depth = 1;
508 pos++;
509 while (pos < input_len && depth > 0) {
510 if (input[pos] == next)
511 depth++;
512 else if (input[pos] == close)
513 depth--;
514 pos++;
515 }
516 }
517 /* Simple $VAR */
518 else if (isalpha((unsigned char)next) || next == '_') {
519 pos++;
520 while (pos < input_len &&
521 (isalnum((unsigned char)input[pos]) ||
522 input[pos] == '_')) {
523 pos++;
524 }
525 }
526
527 add_token(highlighter, vtype, token_start, pos);
528 expect_command = false;
529 continue;
530 }
531 }
532
533 /* Operators */
534 if (c == '|') {
535 pos++;
536 if (pos < input_len && input[pos] == '|') {
537 pos++;
538 add_token(highlighter, LLE_TOKEN_OR, token_start, pos);
539 } else {
540 add_token(highlighter, LLE_TOKEN_PIPE, token_start, pos);
541 }
542 expect_command = true;
543 continue;
544 }
545
546 if (c == '&') {
547 pos++;
548 if (pos < input_len && input[pos] == '&') {
549 pos++;
550 add_token(highlighter, LLE_TOKEN_AND, token_start, pos);
551 expect_command = true;
552 } else {
553 add_token(highlighter, LLE_TOKEN_BACKGROUND, token_start, pos);
554 }
555 continue;
556 }
557
558 if (c == ';') {
559 pos++;
560 add_token(highlighter, LLE_TOKEN_SEMICOLON, token_start, pos);
561 expect_command = true;
562 continue;
563 }
564
565 if (c == '>' || c == '<') {
566 /* Process substitution: >(...) or <(...) */
567 if (pos + 1 < input_len && input[pos + 1] == '(') {
568 lle_syntax_token_type_t pstype =
569 (c == '<') ? LLE_TOKEN_PROCSUB_IN : LLE_TOKEN_PROCSUB_OUT;
570 pos += 2; /* Skip <( or >( */
571 int depth = 1;
572 while (pos < input_len && depth > 0) {
573 if (input[pos] == '(')
574 depth++;
575 else if (input[pos] == ')')
576 depth--;
577 pos++;
578 }
579 add_token(highlighter, pstype, token_start, pos);
580 expect_command = false;
581 continue;
582 }
583
584 /* Here-string: <<< */
585 if (c == '<' && pos + 2 < input_len && input[pos + 1] == '<' &&
586 input[pos + 2] == '<') {
587 pos += 3; /* Skip <<< */
588 add_token(highlighter, LLE_TOKEN_HERESTRING, token_start, pos);
589 expect_command = false;
590 continue;
591 }
592
593 /* Here-document: << or <<- (with optional quoting of delimiter) */
594 if (c == '<' && pos + 1 < input_len && input[pos + 1] == '<' &&
595 (pos + 2 >= input_len || input[pos + 2] != '<')) {
596 pos += 2; /* Skip << */
597 /* Check for <<- (strip leading tabs) */
598 if (pos < input_len && input[pos] == '-') {
599 pos++;
600 }
601 add_token(highlighter, LLE_TOKEN_HEREDOC_OP, token_start, pos);
602 /* Next token will be the delimiter (handled by word parsing) */
603 expect_command = false;
604 continue;
605 }
606
607 /* Regular redirect: >, >>, <, >&, <&, etc. */
608 pos++;
609 while (pos < input_len && (input[pos] == '>' || input[pos] == '&' ||
610 isdigit((unsigned char)input[pos]))) {
611 pos++;
612 }
613 add_token(highlighter, LLE_TOKEN_REDIRECT, token_start, pos);
614 expect_command = false;
615 continue;
616 }
617
618 if (c == '(') {
619 pos++;
620 add_token(highlighter, LLE_TOKEN_SUBSHELL_START, token_start, pos);
621 expect_command = true;
622 continue;
623 }
624
625 if (c == ')') {
626 pos++;
627 add_token(highlighter, LLE_TOKEN_SUBSHELL_END, token_start, pos);
628 expect_command = false;
629 continue;
630 }
631
632 if (c == '{') {
633 pos++;
634 add_token(highlighter, LLE_TOKEN_BRACE_START, token_start, pos);
635 expect_command = true;
636 continue;
637 }
638
639 if (c == '}') {
640 pos++;
641 add_token(highlighter, LLE_TOKEN_BRACE_END, token_start, pos);
642 expect_command = false;
643 continue;
644 }
645
646 /* Word (command, argument, path, option, etc.) */
647 if (is_word_char(c) || c == '\\') {
648 /* Scan the word */
649 bool has_glob = false;
650 bool has_slash = false;
651
652 while (pos < input_len) {
653 char ch = input[pos];
654 if (ch == '\\' && pos + 1 < input_len) {
655 pos += 2;
656 continue;
657 }
658 if (is_glob_char(ch))
659 has_glob = true;
660 if (ch == '/')
661 has_slash = true;
662 if (!is_word_char(ch) && !is_glob_char(ch))
663 break;
664 pos++;
665 }
666
667 size_t word_len = pos - token_start;
668
669 /* Determine token type */
670 lle_syntax_token_type_t type;
671
672 if (expect_command) {
673 /* Extract word for checking */
674 char word[256];
675 size_t copy_len =
676 word_len < sizeof(word) - 1 ? word_len : sizeof(word) - 1;
677 memcpy(word, input + token_start, copy_len);
678 word[copy_len] = '\0';
679
680 /* Check for VAR=value assignment prefix first */
681 if (is_assignment(input + token_start, word_len)) {
682 type = LLE_TOKEN_ASSIGNMENT;
683 /* Keep expect_command = true because command follows */
684 /* e.g., "VAR=value command arg1 arg2" */
685 } else if (is_shell_keyword(input + token_start, word_len)) {
686 type = LLE_TOKEN_KEYWORD;
687 /* Block-ending keywords (done, fi, esac, etc.) don't expect
688 a command after them. Block-starting keywords do. */
689 if (is_block_ending_keyword(input + token_start,
690 word_len)) {
691 expect_command = false;
692 } else {
693 expect_command = true;
694 }
695 } else if (highlighter->validate_commands) {
696 type = lle_syntax_check_command(highlighter, word);
697 expect_command = false;
698 } else {
699 type = LLE_TOKEN_COMMAND_VALID; /* Assume valid if not
700 checking */
701 expect_command = false;
702 }
703 } else {
704 /* Not expecting command - it's an argument */
705 if (is_option_start(input + token_start, word_len)) {
706 type = LLE_TOKEN_OPTION;
707 } else if (has_glob) {
708 type = LLE_TOKEN_GLOB;
709 } else if (has_slash && highlighter->validate_paths) {
710 /* Looks like a path - check if it exists */
711 char path[4096];
712 size_t copy_len = word_len < sizeof(path) - 1
713 ? word_len
714 : sizeof(path) - 1;
715 memcpy(path, input + token_start, copy_len);
716 path[copy_len] = '\0';
717
718 /* Expand ~ if present */
719 if (path[0] == '~') {
720 const char *home = getenv("HOME");
721 if (home) {
722 char expanded[4096];
723 snprintf(expanded, sizeof(expanded), "%s%s", home,
724 path + 1);
725 type = path_exists(expanded)
726 ? LLE_TOKEN_PATH_VALID
727 : LLE_TOKEN_PATH_INVALID;
728 } else {
729 type = LLE_TOKEN_PATH_INVALID;
730 }
731 } else {
732 type = path_exists(path) ? LLE_TOKEN_PATH_VALID
733 : LLE_TOKEN_PATH_INVALID;
734 }
735 } else if (has_slash) {
736 type =
737 LLE_TOKEN_ARGUMENT; /* Path-like but not validating */
738 } else {
739 /* Check if it's a number */
740 bool is_num = true;
741 for (size_t i = token_start; i < pos && is_num; i++) {
742 if (!isdigit((unsigned char)input[i]) &&
743 input[i] != '.' && input[i] != '-' &&
744 input[i] != '+') {
745 is_num = false;
746 }
747 }

892int lle_syntax_color_to_ansi(uint32_t color, uint8_t attributes,
893 int color_depth, char *output,
894 size_t output_size) {
895 if (!output || output_size < 2)
896 return -1;
897
898 char *p = output;
899 char *end = output + output_size - 1;
900
901 /* Start escape sequence */
902 p += snprintf(p, end - p, "\x1b[");
903
904 /* Attributes */
905 bool need_sep = false;
906 if (attributes & LLE_ATTR_BOLD) {
907 p += snprintf(p, end - p, "1");
908 need_sep = true;
909 }
910 if (attributes & LLE_ATTR_DIM) {
911 p += snprintf(p, end - p, "%s2", need_sep ? ";" : "");
912 need_sep = true;
913 }
914 if (attributes & LLE_ATTR_ITALIC) {
915 p += snprintf(p, end - p, "%s3", need_sep ? ";" : "");
916 need_sep = true;
917 }
918 if (attributes & LLE_ATTR_UNDERLINE) {
919 p += snprintf(p, end - p, "%s4", need_sep ? ";" : "");
920 need_sep = true;
921 }
922
923 /* Color */
924 if (color != 0 && color_depth > 0) {
925 uint8_t r = (color >> 16) & 0xFF;
926 uint8_t g = (color >> 8) & 0xFF;
927 uint8_t b = color & 0xFF;
928
929 if (color_depth >= 3) {
930 /* Truecolor */
931 p += snprintf(p, end - p, "%s38;2;%d;%d;%d", need_sep ? ";" : "", r,
932 g, b);
933 } else if (color_depth == 2) {
934 /* 256 color - convert RGB to 256 color index */
935 int idx;
936 if (r == g && g == b) {
937 /* Grayscale */
938 if (r < 8)
939 idx = 16;
940 else if (r > 248)
941 idx = 231;
942 else
943 idx = 232 + (r - 8) / 10;
944 } else {
945 /* Color cube */
946 idx = 16 + (r / 51) * 36 + (g / 51) * 6 + (b / 51);
947 }
948 p += snprintf(p, end - p, "%s38;5;%d", need_sep ? ";" : "", idx);
949 } else {
950 /* 8 color - pick closest basic color */
951 int basic = 0;
952 if (r > 127)
953 basic |= 1;
954 if (g > 127)
955 basic |= 2;
956 if (b > 127)
957 basic |= 4;
958 /* Use bright if any component is very bright */
959 if (r > 200 || g > 200 || b > 200) {
960 p += snprintf(p, end - p, "%s9%d", need_sep ? ";" : "", basic);
961 } else {
962 p += snprintf(p, end - p, "%s3%d", need_sep ? ";" : "", basic);
963 }
964 }
965 }
966
967 p += snprintf(p, end - p, "m");
968
969 return (int)(p - output);
970}
971
972int lle_syntax_render_ansi(lle_syntax_highlighter_t *highlighter,
973 const char *input, char *output,
974 size_t output_size) {
975 if (!highlighter || !input || !output || output_size == 0)
976 return -1;
977
978 char *p = output;
979 char *end = output + output_size - 1;
980
981 for (size_t i = 0; i < highlighter->token_count && p < end; i++) {
982 lle_syntax_token_t *tok = &highlighter->tokens[i];
983
984 /* Skip whitespace and unknown tokens - just copy them */
985 if (tok->type == LLE_TOKEN_WHITESPACE ||
986 tok->type == LLE_TOKEN_UNKNOWN) {
987 size_t len = tok->end - tok->start;
988 if (p + len >= end)
989 len = end - p;
990 memcpy(p, input + tok->start, len);
991 p += len;
992 continue;
993 }
994
995 /* Apply color */
996 if (tok->color != 0 || tok->attributes != 0) {
997 char color_seq[64];
998 int seq_len = lle_syntax_color_to_ansi(
999 tok->color, tok->attributes, highlighter->color_depth,
1000 color_seq, sizeof(color_seq));
1001 if (seq_len > 0 && p + seq_len < end) {
1002 memcpy(p, color_seq, seq_len);
1003 p += seq_len;
1004 }
1005 }
1006
1007 /* Copy token text */
1008 size_t len = tok->end - tok->start;
1009 if (p + len >= end)
1010 len = end - p;
1011 memcpy(p, input + tok->start, len);
1012 p += len;
1013
1014 /* Reset after token */
1015 if (tok->color != 0 || tok->attributes != 0) {
1016 const char *reset = "\x1b[0m";
1017 size_t reset_len = 4;
1018 if (p + reset_len < end) {
1019 memcpy(p, reset, reset_len);
1020 p += reset_len;
1021 }
1022 }
1023 }
1024
1025 *p = '\0';
1026 return (int)(p - output);
1027}
