35/**
36 * Create a new editor context
37 *
38 * Initializes the editor with core subsystems. Subsystems are initialized
39 * in dependency order:
40 * 1. Memory pool (if not provided)
41 * 2. Buffer
42 * 3. Cursor manager (depends on buffer)
43 * 4. Kill ring
44 * 5. Other subsystems initialized on-demand
45 *
46 * @param editor Pointer to receive new editor context
47 * @param pool Memory pool for allocations (NULL = use global pool)
48 * @return LLE_SUCCESS or error code
49 */
50lle_result_t lle_editor_create(lle_editor_t **editor,
51 lusush_memory_pool_t *pool) {
52 if (!editor) {
53 return LLE_ERROR_INVALID_PARAMETER;
54 }
55
56 /* Use global pool if none provided */
57 if (!pool) {
58 pool = global_memory_pool;
59 }
60
61 /* Allocate editor structure */
62 lle_editor_t *ed = (lle_editor_t *)lle_pool_alloc(sizeof(lle_editor_t));
63 if (!ed) {
64 return LLE_ERROR_OUT_OF_MEMORY;
65 }
66
67 /* Zero-initialize entire structure */
68 memset(ed, 0, sizeof(lle_editor_t));
69
70 /* Store Lusush memory pool reference */
71 ed->lusush_pool = pool;
72
73 /* Create LLE memory pool wrapper for unified memory management */
74 lle_result_t result = lle_memory_pool_create_from_lusush(
75 &ed->lle_pool, pool, LLE_POOL_BUFFER /* Editor uses buffer pool type */
76 );
77 if (result != LLE_SUCCESS) {
78 lle_pool_free(ed);
79 return result;
80 }
81
82 /* Initialize editing mode to Emacs (default) */
83 ed->editing_mode = LLE_EDITING_MODE_EMACS;
84 ed->current_keymap = LLE_KEYMAP_EMACS;
85
86 /* Create buffer (core subsystem) */
87 result = lle_buffer_create(&ed->buffer, pool, 0);
88 if (result != LLE_SUCCESS) {
89 lle_pool_free(ed);
90 return result;
91 }
92
93 /* Create cursor manager (depends on buffer) */
94 result = lle_cursor_manager_init(&ed->cursor_manager, ed->buffer);
95 if (result != LLE_SUCCESS) {
96 lle_buffer_destroy(ed->buffer);
97 lle_pool_free(ed);
98 return result;
99 }
100
101 /* Create kill ring with default max entries (100) using unified pool */
102 result = lle_kill_ring_create(&ed->kill_ring, 100, ed->lle_pool);
103 if (result != LLE_SUCCESS) {
104 lle_cursor_manager_destroy(ed->cursor_manager);
105 lle_buffer_destroy(ed->buffer);
106 lle_memory_pool_destroy(ed->lle_pool);
107 lle_pool_free(ed);
108 return result;
109 }
110
111 /* Create change tracker for undo/redo support */
112 result = lle_change_tracker_init(&ed->change_tracker, pool,
113 LLE_BUFFER_MAX_UNDO_LEVELS);
114 if (result != LLE_SUCCESS) {
115 lle_kill_ring_destroy(ed->kill_ring);
116 lle_cursor_manager_destroy(ed->cursor_manager);
117 lle_buffer_destroy(ed->buffer);
118 lle_memory_pool_destroy(ed->lle_pool);
119 lle_pool_free(ed);
120 return result;
121 }
122
123 /* Enable change tracking on the buffer */
124 ed->buffer->change_tracking_enabled = true;
125
126 /* Create completion system (Spec 12) using unified pool */
127 result = lle_completion_system_create(ed->lle_pool, &ed->completion_system);
128 if (result != LLE_SUCCESS) {
129 lle_change_tracker_destroy(ed->change_tracker);
130 lle_kill_ring_destroy(ed->kill_ring);
131 lle_cursor_manager_destroy(ed->cursor_manager);
132 lle_buffer_destroy(ed->buffer);
133 lle_memory_pool_destroy(ed->lle_pool);
134 lle_pool_free(ed);
135 return result;
136 }
137
138 /* Initialize custom completion source subsystem */
139 result = lle_custom_source_init(ed->completion_system->source_manager,
140 ed->lle_pool);
141 if (result != LLE_SUCCESS) {
142 /* Non-fatal - custom sources won't be available but continue */
143 }
144
145 /* Load user completion config (non-fatal if fails) */
146 lle_completion_load_config();
147
148 /* Initialize statistics */
149 ed->total_keystrokes = 0;
150 ed->command_count = 0;
151 ed->edit_session_start = 0; /* Will be set when editing starts */
152
153 /* Other subsystems (history, display, etc.) are initialized on-demand */
154 ed->history_system = NULL;
155 ed->history_buffer_integration = NULL;
156 ed->keybinding_manager = NULL;
157 ed->display_controller = NULL;
158 ed->vi_state = NULL;
159 ed->plugin_manager = NULL;
160 ed->script_integration = NULL;
161 ed->history_edit_callbacks = NULL;
162
163 /* Initialize state flags */
164 ed->history_search_active = false;
165 ed->history_search_direction = 0;
166 ed->quoted_insert_mode = false;
167
168 /* === Widget System Initialization (Spec 07) === */
169 /* Initialize widget registry for ZSH-style widget system */
170 result = lle_widget_registry_init(&ed->widget_registry, ed->lle_pool);
171 if (result != LLE_SUCCESS) {
172 /* Widget system initialization failed - non-fatal, continue without
173 * widgets */
174 ed->widget_registry = NULL;
175 ed->widget_hooks_manager = NULL;
176 } else {
177 /* Register builtin widgets */
178 lle_register_builtin_widgets(ed->widget_registry);
179
180 /* Initialize widget hooks manager */
181 result = lle_widget_hooks_manager_init(
182 &ed->widget_hooks_manager, ed->widget_registry, ed->lle_pool);
183 if (result != LLE_SUCCESS) {
184 /* Hooks manager failed - widgets work but hooks won't trigger */
185 ed->widget_hooks_manager = NULL;
186 } else {
187 /* Register builtin widget hooks (e.g., transient-prompt) */
188 lle_register_builtin_widget_hooks(ed->widget_hooks_manager);
189 }
190 }
191
192 *editor = ed;
193 return LLE_SUCCESS;
194}

196/**
197 * Destroy editor context
198 *
199 * Frees all resources associated with the editor, including all subsystems.
200 * Subsystems are destroyed in reverse dependency order.
201 *
202 * @param editor Editor context to destroy
203 * @return LLE_SUCCESS or error code
204 */
205lle_result_t lle_editor_destroy(lle_editor_t *editor) {
206 if (!editor) {
207 return LLE_ERROR_INVALID_PARAMETER;
208 }
209
210 /* Destroy subsystems in reverse order of dependencies */
211
212 /* Destroy widget hooks manager first (depends on widget registry) */
213 if (editor->widget_hooks_manager) {
214 lle_widget_hooks_manager_destroy(editor->widget_hooks_manager);
215 editor->widget_hooks_manager = NULL;
216 }
217
218 /* Destroy widget registry */
219 if (editor->widget_registry) {
220 lle_widget_registry_destroy(editor->widget_registry);
221 editor->widget_registry = NULL;
222 }
223
224 /* Destroy optional subsystems if allocated */
225 if (editor->vi_state) {
226 lle_pool_free(editor->vi_state);
227 editor->vi_state = NULL;
228 }
229
230 /* Note: history_system, keybinding_manager, display_controller, etc.
231 * should have their own destroy functions called if they're not NULL.
232 * For now, we only handle the core subsystems we initialize. */
233
234 /* Shutdown custom completion sources before destroying completion system */
235 lle_custom_source_shutdown();
236
237 /* Destroy completion system (Spec 12) */
238 if (editor->completion_system) {
239 lle_completion_system_destroy(editor->completion_system);
240 editor->completion_system = NULL;
241 }
242
243 /* Destroy change tracker */
244 if (editor->change_tracker) {
245 lle_change_tracker_destroy(editor->change_tracker);
246 editor->change_tracker = NULL;
247 }
248
249 /* Destroy kill ring */
250 if (editor->kill_ring) {
251 lle_kill_ring_destroy(editor->kill_ring);
252 editor->kill_ring = NULL;
253 }
254
255 /* Destroy cursor manager */
256 if (editor->cursor_manager) {
257 lle_cursor_manager_destroy(editor->cursor_manager);
258 editor->cursor_manager = NULL;
259 }
260
261 /* Destroy buffer */
262 if (editor->buffer) {
263 lle_buffer_destroy(editor->buffer);
264 editor->buffer = NULL;
265 }
266
267 /* Destroy LLE memory pool wrapper */
268 if (editor->lle_pool) {
269 lle_memory_pool_destroy(editor->lle_pool);
270 editor->lle_pool = NULL;
271 }
272
273 /* Note: lusush_pool is not destroyed here - it's managed externally */
274
275 /* Free editor structure */
276 lle_pool_free(editor);
277
278 return LLE_SUCCESS;
279}

281/**
282 * Reset editor to initial state
283 *
284 * Clears the buffer and resets cursor to beginning, but keeps all
285 * subsystems initialized.
286 *
287 * @param editor Editor context to reset
288 * @return LLE_SUCCESS or error code
289 */
290lle_result_t lle_editor_reset(lle_editor_t *editor) {
291 if (!editor || !editor->buffer) {
292 return LLE_ERROR_INVALID_PARAMETER;
293 }
294
295 /* Clear buffer */
296 lle_result_t result = lle_buffer_clear(editor->buffer);
297 if (result != LLE_SUCCESS) {
298 return result;
299 }
300
301 /* Reset cursor to beginning */
302 if (editor->cursor_manager) {
303 result =
304 lle_cursor_manager_move_to_byte_offset(editor->cursor_manager, 0);
305 if (result != LLE_SUCCESS) {
306 return result;
307 }
308 }
309
310 /* Clear history search state */
311 editor->history_search_active = false;
312 editor->history_search_direction = 0;
313
314 /* Clear quoted insert mode */
315 editor->quoted_insert_mode = false;
316
317 return LLE_SUCCESS;
318}

351/**
352 * Cancel current operation
353 *
354 * Used for Ctrl-G (abort) functionality. Resets any active modes
355 * and returns editor to normal state.
356 *
357 * @param editor Editor context
358 * @return LLE_SUCCESS or error code
359 */
360lle_result_t lle_editor_cancel_operation(lle_editor_t *editor) {
361 if (!editor) {
362 return LLE_ERROR_INVALID_PARAMETER;
363 }
364
365 /* Cancel history search if active */
366 editor->history_search_active = false;
367 editor->history_search_direction = 0;
368
369 /* Exit quoted insert mode */
370 editor->quoted_insert_mode = false;
371
372 /* If in vi command mode, return to insert mode */
373 if (editor->editing_mode == LLE_EDITING_MODE_VI_COMMAND) {
374 editor->editing_mode = LLE_EDITING_MODE_VI_INSERT;
375 }
376
377 return LLE_SUCCESS;
378}
