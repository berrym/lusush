133 bool

135 bool onecmd; // -t flag: exit after one command
136 bool notify; // -b flag: asynchronous background job notification
137 bool ignoreeof; // ignoreeof: prevent exit on EOF (Ctrl+D)
138 bool nolog; // nolog: prevent function definitions from entering command
139 // history
140 bool emacs_mode; // emacs: enable emacs-style command line editing (default
141 // true)
142 bool vi_mode; // vi: enable vi-style command line editing
143 bool posix_mode; // posix: enable strict POSIX compliance mode
144 bool pipefail_mode; // pipefail: make pipelines fail if any command fails
145 bool
146 histexpand_mode; // histexpand: enable history expansion (!! !n !string)
147 bool history_mode; // history: enable command history recording
148 bool interactive_comments_mode; // interactive-comments: enable # comments
149 // in interactive mode
150 bool physical_mode; // physical: use physical directory paths (resolve
151 // symlinks)
152 bool privileged_mode; // privileged: restricted shell security mode
