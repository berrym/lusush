46/**
47 * Calculate category positions in result
48 */
49static lle_result_t
50calculate_category_positions(lle_completion_menu_state_t *state) {
51 if (!state || !state->result) {
52 return LLE_ERROR_INVALID_PARAMETER;
53 }
54
55 if (state->result->count == 0) {
56 state->category_count = 0;
57 state->category_positions = NULL;
58 return LLE_SUCCESS;
59 }
60
61 // Count unique categories
62 size_t cat_count = 0;
63 lle_completion_type_t current_type = LLE_COMPLETION_TYPE_UNKNOWN;
64
65 for (size_t i = 0; i < state->result->count; i++) {
66 if (state->result->items[i].type != current_type) {
67 current_type = state->result->items[i].type;
68 cat_count++;
69 }
70 }
71
72 if (cat_count == 0) {
73 state->category_count = 0;
74 state->category_positions = NULL;
75 return LLE_SUCCESS;
76 }
77
78 // Allocate category positions array
79 state->category_positions =
80 (size_t *)lle_pool_alloc(sizeof(size_t) * cat_count);
81 if (!state->category_positions) {
82 return LLE_ERROR_OUT_OF_MEMORY;
83 }
84
85 // Fill category positions
86 size_t cat_index = 0;
87 current_type = LLE_COMPLETION_TYPE_UNKNOWN;
88
89 for (size_t i = 0; i < state->result->count; i++) {
90 if (state->result->items[i].type != current_type) {
91 current_type = state->result->items[i].type;
92 state->category_positions[cat_index++] = i;
93 }
94 }
95
96 state->category_count = cat_count;
97 return LLE_SUCCESS;
98}

104lle_result_t
105lle_completion_menu_state_create(lle_memory_pool_t *memory_pool,
106 lle_completion_result_t *result,
107 const lle_completion_menu_config_t *config,
108 lle_completion_menu_state_t **state) {
109 if (!memory_pool || !result || !state) {
110 return LLE_ERROR_INVALID_PARAMETER;
111 }
112
113 // Allocate state structure
114 lle_completion_menu_state_t *new_state =
115 (lle_completion_menu_state_t *)lle_pool_alloc(
116 sizeof(lle_completion_menu_state_t));
117 if (!new_state) {
118 return LLE_ERROR_OUT_OF_MEMORY;
119 }
120
121 // Initialize state
122 new_state->result = result;
123 new_state->selected_index = 0;
124 new_state->first_visible = 0;
125 new_state->visible_count = 0;
126 new_state->target_column = 0; // Sticky column for UP/DOWN navigation
127 new_state->category_positions = NULL;
128 new_state->category_count = 0;
129 new_state->menu_active = true; // Menu is active when created
130 new_state->memory_pool = memory_pool;
131
132 // Copy configuration or use defaults
133 if (config) {
134 new_state->config = *config;
135 } else {
136 new_state->config = lle_completion_menu_default_config();
137 }
138
139 // Calculate visible count
140 size_t total_items = result->count;
141 if (new_state->config.enable_scrolling) {
142 new_state->visible_count =
143 total_items < new_state->config.max_visible_items
144 ? total_items
145 : new_state->config.max_visible_items;
146 } else {
147 new_state->visible_count = total_items;
148 }
149
150 // Calculate category positions
151 lle_result_t res = calculate_category_positions(new_state);
152 if (res != LLE_SUCCESS) {
153 lle_pool_free(new_state);
154 return res;
155 }
156
157 *state = new_state;
158 return LLE_SUCCESS;
159}

161lle_result_t
162lle_completion_menu_state_free(lle_completion_menu_state_t *state) {
163 if (!state) {
164 return LLE_ERROR_INVALID_PARAMETER;
165 }
166
167 // Free category positions if allocated
168 if (state->category_positions) {
169 lle_pool_free(state->category_positions);
170 }
171
172 // Free state structure
173 // Note: result is owned by caller, so we don't free it
174 lle_pool_free(state);
175
176 return LLE_SUCCESS;
177}

263/**
264 * Calculate visual width of text (excluding ANSI codes)
265 */
266static size_t calc_visual_width(const char *text) {
267 if (!text)
268 return 0;
269
270 size_t width = 0;
271 bool in_escape = false;
272
273 for (const char *p = text; *p; p++) {
274 if (*p == '\033' || *p == '\x1b') {
275 in_escape = true;
276 continue;
277 }
278 if (in_escape) {
279 if ((*p >= 'A' && *p <= 'Z') || (*p >= 'a' && *p <= 'z') ||
280 *p == 'm') {
281 in_escape = false;
282 }
283 continue;
284 }
285 width++;
286 }
287
288 return width;
289}

291lle_result_t
292lle_completion_menu_update_layout(lle_completion_menu_state_t *state,
293 size_t terminal_width) {
294 if (!state) {
295 return LLE_ERROR_INVALID_PARAMETER;
296 }
297
298 // Store terminal width
299 state->terminal_width = terminal_width > 0 ? terminal_width : 80;
300
301 // Calculate max item width
302 size_t max_item_width = 0;
303 if (state->result && state->result->count > 0) {
304 for (size_t i = 0; i < state->result->count; i++) {
305 if (state->result->items[i].text) {
306 size_t len = calc_visual_width(state->result->items[i].text);
307 if (len > max_item_width) {
308 max_item_width = len;
309 }
310 }
311 }
312 }
313
314 // Add padding for selection indicator and spacing
315 const size_t padding = 4; // " " separator + selection indicator space
316 state->column_width = max_item_width + padding;
317
318 // Ensure minimum column width
319 if (state->column_width < 10) {
320 state->column_width = 10;
321 }
322
323 // Calculate number of columns that fit
324 if (state->column_width >= state->terminal_width) {
325 state->num_columns = 1;
326 } else {
327 state->num_columns = state->terminal_width / state->column_width;
328 if (state->num_columns == 0) {
329 state->num_columns = 1;
330 }
331 // Cap at reasonable maximum
332 if (state->num_columns > 6) {
333 state->num_columns = 6;
334 }
335 }
336
337 return LLE_SUCCESS;
338}
