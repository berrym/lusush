31/**
32 * Count newlines in a string
33 *
34 * @param str String to analyze
35 * @return Number of newlines found
36 */
37static size_t count_newlines(const char *str) {
38 if (!str) {
39 return 0;
40 }
41
42 size_t count = 0;
43 for (const char *p = str; *p; p++) {
44 if (*p == '\n') {
45 count++;
46 }
47 }
48 return count;
49}

51/**
52 * Create a flattened version of multiline command
53 *
54 * Replaces newlines with spaces while preserving essential structure.
55 *
56 * @param original Original multiline command
57 * @param flattened Output buffer for flattened command
58 * @param flattened_size Size of output buffer
59 * @return LLE_SUCCESS on success, error code on failure
60 */
61static lle_result_t flatten_command(const char *original, char *flattened,
62 size_t flattened_size) {
63 if (!original || !flattened || flattened_size == 0) {
64 return LLE_ERROR_INVALID_PARAMETER;
65 }
66
67 const char *src = original;
68 char *dst = flattened;
69 size_t remaining = flattened_size - 1; /* Reserve for null terminator */
70 bool prev_was_space = false;
71
72 while (*src && remaining > 0) {
73 char c = *src++;
74
75 /* Replace newlines with spaces */
76 if (c == '\n' || c == '\r') {
77 if (!prev_was_space && dst > flattened) {
78 *dst++ = ' ';
79 remaining--;
80 prev_was_space = true;
81 }
82 continue;
83 }
84
85 /* Skip redundant spaces */
86 if (c == ' ' || c == '\t') {
87 if (!prev_was_space) {
88 *dst++ = ' ';
89 remaining--;
90 prev_was_space = true;
91 }
92 continue;
93 }
94
95 /* Copy normal character */
96 *dst++ = c;
97 remaining--;
98 prev_was_space = false;
99 }
100
101 /* Trim trailing space */
102 if (dst > flattened && dst[-1] == ' ') {
103 dst--;
104 }
105
106 *dst = '\0';
107
108 if (*src && remaining == 0) {
109 return LLE_ERROR_BUFFER_OVERFLOW;
110 }
111
112 return LLE_SUCCESS;
113}

154/**
155 * Detect multiline structure using continuation analysis
156 */
157lle_result_t
158lle_history_detect_multiline_structure(const char *command,
159 lle_history_multiline_info_t *info) {
160 if (!command || !info) {
161 return LLE_ERROR_INVALID_PARAMETER;
162 }
163
164 /* Initialize info structure */
165 memset(info, 0, sizeof(lle_history_multiline_info_t));
166
167 /* Count newlines */
168 info->line_count = count_newlines(command) + 1;
169 info->is_multiline = (info->line_count > 1);
170
171 if (!info->is_multiline) {
172 return LLE_SUCCESS;
173 }
174
175 /* Use Lusush continuation system to analyze structure */
176 continuation_state_t state;
177 continuation_state_init(&state);
178
179 /* Analyze the command */
180 continuation_analyze_line(command, &state);
181
182 /* Extract structural information */
183 info->has_unclosed_quotes =
184 (state.in_single_quote || state.in_double_quote || state.in_backtick);
185 info->has_unclosed_brackets =
186 (state.paren_count > 0 || state.brace_count > 0 ||
187 state.bracket_count > 0);
188 info->is_function_def = state.in_function_definition;
189 info->is_control_structure =
190 (state.in_if_statement || state.in_while_loop || state.in_for_loop ||
191 state.in_until_loop || state.in_case_statement);
192 info->is_here_doc = state.in_here_doc;
193
194 /* Calculate total length */
195 info->total_length = strlen(command);
196
197 /* Cleanup continuation state */
198 continuation_state_cleanup(&state);
199
200 return LLE_SUCCESS;
201}

203/**
204 * Preserve multiline formatting in history entry
205 */
206lle_result_t lle_history_preserve_multiline(lle_history_entry_t *entry,
207 const char *original_multiline) {
208 if (!entry || !original_multiline) {
209 return LLE_ERROR_INVALID_PARAMETER;
210 }
211
212 /* Free existing multiline storage if present */
213 if (entry->original_multiline) {
214 lle_pool_free(entry->original_multiline);
215 entry->original_multiline = NULL;
216 }
217
218 /* Detect if it's actually multiline */
219 bool is_multiline = false;
220 lle_result_t result =
221 lle_history_detect_multiline(original_multiline, &is_multiline);
222 if (result != LLE_SUCCESS) {
223 return result;
224 }
225
226 entry->is_multiline = is_multiline;
227
228 if (!is_multiline) {
229 /* Not multiline - no need to preserve separate original */
230 return LLE_SUCCESS;
231 }
232
233 /* Store original multiline format */
234 entry->original_multiline = pool_strdup(original_multiline);
235 if (!entry->original_multiline) {
236 return LLE_ERROR_OUT_OF_MEMORY;
237 }
238
239 /* Flatten the command into the main command field (for searching, etc.) */
240 char flattened[LLE_HISTORY_MAX_COMMAND_LENGTH];
241 result = flatten_command(original_multiline, flattened, sizeof(flattened));
242 if (result != LLE_SUCCESS) {
243 lle_pool_free(entry->original_multiline);
244 entry->original_multiline = NULL;
245 return result;
246 }
247
248 /* Update the main command field with flattened version */
249 /* Note: entry->command should already be allocated */
250 if (entry->command) {
251 lle_pool_free(entry->command);
252 }
253
254 entry->command = pool_strdup(flattened);
255 if (!entry->command) {
256 lle_pool_free(entry->original_multiline);
257 entry->original_multiline = NULL;
258 return LLE_ERROR_OUT_OF_MEMORY;
259 }
260
261 entry->command_length = strlen(flattened);
262
263 return LLE_SUCCESS;
264}

266/**
267 * Reconstruct multiline command for editing
268 */
269lle_result_t
270lle_history_reconstruct_multiline(const lle_history_entry_t *entry,
271 char *buffer, size_t buffer_size,
272 lle_history_multiline_format_t format) {
273 if (!entry || !buffer || buffer_size == 0) {
274 return LLE_ERROR_INVALID_PARAMETER;
275 }
276
277 const char *source = NULL;
278
279 switch (format) {
280 case LLE_MULTILINE_FORMAT_ORIGINAL:
281 /* Use original formatting if available */
282 if (entry->is_multiline && entry->original_multiline) {
283 source = entry->original_multiline;
284 } else {
285 source = entry->command;
286 }
287 break;
288
289 case LLE_MULTILINE_FORMAT_FLATTENED:
290 /* Use flattened version */
291 source = entry->command;
292 break;
293
294 case LLE_MULTILINE_FORMAT_COMPACT:
295 /* Use flattened version (same as FLATTENED for now) */
296 source = entry->command;
297 break;
298
299 default:
300 return LLE_ERROR_INVALID_PARAMETER;
301 }
302
303 if (!source) {
304 return LLE_ERROR_INVALID_STATE;
305 }
306
307 /* Copy to buffer */
308 size_t source_len = strlen(source);
309 if (source_len >= buffer_size) {
310 return LLE_ERROR_BUFFER_OVERFLOW;
311 }
312
313 memcpy(buffer, source, source_len + 1);
314
315 return LLE_SUCCESS;
316}

371/**
372 * Analyze multiline command lines
373 */
374lle_result_t
375lle_history_analyze_multiline_lines(const char *command,
376 lle_history_multiline_line_t **lines,
377 size_t *line_count) {
378 if (!command || !lines || !line_count) {
379 return LLE_ERROR_INVALID_PARAMETER;
380 }
381
382 /* Count lines first */
383 size_t count = count_newlines(command) + 1;
384
385 /* Allocate line array */
386 lle_history_multiline_line_t *line_array =
387 lle_pool_alloc(sizeof(lle_history_multiline_line_t) * count);
388
389 if (!line_array) {
390 return LLE_ERROR_OUT_OF_MEMORY;
391 }
392
393 /* Parse lines */
394 const char *start = command;
395 const char *end = NULL;
396 size_t index = 0;
397
398 while (*start && index < count) {
399 /* Find end of line */
400 end = strchr(start, '\n');
401 if (!end) {
402 end = start + strlen(start);
403 }
404
405 /* Calculate line length */
406 size_t line_len = end - start;
407
408 /* Store line information */
409 line_array[index].line_text =
410 start; /* Note: Not duplicating, just pointing */
411 line_array[index].line_length = line_len;
412 line_array[index].line_number = index + 1;
413
414 /* Calculate indentation */
415 size_t indent = 0;
416 for (const char *p = start; p < end && (*p == ' ' || *p == '\t'); p++) {
417 indent++;
418 }
419 line_array[index].indentation = indent;
420
421 /* Move to next line */
422 start = (*end == '\n') ? (end + 1) : end;
423 index++;
424 }
425
426 *lines = line_array;
427 *line_count = count;
428
429 return LLE_SUCCESS;
430}

445/**
446 * Format multiline command with indentation
447 */
448lle_result_t lle_history_format_multiline(const char *command, char *formatted,
449 size_t formatted_size,
450 size_t base_indent) {
451 if (!command || !formatted || formatted_size == 0) {
452 return LLE_ERROR_INVALID_PARAMETER;
453 }
454
455 const char *src = command;
456 char *dst = formatted;
457 size_t remaining = formatted_size - 1;
458 bool at_line_start = true;
459 size_t indent_added = 0;
460 (void)indent_added; /* Reserved for indentation tracking */
461
462 while (*src && remaining > 0) {
463 char c = *src++;
464
465 /* Add indentation at start of each line */
466 if (at_line_start && c != '\n') {
467 for (size_t i = 0; i < base_indent && remaining > 0; i++) {
468 *dst++ = ' ';
469 remaining--;
470 indent_added++;
471 }
472 at_line_start = false;
473 }
474
475 /* Copy character */
476 *dst++ = c;
477 remaining--;
478
479 /* Track line starts */
480 if (c == '\n') {
481 at_line_start = true;
482 }
483 }
484
485 *dst = '\0';
486
487 if (*src && remaining == 0) {
488 return LLE_ERROR_BUFFER_OVERFLOW;
489 }
490
491 return LLE_SUCCESS;
492}
