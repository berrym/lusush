27// Auto-correction configuration structure
28typedef struct {
29 bool enabled; // Enable/disable auto-correction
30 int max_suggestions; // Maximum number of suggestions (1-5)
31 int similarity_threshold; // Minimum similarity score (0-100)
32 bool interactive_prompts; // Show "Did you mean?" prompts
33 bool learn_from_history; // Learn from command history
34 bool correct_builtins; // Suggest builtin corrections
35 bool correct_external; // Suggest external command corrections
36 bool case_sensitive; // Case-sensitive matching
37} autocorrect_config_t;

39// Correction suggestion structure
40typedef struct {
41 char *command; // Suggested command
42 int score; // Similarity score (0-100)
43 const char *source; // Source: "builtin", "history", "path", "function"
44} correction_t;

46// Correction results structure
47typedef struct {
48 correction_t suggestions[MAX_CORRECTIONS]; // Array of suggestions
49 int count; // Number of suggestions found
50 char *original_command; // Original misspelled command
