56lle_result_t lle_segment_registry_register(lle_segment_registry_t *registry,
57 lle_prompt_segment_t *segment) {
58 if (!registry || !segment || !registry->initialized) {
59 return LLE_ERROR_INVALID_PARAMETER;
60 }
61
62 if (registry->count >= LLE_SEGMENT_REGISTRY_MAX) {
63 return LLE_ERROR_BUFFER_OVERFLOW;
64 }
65
66 /* Check for duplicate name */
67 for (size_t i = 0; i < registry->count; i++) {
68 if (strcmp(registry->segments[i]->name, segment->name) == 0) {
69 return LLE_ERROR_INVALID_STATE; /* Duplicate */
70 }
71 }
72
73 /* Initialize segment if it has an init function */
74 if (segment->init) {
75 lle_result_t result = segment->init(segment);
76 if (result != LLE_SUCCESS) {
77 return result;
78 }
79 }
80
81 registry->segments[registry->count] = segment;
82 registry->count++;
83
84 return LLE_SUCCESS;
85}

136lle_result_t lle_prompt_context_init(lle_prompt_context_t *ctx) {
137 if (!ctx) {
138 return LLE_ERROR_INVALID_PARAMETER;
139 }
140
141 memset(ctx, 0, sizeof(*ctx));
142
143 /* Get current user */
144 uid_t uid = getuid();
145 ctx->uid = uid;
146 ctx->is_root = (uid == 0);
147
148 struct passwd *pw = getpwuid(uid);
149 if (pw) {
150 snprintf(ctx->username, sizeof(ctx->username), "%s", pw->pw_name);
151 snprintf(ctx->home_dir, sizeof(ctx->home_dir), "%s", pw->pw_dir);
152 } else {
153 snprintf(ctx->username, sizeof(ctx->username), "user");
154 const char *home = getenv("HOME");
155 if (home) {
156 snprintf(ctx->home_dir, sizeof(ctx->home_dir), "%s", home);
157 }
158 }
159
160 /* Get hostname */
161 if (gethostname(ctx->hostname, sizeof(ctx->hostname) - 1) != 0) {
162 snprintf(ctx->hostname, sizeof(ctx->hostname), "localhost");
163 }
164
165 /* Get current directory */
166 lle_prompt_context_refresh_directory(ctx);
167
168 /* Get current time */
169 ctx->current_time = time(NULL);
170 localtime_r(&ctx->current_time, &ctx->current_tm);
171
172 /* Get shell level */
173 const char *shlvl = getenv("SHLVL");
174 ctx->shlvl = shlvl ? atoi(shlvl) : 1;
175
176 /* Check for SSH session */
177 ctx->is_ssh_session =
178 (getenv("SSH_CLIENT") != NULL || getenv("SSH_TTY") != NULL);
179
180 /* Default keymap */
181 snprintf(ctx->keymap, sizeof(ctx->keymap), "emacs");
182
183 /* Terminal capabilities - detect from adaptive terminal system */
184 ctx->has_unicode = true;
185 ctx->terminal_width = 80;
186 ctx->terminal_height = 24;
187
188 /* Use adaptive terminal detection for color capabilities.
189 * Note: The optimized detection returns a cached result that is
190 * managed by the detection system - do NOT destroy it, as that
191 * could cause a double-free if the cache was refreshed. */
192 lle_terminal_detection_result_t *detection = NULL;
193 lle_result_t det_result =
194 lle_detect_terminal_capabilities_optimized(&detection);
195 if (det_result == LLE_SUCCESS && detection) {
196 ctx->has_256_color = detection->supports_256_colors;
197 ctx->has_true_color = detection->supports_truecolor;
198 /* Do NOT call lle_terminal_detection_result_destroy here -
199 * the optimized version manages the cached result internally */
200 } else {
201 /* Default to 256 colors if detection fails */
202 ctx->has_256_color = true;
203 ctx->has_true_color = false;
204 }
205
206 return LLE_SUCCESS;
207}

231lle_result_t lle_prompt_context_refresh_directory(lle_prompt_context_t *ctx) {
232 if (!ctx) {
233 return LLE_ERROR_INVALID_PARAMETER;
234 }
235
236 /* Get current directory */
237 if (!getcwd(ctx->cwd, sizeof(ctx->cwd))) {
238 ctx->cwd[0] = '\0';
239 ctx->cwd_display[0] = '\0';
240 return LLE_ERROR_SYSTEM_CALL;
241 }
242
243 /* Create display version with ~ for home */
244 size_t home_len = strlen(ctx->home_dir);
245 if (home_len > 0 && strncmp(ctx->cwd, ctx->home_dir, home_len) == 0) {
246 if (ctx->cwd[home_len] == '\0') {
247 snprintf(ctx->cwd_display, sizeof(ctx->cwd_display), "~");
248 ctx->cwd_is_home = true;
249 } else if (ctx->cwd[home_len] == '/') {
250 snprintf(ctx->cwd_display, sizeof(ctx->cwd_display), "~%s",
251 ctx->cwd + home_len);
252 ctx->cwd_is_home = false;
253 } else {
254 snprintf(ctx->cwd_display, sizeof(ctx->cwd_display), "%s",
255 ctx->cwd);
256 ctx->cwd_is_home = false;
257 }
258 } else {
259 snprintf(ctx->cwd_display, sizeof(ctx->cwd_display), "%s", ctx->cwd);
260 ctx->cwd_is_home = strcmp(ctx->cwd, ctx->home_dir) == 0;
261 }
262
263 /* Check if at filesystem root */
264 ctx->cwd_is_root = (strcmp(ctx->cwd, "/") == 0);
265
266 /* Check writability */
267 ctx->cwd_is_writable = (access(ctx->cwd, W_OK) == 0);
268
269 /* Check for git repo */
270 char git_dir[PATH_MAX + 8]; /* Extra space for /.git */
271 snprintf(git_dir, sizeof(git_dir), "%s/.git", ctx->cwd);
272 struct stat st;
273 ctx->cwd_is_git_repo = (stat(git_dir, &st) == 0);
274
275 return LLE_SUCCESS;
276}

349static lle_result_t segment_directory_render(const lle_prompt_segment_t *self,
350 const lle_prompt_context_t *ctx,
351 const lle_theme_t *theme,
352 lle_segment_output_t *output) {
353 (void)self;
354
355 const char *display =
356 strlen(ctx->cwd_display) > 0 ? ctx->cwd_display : ctx->cwd;
357
358 /* Check if we should apply path_root color (when at filesystem root) */
359 if (theme && ctx->cwd_is_root &&
360 theme->colors.path_root.mode != LLE_COLOR_MODE_NONE) {
361 static const char *reset = "\033[0m";
362 char color_code[32];
363 lle_color_to_ansi(&theme->colors.path_root, true, color_code,
364 sizeof(color_code));
365
366 snprintf(output->content, sizeof(output->content), "%s%.470s%s", color_code,
367 display, reset);
368 output->content_len = strlen(output->content);
369 output->visual_width = lle_utf8_string_width(display, strlen(display));
370 } else {
371 /* Standard path display */
372 size_t display_len = strlen(display);
373 size_t copy_len = (display_len < sizeof(output->content) - 1)
374 ? display_len
375 : sizeof(output->content) - 1;
376 memcpy(output->content, display, copy_len);
377 output->content[copy_len] = '\0';
378
379 output->content_len = copy_len;
380 output->visual_width = lle_utf8_string_width(output->content, copy_len);
381 }
382
383 output->is_empty = (output->content_len == 0);
384 output->needs_separator = true;
385
386 return LLE_SUCCESS;
387}

842/**
843 * @brief Fetch git status and populate state
844 */
845static void fetch_git_status(segment_git_state_t *state) {
846 if (!state)
847 return;
848
849 /* Check if in git repo */
850 state->is_repo = is_in_git_repo();
851 if (!state->is_repo) {
852 state->branch[0] = '\0';
853 state->staged = 0;
854 state->unstaged = 0;
855 state->untracked = 0;
856 state->ahead = 0;
857 state->behind = 0;
858 return;
859 }
860
861 /* Get branch name */
862 run_git_command("symbolic-ref --short HEAD", state->branch,
863 sizeof(state->branch));
864 if (state->branch[0] == '\0') {
865 /* Detached HEAD - get short commit hash */
866 run_git_command("rev-parse --short HEAD", state->branch,
867 sizeof(state->branch));
868 }
869
870 /* Get status counts using git status --porcelain */
871 FILE *fp = popen("git status --porcelain 2>/dev/null", "r");
872 if (fp) {
873 char line[512];
874 state->staged = 0;
875 state->unstaged = 0;
876 state->untracked = 0;
877
878 while (fgets(line, sizeof(line), fp)) {
879 if (line[0] == '?') {
880 state->untracked++;
881 } else {
882 if (line[0] != ' ' && line[0] != '?') {
883 state->staged++;
884 }
885 if (line[1] != ' ' && line[1] != '?') {
886 state->unstaged++;
887 }
888 }
889 }
890 pclose(fp);
891 }
892
893 /* Get ahead/behind counts */
894 char ab_output[64];
895 if (run_git_command("rev-list --left-right --count @{upstream}...HEAD",
896 ab_output, sizeof(ab_output)) == 0) {
897 sscanf(ab_output, "%d\t%d", &state->behind, &state->ahead);
898 } else {
899 state->ahead = 0;
900 state->behind = 0;
901 }
902
903 /* Get stash count */
904 FILE *stash_fp = popen("git stash list 2>/dev/null | wc -l", "r");
905 if (stash_fp) {
906 char stash_buf[16];
907 if (fgets(stash_buf, sizeof(stash_buf), stash_fp)) {
908 state->stash_count = atoi(stash_buf);
909 }
910 pclose(stash_fp);
911 }
912
913 /* Check for merge conflicts (unmerged files) */
914 state->has_conflicts = false;
915 FILE *conflict_fp = popen("git ls-files -u 2>/dev/null | head -1", "r");
916 if (conflict_fp) {
917 char conflict_buf[8];
918 if (fgets(conflict_buf, sizeof(conflict_buf), conflict_fp) &&
919 conflict_buf[0]) {
920 state->has_conflicts = true;
921 }
922 /* Drain any remaining output to prevent child from blocking */
923 while (fgets(conflict_buf, sizeof(conflict_buf), conflict_fp)) {
924 }
925 pclose(conflict_fp);
926 }
927
928 state->cache_valid = true;
929}

981/**
982 * @brief Queue an async git status fetch
983 *
984 * Non-blocking - queues request and returns immediately.
985 * If async is not available or queue fails, returns false.
986 */
987static bool queue_async_git_fetch(segment_git_state_t *state) {
988 if (!state || !state->async_initialized || !state->async_worker) {
989 return false;
990 }
991
992 if (!lle_async_worker_is_running(state->async_worker)) {
993 return false;
994 }
995
996 char cwd[PATH_MAX];
997 if (getcwd(cwd, sizeof(cwd)) == NULL) {
998 return false;
999 }
1000
1001 pthread_mutex_lock(&state->async_mutex);
1002
1003 /* Don't queue if already pending for same directory */
1004 if (state->async_pending && strcmp(state->async_cwd, cwd) == 0) {
1005 pthread_mutex_unlock(&state->async_mutex);
1006 return true; /* Already queued */
1007 }
1008
1009 lle_async_request_t *req = lle_async_request_create(LLE_ASYNC_GIT_STATUS);
1010 if (!req) {
1011 pthread_mutex_unlock(&state->async_mutex);
1012 return false;
1013 }
1014
1015 snprintf(req->cwd, sizeof(req->cwd), "%s", cwd);
1016
1017 if (lle_async_worker_submit(state->async_worker, req) == LLE_SUCCESS) {
1018 state->async_pending = true;
1019 snprintf(state->async_cwd, sizeof(state->async_cwd), "%s", cwd);
1020 pthread_mutex_unlock(&state->async_mutex);
1021 return true;
1022 }
1023
1024 lle_async_request_free(req);
1025 pthread_mutex_unlock(&state->async_mutex);
1026 return false;
1027}

1029/**
1030 * @brief Helper to append colored text to buffer
1031 *
1032 * Appends text with optional ANSI color prefix and reset suffix.
1033 *
1034 * @param buf Output buffer
1035 * @param buf_size Buffer size
1036 * @param pos Current position in buffer (updated)
1037 * @param text Text to append
1038 * @param color Color to use (NULL for no color)
1039 * @param reset Reset sequence (NULL to skip reset)
1040 * @return Number of visible characters added (excluding ANSI codes)
1041 */
1042static size_t append_colored(char *buf, size_t buf_size, size_t *pos,
1043 const char *text, const lle_color_t *color,
1044 const char *reset) {
1045 if (*pos >= buf_size - 1)
1046 return 0;
1047
1048 size_t remaining = buf_size - *pos - 1;
1049 size_t visible = 0;
1050
1051 /* Add color code if provided */
1052 if (color && color->mode != LLE_COLOR_MODE_NONE) {
1053 char color_code[32];
1054 lle_color_to_ansi(color, true, color_code, sizeof(color_code));
1055 size_t code_len = strlen(color_code);
1056 if (code_len < remaining) {
1057 memcpy(buf + *pos, color_code, code_len);
1058 *pos += code_len;
1059 remaining -= code_len;
1060 }
1061 }
1062
1063 /* Add text */
1064 size_t text_len = strlen(text);
1065 if (text_len < remaining) {
1066 memcpy(buf + *pos, text, text_len);
1067 *pos += text_len;
1068 remaining -= text_len;
1069 visible = lle_utf8_string_width(text, text_len);
1070 }
1071
1072 /* Add reset if color was used */
1073 if (color && color->mode != LLE_COLOR_MODE_NONE && reset) {
1074 size_t reset_len = strlen(reset);
1075 if (reset_len < remaining) {
1076 memcpy(buf + *pos, reset, reset_len);
1077 *pos += reset_len;
1078 }
1079 }
1080
1081 buf[*pos] = '\0';
1082 return visible;
1083}
