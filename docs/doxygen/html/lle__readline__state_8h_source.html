1/**
2 * @file lle_readline_state.h
3 * @brief LLE Readline State Machine
4 *
5 * Provides explicit state management for the readline input loop with
6 * guaranteed exit paths. This replaces implicit flag-based state tracking
7 * with a formal state machine that ensures Ctrl+C and Ctrl+G can always
8 * exit from any state.
9 *
10 * State Machine Design:
11 * - Normal states: IDLE, EDITING, COMPLETION, SEARCH, MULTILINE, QUOTED_INSERT
12 * - Terminal states: DONE, ABORT, EOF, TIMEOUT, ERROR
13 * - INVARIANT: Transitions to terminal states ALWAYS succeed
14 * - INVARIANT: Ctrl+C/Ctrl+G can exit from ANY non-terminal state
15 *
16 * @see src/lle/lle_readline_state.c for implementation
17 * @see src/lle/lle_readline.c for integration
18 */
19

27/**
28 * @brief Readline state machine states
29 *
30 * States are ordered such that terminal states come last, allowing
31 * a simple >= comparison to check if readline should exit.
32 */
33typedef enum {
34 /* ========== Normal (Non-Terminal) States ========== */
35
36 /** Initial state before any input received */
37 LLE_READLINE_STATE_IDLE = 0,
38
39 /** Normal text editing mode (default operating state) */
40 LLE_READLINE_STATE_EDITING,
41
42 /** Completion menu is visible and accepting navigation */
43 LLE_READLINE_STATE_COMPLETION,
44
45 /** Interactive history search active (Ctrl+R) */
46 LLE_READLINE_STATE_SEARCH,
47
48 /** Multiline input mode (incomplete command, awaiting more input) */
49 LLE_READLINE_STATE_MULTILINE,
50
51 /** Quoted insert mode (Ctrl+V - next char inserted literally) */
52 LLE_READLINE_STATE_QUOTED_INSERT,
53
54 /* ========== Terminal States (Readline Exits) ========== */
55
56 /** Line accepted successfully (Enter pressed, input complete) */
57 LLE_READLINE_STATE_DONE,
58
59 /** Line aborted by user (Ctrl+G or Ctrl+C) */
60 LLE_READLINE_STATE_ABORT,
61
62 /** EOF received (Ctrl+D on empty line) */
63 LLE_READLINE_STATE_EOF,
64
65 /** Watchdog timeout - no input for too long */
66 LLE_READLINE_STATE_TIMEOUT,
67
68 /** Unrecoverable error occurred */
69 LLE_READLINE_STATE_ERROR,
70
71 /** Sentinel for iteration/validation */
72 LLE_READLINE_STATE_COUNT
73} lle_readline_state_t;
74
75/**
76 * @brief First terminal state (for range checking)
77 */
78#define LLE_READLINE_STATE_FIRST_TERMINAL LLE_READLINE_STATE_DONE
79
80/**
81 * @brief Check if state is a terminal state (readline should exit)
82 *
83 * Terminal states indicate that the readline loop should exit and
84 * return control to the caller.
85 *
86 * @param state State to check
87 * @return true if state is terminal, false otherwise
88 */
89static inline bool lle_readline_state_is_terminal(lle_readline_state_t state) {
90 return state >= LLE_READLINE_STATE_FIRST_TERMINAL;
91}
92
93/**
94 * @brief Check if state is a normal (non-terminal) state
95 *
96 * @param state State to check
97 * @return true if state is normal, false if terminal
98 */
99static inline bool lle_readline_state_is_normal(lle_readline_state_t state) {
100 return state < LLE_READLINE_STATE_FIRST_TERMINAL;
101}
102
103/**
104 * @brief Check if state allows text input
105 *
106 * Some states like COMPLETION and SEARCH have special input handling
107 * but still allow transitioning back to EDITING.
108 *
109 * @param state State to check
110 * @return true if text input is accepted
111 */
112static inline bool
113lle_readline_state_accepts_input(lle_readline_state_t state) {
114 switch (state) {
115 case LLE_READLINE_STATE_EDITING:
116 case LLE_READLINE_STATE_COMPLETION:
117 case LLE_READLINE_STATE_SEARCH:
118 case LLE_READLINE_STATE_MULTILINE:
119 case LLE_READLINE_STATE_QUOTED_INSERT:
120 return true;
121 default:
122 return false;
123 }
124}

132const char *lle_readline_state_name(lle_readline_state_t state);

134/**
135 * @brief Validate that a state value is within valid range
136 *
137 * @param state State to validate
138 * @return true if valid, false if out of range
139 */
140static inline bool lle_readline_state_is_valid(lle_readline_state_t state) {
141 return state >= LLE_READLINE_STATE_IDLE && state < LLE_READLINE_STATE_COUNT;
142}

147/**
148 * @brief Transition to a new state with validation
149 *
150 * Performs the state transition and any necessary cleanup/setup for
151 * the new state. Validates that the transition is legal.
152 *
153 * INVARIANT: Transitions to terminal states (ABORT, EOF, TIMEOUT, ERROR)
154 * ALWAYS succeed regardless of current state. This is the escape hatch.
155 *

160 */
