1/*
2 * unicode_compare.h - Unicode-aware string comparison for LLE
3 *
4 * Provides Unicode-normalized string comparison using NFC (Canonical
5 * Composition) normalization. This ensures that equivalent Unicode sequences
6 * compare as equal regardless of their byte representation.
7 *
8 * Example: "é" (U+00E9 precomposed) == "e" + "́" (U+0065 + U+0301 decomposed)
9 *
10 * For shell commands, most input is ASCII, but this handles edge cases where
11 * users type accented characters, emoji, or other Unicode that may have
12 * multiple valid representations.
13 */
14

41/*
42 * Compare two UTF-8 strings with Unicode normalization
43 *
44 * This function compares strings using NFC normalization to ensure
45 * equivalent Unicode sequences compare as equal. It handles:
46 * - Precomposed vs decomposed characters (é vs e + combining acute)
47 * - Different byte representations of the same codepoint
48 * - Optional case-insensitive comparison
49 * - Optional whitespace trimming
50 *
51 * Parameters:
52 * str1 - First UTF-8 string
53 * str2 - Second UTF-8 string
54 * options - Comparison options (NULL for defaults)
55 *
56 * Returns:
57 * true if strings are equal after normalization, false otherwise
58 */

164/*
165 * Check if one UTF-8 string is a prefix of another (Unicode-aware)
166 *
167 * This function performs Unicode-normalized prefix matching using NFC

169 * - Precomposed vs decomposed characters (café vs cafe + combining acute)
170 * - Different byte representations of equivalent Unicode sequences
171 * - Grapheme cluster boundaries (won't match partial graphemes)

173 * For shell autosuggestions, this ensures that typing "café" will match

175 *
176 * Parameters:
177 * prefix - The prefix string to check (UTF-8)
178 * prefix_len - Length of prefix in bytes
179 * str - The string to check against (UTF-8)
180 * str_len - Length of string in bytes
181 * options - Comparison options (NULL for defaults with normalization)
182 *
183 * Returns:
184 * true if prefix is a Unicode-normalized prefix of str, false otherwise
185 */

187 const char *str, size_t str_len,
188 const lle_unicode_compare_options_t *options);
