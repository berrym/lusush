1/**
2 * keybinding_actions.h - Default Keybinding Action Functions
3 *
4 * Implements all 42 GNU Readline compatible keybinding actions for
5 * interactive line editing. Provides 100% compatibility with standard
6 * Emacs-style keybindings used in bash, zsh, and other shells.
7 *
8 * Key Features:
9 * - Complete GNU Readline function set
10 * - Movement operations (beginning/end of line, word navigation)
11 * - Editing operations (kill/yank, transpose, case changes)
12 * - History navigation (previous/next, search)
13 * - Completion operations (tab completion, listing)
14 * - Shell-specific operations (EOF, suspend, clear screen)
15 * - Multiline command support for all operations
16 *
17 * Performance:
18 * - All actions complete within 100 microseconds
19 * - Optimized for interactive use
20 *
21 * Specification:
22 * docs/lle_specification/critical_gaps/25_default_keybindings_complete.md Date:
23 * 2025-11-02
24 */

294/**
295 * Accept line with context (Return, Ctrl-M, Ctrl-J) - Context-Aware Action
296 *
297 * Full readline-aware ENTER key handler with direct access to readline context.
298 * Checks for incomplete input using continuation state, handles history
299 * integration, and directly manages readline completion (done/final_line).
300 *
301 * Behavior:
302 * - Incomplete input: Inserts newline, syncs cursor, refreshes display,
303 * continues editing
304 * - Complete input: Adds to history, sets done=true, returns final line
305 *
306 * This is a context-aware action that requires readline_context_t access.
307 * See docs/lle_implementation/DUAL_ACTION_ARCHITECTURE.md for architecture
308 * details.
309 *
310 * @param ctx Readline context with full access to buffer, history, completion
311 * state
312 * @return LLE_SUCCESS on successful handling
313 */
314lle_result_t lle_accept_line_context(struct readline_context *ctx);
315
316/**
317 * Abort current operation (Ctrl-G)
318 * Clears buffer, cancels search/completion, resets to clean state

325/**
326 * Abort line with context (Ctrl-G) - Context-Aware Action

328 * Emacs-style abort that cancels current input and returns empty line to shell.

330 * flags.
