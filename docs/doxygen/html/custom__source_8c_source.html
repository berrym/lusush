38/**
39 * Custom source entry - wraps user source with internal state
40 */
41typedef struct custom_source_entry {
42 lle_custom_completion_source_t
43 source; /* Copy of user's source definition */
44 char *name_copy; /* Owned copy of name string */
45 char *description_copy; /* Owned copy of description */
46 bool registered; /* Currently registered in manager */
47} custom_source_entry_t;
48
49/**
50 * Global custom source registry
51 */
52static struct {
53 custom_source_entry_t entries[MAX_CUSTOM_SOURCES];
54 size_t count;
55 lle_source_manager_t *source_manager; /* Active source manager */
56 lle_memory_pool_t *pool; /* Memory pool for allocations */
57 pthread_mutex_t mutex; /* Thread safety */
58 bool initialized;
59} g_custom_registry = {.count = 0,
60 .source_manager = NULL,
61 .pool = NULL,
62 .mutex = PTHREAD_MUTEX_INITIALIZER,
63 .initialized = false};

70/**
71 * Find entry by name
72 */
73static custom_source_entry_t *find_entry_by_name(const char *name) {
74 for (size_t i = 0; i < g_custom_registry.count; i++) {
75 if (g_custom_registry.entries[i].registered &&
76 strcmp(g_custom_registry.entries[i].name_copy, name) == 0) {
77 return &g_custom_registry.entries[i];
78 }
79 }
80 return NULL;
81}

83/**
84 * Wrapper generate function - adapts custom source callback to internal format
85 *
86 * The internal source manager passes (pool, context, prefix, result) but
87 * custom sources expect (user_data, context, prefix, result).
88 */
89static lle_result_t
90custom_generate_wrapper(lle_memory_pool_t *pool,
91 const lle_context_analyzer_t *context,
92 const char *prefix, lle_completion_result_t *result) {
93
94 (void)pool; /* Pool is available in result->pool if needed */
95
96 /*
97 * We need to find which custom source this call is for.
98 * The internal source manager doesn't pass user_data through its callback,
99 * so we use a different approach: the custom source stores its user_data
100 * in the lle_completion_source_t.user_data field.
101 *
102 * For now, we iterate through registered sources to find a match.
103 * This is O(n) but n is small (max 32).
104 */
105
106 /* Find which custom source is being queried based on the context */
107 pthread_mutex_lock(&g_custom_registry.mutex);
108
109 for (size_t i = 0; i < g_custom_registry.count; i++) {
110 custom_source_entry_t *entry = &g_custom_registry.entries[i];
111 if (!entry->registered) {
112 continue;
113 }
114
115 /* Check if this source is applicable */
116 if (entry->source.is_applicable) {
117 if (!entry->source.is_applicable(entry->source.user_data,
118 context)) {
119 continue;
120 }
121 }
122
123 /* Generate completions from this source */
124 lle_result_t res = entry->source.generate(entry->source.user_data,
125 context, prefix, result);
126
127 if (res != LLE_SUCCESS) {
128 /* Log but continue - don't fail the whole query */
129 }
130 }
131
132 pthread_mutex_unlock(&g_custom_registry.mutex);
133 return LLE_SUCCESS;
134}

136/**
137 * Wrapper applicability function - custom sources are always checked
138 *
139 * Individual applicability is checked in custom_generate_wrapper.
140 */
141static bool custom_applicable_wrapper(const lle_context_analyzer_t *context) {
142 (void)context;
143 return true; /* Always query custom sources; they filter internally */
144}

151lle_result_t lle_custom_source_init(lle_source_manager_t *manager,
152 lle_memory_pool_t *pool) {
153 if (!manager || !pool) {
154 return LLE_ERROR_INVALID_PARAMETER;
155 }
156
157 pthread_mutex_lock(&g_custom_registry.mutex);
158
159 if (g_custom_registry.initialized) {
160 /* Already initialized - update manager reference */
161 g_custom_registry.source_manager = (lle_source_manager_t *)manager;
162 g_custom_registry.pool = (lle_memory_pool_t *)pool;
163 pthread_mutex_unlock(&g_custom_registry.mutex);
164 return LLE_SUCCESS;
165 }
166
167 g_custom_registry.source_manager = (lle_source_manager_t *)manager;
168 g_custom_registry.pool = (lle_memory_pool_t *)pool;
169 g_custom_registry.count = 0;
170 g_custom_registry.initialized = true;
171
172 /*
173 * Register a single "custom" meta-source in the manager.
174 * This source dispatches to all registered custom sources.
175 */
176 lle_result_t res = lle_source_manager_register(
177 manager, LLE_SOURCE_CUSTOM, "custom", custom_generate_wrapper,
178 custom_applicable_wrapper);
179
180 pthread_mutex_unlock(&g_custom_registry.mutex);
181 return res;
182}

184void lle_custom_source_shutdown(void) {
185 pthread_mutex_lock(&g_custom_registry.mutex);
186
187 /* Call cleanup callbacks for all registered sources */
188 for (size_t i = 0; i < g_custom_registry.count; i++) {
189 custom_source_entry_t *entry = &g_custom_registry.entries[i];
190 if (entry->registered) {
191 if (entry->source.cleanup) {
192 entry->source.cleanup(entry->source.user_data);
193 }
194 free(entry->name_copy);
195 free(entry->description_copy);
196 entry->registered = false;
197 }
198 }
199
200 g_custom_registry.count = 0;
201 g_custom_registry.source_manager = NULL;
202 g_custom_registry.pool = NULL;
203 g_custom_registry.initialized = false;
204
205 pthread_mutex_unlock(&g_custom_registry.mutex);
206}

213lle_result_t
214lle_completion_register_source(const lle_custom_completion_source_t *source) {
215
216 if (!source || !source->name || !source->generate) {
217 return LLE_ERROR_INVALID_PARAMETER;
218 }
219
220 pthread_mutex_lock(&g_custom_registry.mutex);
221
222 if (!g_custom_registry.initialized) {
223 pthread_mutex_unlock(&g_custom_registry.mutex);
224 return LLE_ERROR_NOT_INITIALIZED;
225 }
226
227 /* Check for duplicate name */
228 if (find_entry_by_name(source->name)) {
229 pthread_mutex_unlock(&g_custom_registry.mutex);
230 return LLE_ERROR_ALREADY_EXISTS;
231 }
232
233 /* Check capacity */
234 if (g_custom_registry.count >= MAX_CUSTOM_SOURCES) {
235 pthread_mutex_unlock(&g_custom_registry.mutex);
236 return LLE_ERROR_BUFFER_OVERFLOW;
237 }
238
239 /* Create entry */
240 custom_source_entry_t *entry =
241 &g_custom_registry.entries[g_custom_registry.count];
242
243 /* Copy strings */
244 entry->name_copy = strdup(source->name);
245 if (!entry->name_copy) {
246 pthread_mutex_unlock(&g_custom_registry.mutex);
247 return LLE_ERROR_OUT_OF_MEMORY;
248 }
249
250 if (source->description) {
251 entry->description_copy = strdup(source->description);
252 if (!entry->description_copy) {
253 free(entry->name_copy);
254 pthread_mutex_unlock(&g_custom_registry.mutex);
255 return LLE_ERROR_OUT_OF_MEMORY;
256 }
257 } else {
258 entry->description_copy = NULL;
259 }
260
261 /* Copy source definition */
262 entry->source = *source;
263 entry->source.name = entry->name_copy;
264 entry->source.description = entry->description_copy;
265 entry->source.priority = source->priority > 0 ? source->priority : 500;
266 entry->registered = true;
267
268 g_custom_registry.count++;
269
270 pthread_mutex_unlock(&g_custom_registry.mutex);
271 return LLE_SUCCESS;
272}

274lle_result_t lle_completion_unregister_source(const char *name) {
275 if (!name) {
276 return LLE_ERROR_INVALID_PARAMETER;
277 }
278
279 pthread_mutex_lock(&g_custom_registry.mutex);
280
281 custom_source_entry_t *entry = find_entry_by_name(name);
282 if (!entry) {
283 pthread_mutex_unlock(&g_custom_registry.mutex);
284 return LLE_ERROR_NOT_FOUND;
285 }
286
287 /* Call cleanup if provided */
288 if (entry->source.cleanup) {
289 entry->source.cleanup(entry->source.user_data);
290 }
291
292 /* Free strings */
293 free(entry->name_copy);
294 free(entry->description_copy);
295
296 /* Mark as unregistered */
297 entry->registered = false;
298
299 pthread_mutex_unlock(&g_custom_registry.mutex);
300 return LLE_SUCCESS;
301}

448lle_result_t lle_completion_add_item(lle_completion_result_t *result,
449 const char *text, const char *suffix,
450 const char *description, int score) {
451 if (!result || !text) {
452 return LLE_ERROR_INVALID_PARAMETER;
453 }
454
455 /* Check capacity */
456 if (result->count >= result->capacity) {
457 return LLE_ERROR_BUFFER_OVERFLOW;
458 }
459
460 lle_completion_item_t *item = &result->items[result->count];
461
462 /* Copy text - use pool if available */
463 if (result->memory_pool) {
464 char *text_copy = lle_pool_alloc(strlen(text) + 1);
465 if (!text_copy) {
466 return LLE_ERROR_OUT_OF_MEMORY;
467 }
468 strcpy(text_copy, text);
469 item->text = text_copy;
470 item->owns_text = false; /* Pool-allocated, not individually owned */
471
472 if (suffix) {
473 char *suffix_copy = lle_pool_alloc(strlen(suffix) + 1);
474 if (suffix_copy) {
475 strcpy(suffix_copy, suffix);
476 item->suffix = suffix_copy;
477 } else {
478 item->suffix = strdup(" "); /* Fallback */
479 item->owns_suffix = true;
480 }
481 } else {
482 item->suffix = strdup(" ");
483 item->owns_suffix = true;
484 }
485 item->owns_suffix = false;
486
487 if (description) {
488 char *desc_copy = lle_pool_alloc(strlen(description) + 1);
489 if (desc_copy) {
490 strcpy(desc_copy, description);
491 item->description = desc_copy;
492 } else {
493 item->description = NULL;
494 }
495 } else {
496 item->description = NULL;
497 }
498 item->owns_description = false;
499 } else {
500 /* No pool - duplicate strings (caller can free theirs) */
501 item->text = strdup(text);
502 item->owns_text = true;
503 item->suffix = strdup(suffix ? suffix : " ");
504 item->owns_suffix = true;
505 item->description = description ? strdup(description) : NULL;
506 item->owns_description = (description != NULL);
507 }
508
509 item->type = LLE_COMPLETION_TYPE_CUSTOM;
510 item->relevance_score = score > 0 ? score : 500;
511 item->type_indicator = NULL; /* Will be set by type system */
512
513 result->count++;
514 result->custom_count++;
515
516 return LLE_SUCCESS;
517}
