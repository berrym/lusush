17 if (!queue || capacity == 0) {

73/*
74 * Enqueue event
75 */
76lle_result_t lle_event_enqueue(lle_event_system_t *system, lle_event_t *event) {
77 if (!system || !system->queue || !event) {
78 return LLE_ERROR_INVALID_PARAMETER;
79 }
80
81 /* Phase 2: Route to priority queue for CRITICAL priority events only */
82 lle_event_queue_t *queue;
83 if (system->use_priority_queue &&
84 event->priority == LLE_PRIORITY_CRITICAL) {
85 queue = system->priority_queue;
86 __atomic_fetch_add(&system->priority_events_queued, 1,
87 __ATOMIC_SEQ_CST);
88 } else {
89 queue = system->queue;
90 }
91
92 pthread_mutex_lock(&queue->mutex);
93
94 /* Check if queue is full */
95 if (queue->count >= queue->capacity) {
96 pthread_mutex_unlock(&queue->mutex);
97 __atomic_fetch_add(&system->events_dropped, 1, __ATOMIC_SEQ_CST);
98 return LLE_ERROR_QUEUE_FULL;
99 }
100
101 /* Add event to tail */
102 queue->events[queue->tail] = event;
103 queue->tail = (queue->tail + 1) % queue->capacity;
104 queue->count++;
105
106 /* Phase 2: Mark event as queued and update statistics */
107 event->flags |= LLE_EVENT_FLAG_QUEUED;
108 __atomic_fetch_add(&system->events_by_priority[event->priority], 1,
109 __ATOMIC_SEQ_CST);
110
111 pthread_mutex_unlock(&queue->mutex);
112
113 return LLE_SUCCESS;
114}

116/*
117 * Dequeue event
118 */
119lle_result_t lle_event_dequeue(lle_event_system_t *system,
120 lle_event_t **event) {
121 if (!system || !system->queue || !event) {
122 return LLE_ERROR_INVALID_PARAMETER;
123 }
124
125 /* Phase 2: Check priority queue first if enabled */
126 lle_event_queue_t *queue;
127 bool from_priority_queue = false;
128
129 if (system->use_priority_queue && system->priority_queue) {
130 pthread_mutex_lock(&system->priority_queue->mutex);
131 if (system->priority_queue->count > 0) {
132 queue = system->priority_queue;
133 from_priority_queue = true;
134 } else {
135 pthread_mutex_unlock(&system->priority_queue->mutex);
136 queue = system->queue;
137 pthread_mutex_lock(&queue->mutex);
138 }
139 } else {
140 queue = system->queue;
141 pthread_mutex_lock(&queue->mutex);
142 }
143
144 /* Check if queue is empty */
145 if (queue->count == 0) {
146 pthread_mutex_unlock(&queue->mutex);
147 *event = NULL;
148 return LLE_ERROR_QUEUE_EMPTY;
149 }
150
151 /* Remove event from head */
152 *event = queue->events[queue->head];
153 queue->events[queue->head] = NULL;
154 queue->head = (queue->head + 1) % queue->capacity;
155 queue->count--;
156
157 /* Phase 2: Update event flags and statistics */
158 if (*event) {
159 (*event)->flags &= ~LLE_EVENT_FLAG_QUEUED; /* Clear queued flag */
160 if (from_priority_queue) {
161 __atomic_fetch_add(&system->priority_events_processed, 1,
162 __ATOMIC_SEQ_CST);
163 }
164 }
165
166 pthread_mutex_unlock(&queue->mutex);
167
168 return LLE_SUCCESS;
169}

174size_t lle_event_queue_size(lle_event_system_t *system) {
175 if (!system || !system->queue) {

189 count += system->priority_queue->count;
190 pthread_mutex_unlock(&system->priority_queue->mutex);
191 }
192

206bool lle_event_queue_full(lle_event_system_t *system) {
207 if (!system || !system->queue) {
208 return false;
209 }
210
211 bool main_full = false;
212 bool priority_full = false;
213
214 /* Check main queue */
215 pthread_mutex_lock(&system->queue->mutex);
216 main_full = (system->queue->count >= system->queue->capacity);
217 pthread_mutex_unlock(&system->queue->mutex);
218
219 /* Phase 2: Check priority queue if enabled */
220 if (system->use_priority_queue && system->priority_queue) {
221 pthread_mutex_lock(&system->priority_queue->mutex);
222 priority_full =
223 (system->priority_queue->count >= system->priority_queue->capacity);
224 pthread_mutex_unlock(&system->priority_queue->mutex);
225
226 /* Both queues must be full for system to be considered full */
227 return main_full && priority_full;
228 }
