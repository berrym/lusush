1/**
2 * @file builtin_widgets.c
3 * @brief LLE Built-in Widget Implementations
4 *
5 * Provides core widgets that are automatically registered when the editor
6 * is created. These widgets provide essential editing operations that can
7 * be bound to keys, triggered by hooks, or invoked programmatically.
8 *
9 * Specification: docs/lle_specification/07_extensibility_framework_complete.md
10 * Section: 4 - Widget System Architecture
11 *
12 * Widget Categories:
13 * - Movement widgets: cursor navigation
14 * - Editing widgets: text manipulation
15 * - History widgets: command history navigation
16 * - Display widgets: screen refresh and clearing
17 * - Completion widgets: tab completion operations
18 *
19 * Naming Convention (ZSH-style):
20 * - lowercase-with-hyphens for all widget names
21 * - e.g., "forward-char", "backward-word", "kill-line"
22 */

299/**
300 * @brief Apply transient prompt - simplify prompt in scrollback
301 *
302 * This widget is triggered by the LINE_ACCEPTED hook, which fires after
303 * the user presses Enter but before the cursor moves to the output area.
304 * At this point:
305 * - Cursor is at end of command line
306 * - Screen buffer still has valid prompt metrics
307 * - Display controller can re-render with transient prompt
308 *
309 * The transient prompt replaces the fancy multi-line prompt with a minimal
310 * version (e.g., " ") in the terminal scrollback, reducing visual clutter.
311 *
312 * This implementation uses dc_apply_transient_prompt() which goes through
313 * the screen buffer system, respecting the LLE architecture.
314 */
315static lle_result_t widget_transient_prompt(lle_editor_t *editor,
316 void *user_data) {
317 (void)user_data;
318
319 /* Get shell integration for composer access */
320 if (!g_lle_integration || !g_lle_integration->prompt_composer) {
321 return LLE_SUCCESS; /* Graceful degradation */
322 }
323
324 lle_prompt_composer_t *composer = g_lle_integration->prompt_composer;
325
326 /* Check if transient prompts are enabled */
327 if (!composer->config.enable_transient) {
328 return LLE_SUCCESS;
329 }
330
331 /* Get active theme */
332 const lle_theme_t *theme = lle_theme_registry_get_active(composer->themes);
333 if (!theme || !theme->layout.enable_transient ||
334 theme->layout.transient_format[0] == '\0') {
335 return LLE_SUCCESS;
336 }
337
338 /* Render transient format using template engine */
339 char transient_output[LLE_TRANSIENT_OUTPUT_MAX];
340 lle_result_t result = lle_composer_render_template(
341 composer, theme->layout.transient_format, transient_output,
342 sizeof(transient_output));
343
344 if (result != LLE_SUCCESS) {
345 return LLE_SUCCESS; /* Graceful degradation on render failure */
346 }
347
348 /* Get command text from editor buffer */
349 const char *command_text = NULL;
350 if (editor && editor->buffer && editor->buffer->data) {
351 command_text = editor->buffer->data;
352 }
353
354 /* Apply transient prompt through display controller (screen buffer) */
355 dc_apply_transient_prompt(transient_output, command_text);
356
357 return LLE_SUCCESS;
358}

544/**
545 * @brief Register built-in widget hooks
546 *
547 * Registers widgets for their associated lifecycle hooks. Called after
548 * both widget registry and hooks manager are initialized.
549 *
550 * @param hooks_manager Widget hooks manager
551 * @return LLE_SUCCESS on success, error code if any registration fails
552 */
553lle_result_t
554lle_register_builtin_widget_hooks(lle_widget_hooks_manager_t *hooks_manager) {
555 if (!hooks_manager) {
556 return LLE_ERROR_INVALID_PARAMETER;
557 }
558
559 lle_result_t result;
560
561 /* Register transient-prompt widget for LINE_ACCEPTED hook
562 * This applies transient prompt when user presses Enter, before
563 * cursor moves to output area (Spec 25 Section 12) */
564 result = lle_widget_hook_register(hooks_manager, LLE_HOOK_LINE_ACCEPTED,
565 "transient-prompt");
566 if (result != LLE_SUCCESS && result != LLE_ERROR_NOT_FOUND) {
567 /* NOT_FOUND is acceptable if shell integration not available */
568 return result;
569 }
570
571 return LLE_SUCCESS;
572}
