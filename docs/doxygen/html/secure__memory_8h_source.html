60/**
61 * Securely wipe memory contents
62 *
63 * Guarantees that the memory wipe will not be optimized away by the compiler.
64 * Uses explicit_bzero() if available, falls back to volatile pointer trick.
65 *
66 * @param ptr Pointer to memory to wipe
67 * @param len Length in bytes
68 */
69static inline void lle_secure_wipe(void *ptr, size_t len) {
70 if (!ptr || len == 0) {
71 return;
72 }
73
74#if LLE_HAVE_EXPLICIT_BZERO
75 /* Use explicit_bzero if available - guaranteed not to be optimized away */
76 explicit_bzero(ptr, len);
77#else
78 /* Portable fallback: volatile pointer prevents compiler optimization */
79 volatile unsigned char *p = (volatile unsigned char *)ptr;
80 while (len--) {
81 *p++ = 0;
82 }
83#endif
84}

86/* ============================================================================
87 * MEMORY LOCKING (PREVENT SWAPPING)
88 * ============================================================================

91/**
92 * Lock memory to prevent swapping to disk
93 *
94 * Uses mlock() on POSIX systems. On systems without mlock or when
95 * mlock fails (insufficient privileges), returns false but this is
96 * not a fatal error - the buffer can still be used, just without
97 * the anti-swap protection.
98 *
99 * @param addr Address of memory to lock
100 * @param len Length in bytes
101 * @return true if locked successfully, false otherwise
102 */
103static inline bool lle_memory_lock(void *addr, size_t len) {
104 if (!addr || len == 0) {
105 return false;
106 }
107
108#if LLE_HAVE_MLOCK
109 /* Attempt to lock - may fail due to:
110 * - Insufficient privileges (need CAP_IPC_LOCK on Linux)
111 * - Exceeding RLIMIT_MEMLOCK
112 * - System doesn't support mlock
113 */
114 return mlock(addr, len) == 0;
115#else
116 /* Platform doesn't have mlock - not an error, just unavailable */
117 (void)addr;
118 (void)len;
119 return false;
120#endif
121}
122
123/**
124 * Unlock previously locked memory
125 *
126 * Uses munlock() on POSIX systems. Should be called when memory
127 * no longer needs to be protected from swapping.
128 *
129 * @param addr Address of memory to unlock
130 * @param len Length in bytes
131 * @return true if unlocked successfully, false otherwise
132 */
133static inline bool lle_memory_unlock(void *addr, size_t len) {
134 if (!addr || len == 0) {
135 return true; /* Nothing to unlock */
136 }
137
138#if LLE_HAVE_MLOCK
139 return munlock(addr, len) == 0;
140#else
141 /* Platform doesn't have munlock - no-op, return success */
142 (void)addr;
143 (void)len;
144 return true;
145#endif
146}
