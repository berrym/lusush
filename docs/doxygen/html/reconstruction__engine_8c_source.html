110lle_result_t lle_reconstruction_engine_normalize_whitespace(
111 lle_reconstruction_engine_t *engine, const char *command_text,
112 size_t command_length, char **normalized_text, size_t *normalized_length) {
113 if (!engine || !engine->active || !command_text || !normalized_text ||
114 !normalized_length) {
115 return LLE_ERROR_INVALID_PARAMETER;
116 }
117
118 /* Allocate output buffer (same size as input is sufficient) */
119 char *output = lle_pool_alloc(command_length + 1);
120 if (!output) {
121 return LLE_ERROR_OUT_OF_MEMORY;
122 }
123
124 size_t out_pos = 0;
125 bool last_was_space = false;
126 bool in_quote = false;
127 char quote_char = '\0';
128
129 for (size_t i = 0; i < command_length; i++) {
130 char c = command_text[i];
131
132 /* Handle quotes */
133 if (!in_quote && (c == '"' || c == '\'')) {
134 in_quote = true;
135 quote_char = c;
136 output[out_pos++] = c;
137 last_was_space = false;
138 continue;
139 } else if (in_quote && c == quote_char) {
140 in_quote = false;
141 quote_char = '\0';
142 output[out_pos++] = c;
143 last_was_space = false;
144 continue;
145 }
146
147 /* Don't normalize inside quotes */
148 if (in_quote) {
149 output[out_pos++] = c;
150 last_was_space = false;
151 continue;
152 }
153
154 /* Normalize whitespace outside quotes */
155 if (isspace(c)) {
156 if (c == '\n') {
157 /* Preserve newlines */
158 output[out_pos++] = '\n';
159 last_was_space = false;
160 } else if (!last_was_space) {
161 /* Replace multiple spaces with single space */
162 output[out_pos++] = ' ';
163 last_was_space = true;
164 }
165 } else {
166 output[out_pos++] = c;
167 last_was_space = false;
168 }
169 }
170
171 output[out_pos] = '\0';
172
173 *normalized_text = output;
174 *normalized_length = out_pos;
175
176 return LLE_SUCCESS;
177}

179lle_result_t lle_reconstruction_engine_apply_indentation(
180 lle_reconstruction_engine_t *engine, lle_command_structure_t *structure,
181 const char *original_text, size_t original_length, char **indented_text,
182 size_t *indented_length) {
183 if (!engine || !engine->active || !structure || !original_text ||
184 !indented_text || !indented_length) {
185 return LLE_ERROR_INVALID_PARAMETER;
186 }
187
188 if (!engine->options.apply_indentation) {
189 /* No indentation requested, just copy */
190 char *output = lle_pool_alloc(original_length + 1);
191 if (!output) {
192 return LLE_ERROR_OUT_OF_MEMORY;
193 }
194 memcpy(output, original_text, original_length);
195 output[original_length] = '\0';
196 *indented_text = output;
197 *indented_length = original_length;
198 return LLE_SUCCESS;
199 }
200
201 /* Parse into lines */
202 lle_multiline_parse_result_t *parse_result = NULL;
203 lle_result_t res = lle_multiline_parser_parse(
204 engine->parser, original_text, original_length, &parse_result);
205
206 if (res != LLE_SUCCESS) {
207 return res;
208 }
209
210 /* Calculate output size */
211 size_t output_size = calculate_indented_size(original_text, original_length,
212 structure, &engine->options);
213
214 if (output_size > engine->options.max_output_length) {
215 lle_multiline_parser_free_result(engine->parser, parse_result);
216 return LLE_ERROR_INVALID_PARAMETER;
217 }
218
219 /* Allocate output buffer */
220 char *output = lle_pool_alloc(output_size + 1);
221 if (!output) {
222 lle_multiline_parser_free_result(engine->parser, parse_result);
223 return LLE_ERROR_OUT_OF_MEMORY;
224 }
225
226 /* Apply indentation to lines */
227 size_t out_len = 0;
228 res = apply_indentation_to_lines(engine, parse_result->first_line,
229 structure, output, &out_len);
230
231 lle_multiline_parser_free_result(engine->parser, parse_result);
232
233 if (res != LLE_SUCCESS) {
234 return res;
235 }
236
237 output[out_len] = '\0';
238
239 *indented_text = output;
240 *indented_length = out_len;
241
242 return LLE_SUCCESS;
243}

245lle_result_t lle_reconstruction_engine_reconstruct(
246 lle_reconstruction_engine_t *engine, const char *command_text,
247 size_t command_length, lle_reconstructed_command_t **result) {
248 if (!engine || !engine->active || !command_text || !result) {
249 return LLE_ERROR_INVALID_PARAMETER;
250 }
251
252 /* Allocate result structure */
253 lle_reconstructed_command_t *recon =
254 lle_pool_alloc(sizeof(lle_reconstructed_command_t));
255 if (!recon) {
256 return LLE_ERROR_OUT_OF_MEMORY;
257 }
258
259 memset(recon, 0, sizeof(lle_reconstructed_command_t));
260
261 /* Analyze structure */
262 lle_command_structure_t *structure = NULL;
263 lle_result_t res = lle_structure_analyzer_analyze(
264 engine->analyzer, command_text, command_length, &structure);
265
266 if (res != LLE_SUCCESS) {
267 return res;
268 }
269
270 /* Normalize whitespace if requested */
271 const char *input_text = command_text;
272 size_t input_length = command_length;
273 char *normalized = NULL;
274
275 if (engine->options.normalize_whitespace) {
276 res = lle_reconstruction_engine_normalize_whitespace(
277 engine, command_text, command_length, &normalized, &input_length);
278
279 if (res == LLE_SUCCESS) {
280 input_text = normalized;
281 }
282 }
283
284 /* Apply indentation if requested */
285 char *output_text = NULL;
286 size_t output_length = 0;
287
288 res = lle_reconstruction_engine_apply_indentation(
289 engine, structure, input_text, input_length, &output_text,
290 &output_length);
291
292 if (res != LLE_SUCCESS) {
293 lle_command_structure_destroy(structure);
294 return res;
295 }
296
297 recon->text = output_text;
298 recon->length = output_length;
299 recon->indentation_applied = engine->options.apply_indentation;
300
301 /* Count lines and build line offset array */
302 size_t line_count = 1;
303 for (size_t i = 0; i < output_length; i++) {
304 if (output_text[i] == '\n') {
305 line_count++;
306 }
307 }
308
309 recon->line_count = line_count;
310 recon->line_offsets = lle_pool_alloc(line_count * sizeof(size_t));
311
312 if (recon->line_offsets) {
313 size_t line_idx = 0;
314 recon->line_offsets[line_idx++] = 0;
315
316 for (size_t i = 0; i < output_length; i++) {
317 if (output_text[i] == '\n' && line_idx < line_count) {
318 recon->line_offsets[line_idx++] = i + 1;
319 }
320 }
321 }
322
323 lle_command_structure_destroy(structure);
324
325 *result = recon;
326 return LLE_SUCCESS;
327}

343static size_t
344calculate_indented_size(const char *text, size_t length,
345 lle_command_structure_t *structure,
346 const lle_reconstruction_options_t *options) {
347 (void)text; /* Length used instead of text scanning */
348 /* Estimate: original size + (max_depth * spaces_per_level * line_count) */
349 size_t line_count = structure->total_lines > 0 ? structure->total_lines : 1;
350 size_t max_indent = structure->max_depth * options->spaces_per_level;
351
352 return length + (max_indent * line_count) +
353 line_count; /* +line_count for newlines */
354}

356static size_t get_indent_for_line(lle_command_structure_t *structure,
357 size_t line_number,
358 const lle_reconstruction_options_t *options) {
359 if (!structure->indentation ||
360 line_number >= structure->indentation->line_count) {
361 return 0;
362 }
363
364 size_t indent_spaces = structure->indentation->level_per_line[line_number];
365
366 /* Convert to indentation levels */
367 size_t indent_level =
368 indent_spaces / (structure->indentation->spaces_per_level > 0
369 ? structure->indentation->spaces_per_level
370 : 2);
371
372 return indent_level * options->spaces_per_level;
373}

375static lle_result_t apply_indentation_to_lines(
376 lle_reconstruction_engine_t *engine, lle_parsed_line_t *first_line,
377 lle_command_structure_t *structure, char *output, size_t *output_length) {
378 size_t out_pos = 0;
379 lle_parsed_line_t *current_line = first_line;
380
381 while (current_line) {
382 /* Get indentation for this line */
383 size_t indent_spaces = get_indent_for_line(
384 structure, current_line->line_number, &engine->options);
385
386 /* Add indentation */
387 for (size_t i = 0; i < indent_spaces; i++) {
388 output[out_pos++] = engine->options.indent_char;
389 }
390
391 /* Find start of non-whitespace content */
392 size_t content_start = 0;
393 while (content_start < current_line->length &&
394 isspace(current_line->content[content_start])) {
395 content_start++;
396 }
397
398 /* Copy line content (without leading whitespace) */
399 size_t content_len = current_line->length - content_start;
400 if (content_len > 0) {
401 memcpy(&output[out_pos], &current_line->content[content_start],
402 content_len);
403 out_pos += content_len;
404 }
405
406 /* Add newline if not last line or if preserve_line_breaks is enabled */
407 if (current_line->next || engine->options.preserve_line_breaks) {
408 output[out_pos++] = '\n';
409 }
410
411 current_line = current_line->next;
412 }
413
414 *output_length = out_pos;
415 return LLE_SUCCESS;
416}
