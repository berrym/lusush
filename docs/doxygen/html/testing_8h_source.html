64/* Assertion Macros */
65#define LLE_ASSERT_EQ(expected, actual, message) \
66 do { \
67 if ((expected) != (actual)) { \
68 lle_test_record_assertion_failure( \
69 ctx, __FILE__, __LINE__, \
70 "Expected: %s, Actual: %s, Message: %s", #expected, #actual, \
71 message); \
72 return LLE_TEST_RESULT_ASSERTION_FAILED; \
73 } \
74 lle_test_record_assertion_success(ctx, __FILE__, __LINE__); \
75 } while (0)
76
77#define LLE_ASSERT_NULL(ptr, message) LLE_ASSERT_EQ(NULL, ptr, message)
78
79#define LLE_ASSERT_NOT_NULL(ptr, message) \
80 do { \
81 if ((ptr) == NULL) { \
82 lle_test_record_assertion_failure( \
83 ctx, __FILE__, __LINE__, \
84 "Expected non-null pointer, Message: %s", message); \
85 return LLE_TEST_RESULT_ASSERTION_FAILED; \
86 } \
87 lle_test_record_assertion_success(ctx, __FILE__, __LINE__); \
88 } while (0)
89
90#define LLE_ASSERT_PERFORMANCE(operation, max_duration_us, message) \
91 do { \
92 uint64_t start_time = lle_get_microsecond_timestamp(); \
93 operation; \
94 uint64_t duration = lle_get_microsecond_timestamp() - start_time; \
95 if (duration > (max_duration_us)) { \
96 lle_test_record_performance_failure( \
97 ctx, __FILE__, __LINE__, \
98 "Performance exceeded: %llu us > %llu us, Message: %s", \
99 duration, (uint64_t)(max_duration_us), message); \
100 return LLE_TEST_RESULT_PERFORMANCE_FAILED; \
101 } \
102 lle_test_record_performance_success(ctx, __FILE__, __LINE__, \
103 duration); \
104 } while (0)
