19/*
20 * Grapheme_Cluster_Break property values as defined in UAX #29
21 */
22typedef enum {
23 GCB_OTHER = 0, // Any (default)
24 GCB_CR, // Carriage Return (U+000D)
25 GCB_LF, // Line Feed (U+000A)
26 GCB_CONTROL, // Control characters
27 GCB_EXTEND, // Grapheme_Extend=Yes or Emoji_Modifier=Yes
28 GCB_ZWJ, // Zero Width Joiner (U+200D)
29 GCB_REGIONAL_INDICATOR, // Regional Indicator (U+1F1E6..U+1F1FF)
30 GCB_PREPEND, // Prepended concatenation marks
31 GCB_SPACING_MARK, // Spacing marks
32 GCB_L, // Hangul Leading Jamo
33 GCB_V, // Hangul Vowel Jamo
34 GCB_T, // Hangul Trailing Jamo
35 GCB_LV, // Hangul LV syllable
36 GCB_LVT, // Hangul LVT syllable
37 GCB_EXTENDED_PICTOGRAPHIC // Extended_Pictographic property
38} grapheme_cluster_break_t;

61 // GB3: CR
62 if (codepoint == 0x000D) {

568/*
569 * Check if there's a grapheme cluster boundary between two codepoints
570 *
571 * This implements the complete UAX #29 grapheme cluster boundary rules.
572 * Returns true if there's a boundary, false otherwise.
573 *
574 * Parameters:
575 * prev_cp - Previous codepoint
576 * curr_cp - Current codepoint
577 * prev_ri_count - Count of preceding Regional Indicator codepoints (for
578 * GB12/13)
579 *
580 * Returns:
581 * true if boundary exists, false if no break
582 */
583static bool lle_check_grapheme_break(uint32_t prev_cp, uint32_t curr_cp,
584 int prev_ri_count) {
585 grapheme_cluster_break_t prev_gcb = lle_get_gcb_property(prev_cp);
586 grapheme_cluster_break_t curr_gcb = lle_get_gcb_property(curr_cp);
587
588 // GB3: Do not break between CR and LF
589 if (prev_gcb == GCB_CR && curr_gcb == GCB_LF) {
590 return false; // No break
591 }
592
593 // GB4: Break after controls (CR, LF, Control)
594 if (prev_gcb == GCB_CR || prev_gcb == GCB_LF || prev_gcb == GCB_CONTROL) {
595 return true; // Break
596 }
597
598 // GB5: Break before controls (CR, LF, Control)
599 if (curr_gcb == GCB_CR || curr_gcb == GCB_LF || curr_gcb == GCB_CONTROL) {
600 return true; // Break
601 }
602
603 // GB6: Do not break Hangul syllable sequences (L (L | V | LV | LVT))
604 if (prev_gcb == GCB_L && (curr_gcb == GCB_L || curr_gcb == GCB_V ||
605 curr_gcb == GCB_LV || curr_gcb == GCB_LVT)) {
606 return false; // No break
607 }
608
609 // GB7: Do not break Hangul syllable sequences ((LV | V) (V | T))
610 if ((prev_gcb == GCB_LV || prev_gcb == GCB_V) &&
611 (curr_gcb == GCB_V || curr_gcb == GCB_T)) {
612 return false; // No break
613 }
614
615 // GB8: Do not break Hangul syllable sequences ((LVT | T) T)
616 if ((prev_gcb == GCB_LVT || prev_gcb == GCB_T) && curr_gcb == GCB_T) {
617 return false; // No break
618 }
619
620 // GB9: Do not break before extending characters or ZWJ
621 if (curr_gcb == GCB_EXTEND || curr_gcb == GCB_ZWJ) {
622 return false; // No break
623 }
624
625 // GB9a: Do not break before SpacingMarks
626 if (curr_gcb == GCB_SPACING_MARK) {
627 return false; // No break
628 }
629
630 // GB9b: Do not break after Prepend characters
631 if (prev_gcb == GCB_PREPEND) {
632 return false; // No break
633 }
634
635 // GB11: Do not break within emoji modifier sequences or emoji ZWJ sequences
636 // (ZWJ Extended_Pictographic)
637 if (prev_gcb == GCB_ZWJ && curr_gcb == GCB_EXTENDED_PICTOGRAPHIC) {
638 return false; // No break
639 }
640
641 // GB12/GB13: Regional Indicator pairs (emoji flags)
642 // Do not break between regional indicator symbols if there is an odd
643 // number of regional indicator symbols before the break point
644 if (prev_gcb == GCB_REGIONAL_INDICATOR &&
645 curr_gcb == GCB_REGIONAL_INDICATOR) {
646 // If we have an odd number of RI codepoints before this pair,
647 // we don't break (forming a pair). If even, we break (starting new
648 // pair).
649 if (prev_ri_count % 2 == 1) {
650 return false; // Odd count before - no break (complete the pair)
651 } else {
652 return true; // Even count - break (start new pair)
653 }
654 }
655
656 // GB999: Otherwise, break everywhere
657 return true;
658}

660/*
661 * Determine if a given UTF-8 position is a grapheme cluster boundary
662 *
663 * This is the main public function for grapheme boundary detection.
664 * It examines the current position and the previous codepoint to determine
665 * if there's a boundary at this position.
666 *
667 * Parameters:
668 * ptr - Pointer to current position in UTF-8 text
669 * start - Pointer to start of text (for context)
670 * end - Pointer to end of text (for bounds checking)
671 *
672 * Returns:
673 * true if this position starts a new grapheme cluster, false otherwise
674 */
675bool lle_is_grapheme_boundary(const char *ptr, const char *start,
676 const char *end) {
677 if (!ptr || !start || !end || ptr < start || ptr >= end) {
678 return true; // Invalid input - treat as boundary
679 }
680
681 // GB1: Start of text is always a boundary
682 if (ptr == start) {
683 return true;
684 }
685
686 // Get current codepoint
687 uint32_t curr_cp = 0;
688 int curr_len = lle_utf8_decode_codepoint(ptr, end - ptr, &curr_cp);
689 if (curr_len <= 0) {
690 return true; // Invalid UTF-8 - treat as boundary
691 }
692
693 // Find previous codepoint by scanning backward
694 const char *prev_ptr = ptr - 1;
695 while (prev_ptr >= start) {
696 // Check if this is a UTF-8 start byte
697 if ((*prev_ptr & 0xC0) != 0x80) {
698 // Found start of previous codepoint
699 break;
700 }
701 prev_ptr--;
702 }
703
704 if (prev_ptr < start) {
705 return true; // No previous codepoint - boundary
706 }
707
708 // Get previous codepoint
709 uint32_t prev_cp = 0;
710 int prev_len =
711 lle_utf8_decode_codepoint(prev_ptr, ptr - prev_ptr, &prev_cp);
712 if (prev_len <= 0) {
713 return true; // Invalid UTF-8 - treat as boundary
714 }
715
716 // For Regional Indicators (GB12/GB13), we need to count preceding RI
717 // symbols
718 int prev_ri_count = 0;
719 if (lle_get_gcb_property(curr_cp) == GCB_REGIONAL_INDICATOR) {
720 // Count consecutive Regional Indicators before current position
721 const char *scan_ptr = prev_ptr;
722 while (scan_ptr >= start) {
723 uint32_t scan_cp = 0;
724 int scan_len =
725 lle_utf8_decode_codepoint(scan_ptr, end - scan_ptr, &scan_cp);
726 if (scan_len <= 0)
727 break;
728
729 if (lle_get_gcb_property(scan_cp) == GCB_REGIONAL_INDICATOR) {
730 prev_ri_count++;
731 // Move to previous codepoint
732 scan_ptr--;
733 while (scan_ptr >= start && (*scan_ptr & 0xC0) == 0x80) {
734 scan_ptr--;
735 }
736 } else {
737 break; // Not an RI, stop counting
738 }
739 }
740 }
741
742 // Check if there's a break between previous and current codepoint
743 return lle_check_grapheme_break(prev_cp, curr_cp, prev_ri_count);
744}

746/*
747 * Count grapheme clusters in a UTF-8 string
748 *
749 * This function iterates through the text and counts the number of
750 * grapheme clusters using the boundary detection algorithm.
751 *
752 * Parameters:
753 * text - UTF-8 text to analyze
754 * length - Length of text in bytes
755 *
756 * Returns:
757 * Number of grapheme clusters in the text
758 */
759size_t lle_utf8_count_graphemes(const char *text, size_t length) {
760 if (!text || length == 0) {
761 return 0;
762 }
763
764 const char *ptr = text;
765 const char *end = text + length;
766 size_t grapheme_count = 0;
767
768 while (ptr < end) {
769 // Check if this is a grapheme boundary
770 if (lle_is_grapheme_boundary(ptr, text, end)) {
771 grapheme_count++;
772 }
773
774 // Move to next codepoint
775 int seq_len = lle_utf8_sequence_length(*ptr);
776 if (seq_len <= 0 || ptr + seq_len > end) {
777 break; // Invalid UTF-8 or end of string
778 }
779 ptr += seq_len;
780 }
781
782 return grapheme_count;
783}
