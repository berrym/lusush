75/**
76 * Get current timestamp in nanoseconds
77 */
78static uint64_t get_timestamp_ns(void) {
79 struct timespec ts;
80 if (clock_gettime(CLOCK_MONOTONIC, &ts) == 0) {
81 return ts.tv_sec * 1000000000L + ts.tv_nsec;
82 }
83 return 0;
84}

113/**
114 * Calculate cache hash for input
115 */
116MAYBE_UNUSED
117static uint32_t calculate_cache_hash(const char *input) {
118 if (!input)
119 return 0;
120
121 uint32_t hash = 5381;
122 for (const char *p = input; *p; p++) {
123 hash = ((hash << 5) + hash) + (unsigned char)*p;
124 }
125 return hash;
126}

128/**
129 * Find cache entry for input
130 */
131static autosuggestions_cache_entry_t *
132find_cache_entry(autosuggestions_layer_t *layer, const char *input) {
133 if (!validate_layer(layer) || !input)
134 return NULL;
135
136 // uint32_t hash = calculate_cache_hash(input); // Unused for now
137
138 for (int i = 0; i < layer->cache_size; i++) {
139 if (layer->cache[i].valid && layer->cache[i].input_text &&
140 strcmp(layer->cache[i].input_text, input) == 0) {
141
142 // Update access time and count
143 layer->cache[i].last_used_timestamp = get_timestamp_ns();
144 layer->cache[i].use_count++;
145 layer->metrics.cache_hits++;
146
147 // Enhanced Performance Monitoring: Record cache hit
148 display_integration_record_layer_cache_operation("autosuggestions",
149 true);
150
151 return &layer->cache[i];
152 }
153 }
154
155 layer->metrics.cache_misses++;
156
157 // Enhanced Performance Monitoring: Record cache miss
158 display_integration_record_layer_cache_operation("autosuggestions", false);
159
160 return NULL;
161}

163/**
164 * Add entry to cache
165 */
166static autosuggestions_layer_error_t
167add_cache_entry(autosuggestions_layer_t *layer, const char *input,
168 const char *suggestion, int score, uint64_t generation_time) {
169 if (!validate_layer(layer) || !input || !suggestion) {
170 return AUTOSUGGESTIONS_LAYER_ERROR_INVALID_PARAM;
171 }
172
173 // Find slot to use (LRU replacement if full)
174 int slot = layer->cache_next_index;
175 if (layer->cache_size < AUTOSUGGESTIONS_LAYER_MAX_CACHE_ENTRIES) {
176 slot = layer->cache_size;
177 layer->cache_size++;
178 } else {
179 // Find LRU entry
180 uint64_t oldest_time = UINT64_MAX;
181 for (int i = 0; i < layer->cache_size; i++) {
182 if (layer->cache[i].last_used_timestamp < oldest_time) {
183 oldest_time = layer->cache[i].last_used_timestamp;
184 slot = i;
185 }
186 }
187 }
188
189 // Free existing data
190 if (layer->cache[slot].input_text) {
191 free(layer->cache[slot].input_text);
192 }
193 if (layer->cache[slot].suggestion_text) {
194 free(layer->cache[slot].suggestion_text);
195 }
196
197 // Set new data
198 layer->cache[slot].input_text = strdup(input);
199 layer->cache[slot].suggestion_text = strdup(suggestion);
200 layer->cache[slot].suggestion_score = score;
201 layer->cache[slot].generation_time_ns = generation_time;
202 layer->cache[slot].last_used_timestamp = get_timestamp_ns();
203 layer->cache[slot].use_count = 1;
204 layer->cache[slot].valid = true;
205
206 layer->cache_next_index =
207 (slot + 1) % AUTOSUGGESTIONS_LAYER_MAX_CACHE_ENTRIES;
208
209 return AUTOSUGGESTIONS_LAYER_SUCCESS;
210}

240/**
241 * Update performance metrics
242 */
243static void update_performance_metrics(autosuggestions_layer_t *layer,
244 uint64_t generation_time_ns,
245 uint64_t display_time_ns) {
246 if (!validate_layer(layer))
247 return;
248
249 // Update generation time average
250 double generation_ms = generation_time_ns / 1000000.0;
251 layer->metrics.avg_generation_time_ms =
252 (layer->metrics.avg_generation_time_ms *
253 layer->metrics.suggestions_generated +
254 generation_ms) /
255 (layer->metrics.suggestions_generated + 1);
256
257 // Update display time average
258 if (display_time_ns > 0) {
259 double display_ms = display_time_ns / 1000000.0;
260 layer->metrics.avg_display_time_ms =
261 (layer->metrics.avg_display_time_ms *
262 layer->metrics.suggestions_displayed +
263 display_ms) /
264 (layer->metrics.suggestions_displayed + 1);
265 }
266
267 // Update cache hit rate
268 uint64_t total_requests =
269 layer->metrics.cache_hits + layer->metrics.cache_misses;
270 if (total_requests > 0) {
271 layer->metrics.cache_hit_rate =
272 (double)layer->metrics.cache_hits / total_requests;
273 }
274
275 layer->metrics.last_performance_update = get_timestamp_ns();
276}

278/**
279 * Compose suggestion display using terminal control
280 */
281static autosuggestions_layer_error_t
282compose_suggestion_display(autosuggestions_layer_t *layer,
283 const char *suggestion, char *buffer,
284 size_t buffer_size, size_t *bytes_written) {
285 if (!validate_layer(layer) || !suggestion || !buffer || !bytes_written) {
286 return AUTOSUGGESTIONS_LAYER_ERROR_INVALID_PARAM;
287 }
288
289 *bytes_written = 0;
290
291 // Check terminal capabilities
292 if (!layer->display_config.enable_color ||
293 !terminal_control_has_capability(layer->terminal_control,
294 TERMINAL_CAP_COLOR_8)) {
295 // Plain text display
296 int written = snprintf(buffer, buffer_size, "%s", suggestion);
297 if (written < 0 || (size_t)written >= buffer_size) {
298 return AUTOSUGGESTIONS_LAYER_ERROR_DISPLAY_FAILED;
299 }
300 *bytes_written = written;
301 return AUTOSUGGESTIONS_LAYER_SUCCESS;
302 }
303
304 // Generate color sequence
305 char color_sequence[64] = {0};
306 if (terminal_control_generate_color_sequence(
307 layer->terminal_control, layer->display_config.suggestion_color,
308 true, // foreground
309 color_sequence, sizeof(color_sequence)) <= 0) {
310 // Fallback to plain text
311 int written = snprintf(buffer, buffer_size, "%s", suggestion);
312 if (written < 0 || (size_t)written >= buffer_size) {
313 return AUTOSUGGESTIONS_LAYER_ERROR_DISPLAY_FAILED;
314 }
315 *bytes_written = written;
316 return AUTOSUGGESTIONS_LAYER_SUCCESS;
317 }
318
319 // Generate reset sequence
320 char reset_sequence[32] = {0};
321 if (terminal_control_generate_style_sequence(
322 layer->terminal_control, TERMINAL_STYLE_NONE, reset_sequence,
323 sizeof(reset_sequence)) <= 0) {
324 strcpy(reset_sequence, "\033[0m"); // Fallback
325 }
326
327 // Compose full display
328 int written = snprintf(buffer, buffer_size, "%s%s%s", color_sequence,
329 suggestion, reset_sequence);
330
331 if (written < 0 || (size_t)written >= buffer_size) {
332 return AUTOSUGGESTIONS_LAYER_ERROR_DISPLAY_FAILED;
333 }
334
335 *bytes_written = written;
336 return AUTOSUGGESTIONS_LAYER_SUCCESS;
337}

344autosuggestions_layer_create(layer_event_system_t *event_system,
345 terminal_control_t *terminal_control) {
346 if (!event_system || !terminal_control) {
347 return NULL;
348 }
349
350 autosuggestions_layer_t *layer = calloc(1, sizeof(autosuggestions_layer_t));
351 if (!layer) {
352 return NULL;

354

356 layer->layer_id = LAYER_ID_AUTOSUGGESTIONS;
357 layer->initialized = false;
358 layer->enabled = true;
359 layer->last_error = AUTOSUGGESTIONS_LAYER_SUCCESS;
360
361 // Store references
362 layer->event_system = event_system;
363 layer->terminal_control = terminal_control;
364
365 // Initialize state
366 layer->current_input = NULL;
367 layer->current_suggestion = NULL;
368 layer->cursor_position = 0;
369 layer->suggestion_displayed = false;
370 layer->suggestion_timestamp = 0;
371
372 // Initialize cache
373 layer->cache_size = 0;
374 layer->cache_next_index = 0;
375 memset(layer->cache, 0, sizeof(layer->cache));
376
377 // Initialize performance metrics
378 memset(&layer->metrics, 0, sizeof(layer->metrics));
379 layer->metrics.last_performance_update = get_timestamp_ns();
380
381 // Initialize coordination flags
382 layer->in_display_operation = false;
383 layer->needs_refresh = false;

388autosuggestions_layer_error_t
389autosuggestions_layer_init(autosuggestions_layer_t *layer,
390 const autosuggestions_display_config_t *config) {
391 if (!layer) {
392 return AUTOSUGGESTIONS_LAYER_ERROR_NULL_POINTER;
393 }
394
395 if (layer->initialized) {
396 return AUTOSUGGESTIONS_LAYER_SUCCESS; // Already initialized
397 }
398
399 // Set configuration
400 if (config) {
401 layer->display_config = *config;
402 } else {
403 autosuggestions_layer_create_default_config(&layer->display_config);
404 }
405
406 // Get terminal capabilities
407 layer->terminal_caps =
408 terminal_control_get_capabilities(layer->terminal_control);
409
410 // Use LLE for comprehensive terminal capability detection
411 lle_terminal_detection_result_t *detection = NULL;
412 lle_detect_terminal_capabilities_optimized(&detection);
413
414 // For autosuggestions, we need stdout to be a TTY for display, not
415 // necessarily stdin
416 if (!detection || !detection->stdout_is_tty) {
417 set_layer_error(layer,
418 AUTOSUGGESTIONS_LAYER_ERROR_UNSUPPORTED_TERMINAL);
419 return AUTOSUGGESTIONS_LAYER_ERROR_UNSUPPORTED_TERMINAL;
420 }
421
422 // Check if terminal has basic capabilities needed for autosuggestions
423 // We need colors and unicode support for professional display
424 if (!detection->supports_colors || !detection->supports_unicode) {
425 set_layer_error(layer,
426 AUTOSUGGESTIONS_LAYER_ERROR_UNSUPPORTED_TERMINAL);
427 return AUTOSUGGESTIONS_LAYER_ERROR_UNSUPPORTED_TERMINAL;
428 }
429
430 // NOTE: In v1.3.0, suggestion generation is handled externally by LLE's
431 // history system. LLE's readline calls
432 // autosuggestions_layer_set_suggestion() directly with history-based
433 // suggestions. No legacy init needed.
434
435 // Subscribe to layer events
436 autosuggestions_layer_error_t error =
437 autosuggestions_layer_subscribe_events(layer);
438 if (error != AUTOSUGGESTIONS_LAYER_SUCCESS) {
439 set_layer_error(layer, error);
440 return error;
441 }
442
443 layer->initialized = true;
444 set_layer_error(layer, AUTOSUGGESTIONS_LAYER_SUCCESS);
445
446 return AUTOSUGGESTIONS_LAYER_SUCCESS;
447}

449autosuggestions_layer_error_t
450autosuggestions_layer_destroy(autosuggestions_layer_t **layer) {
451 if (!layer || !*layer) {
452 return AUTOSUGGESTIONS_LAYER_ERROR_NULL_POINTER;
453 }
454
455 autosuggestions_layer_t *l = *layer;
456
457 // Unsubscribe from events
458 if (l->initialized) {
459 autosuggestions_layer_unsubscribe_events(l);
460 }
461
462 // Clear display
463 autosuggestions_layer_clear(l);
464
465 // Free current state
466 if (l->current_input) {
467 free(l->current_input);
468 }
469 if (l->current_suggestion) {
470 free(l->current_suggestion);
471 }
472
473 // Free cache entries
474 for (int i = 0; i < l->cache_size; i++) {
475 if (l->cache[i].input_text) {
476 free(l->cache[i].input_text);
477 }
478 if (l->cache[i].suggestion_text) {
479 free(l->cache[i].suggestion_text);
480 }
481 }
482
483 // Free layer structure
484 free(l);
485 *layer = NULL;
486
487 return AUTOSUGGESTIONS_LAYER_SUCCESS;
488}

518autosuggestions_layer_error_t
519autosuggestions_layer_update(autosuggestions_layer_t *layer,
520 const autosuggestions_context_t *context) {
521 if (!validate_layer(layer) || !context) {
522 return AUTOSUGGESTIONS_LAYER_ERROR_INVALID_PARAM;
523 }
524
525 if (!layer->enabled || layer->in_display_operation) {
526 return AUTOSUGGESTIONS_LAYER_SUCCESS;
527 }
528
529 layer->in_display_operation = true;
530 layer->metrics.total_suggestions_requested++;
531
532 START_TIMING();
533
534 autosuggestions_layer_error_t result = AUTOSUGGESTIONS_LAYER_SUCCESS;
535
536 // Check if we should suggest
537 if (!autosuggestions_layer_should_suggest(layer, context)) {
538 autosuggestions_layer_clear(layer);
539 goto cleanup;
540 }
541
542 // Check cache first
543 autosuggestions_cache_entry_t *cached =
544 find_cache_entry(layer, context->input_line);
545 char *suggestion_text = NULL;
546 uint64_t generation_time_ns = 0;
547
548 if (cached) {
549 suggestion_text = strdup(cached->suggestion_text);
550 generation_time_ns = cached->generation_time_ns;
551 } else {
552 // Generate new suggestion
553 suggestion_text =
554 generate_suggestion(layer, context, &generation_time_ns);
555
556 if (suggestion_text) {
557 // Add to cache
558 add_cache_entry(layer, context->input_line, suggestion_text, 100,
559 generation_time_ns);
560 }
561 }
562
563 // Update current state
564 if (layer->current_input) {
565 free(layer->current_input);
566 }
567 layer->current_input = strdup(context->input_line);
568 layer->cursor_position = context->cursor_position;
569
570 if (suggestion_text) {
571 if (layer->current_suggestion) {
572 free(layer->current_suggestion);
573 }
574 layer->current_suggestion = suggestion_text;
575 layer->suggestion_timestamp = get_timestamp_ns();
576 layer->suggestion_displayed = true;
577 layer->metrics.suggestions_displayed++;
578
579 // Publish change event
580 autosuggestions_layer_publish_change(layer, NULL, suggestion_text);
581 } else {
582 autosuggestions_layer_clear(layer);
583 }
584
585cleanup:;
586 END_TIMING(generation_time_ns);
587 update_performance_metrics(layer, generation_time_ns, 0);
588
589 layer->in_display_operation = false;
590 set_layer_error(layer, result);
591 return result;
592}

620autosuggestions_layer_error_t
621autosuggestions_layer_accept(autosuggestions_layer_t *layer,
622 char *accepted_text, size_t buffer_size) {
623 if (!validate_layer(layer) || !accepted_text) {
624 return AUTOSUGGESTIONS_LAYER_ERROR_INVALID_PARAM;
625 }
626
627 if (!layer->suggestion_displayed || !layer->current_suggestion) {
628 accepted_text[0] = '\0';
629 return AUTOSUGGESTIONS_LAYER_SUCCESS;
630 }
631
632 size_t suggestion_len = strlen(layer->current_suggestion);
633 if (suggestion_len >= buffer_size) {
634 set_layer_error(layer, AUTOSUGGESTIONS_LAYER_ERROR_INVALID_PARAM);
635 return AUTOSUGGESTIONS_LAYER_ERROR_INVALID_PARAM;
636 }
637
638 strcpy(accepted_text, layer->current_suggestion);
639 layer->metrics.suggestions_accepted++;
640
641 // Clear after acceptance
642 autosuggestions_layer_clear(layer);
643
644 set_layer_error(layer, AUTOSUGGESTIONS_LAYER_SUCCESS);
645 return AUTOSUGGESTIONS_LAYER_SUCCESS;
646}

657autosuggestions_layer_error_t
658autosuggestions_layer_set_suggestion(autosuggestions_layer_t *layer,
659 const char *suggestion) {
660
661 if (!layer) {
662 return AUTOSUGGESTIONS_LAYER_ERROR_NULL_POINTER;
663 }
664
665 /* Handle NULL or empty suggestion as clear */
666 if (!suggestion || !*suggestion) {
667 return autosuggestions_layer_clear(layer);
668 }
669
670 /* Validate suggestion length */
671 size_t len = strlen(suggestion);
672 if (len > AUTOSUGGESTIONS_LAYER_MAX_SUGGESTION_LENGTH) {
673 set_layer_error(layer, AUTOSUGGESTIONS_LAYER_ERROR_INVALID_PARAM);
674 return AUTOSUGGESTIONS_LAYER_ERROR_INVALID_PARAM;
675 }
676
677 /* Free existing suggestion if any */
678 if (layer->current_suggestion) {
679 free(layer->current_suggestion);
680 layer->current_suggestion = NULL;
681 }
682
683 /* Set new suggestion */
684 layer->current_suggestion = strdup(suggestion);
685 if (!layer->current_suggestion) {
686 set_layer_error(layer, AUTOSUGGESTIONS_LAYER_ERROR_MEMORY_ALLOCATION);
687 return AUTOSUGGESTIONS_LAYER_ERROR_MEMORY_ALLOCATION;
688 }
689
690 layer->suggestion_displayed = true;
691 layer->suggestion_timestamp = get_timestamp_ns();
692 layer->metrics.suggestions_displayed++;
693
694 set_layer_error(layer, AUTOSUGGESTIONS_LAYER_SUCCESS);
695 return AUTOSUGGESTIONS_LAYER_SUCCESS;
696}

730
731layer_events_error_t
732autosuggestions_layer_handle_event(const layer_event_t *event,
733 void *user_data) {
734 autosuggestions_layer_t *layer = (autosuggestions_layer_t *)user_data;
735
736 if (!validate_layer(layer) || !event) {
737 return LAYER_EVENTS_ERROR_INVALID_PARAM;
738 }
739
740 switch (event->type) {
741 case LAYER_EVENT_CONTENT_CHANGED:
742 // Command layer content changed - may need to update suggestions
743 if (event->source_layer == LAYER_ID_COMMAND_LAYER) {
744 layer->needs_refresh = true;
745 }
746 break;
747
748 case LAYER_EVENT_SIZE_CHANGED:
749 // Terminal size changed - update capabilities
750 layer->terminal_caps =
751 terminal_control_get_capabilities(layer->terminal_control);
752 break;
753
754 default:
755 return LAYER_EVENTS_ERROR_INVALID_PARAM;
756 }
757
758 return LAYER_EVENTS_SUCCESS;
759}

761autosuggestions_layer_error_t
762autosuggestions_layer_publish_change(autosuggestions_layer_t *layer,
763 const char *old_suggestion
764 __attribute__((unused)),
765 const char *new_suggestion) {
766 if (!validate_layer(layer)) {
767 return AUTOSUGGESTIONS_LAYER_ERROR_NOT_INITIALIZED;
768 }
769
770 // Publish content changed event
771 if (layer->event_system) {
772 const char *content = new_suggestion ? new_suggestion : "";
773 size_t content_len = strlen(content);
774
775 layer_events_publish_content_changed(layer->event_system,
776 LAYER_ID_AUTOSUGGESTIONS, content,
777 content_len, false);
778
779 /* Also publish REDRAW_NEEDED so display_controller renders the change.
780 * This is critical when clearing autosuggestion - display_controller
781 * subscribes to REDRAW_NEEDED events to trigger
782 * dc_handle_redraw_needed() which handles ghost text cleanup. Without
783 * this, clearing autosuggestion doesn't trigger a visual update. */
784 layer_events_publish_simple(
785 layer->event_system, LAYER_EVENT_REDRAW_NEEDED,
786 LAYER_ID_AUTOSUGGESTIONS, 0, /* broadcast to all */
787 LAYER_EVENT_PRIORITY_HIGH);
788 }
789
790 return AUTOSUGGESTIONS_LAYER_SUCCESS;
791}

793autosuggestions_layer_error_t
794autosuggestions_layer_subscribe_events(autosuggestions_layer_t *layer) {
795 if (!validate_layer_for_init(layer)) {
796 return AUTOSUGGESTIONS_LAYER_ERROR_NOT_INITIALIZED;
797 }
798
799 // Subscribe to content changes from command layer
800 layer_events_error_t error = layer_events_subscribe(
801 layer->event_system, LAYER_EVENT_CONTENT_CHANGED,
802 LAYER_ID_AUTOSUGGESTIONS, autosuggestions_layer_handle_event, layer,
803 LAYER_EVENT_PRIORITY_NORMAL);
804
805 if (error != LAYER_EVENTS_SUCCESS) {
806 return AUTOSUGGESTIONS_LAYER_ERROR_EVENT_FAILED;
807 }
808
809 // Subscribe to size changes
810 error = layer_events_subscribe(
811 layer->event_system, LAYER_EVENT_SIZE_CHANGED, LAYER_ID_AUTOSUGGESTIONS,
812 autosuggestions_layer_handle_event, layer, LAYER_EVENT_PRIORITY_LOW);
813
814 if (error != LAYER_EVENTS_SUCCESS) {
815 return AUTOSUGGESTIONS_LAYER_ERROR_EVENT_FAILED;
816 }
817
818 return AUTOSUGGESTIONS_LAYER_SUCCESS;
819}

921bool autosuggestions_layer_should_suggest(
922 const autosuggestions_layer_t *layer,
923 const autosuggestions_context_t *context) {
924 if (!validate_layer(layer) || !context) {
925 return false;
926 }
927
928 // Don't suggest if disabled
929 if (!layer->enabled) {
930 return false;
931 }
932
933 // Don't suggest if no input
934 if (!context->input_line || !*context->input_line) {
935 return false;
936 }
937
938 // Don't suggest if cursor not at end
939 if (context->cursor_position != context->line_length) {
940 return false;
941 }
942
943 // Don't suggest for very short commands unless configured
944 if (!layer->display_config.show_for_short_commands &&
945 context->line_length < 3) {
946 return false;
947 }
948
949 // Don't suggest in multiline context unless configured
950 if (context->is_multiline_context &&
951 !layer->display_config.show_in_multiline) {
952 return false;
953 }
954
955 // Check terminal size
956 if (layer->terminal_caps.terminal_width <
957 AUTOSUGGESTIONS_LAYER_MIN_TERMINAL_WIDTH) {
958 return false;
959 }
960
961 return true;
962}
