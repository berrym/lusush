1/**
2 * keybinding.h - Keybinding Engine for GNU Readline Compatibility
3 *
4 * Implements fast key sequence lookup and binding management for interactive
5 * line editing. Uses hashtable for O(1) lookup performance (<50Î¼s requirement).
6 *
7 * Key Features:
8 * - Fast key sequence lookup using libhashtable
9 * - Multi-key sequences (chords) support
10 * - GNU Readline key notation (C-a, M-f, etc.)
11 * - Mode-specific bindings (emacs/vi)
12 * - Function pointer dispatch
13 *
14 * GNU Readline Compatibility:
15 * - C-a, C-e (beginning/end of line)
16 * - M-f, M-b (forward/backward word)
17 * - C-k, C-y, M-y (kill/yank)
18 * - C-r, C-s (reverse/forward search)
19 * - Multi-key sequences (C-x C-s, etc.)
20 *
21 * Specification:
22 * docs/lle_specification/critical_gaps/25_default_keybindings_complete.md
23 * Implementation Plan:
24 * docs/lle_specification/critical_gaps/25_IMPLEMENTATION_PLAN.md Date:
25 * 2025-11-02
26 */

82/**
83 * Simple action function signature
84 *
85 * Simple actions take an editor context only and operate on editor state.
86 * The editor context provides access to buffer, history, kill ring, etc.
87 * Most keybindings (navigation, editing, kill/yank) are simple actions.
88 *
89 * @param editor Editor context (contains buffer, history, kill ring, etc.)
90 * @return LLE_SUCCESS or error code
91 */
92typedef lle_result_t (*lle_action_simple_t)(lle_editor_t *editor);
93
94/**
95 * Context-aware action function signature
96 *
97 * Context-aware actions have full access to readline context including
98 * continuation state, display controller, done flag, and final_line.
99 * Used for complex actions like ENTER (accept line), incremental search,
100 * completion menus, etc.
101 *
102 * @param ctx Readline context (full access to readline state)
103 * @return LLE_SUCCESS or error code
104 */
105typedef lle_result_t (*lle_action_context_t)(readline_context_t *ctx);

115/**
116 * Unified keybinding action structure
117 *
118 * Supports both simple and context-aware actions through a tagged union.
119 * The type field indicates which function pointer is valid.
120 */
121typedef struct {
122 lle_action_type_t type; /* Action type (simple or context-aware) */
123 union {
124 lle_action_simple_t simple; /* Simple action function */
125 lle_action_context_t context; /* Context-aware action function */
126 } func;
127 const char *name; /* Function name (for debugging/introspection) */
128} lle_keybinding_action_t;
