118/**
119 * Calculate FNV-1a hash for content validation
120 */
121static uint64_t calculate_content_hash(const char *content) {
122 if (!content)
123 return 0;
124
125 uint64_t hash = HASH_INITIAL_VALUE;
126 const unsigned char *data = (const unsigned char *)content;
127
128 while (*data) {
129 hash ^= *data++;
130 hash *= HASH_PRIME;
131 }
132
133 return hash;
134}

136/**
137 * Get current timestamp with nanosecond precision
138 */
139static uint64_t get_current_time_ns(void) {
140 struct timespec ts = {0};
141 clock_gettime(CLOCK_MONOTONIC, &ts);
142 return ts.tv_sec * NSEC_PER_SEC + ts.tv_nsec;
143}

162/**
163 * Calculate prompt metrics from content
164 */
165static void calculate_prompt_metrics(const char *content,
166 prompt_metrics_t *metrics) {
167 if (!content || !metrics)
168 return;
169
170 memset(metrics, 0, sizeof(*metrics));
171
172 const char *line_start = content;
173 const char *current = content;
174 int current_line_width = 0;
175 bool in_ansi_sequence = false;
176
177 while (*current) {
178 /* Skip readline's prompt markers: \001 (RL_PROMPT_START_IGNORE) and
179 * \002 (RL_PROMPT_END_IGNORE) */
180 if (*current == '\001' || *current == '\002') {
181 /* Don't count these control characters */
182 current++;
183 continue;
184 }
185
186 if (*current == '\033') {
187 in_ansi_sequence = true;
188 metrics->has_ansi_sequences = true;
189 } else if (in_ansi_sequence) {
190 /* While in ANSI sequence, check if this is the terminator */
191 if ((*current >= 'A' && *current <= 'Z') ||
192 (*current >= 'a' && *current <= 'z')) {
193 /* ANSI sequences end with a letter (A-Z or a-z) */
194 in_ansi_sequence = false;
195 }
196 /* Don't count ANY characters while in_ansi_sequence (including
197 * terminators) */
198 } else {
199 /* Not in ANSI sequence - count this character */
200 if (*current == '\n') {
201 metrics->line_count++;
202 if (current_line_width > metrics->max_line_width) {
203 metrics->max_line_width = current_line_width;
204 }
205 current_line_width = 0;
206 line_start = current + 1;
207 } else {
208 /* Only count UTF-8 character start bytes, not continuation
209 * bytes */
210 /* UTF-8 continuation bytes have the form 10xxxxxx (0x80-0xBF)
211 */
212 unsigned char byte = (unsigned char)*current;
213 if ((byte & 0xC0) != 0x80) {
214 /* This is a character start byte (ASCII or UTF-8 lead byte)
215 */
216 current_line_width++;
217
218 // Check for Unicode characters
219 if (byte > 127) {
220 metrics->has_unicode = true;
221 }
222 }
223 /* Skip UTF-8 continuation bytes - don't increment counter */
224 }
225 }
226 current++;
227 }
228
229 // Handle final line if no trailing newline
230 if (current > line_start) {
231 metrics->line_count++;
232 if (current_line_width > metrics->max_line_width) {
233 metrics->max_line_width = current_line_width;
234 }
235 }
236
237 // Ensure at least one line
238 if (metrics->line_count == 0) {
239 metrics->line_count = 1;
240 }
241
242 metrics->is_multiline = (metrics->line_count > 1);
243 metrics->total_visual_width = metrics->max_line_width;
244
245 // Estimate command position (best effort)
246 // Command always starts after the LAST line, so use current_line_width
247 metrics->estimated_command_row = metrics->line_count;
248 metrics->estimated_command_column =
249 current_line_width + 1; /* +1 for 1-indexed columns */
250}

252/**
253 * Find cached entry for content and theme
254 */
255static prompt_cache_entry_t *find_cache_entry(prompt_layer_t *layer,
256 const char *content,
257 const char *theme_name) {
258 if (!layer || !content)
259 return NULL;
260
261 uint64_t content_hash = calculate_content_hash(content);
262 uint64_t theme_hash = theme_name ? calculate_content_hash(theme_name) : 0;
263 uint64_t now = get_current_time_ns();
264
265 for (int i = 0; i < PROMPT_LAYER_CACHE_SIZE; i++) {
266 prompt_cache_entry_t *entry = &layer->cache[i];
267
268 if (entry->is_valid && entry->content_hash == content_hash &&
269 entry->theme_hash == theme_hash) {
270
271 // Check cache expiry
272 uint64_t age_ms =
273 time_diff_ns(entry->creation_time_ns, now) / MSEC_TO_NSEC;
274 if (age_ms < PROMPT_LAYER_CACHE_EXPIRY_MS) {
275 return entry;
276 } else {
277 // Expire old entry
278 entry->is_valid = false;
279 }
280 }
281 }
282
283 return NULL;
284}

286/**
287 * Create new cache entry
288 */
289static prompt_cache_entry_t *
290create_cache_entry(prompt_layer_t *layer, const char *content,
291 const char *rendered_content, const char *theme_name,
292 const prompt_metrics_t *metrics) {
293 if (!layer || !content || !rendered_content)
294 return NULL;
295
296 prompt_cache_entry_t *entry = &layer->cache[layer->cache_next_index];
297 layer->cache_next_index =
298 (layer->cache_next_index + 1) % PROMPT_LAYER_CACHE_SIZE;
299
300 // Free existing content
301 if (entry->raw_content) {
302 free(entry->raw_content);
303 entry->raw_content = NULL;
304 }
305 if (entry->rendered_content) {
306 free(entry->rendered_content);
307 entry->rendered_content = NULL;
308 }
309 if (entry->theme_name) {
310 free(entry->theme_name);
311 entry->theme_name = NULL;
312 }
313
314 // Allocate and copy new content
315 entry->raw_content = strdup(content);
316 entry->rendered_content = strdup(rendered_content);
317 entry->theme_name = theme_name ? strdup(theme_name) : NULL;
318
319 if (!entry->raw_content || !entry->rendered_content) {
320 // Allocation failed - invalidate entry
321 entry->is_valid = false;
322 return NULL;
323 }
324
325 entry->content_hash = calculate_content_hash(content);
326 entry->theme_hash = theme_name ? calculate_content_hash(theme_name) : 0;
327 entry->creation_time_ns = get_current_time_ns();
328 entry->metrics = *metrics;
329 entry->is_valid = true;
330
331 return entry;
332}

360/**
361 * Update performance statistics
362 */
363static void update_performance_stats(prompt_layer_t *layer,
364 uint64_t render_time_ns, bool cache_hit) {
365 if (!layer)
366 return;
367
368 prompt_performance_t *perf = &layer->performance;
369
370 perf->render_count++;
371 perf->total_render_time_ns += render_time_ns;
372
373 if (cache_hit) {
374 perf->cache_hits++;
375 } else {
376 perf->cache_misses++;
377 }
378
379 // Update min/max times
380 if (perf->render_count == 1 || render_time_ns < perf->min_render_time_ns) {
381 perf->min_render_time_ns = render_time_ns;
382 }
383 if (render_time_ns > perf->max_render_time_ns) {
384 perf->max_render_time_ns = render_time_ns;
385 }
386
387 // Calculate running average
388 perf->avg_render_time_ns = perf->total_render_time_ns / perf->render_count;
389
390 // Store in recent times circular buffer
391 perf->recent_render_times[perf->recent_times_index] = render_time_ns;
392 perf->recent_times_index =
393 (perf->recent_times_index + 1) % PROMPT_LAYER_METRICS_HISTORY_SIZE;
394
395 perf->last_render_time_ns = get_current_time_ns();
396 perf->last_metrics_update_ns = perf->last_render_time_ns;
397}

442/**
443 * Render prompt content with current theme
444 */
445static prompt_layer_error_t render_prompt_content(prompt_layer_t *layer) {
446 if (!layer || !validate_layer_memory(layer)) {
447 return PROMPT_LAYER_ERROR_INVALID_PARAM;
448 }
449
450 if (!layer->raw_content) {
451 return PROMPT_LAYER_ERROR_INVALID_STATE;
452 }
453
454 uint64_t start_time = get_current_time_ns();
455
456 // Get current theme information from LLE
457 const char *theme_name = get_active_theme_name();
458
459 // Check cache first
460 prompt_cache_entry_t *cached =
461 find_cache_entry(layer, layer->raw_content, theme_name);
462 if (cached) {
463 // Cache hit - use cached content
464 if (layer->rendered_content) {
465 free(layer->rendered_content);
466 }
467 layer->rendered_content = strdup(cached->rendered_content);
468 layer->current_metrics = cached->metrics;
469
470 uint64_t end_time = get_current_time_ns();
471 uint64_t render_time = time_diff_ns(start_time, end_time);
472 update_performance_stats(layer, render_time, true);
473
474 DEBUG_PRINT("Cache hit for theme '%s'", theme_name);
475 return PROMPT_LAYER_SUCCESS;
476 }
477
478 // Cache miss - need to render
479 DEBUG_PRINT("Cache miss - rendering with theme '%s'", theme_name);
480
481 char rendered_buffer[PROMPT_LAYER_MAX_CONTENT_SIZE];
482
483 // Use raw_content directly - it's already been rendered by the appropriate
484 // prompt system (Spec 25 composer or legacy theme system)
485 // Do NOT re-render here as that would override Spec 25 prompts with legacy
486 strncpy(rendered_buffer, layer->raw_content, sizeof(rendered_buffer) - 1);
487 rendered_buffer[sizeof(rendered_buffer) - 1] = '\0';
488 DEBUG_PRINT("Using raw content as prompt");
489
490 // Calculate metrics
491 calculate_prompt_metrics(rendered_buffer, &layer->current_metrics);
492
493 // Update rendered content
494 if (layer->rendered_content) {
495 free(layer->rendered_content);
496 }
497 layer->rendered_content = strdup(rendered_buffer);
498
499 if (!layer->rendered_content) {
500 return PROMPT_LAYER_ERROR_MEMORY_ALLOCATION;
501 }
502
503 // Create cache entry
504 create_cache_entry(layer, layer->raw_content, rendered_buffer, theme_name,
505 &layer->current_metrics);
506
507 // Update performance statistics
508 uint64_t end_time = get_current_time_ns();
509 uint64_t render_time = time_diff_ns(start_time, end_time);
510 update_performance_stats(layer, render_time, false);
511
512 layer->content_dirty = false;
513 layer->metrics_dirty = false;
514
515 return PROMPT_LAYER_SUCCESS;
516}

522prompt_layer_t *prompt_layer_create(void) {
523 prompt_layer_t *layer = calloc(1, sizeof(prompt_layer_t));
524 if (!layer) {
525 DEBUG_PRINT("Failed to allocate memory for prompt layer");

528
529 // Set magic numbers for memory corruption detection

531 layer->magic_footer = PROMPT_LAYER_MAGIC_FOOTER;
532
533 // Initialize timestamps
534 layer->creation_time_ns = get_current_time_ns();
535 layer->last_update_time_ns = layer->creation_time_ns;
536
537 // Initialize performance metrics
538 layer->performance.min_render_time_ns = UINT64_MAX;
539
540 // Set initial state
541 layer->initialized = false;
542 layer->enabled = false;
543 layer->content_dirty = true;
544 layer->metrics_dirty = true;
545
546 layer->allocated_size = sizeof(prompt_layer_t);
547
548 DEBUG_PRINT("Prompt layer created successfully");

552prompt_layer_error_t prompt_layer_init(prompt_layer_t *layer,
553 layer_event_system_t *events) {
554 if (!layer || !validate_layer_memory(layer)) {
555 return PROMPT_LAYER_ERROR_INVALID_PARAM;
556 }
557
558 if (!events) {
559 return PROMPT_LAYER_ERROR_INVALID_PARAM;
560 }
561
562 if (layer->initialized) {
563 return PROMPT_LAYER_SUCCESS; // Already initialized
564 }
565
566 DEBUG_PRINT("Initializing prompt layer");
567
568 // Initialize event system integration
569 layer->events_context.events = events;
570 layer->events_context.events_initialized = true;
571
572 // Subscribe to relevant events
573 // TODO: Event subscription will be implemented when event handlers are
574 // ready
575 layer_events_error_t event_result;
576
577 // Subscribe to theme change events
578 event_result = layer_events_subscribe(
579 events, LAYER_EVENT_THEME_CHANGED, LAYER_ID_PROMPT_LAYER,
580 handle_theme_change_event, layer, LAYER_EVENT_PRIORITY_HIGH);
581
582 if (event_result == LAYER_EVENTS_SUCCESS) {
583 layer->events_context.subscription_count++;
584 DEBUG_PRINT("Subscribed to theme change events");
585 }
586
587 // Subscribe to content refresh events
588 event_result = layer_events_subscribe(
589 events, LAYER_EVENT_CONTENT_CHANGED, LAYER_ID_PROMPT_LAYER,
590 handle_content_refresh_event, layer, LAYER_EVENT_PRIORITY_NORMAL);
591
592 if (event_result == LAYER_EVENTS_SUCCESS) {
593 layer->events_context.subscription_count++;
594 DEBUG_PRINT("Subscribed to content refresh events");
595 }
596
597 // Initialize theme context
598 layer->theme_context.theme_available = is_theme_available();
599 if (layer->theme_context.theme_available) {
600 const char *theme_name = get_active_theme_name();
601 layer->theme_context.current_theme_name = strdup(theme_name);
602 layer->theme_context.theme_hash = calculate_content_hash(theme_name);
603 }
604 layer->theme_context.last_theme_check_ns = get_current_time_ns();
605
606 // Set initialization complete
607 layer->initialized = true;
608 layer->enabled = true;
609 layer->last_update_time_ns = get_current_time_ns();
610
611 DEBUG_PRINT("Prompt layer initialized successfully");
612 return PROMPT_LAYER_SUCCESS;
613}

615prompt_layer_error_t prompt_layer_cleanup(prompt_layer_t *layer) {
616 if (!layer || !validate_layer_memory(layer)) {
617 return PROMPT_LAYER_ERROR_INVALID_PARAM;
618 }
619
620 DEBUG_PRINT("Cleaning up prompt layer");
621
622 // Disable layer
623 layer->enabled = false;
624
625 // Unsubscribe from events if initialized
626 if (layer->events_context.events_initialized &&
627 layer->events_context.events) {
628 // Note: In a real implementation, we would need unsubscribe functions
629 // For now, just clear the event context
630 layer->events_context.events = NULL;
631 layer->events_context.subscription_count = 0;
632 layer->events_context.events_initialized = false;
633 }
634
635 // Clear cache
636 clear_cache(layer);
637
638 // Free content strings
639 if (layer->raw_content) {
640 free(layer->raw_content);
641 layer->raw_content = NULL;
642 }
643
644 if (layer->rendered_content) {
645 free(layer->rendered_content);
646 layer->rendered_content = NULL;
647 }
648
649 // Free theme context
650 if (layer->theme_context.current_theme_name) {
651 free(layer->theme_context.current_theme_name);
652 layer->theme_context.current_theme_name = NULL;
653 }
654
655 // Reset state
656 layer->initialized = false;
657 layer->content_dirty = true;
658 layer->metrics_dirty = true;
659
660 DEBUG_PRINT("Prompt layer cleanup complete");
661 return PROMPT_LAYER_SUCCESS;
662}

694prompt_layer_error_t prompt_layer_set_content(prompt_layer_t *layer,

700 if (!content) {

707
708 size_t content_len = strlen(content);
709 if (content_len >= PROMPT_LAYER_MAX_CONTENT_SIZE) {
710 return PROMPT_LAYER_ERROR_CONTENT_TOO_LARGE;
711 }
712
713 DEBUG_PRINT("Setting prompt content (%zu bytes)", content_len);

715 // Free existing content

719

723 return PROMPT_LAYER_ERROR_MEMORY_ALLOCATION;
724 }
725
726 // Mark content as dirty for re-rendering
727 layer->content_dirty = true;
728 layer->metrics_dirty = true;
729 layer->last_update_time_ns = get_current_time_ns();
730
731 return PROMPT_LAYER_SUCCESS;

734prompt_layer_error_t prompt_layer_get_rendered_content(prompt_layer_t *layer,
735 char *output,
736 size_t output_size) {
737 if (!layer || !validate_layer_memory(layer)) {
738 return PROMPT_LAYER_ERROR_INVALID_PARAM;
739 }
740
741 if (!output || output_size < 1) {
742 return PROMPT_LAYER_ERROR_INVALID_PARAM;
743 }
744
745 if (!layer->initialized || !layer->enabled) {
746 return PROMPT_LAYER_ERROR_INVALID_STATE;
747 }
748
749 // Render content if dirty or not yet rendered
750 if (layer->content_dirty || !layer->rendered_content) {
751 prompt_layer_error_t result = render_prompt_content(layer);
752 if (result != PROMPT_LAYER_SUCCESS) {
753 return result;
754 }
755 }
756
757 if (!layer->rendered_content) {
758 return PROMPT_LAYER_ERROR_INVALID_STATE;
759 }
760
761 size_t content_len = strlen(layer->rendered_content);
762 if (content_len >= output_size) {
763 return PROMPT_LAYER_ERROR_BUFFER_TOO_SMALL;
764 }
765
766 strcpy(output, layer->rendered_content);
