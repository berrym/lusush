40/**
41 * @brief Insert Unicode replacement character for invalid UTF-8
42 *
43 * When invalid UTF-8 is encountered, insert U+FFFD (replacement character)
44 * to maintain zero data loss while indicating the error visually.
45 */
46static lle_result_t
47insert_replacement_character(lle_input_parser_system_t *parser_sys) {
48 if (!parser_sys) {
49 return LLE_ERROR_INVALID_PARAMETER;
50 }
51
52 /* If no event system, recovery still succeeds (just no event generated) */
53 if (!parser_sys->event_system) {
54 return LLE_SUCCESS;
55 }
56
57 /* Unicode replacement character U+FFFD in UTF-8 */
58 const char replacement_utf8[] = "\xEF\xBF\xBD";
59 const size_t replacement_len = 3;
60
61 /* Create text input event with replacement character */
62 lle_parsed_input_t replacement_input = {
63 .type = LLE_PARSED_INPUT_TYPE_TEXT,
64 .handled = false,
65 .parse_time_us = 0,
66 };
67
68 /* Fill text input info in union */
69 replacement_input.data.text_info.codepoint = 0xFFFD;
70 replacement_input.data.text_info.utf8_length = replacement_len;
71 memcpy(replacement_input.data.text_info.utf8_bytes, replacement_utf8,
72 replacement_len);
73 replacement_input.data.text_info.is_grapheme_start = true;
74 replacement_input.data.text_info.display_width = 1;
75 replacement_input.data.text_info.timestamp = lle_event_get_timestamp_us();
76
77 /* Generate event for replacement character */
78 lle_result_t result =
79 lle_input_parser_generate_text_events(parser_sys, &replacement_input);
80
81 /* Note: Error statistics could be tracked in error_ctx if needed */
82
83 return result;
84}

86/**
87 * @brief Process problematic data as individual text characters
88 *
89 * When a sequence is too long or cannot be parsed, fall back to treating
90 * each byte as individual text input to ensure zero data loss.
91 */
92static lle_result_t process_as_text(lle_input_parser_system_t *parser_sys,
93 const char *data, size_t data_len) {
94 if (!parser_sys || !data || data_len == 0) {
95 return LLE_ERROR_INVALID_PARAMETER;
96 }
97
98 /* Process each byte as text input */
99 for (size_t i = 0; i < data_len; i++) {
100 /* Create text input event for single byte */
101 lle_parsed_input_t text_input = {
102 .type = LLE_PARSED_INPUT_TYPE_TEXT,
103 .handled = false,
104 .parse_time_us = 0,
105 };
106
107 /* Fill basic text info */
108 text_input.data.text_info.codepoint = (uint8_t)data[i];
109 text_input.data.text_info.utf8_length = 1;
110 text_input.data.text_info.utf8_bytes[0] = data[i];
111 text_input.data.text_info.is_grapheme_start = true;
112 text_input.data.text_info.display_width = 1;
113 text_input.data.text_info.timestamp = lle_event_get_timestamp_us();
114
115 /* Generate event for this character */
116 lle_result_t result =
117 lle_input_parser_generate_text_events(parser_sys, &text_input);
118 if (result != LLE_SUCCESS) {
119 return result;
120 }
121 }
122
123 return LLE_SUCCESS;
124}

149/**
150 * @brief Force resolution of ambiguous key sequence
151 *
152 * When a key sequence is ambiguous (could be multiple keys), force
153 * resolution based on timeout or current buffer state.
154 */
155static lle_result_t
156force_key_resolution(lle_input_parser_system_t *parser_sys) {
157 if (!parser_sys || !parser_sys->key_detector) {
158 return LLE_ERROR_INVALID_PARAMETER;
159 }
160
161 lle_key_detector_t *detector = parser_sys->key_detector;
162
163 /* If we have buffered key data, process it now */
164 if (detector->sequence_pos > 0) {
165 /* Create key event from current buffer */
166 lle_parsed_input_t key_input = {
167 .type = LLE_PARSED_INPUT_TYPE_KEY,
168 .handled = false,
169 .parse_time_us = 0,
170 };
171
172 /* Fill key info with best guess */
173 key_input.data.key_info.type = LLE_KEY_TYPE_SPECIAL;
174 key_input.data.key_info.keycode = 27; /* ESC */
175 key_input.data.key_info.modifiers = LLE_KEY_MOD_NONE;
176 strcpy(key_input.data.key_info.key_name, "Escape");
177 memcpy(key_input.data.key_info.sequence, detector->sequence_buffer,
178 detector->sequence_pos);
179 key_input.data.key_info.sequence_length = detector->sequence_pos;
180 key_input.data.key_info.is_repeat = false;
181 key_input.data.key_info.timestamp = lle_event_get_timestamp_us();
182
183 /* Generate event */
184 lle_result_t result =
185 lle_input_parser_generate_key_events(parser_sys, &key_input);
186
187 /* Clear buffer */
188 detector->sequence_pos = 0;
189 memset(detector->sequence_buffer, 0, sizeof(detector->sequence_buffer));
190
191 return result;
192 }
193
194 return LLE_SUCCESS;

228 parser_sys->utf8_processor->current_codepoint = 0;

242/**
243 * @brief Recover from input parsing error
244 *
245 * Main entry point for error recovery. Dispatches to appropriate recovery
246 * strategy based on error type while ensuring zero data loss.
247 *
248 * @param parser_sys Parser system
249 * @param error_code Type of error that occurred
250 * @param problematic_data Data that caused error (may be NULL)
251 * @param data_len Length of problematic data
252 * @return LLE_SUCCESS on successful recovery, error code on failure
253 */
254lle_result_t lle_input_parser_recover_from_error(
255 lle_input_parser_system_t *parser_sys, lle_result_t error_code,
256 const char *problematic_data, size_t data_len) {
257 if (!parser_sys) {
258 return LLE_ERROR_INVALID_PARAMETER;
259 }
260
261 /* Record recovery attempt start time */
262 uint64_t start_time = lle_event_get_timestamp_us();
263
264 lle_result_t result = LLE_SUCCESS;
265
266 /* Dispatch to appropriate recovery strategy */
267 switch (error_code) {
268 case LLE_ERROR_INVALID_ENCODING:
269 /* Invalid UTF-8: Replace with Unicode replacement character U+FFFD */
270 result = insert_replacement_character(parser_sys);
271 break;
272
273 case LLE_ERROR_BUFFER_OVERFLOW:
274 /* Buffer overflow: Reset sequence parser and process as individual
275 * characters */
276 result = reset_sequence_parser(parser_sys);
277 if (result == LLE_SUCCESS && problematic_data && data_len > 0) {
278 result = process_as_text(parser_sys, problematic_data, data_len);
279 }
280 break;
281
282 case LLE_ERROR_INPUT_PARSING:
283 /* Generic input parsing error: Try to recover data */
284 if (problematic_data && data_len > 0) {
285 result = process_as_text(parser_sys, problematic_data, data_len);
286 } else {
287 result = reset_sequence_parser(parser_sys);
288 }
289 break;
290
291 case LLE_ERROR_INVALID_FORMAT:
292 /* Invalid format (mouse, sequence, etc): Skip and continue */
293 if (parser_sys->mouse_parser) {
294 __atomic_fetch_add(
295 &parser_sys->mouse_parser->invalid_mouse_sequences, 1,
296 __ATOMIC_SEQ_CST);
297 }
298 result = LLE_SUCCESS;
299 break;
300
301 case LLE_ERROR_TIMEOUT:
302 /* Timeout: Force resolution of ambiguous sequences and reset parsers */
303 result = force_key_resolution(parser_sys);
304 if (result == LLE_SUCCESS) {
305 result = reset_all_parsers(parser_sys);
306 }
307 break;
308
309 case LLE_ERROR_INVALID_STATE:
310 /* Invalid state: Reset all parsers to ground state */
311 result = reset_all_parsers(parser_sys);
312 break;
313
314 default:
315 /* Unknown error - reset all parsers to recover */
316 result = reset_all_parsers(parser_sys);
317 break;
318 }
319
320 /* Note: Recovery performance statistics could be tracked in perf_monitor if
321 * needed */
322 (void)start_time; /* Suppress unused variable warning */
323
324 return result;
325}

331/**
332 * @brief Validate UTF-8 sequence before processing
333 *
334 * Pre-validation to catch errors early before they propagate.
335 */
336lle_result_t lle_input_parser_validate_utf8(const char *data, size_t len,
337 size_t *valid_len) {
338 if (!data || !valid_len) {
339 return LLE_ERROR_INVALID_PARAMETER;
340 }
341
342 *valid_len = 0;
343
344 for (size_t i = 0; i < len;) {
345 uint8_t byte = (uint8_t)data[i];
346 size_t char_len = 0;
347
348 /* Determine expected UTF-8 character length */
349 if ((byte & 0x80) == 0) {
350 char_len = 1; /* ASCII */
351 } else if ((byte & 0xE0) == 0xC0) {
352 char_len = 2; /* 2-byte UTF-8 */
353 } else if ((byte & 0xF0) == 0xE0) {
354 char_len = 3; /* 3-byte UTF-8 */
355 } else if ((byte & 0xF8) == 0xF0) {
356 char_len = 4; /* 4-byte UTF-8 */
357 } else {
358 /* Invalid start byte */
359 return LLE_ERROR_INVALID_ENCODING;
360 }
361
362 /* Check if we have enough bytes */
363 if (i + char_len > len) {
364 /* Incomplete sequence at end of buffer */
365 return LLE_ERROR_INVALID_ENCODING;
366 }
367
368 /* Validate continuation bytes */
369 for (size_t j = 1; j < char_len; j++) {
370 if ((data[i + j] & 0xC0) != 0x80) {
371 /* Invalid continuation byte */
372 return LLE_ERROR_INVALID_ENCODING;
373 }
374 }
375
376 i += char_len;
377 *valid_len = i;
378 }
379
380 return LLE_SUCCESS;
381}

383/**
384 * @brief Check for sequence timeout
385 *
386 * Detect when a partial sequence has exceeded timeout threshold.
387 */
388bool lle_input_parser_check_sequence_timeout(
389 lle_input_parser_system_t *parser_sys, uint64_t current_time_us) {
390 if (!parser_sys) {
391 return false;
392 }
393
394 /* Check if sequence parser has partial data */
395 if (parser_sys->sequence_parser &&
396 parser_sys->sequence_parser->buffer_pos > 0) {
397 uint64_t elapsed =
398 current_time_us - parser_sys->sequence_parser->sequence_start_time;
399 if (elapsed > LLE_MAX_SEQUENCE_TIMEOUT_US) {
400 return true;
401 }
402 }
403
404 /* Check if key detector has partial data */
405 if (parser_sys->key_detector &&
406 parser_sys->key_detector->sequence_pos > 0) {
407 uint64_t elapsed =
408 current_time_us - parser_sys->key_detector->sequence_start_time;
409 if (elapsed > LLE_MAX_SEQUENCE_TIMEOUT_US) {
410 return true;
411 }
412 }
413
414 return false;
415}
