51/**
52 * Completion type classification for categorization and display
53 */
54typedef enum {
55 LLE_COMPLETION_TYPE_BUILTIN, // Shell built-in commands (cd, echo, etc.)
56 LLE_COMPLETION_TYPE_COMMAND, // External commands from PATH
57 LLE_COMPLETION_TYPE_FILE, // Regular files
58 LLE_COMPLETION_TYPE_DIRECTORY, // Directories
59 LLE_COMPLETION_TYPE_VARIABLE, // Shell/environment variables
60 LLE_COMPLETION_TYPE_ALIAS, // Command aliases
61 LLE_COMPLETION_TYPE_HISTORY, // History entries
62 LLE_COMPLETION_TYPE_CUSTOM, // User-defined custom completions
63 LLE_COMPLETION_TYPE_UNKNOWN, // Unclassified/fallback
64 LLE_COMPLETION_TYPE_COUNT // Number of types (for iteration)
65} lle_completion_type_t;

71/**
72 * Completion item with type metadata for categorization and display
73 */
74typedef struct lle_completion_item {
75 char *text; // Completion text (owned by this struct)
76 char *suffix; // Suffix to append (space, /, etc.)
77 lle_completion_type_t type; // Completion type
78 const char *type_indicator; // Visual indicator (symbol/emoji)
79 int32_t relevance_score; // Relevance ranking (0-1000)
80 char *description; // Optional description (may be NULL)
81
82 // Memory management flags
83 bool owns_text; // Whether this struct owns text memory
84 bool owns_suffix; // Whether this struct owns suffix memory
85 bool owns_description; // Whether this struct owns description memory
86} lle_completion_item_t;
87
88/**
89 * Completion result with classified items

91typedef struct lle_completion_result {

98 size_t command_count; // Number of command completions
99 size_t file_count; // Number of file completions
100 size_t directory_count; // Number of directory completions
101 size_t variable_count; // Number of variable completions
102 size_t alias_count; // Number of alias completions
103 size_t history_count; // Number of history completions
104 size_t custom_count; // Number of custom completions
105
106 // Memory pool for allocations
107 lle_memory_pool_t *memory_pool; // Memory pool for allocations
108} lle_completion_result_t;
