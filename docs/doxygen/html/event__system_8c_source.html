17/*
18 * Get current timestamp in microseconds
19 */
20uint64_t lle_event_get_timestamp_us(void) {
21 struct timespec ts;
22 clock_gettime(CLOCK_MONOTONIC, &ts);
23 return (uint64_t)ts.tv_sec * 1000000 + (uint64_t)ts.tv_nsec / 1000;
24}

168/*
169 * Initialize event system (Phase 1 + Phase 2)
170 */
171lle_result_t lle_event_system_init(lle_event_system_t **system,
172 lle_memory_pool_t *pool) {
173 if (!system || !pool) {
174 return LLE_ERROR_INVALID_PARAMETER;
175 }
176
177 /* Allocate system structure */
178 lle_event_system_t *sys = lle_pool_alloc(sizeof(lle_event_system_t));
179 if (!sys) {
180 return LLE_ERROR_OUT_OF_MEMORY;
181 }
182
183 memset(sys, 0, sizeof(lle_event_system_t));
184
185 /* Initialize main queue */
186 lle_result_t result =
187 lle_event_queue_init(&sys->queue, LLE_EVENT_QUEUE_DEFAULT_CAPACITY);
188 if (result != LLE_SUCCESS) {
189 lle_pool_free(sys);
190 return result;
191 }
192
193 /* Phase 2: Initialize priority queue */
194 result = lle_event_queue_init(&sys->priority_queue,
195 LLE_EVENT_QUEUE_DEFAULT_CAPACITY / 2);
196 if (result != LLE_SUCCESS) {
197 lle_event_queue_destroy(sys->queue);
198 lle_pool_free(sys);
199 return result;
200 }
201
202 /* Initialize handler array */
203 sys->handler_capacity = LLE_EVENT_HANDLER_INITIAL_CAPACITY;
204 sys->handlers =
205 lle_pool_alloc(sizeof(lle_event_handler_t *) * sys->handler_capacity);
206 if (!sys->handlers) {
207 lle_event_queue_destroy(sys->priority_queue);
208 lle_event_queue_destroy(sys->queue);
209 lle_pool_free(sys);
210 return LLE_ERROR_OUT_OF_MEMORY;
211 }
212
213 sys->handler_count = 0;
214 sys->event_pool = pool;
215 sys->sequence_counter = 1; /* Start at 1 so first event gets sequence 1 */
216 sys->active = true; /* System is active after successful initialization */
217
218 /* Phase 1 statistics */
219 sys->events_created = 0;
220 sys->events_dispatched = 0;
221 sys->events_dropped = 0;
222
223 /* Phase 2A configuration and statistics */
224 sys->use_priority_queue = true; /* Enable priority queue by default */
225 sys->priority_events_queued = 0;
226 sys->priority_events_processed = 0;
227 for (int i = 0; i < LLE_PRIORITY_COUNT; i++) {
228 sys->events_by_priority[i] = 0;
229 }
230
231 /* Phase 2B: Initialize processing configuration with defaults */
232 sys->processing_config.max_events_per_cycle =
233 100; /* Default: 100 events per cycle */
234 sys->processing_config.cycle_time_limit_us =
235 10000; /* Default: 10ms per cycle */
236 sys->processing_config.auto_process =
237 false; /* Default: manual processing */
238 sys->processing_config.record_detailed_stats =
239 false; /* Default: disabled (opt-in) */
240
241 /* Phase 2B: Initialize processing state */
242 sys->processing_state = LLE_PROCESSING_RUNNING; /* Default: running */
243
244 /* Phase 2B: Enhanced statistics starts NULL (created on demand) */
245 sys->enhanced_stats = NULL;
246
247 /* Phase 2C: Initialize event filtering and hooks */
248 sys->filter_system = NULL; /* Filter system created on demand */
249 sys->pre_dispatch_hook = NULL;
250 sys->pre_dispatch_data = NULL;
251 sys->post_dispatch_hook = NULL;
252 sys->post_dispatch_data = NULL;
253
254 /* Phase 2C: Initialize system state tracking */
255 sys->current_state = LLE_STATE_INITIALIZING;
256 sys->previous_state = LLE_STATE_INITIALIZING;
257 sys->state_changed_time = lle_event_get_timestamp_us();
258
259 /* Phase 2D: Timer system starts NULL (created on demand) */
260 sys->timer_system = NULL;
261
262 /* Initialize mutex */
263 pthread_mutex_init(&sys->system_mutex, NULL);
264
265 /* Phase 2C: Transition to IDLE state after successful initialization */
266 sys->current_state = LLE_STATE_IDLE;
267 sys->state_changed_time = lle_event_get_timestamp_us();
268
269 *system = sys;
270 return LLE_SUCCESS;
271}

273/*
274 * Destroy event system (Phase 1 + Phase 2A + Phase 2B + Phase 2C + Phase 2D)
275 */
276void lle_event_system_destroy(lle_event_system_t *system) {
277 if (!system) {
278 return;
279 }
280
281 /* Phase 2C: Set state to shutting down */
282 system->current_state = LLE_STATE_SHUTTING_DOWN;
283
284 /* Stop system if active */
285 if (system->active) {
286 lle_event_system_stop(system);
287 }
288
289 /* Phase 2D: Destroy timer system if allocated */
290 if (system->timer_system) {
291 lle_event_timer_system_destroy(system);
292 }
293
294 /* Phase 2C: Destroy filter system if allocated */
295 if (system->filter_system) {
296 lle_event_filter_system_destroy(system);
297 }
298
299 /* Phase 2B: Destroy enhanced statistics if allocated */
300 if (system->enhanced_stats) {
301 lle_event_enhanced_stats_destroy(system);
302 }
303
304 /* Destroy main queue */
305 if (system->queue) {
306 lle_event_queue_destroy(system->queue);
307 }
308
309 /* Phase 2A: Destroy priority queue */
310 if (system->priority_queue) {
311 lle_event_queue_destroy(system->priority_queue);
312 }
313
314 /* Free handlers */
315 if (system->handlers) {
316 for (size_t i = 0; i < system->handler_count; i++) {
317 if (system->handlers[i]) {
318 lle_pool_free(system->handlers[i]);
319 }
320 }
321 lle_pool_free(system->handlers);
322 }
323
324 /* Destroy mutex */
325 pthread_mutex_destroy(&system->system_mutex);
326
327 /* Free system structure */
328 lle_pool_free(system);
329}

375/*
376 * Get event priority based on type (Phase 2)
377 */
378static lle_event_priority_t
379lle_event_get_priority_for_type(lle_event_kind_t type) {
380 /* Determine priority based on event type category */
381 if (type >= 0x1000 && type < 0x2000) {
382 /* Terminal Input Events - HIGH */
383 return LLE_PRIORITY_HIGH;
384 } else if (type >= 0x2000 && type < 0x3000) {
385 /* Mouse Events - MEDIUM */
386 return LLE_PRIORITY_MEDIUM;
387 } else if (type >= 0x3000 && type < 0x4000) {
388 /* Terminal State Events - CRITICAL */
389 return LLE_PRIORITY_CRITICAL;
390 } else if (type >= 0x4000 && type < 0x5000) {
391 /* Paste Events - HIGH */
392 return LLE_PRIORITY_HIGH;
393 } else if (type >= 0x5000 && type < 0x6000) {
394 /* Buffer Events - MEDIUM */
395 return LLE_PRIORITY_MEDIUM;
396 } else if (type >= 0x6000 && type < 0x7000) {
397 /* History Events - LOW */
398 return LLE_PRIORITY_LOW;
399 } else if (type >= 0x7000 && type < 0x8000) {
400 /* Completion Events - MEDIUM */
401 return LLE_PRIORITY_MEDIUM;
402 } else if (type >= 0x8000 && type < 0x9000) {
403 /* Suggestion Events - LOW */
404 return LLE_PRIORITY_LOW;
405 } else if (type >= 0x9000 && type < 0xA000) {
406 /* System Events - CRITICAL */
407 return LLE_PRIORITY_CRITICAL;
408 } else if (type >= 0xA000 && type < 0xB000) {
409 /* Timer Events - LOW */
410 return LLE_PRIORITY_LOW;
411 } else if (type >= 0xB000 && type < 0xC000) {
412 /* Plugin Events - MEDIUM (configurable) */
413 return LLE_PRIORITY_MEDIUM;
414 } else if (type >= 0xC000 && type < 0xD000) {
415 /* Display Events - HIGH */
416 return LLE_PRIORITY_HIGH;
417 } else if (type >= 0xF000 && type <= 0xFFFF) {
418 /* Debug/Test Events - LOWEST */
419 return LLE_PRIORITY_LOWEST;
420 }
421
422 /* Default to MEDIUM priority */
423 return LLE_PRIORITY_MEDIUM;
424}

426/*
427 * Create event (Phase 1 + Phase 2)
428 */
429lle_result_t lle_event_create(lle_event_system_t *system, lle_event_kind_t type,
430 void *data, size_t data_size,
431 lle_event_t **event) {
432 if (!system || !event) {
433 return LLE_ERROR_INVALID_PARAMETER;
434 }
435
436 /* Allocate event structure */
437 lle_event_t *evt = lle_pool_alloc(sizeof(lle_event_t));
438 if (!evt) {
439 return LLE_ERROR_OUT_OF_MEMORY;
440 }
441
442 memset(evt, 0, sizeof(lle_event_t));
443
444 /* Allocate and copy data if provided */
445 if (data && data_size > 0) {
446 evt->data = lle_pool_alloc(data_size);
447 if (!evt->data) {
448 lle_pool_free(evt);
449 return LLE_ERROR_OUT_OF_MEMORY;
450 }
451 memcpy(evt->data, data, data_size);
452 evt->data_size = data_size;
453 }
454
455 /* Set Phase 1 event fields */
456 evt->type = type;
457 evt->sequence_number =
458 __atomic_fetch_add(&system->sequence_counter, 1, __ATOMIC_SEQ_CST);
459 evt->timestamp = lle_event_get_timestamp_us();
460 evt->next = NULL;
461
462 /* Set Phase 2 event fields */
463 evt->source = LLE_EVENT_SOURCE_INTERNAL; /* Default source */
464 evt->priority = lle_event_get_priority_for_type(type);
465 evt->flags = LLE_EVENT_FLAG_NONE;
466 evt->handler_count = 0;
467 evt->processing_start_time = 0;
468 evt->processing_end_time = 0;
469 evt->prev = NULL;
470
471 /* Update statistics */
472 __atomic_fetch_add(&system->events_created, 1, __ATOMIC_SEQ_CST);
473
474 *event = evt;
475 return LLE_SUCCESS;
476}

542 *
543 * This is the core event for fixing stale git prompts (Issue #16).
544 * When the working directory changes, this event allows handlers to
545 * invalidate cached data that depends on the current directory.
546 */
547lle_result_t lle_event_fire_directory_changed(lle_event_system_t *system,
548 const char *old_dir,

550 if (!system || !new_dir) {
551 return LLE_ERROR_INVALID_PARAMETER;

557 if (result != LLE_SUCCESS) {
558 return result;
559 }
560
561 /* Populate shell event data */
562 event->source = LLE_EVENT_SOURCE_SHELL;
563 event->priority = LLE_PRIORITY_HIGH;
564
565 if (old_dir) {

567 sizeof(event->event_data.shell.old_directory) - 1);
568 event->event_data.shell
569 .old_directory[sizeof(event->event_data.shell.old_directory) - 1] =
570 '\0';
571 } else {
572 event->event_data.shell.old_directory[0] = '\0';

574

576 sizeof(event->event_data.shell.new_directory) - 1);
577 event->event_data.shell
578 .new_directory[sizeof(event->event_data.shell.new_directory) - 1] =
579 '\0';

582 result = lle_event_enqueue(system, event);
583 if (result != LLE_SUCCESS) {
