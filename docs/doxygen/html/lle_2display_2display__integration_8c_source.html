45/**
46 * @brief Initialize the display integration system
47 *
48 * Creates and initializes the global display integration instance that connects
49 * LLE's editing system with Lusush's display layers.
50 *
51 * @param integration Output pointer for created integration instance
52 * @param editor LLE editor instance (opaque)
53 * @param lusush_display Lusush display controller
54 * @param memory_pool Memory pool for allocations
55 * @return LLE_SUCCESS on success, error code on failure
56 */
57lle_result_t
58lle_display_integration_init(lle_display_integration_t **integration,
59 void *editor, display_controller_t *lusush_display,
60 lle_memory_pool_t *memory_pool) {
61 if (!integration || !lusush_display) {
62 return LLE_ERROR_INVALID_PARAMETER;
63 }
64
65 /* Allocate main integration structure */
66 lle_display_integration_t *integ =
67 calloc(1, sizeof(lle_display_integration_t));
68 if (!integ) {
69 return LLE_ERROR_OUT_OF_MEMORY;
70 }
71
72 /* Store Lusush display controller reference */
73 integ->lusush_display = lusush_display;
74 integ->memory_pool = memory_pool;
75 integ->integration_active = true;
76 integ->api_version = 1;
77 integ->frame_counter = 0;
78
79 /* Initialize integration lock */
80 if (pthread_rwlock_init(&integ->integration_lock, NULL) != 0) {
81 free(integ);
82 return LLE_ERROR_SYSTEM_CALL;
83 }
84
85 /* Initialize display bridge */
86 lle_result_t result = lle_display_bridge_init(
87 &integ->display_bridge, editor, lusush_display, memory_pool);
88
89 if (result != LLE_SUCCESS) {
90 pthread_rwlock_destroy(&integ->integration_lock);
91 free(integ);
92 return result;
93 }
94
95 /* Initialize render controller */
96 result = lle_render_controller_init(&integ->render_controller,
97 integ->display_bridge, memory_pool);
98
99 if (result != LLE_SUCCESS) {
100 lle_display_bridge_cleanup(integ->display_bridge);
101 pthread_rwlock_destroy(&integ->integration_lock);
102 free(integ);
103 return result;
104 }
105
106 /* Allocate display metrics */
107 integ->perf_metrics = calloc(1, sizeof(lle_display_metrics_t));
108 if (!integ->perf_metrics) {
109 lle_render_controller_cleanup(integ->render_controller);
110 lle_display_bridge_cleanup(integ->display_bridge);
111 pthread_rwlock_destroy(&integ->integration_lock);
112 free(integ);
113 return LLE_ERROR_OUT_OF_MEMORY;
114 }
115
116 /* Allocate display config */
117 integ->config = calloc(1, sizeof(lle_display_config_t));
118 if (!integ->config) {
119 free(integ->perf_metrics);
120 lle_render_controller_cleanup(integ->render_controller);
121 lle_display_bridge_cleanup(integ->display_bridge);
122 pthread_rwlock_destroy(&integ->integration_lock);
123 free(integ);
124 return LLE_ERROR_OUT_OF_MEMORY;
125 }
126
127 /* Set default config */
128 integ->config->enable_syntax_highlighting = true;
129 integ->config->enable_caching = false; /* Disabled for now */
130 integ->config->enable_performance_monitoring = false;
131
132 /* Allocate display state */
133 integ->current_state = calloc(1, sizeof(lle_display_state_t));
134 if (!integ->current_state) {
135 free(integ->config);
136 free(integ->perf_metrics);
137 lle_render_controller_cleanup(integ->render_controller);
138 lle_display_bridge_cleanup(integ->display_bridge);
139 pthread_rwlock_destroy(&integ->integration_lock);
140 free(integ);
141 return LLE_ERROR_OUT_OF_MEMORY;
142 }
143
144 /* Set as global singleton */
145 global_display_integration = integ;
146 *integration = integ;
147
148 return LLE_SUCCESS;
149}

151/**
152 * @brief Cleanup the display integration system
153 *
154 * @param integration Display integration instance to cleanup
155 * @return LLE_SUCCESS on success, error code on failure
156 */
157lle_result_t
158lle_display_integration_cleanup(lle_display_integration_t *integration) {
159 if (!integration) {
160 return LLE_ERROR_INVALID_PARAMETER;
161 }
162
163 /* Clear global singleton if this is it */
164 if (global_display_integration == integration) {
165 global_display_integration = NULL;
166 }
167
168 /* Cleanup render controller */
169 if (integration->render_controller) {
170 lle_render_controller_cleanup(integration->render_controller);
171 }
172
173 /* Cleanup display bridge */
174 if (integration->display_bridge) {
175 lle_display_bridge_cleanup(integration->display_bridge);
176 }
177
178 /* Free allocated structures */
179 free(integration->current_state);
180 free(integration->config);
181 free(integration->perf_metrics);
182
183 /* Destroy lock */
184 pthread_rwlock_destroy(&integration->integration_lock);
185
186 /* Free main structure */
187 free(integration);
188
189 return LLE_SUCCESS;
190}
