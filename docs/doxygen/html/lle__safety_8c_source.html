29/**
30 * @brief Get human-readable initialization state summary
31 *
32 * Returns a static buffer with the current init state for debugging.
33 * Not thread-safe.
34 *
35 * @return Static string describing initialization state
36 */
37const char *lle_safety_get_init_state_summary(void) {
38 static char summary[256];
39
40 if (!g_lle_integration) {
41 return "LLE not initialized";
42 }
43
44 lle_init_state_t *state = &g_lle_integration->init_state;
45
46 snprintf(summary, sizeof(summary),
47 "pool:%s term:%s hub:%s edit:%s hist:%s hooks:%s",
48 state->memory_pool_verified ? "OK" : "NO",
49 state->terminal_detected ? "OK" : "NO",
50 state->event_hub_initialized ? "OK" : "NO",
51 state->editor_initialized ? "OK" : "NO",
52 state->history_initialized ? "OK" : "NO",
53 state->shell_hooks_installed ? "OK" : "NO");
54
55 return summary;
56}

58/**
59 * @brief Get safety statistics summary
60 *
61 * Returns statistics about errors, resets, and recovery operations.
62 * Not thread-safe.
63 *
64 * @return Static string with safety statistics
65 */
66const char *lle_safety_get_stats_summary(void) {
67 static char summary[512];
68
69 if (!g_lle_integration) {
70 return "LLE not initialized";
71 }
72
73 lle_shell_integration_t *integ = g_lle_integration;
74
75 snprintf(summary, sizeof(summary),
76 "errors:%u ctrl_g:%u recovery:%s "
77 "readline_calls:%lu hard_resets:%lu nuclear_resets:%lu",
78 integ->error_count, integ->ctrl_g_count,
79 integ->recovery_mode ? "active" : "inactive",
80 (unsigned long)integ->total_readline_calls,
81 (unsigned long)integ->hard_reset_count,
82 (unsigned long)integ->nuclear_reset_count);
83
84 return summary;

102/**
103 * @brief Mark a successful readline operation
104 *
105 * Called after lle_readline() completes successfully. Resets error
106 * counters and clears recovery mode, preventing spurious resets.
107 */
108void lle_safety_mark_success(void) {
109 if (!g_lle_integration) {
110 return;
111 }
112
113 g_lle_integration->successful_reads++;
114 g_lle_integration->total_readline_calls++;
115
116 /* Reset error counter on success */
117 lle_reset_error_counter();
118}

120/**
121 * @brief Mark a failed readline operation
122 *
123 * Called when lle_readline() encounters an error. Records the error
124 * for threshold tracking.
125 *
126 * @param error The error that occurred
127 */
128void lle_safety_mark_failure(lle_result_t error) {
129 if (!g_lle_integration) {
130 return;
131 }
132
133 g_lle_integration->total_readline_calls++;
134 lle_record_error(error);
135}

142/**
143 * @brief Get shell event hub statistics
144 *
145 * @param events_fired Output: total events fired (NULL to skip)
146 * @param dir_changes Output: directory change count (NULL to skip)
147 * @param commands Output: commands executed (NULL to skip)
148 */
149void lle_safety_get_event_stats(uint64_t *events_fired, uint64_t *dir_changes,
150 uint64_t *commands) {
151 if (!g_lle_integration || !g_lle_integration->event_hub) {
152 if (events_fired)
153 *events_fired = 0;
154 if (dir_changes)
155 *dir_changes = 0;
156 if (commands)
157 *commands = 0;
158 return;
159 }
160
161 lle_shell_event_hub_t *hub = g_lle_integration->event_hub;
162
163 if (events_fired)
164 *events_fired = hub->events_fired;
165 if (dir_changes)
166 *dir_changes = hub->directory_changes;
167 if (commands)
168 *commands = hub->commands_executed;
169}
