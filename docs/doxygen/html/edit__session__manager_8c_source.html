59lle_edit_session_manager_create(lle_edit_session_manager_t **manager,
60 lle_memory_pool_t *memory_pool,
61 lle_history_core_t *history_core,
62 const lle_session_manager_config_t *config) {
63 if (!manager || !history_core) /* memory_pool can be NULL */ {
64 return LLE_ERROR_INVALID_PARAMETER;
65 }
66
67 lle_edit_session_manager_t *new_manager =
68 lle_pool_alloc(sizeof(lle_edit_session_manager_t));
69 if (!new_manager) {
70 return LLE_ERROR_OUT_OF_MEMORY;
71 }
72
73 memset(new_manager, 0, sizeof(lle_edit_session_manager_t));
74
75 new_manager->memory_pool = memory_pool;
76 new_manager->history_core = history_core;
77
78 if (config) {
79 memcpy(&new_manager->config, config,
80 sizeof(lle_session_manager_config_t));
81 } else {
82 lle_edit_session_manager_get_default_config(&new_manager->config);
83 }
84
85 /* Allocate sessions array */
86 new_manager->sessions = lle_pool_alloc(new_manager->config.max_sessions *
87 sizeof(lle_edit_session_t *));
88 if (!new_manager->sessions) {
89 return LLE_ERROR_OUT_OF_MEMORY;
90 }
91
92 memset(new_manager->sessions, 0,
93 new_manager->config.max_sessions * sizeof(lle_edit_session_t *));
94
95 new_manager->session_count = 0;
96 new_manager->next_session_id = 1;
97 new_manager->current_session = NULL;
98 new_manager->active = true;
99
100 *manager = new_manager;
101 return LLE_SUCCESS;

108 }
109

125lle_result_t
126lle_edit_session_manager_start_session(lle_edit_session_manager_t *manager,
127 size_t entry_index,
128 lle_edit_session_t **session) {
129 if (!manager || !manager->active || !session) {
130 return LLE_ERROR_INVALID_PARAMETER;
131 }
132
133 /* Check if already at max sessions */
134 if (manager->session_count >= manager->config.max_sessions) {
135 return LLE_ERROR_INVALID_STATE;
136 }
137
138 /* Verify entry exists */
139 size_t entry_count = 0;
140 lle_result_t result =
141 lle_history_get_entry_count(manager->history_core, &entry_count);
142 if (result != LLE_SUCCESS || entry_index >= entry_count) {
143 return LLE_ERROR_INVALID_PARAMETER;
144 }
145
146 /* Create new session */
147 lle_edit_session_t *new_session = create_edit_session(manager, entry_index);
148 if (!new_session) {
149 return LLE_ERROR_OUT_OF_MEMORY;
150 }
151
152 /* Add to sessions array */
153 for (size_t i = 0; i < manager->config.max_sessions; i++) {
154 if (!manager->sessions[i]) {
155 manager->sessions[i] = new_session;
156 manager->session_count++;
157 break;
158 }
159 }
160
161 /* Set as current session */
162 manager->current_session = new_session;
163
164 *session = new_session;
165 return LLE_SUCCESS;
166}

178lle_result_t lle_edit_session_manager_record_operation(
179 lle_edit_session_manager_t *manager, lle_edit_session_t *session,
180 const lle_edit_operation_t *operation) {
181 if (!manager || !manager->active || !session || !operation) {
182 return LLE_ERROR_INVALID_PARAMETER;
183 }
184
185 if (!manager->config.track_operations) {
186 return LLE_SUCCESS; /* Operation tracking disabled */
187 }
188
189 /* Check operation limit */
190 if (session->operation_count >= manager->config.max_operations) {
191 return LLE_ERROR_INVALID_STATE;
192 }
193
194 /* Allocate operation record */
195 lle_edit_operation_t *op = lle_pool_alloc(sizeof(lle_edit_operation_t));
196 if (!op) {
197 return LLE_ERROR_OUT_OF_MEMORY;
198 }
199
200 memcpy(op, operation, sizeof(lle_edit_operation_t));
201 op->next = NULL;
202
203 /* Copy text if present */
204 if (operation->text && operation->text_length > 0) {
205 op->text = lle_pool_alloc(operation->text_length + 1);
206 if (!op->text) {
207 return LLE_ERROR_OUT_OF_MEMORY;
208 }
209 memcpy(op->text, operation->text, operation->text_length);
210 op->text[operation->text_length] = '\0';
211 }
212
213 /* Get timestamp */
214 clock_gettime(CLOCK_MONOTONIC, &op->timestamp);
215
216 /* Add to session operation list */
217 if (!session->first_operation) {
218 session->first_operation = op;
219 session->last_operation = op;
220 } else {
221 session->last_operation->next = op;
222 session->last_operation = op;
223 }
224
225 session->operation_count++;
226
227 return LLE_SUCCESS;
228}

231lle_edit_session_manager_update_text(lle_edit_session_manager_t *manager,

249 session->current_length = new_length;
250 session->has_modifications = true;

259lle_result_t
260lle_edit_session_manager_complete_session(lle_edit_session_manager_t *manager,
261 lle_edit_session_t *session) {
262 if (!manager || !manager->active || !session) {
263 return LLE_ERROR_INVALID_PARAMETER;
264 }
265
266 if (session->state != LLE_EDIT_SESSION_ACTIVE &&
267 session->state != LLE_EDIT_SESSION_MODIFIED) {
268 return LLE_ERROR_INVALID_STATE;
269 }
270
271 session->state = LLE_EDIT_SESSION_COMPLETED;
272
273 /* If modifications were made, update the history entry */
274 if (session->has_modifications && session->current_text) {
275 /* Get the history entry */
276 lle_history_entry_t *entry = NULL;
277 lle_result_t result = lle_history_get_entry_by_index(
278 manager->history_core, session->entry_index, &entry);
279
280 if (result == LLE_SUCCESS && entry) {
281 /* Update entry command text */
282 /* Note: This would normally call lle_history_update_entry() */
283 /* For now, we just mark the session as complete */
284 (void)entry; /* Suppress unused warning */
285 }
286 }
287
288 /* Remove from sessions array */
289 for (size_t i = 0; i < manager->config.max_sessions; i++) {
290 if (manager->sessions[i] == session) {
291 manager->sessions[i] = NULL;
292 manager->session_count--;
293 break;
294 }
295 }
296
297 /* Clear current session if this was it */
298 if (manager->current_session == session) {
299 manager->current_session = NULL;
300 }
301
302 /* Free session resources */
303 free_edit_session(session);
304
305 return LLE_SUCCESS;
306}

337lle_result_t
338lle_edit_session_manager_check_timeout(lle_edit_session_manager_t *manager,
339 lle_edit_session_t *session,
340 bool *timed_out) {
341 if (!manager || !manager->active || !session || !timed_out) {
342 return LLE_ERROR_INVALID_PARAMETER;
343 }
344
345 *timed_out = false;
346
347 if (manager->config.session_timeout_ms == 0) {
348 return LLE_SUCCESS; /* No timeout configured */
349 }
350
351 struct timespec now;
352 clock_gettime(CLOCK_MONOTONIC, &now);
353
354 int64_t elapsed_ms = timespec_diff_ms(&session->last_modified, &now);
355
356 if (elapsed_ms >= (int64_t)manager->config.session_timeout_ms) {
357 *timed_out = true;
358 session->state = LLE_EDIT_SESSION_ERROR;
359 }
360
361 return LLE_SUCCESS;
362}

366static lle_edit_session_t *
367create_edit_session(lle_edit_session_manager_t *manager, size_t entry_index) {
368 lle_edit_session_t *session = lle_pool_alloc(sizeof(lle_edit_session_t));
369 if (!session) {
370 return NULL;
371 }
372
373 memset(session, 0, sizeof(lle_edit_session_t));
374
375 session->session_id = manager->next_session_id++;
376 session->state = LLE_EDIT_SESSION_ACTIVE;
377 session->entry_index = entry_index;
378
379 /* Get original entry text */
380 char *text = NULL;
381 size_t length = 0;
382 if (get_entry_text(manager->history_core, entry_index, &text, &length) ==
383 LLE_SUCCESS) {
384 session->original_text = text;
385 session->original_length = length;
386
387 /* Initialize current text as copy of original */
388 session->current_text = text;
389 session->current_length = length;
390 }
391
392 /* Initialize timestamps */
393 clock_gettime(CLOCK_MONOTONIC, &session->start_time);
394 session->last_modified = session->start_time;
395
396 session->has_modifications = false;
397 session->multiline_mode = false;
398 session->first_operation = NULL;
399 session->last_operation = NULL;
400 session->operation_count = 0;
401 session->structure = NULL;
402
403 return session;
404}

412static lle_result_t get_entry_text(lle_history_core_t *history, size_t index,

424 if (!text_copy) {
425 return LLE_ERROR_OUT_OF_MEMORY;
426 }
427
428 memcpy(text_copy, entry->command, cmd_len);
429 text_copy[cmd_len] = '\0';
430
431 *text = text_copy;
432 *length = cmd_len;
433
434 return LLE_SUCCESS;
435}

437static int64_t timespec_diff_ms(const struct timespec *start,
438 const struct timespec *end) {
439 int64_t sec_diff = end->tv_sec - start->tv_sec;
440 int64_t nsec_diff = end->tv_nsec - start->tv_nsec;
