20/*
21 * Initialize event filter system
22 */
23lle_result_t lle_event_filter_system_init(lle_event_system_t *system) {
24 if (!system) {
25 return LLE_ERROR_INVALID_PARAMETER;
26 }
27
28 /* Already initialized */
29 if (system->filter_system) {
30 return LLE_SUCCESS;
31 }
32
33 /* Allocate filter system */
34 lle_event_filter_system_t *filter_sys =
35 lle_pool_alloc(sizeof(lle_event_filter_system_t));
36 if (!filter_sys) {
37 return LLE_ERROR_OUT_OF_MEMORY;
38 }
39
40 memset(filter_sys, 0, sizeof(lle_event_filter_system_t));
41
42 /* Allocate filter array */
43 filter_sys->filter_capacity = LLE_FILTER_INITIAL_CAPACITY;
44 filter_sys->filters = lle_pool_alloc(sizeof(lle_event_filter_t *) *
45 filter_sys->filter_capacity);
46 if (!filter_sys->filters) {
47 lle_pool_free(filter_sys);
48 return LLE_ERROR_OUT_OF_MEMORY;
49 }
50
51 memset(filter_sys->filters, 0,
52 sizeof(lle_event_filter_t *) * filter_sys->filter_capacity);
53 filter_sys->filter_count = 0;
54
55 /* Initialize mutex */
56 pthread_mutex_init(&filter_sys->filter_mutex, NULL);
57
58 system->filter_system = filter_sys;
59 return LLE_SUCCESS;
60}

62/*
63 * Destroy event filter system
64 */
65void lle_event_filter_system_destroy(lle_event_system_t *system) {
66 if (!system || !system->filter_system) {
67 return;
68 }
69
70 lle_event_filter_system_t *filter_sys = system->filter_system;
71
72 /* Destroy mutex */
73 pthread_mutex_destroy(&filter_sys->filter_mutex);
74
75 /* Free all filters */
76 if (filter_sys->filters) {
77 for (size_t i = 0; i < filter_sys->filter_count; i++) {
78 if (filter_sys->filters[i]) {
79 lle_pool_free(filter_sys->filters[i]);
80 }
81 }
82 lle_pool_free(filter_sys->filters);
83 }
84
85 /* Free filter system */
86 lle_pool_free(filter_sys);
87
88 system->filter_system = NULL;
89}

91/*
92 * Add event filter
93 */
94lle_result_t lle_event_filter_add(lle_event_system_t *system, const char *name,
95 lle_event_filter_fn filter, void *user_data) {
96 if (!system || !name || !filter) {
97 return LLE_ERROR_INVALID_PARAMETER;
98 }
99
100 /* Initialize filter system if needed */
101 if (!system->filter_system) {
102 lle_result_t result = lle_event_filter_system_init(system);
103 if (result != LLE_SUCCESS) {
104 return result;
105 }
106 }
107
108 lle_event_filter_system_t *filter_sys = system->filter_system;
109
110 pthread_mutex_lock(&filter_sys->filter_mutex);
111
112 /* Check if filter with this name already exists */
113 for (size_t i = 0; i < filter_sys->filter_count; i++) {
114 if (strcmp(filter_sys->filters[i]->name, name) == 0) {
115 pthread_mutex_unlock(&filter_sys->filter_mutex);
116 return LLE_ERROR_ALREADY_INITIALIZED; /* Filter with this name
117 already exists */
118 }
119 }
120
121 /* Check capacity */
122 if (filter_sys->filter_count >= filter_sys->filter_capacity) {
123 /* Need to grow array */
124 size_t new_capacity = filter_sys->filter_capacity * 2;
125 lle_event_filter_t **new_array =
126 lle_pool_alloc(sizeof(lle_event_filter_t *) * new_capacity);
127 if (!new_array) {
128 pthread_mutex_unlock(&filter_sys->filter_mutex);
129 return LLE_ERROR_OUT_OF_MEMORY;
130 }
131
132 /* Copy existing filters */
133 memcpy(new_array, filter_sys->filters,
134 sizeof(lle_event_filter_t *) * filter_sys->filter_count);
135 memset(new_array + filter_sys->filter_count, 0,
136 sizeof(lle_event_filter_t *) *
137 (new_capacity - filter_sys->filter_count));
138
139 /* Free old array and update */
140 lle_pool_free(filter_sys->filters);
141 filter_sys->filters = new_array;
142 filter_sys->filter_capacity = new_capacity;
143 }
144
145 /* Allocate new filter */
146 lle_event_filter_t *new_filter = lle_pool_alloc(sizeof(lle_event_filter_t));
147 if (!new_filter) {
148 pthread_mutex_unlock(&filter_sys->filter_mutex);
149 return LLE_ERROR_OUT_OF_MEMORY;
150 }
151
152 /* Initialize filter */
153 memset(new_filter, 0, sizeof(lle_event_filter_t));
154 new_filter->filter = filter;
155 new_filter->user_data = user_data;
156 new_filter->enabled = true;
157 strncpy(new_filter->name, name, sizeof(new_filter->name) - 1);
158 new_filter->name[sizeof(new_filter->name) - 1] = '\0';
159
160 /* Add to array */
161 filter_sys->filters[filter_sys->filter_count] = new_filter;
162 filter_sys->filter_count++;
163
164 pthread_mutex_unlock(&filter_sys->filter_mutex);
165
166 return LLE_SUCCESS;
167}

169/*
170 * Remove event filter by name
171 */
172lle_result_t lle_event_filter_remove(lle_event_system_t *system,
173 const char *name) {
174 if (!system || !name) {
175 return LLE_ERROR_INVALID_PARAMETER;
176 }
177
178 if (!system->filter_system) {
179 return LLE_ERROR_NOT_FOUND;
180 }
181
182 lle_event_filter_system_t *filter_sys = system->filter_system;
183
184 pthread_mutex_lock(&filter_sys->filter_mutex);
185
186 /* Find filter */
187 for (size_t i = 0; i < filter_sys->filter_count; i++) {
188 if (strcmp(filter_sys->filters[i]->name, name) == 0) {
189 /* Free filter */
190 lle_pool_free(filter_sys->filters[i]);
191
192 /* Shift remaining filters down */
193 for (size_t j = i; j < filter_sys->filter_count - 1; j++) {
194 filter_sys->filters[j] = filter_sys->filters[j + 1];
195 }
196
197 filter_sys->filters[filter_sys->filter_count - 1] = NULL;
198 filter_sys->filter_count--;
199
200 pthread_mutex_unlock(&filter_sys->filter_mutex);
201 return LLE_SUCCESS;
202 }
203 }
204
205 pthread_mutex_unlock(&filter_sys->filter_mutex);
206 return LLE_ERROR_NOT_FOUND;
207}

269/*
270 * Apply filters to event (internal helper)
271 */
272lle_filter_result_t lle_event_filter_apply(lle_event_system_t *system,
273 lle_event_t *event) {
274 if (!system || !event) {
275 return LLE_FILTER_ERROR;
276 }
277
278 if (!system->filter_system) {
279 return LLE_FILTER_PASS; /* No filters, pass through */
280 }
281
282 lle_event_filter_system_t *filter_sys = system->filter_system;
283
284 pthread_mutex_lock(&filter_sys->filter_mutex);
285
286 filter_sys->total_events_filtered++;
287
288 /* Apply each enabled filter in sequence */
289 for (size_t i = 0; i < filter_sys->filter_count; i++) {
290 lle_event_filter_t *filter = filter_sys->filters[i];
291
292 if (!filter->enabled) {
293 continue;
294 }
295
296 /* Call filter function */
297 lle_filter_result_t result = filter->filter(event, filter->user_data);
298
299 /* Update filter statistics */
300 filter->events_filtered++;
301
302 switch (result) {
303 case LLE_FILTER_PASS:
304 filter->events_passed++;
305 break;
306
307 case LLE_FILTER_BLOCK:
308 filter->events_blocked++;
309 filter_sys->total_events_blocked++;
310 pthread_mutex_unlock(&filter_sys->filter_mutex);
311 return LLE_FILTER_BLOCK; /* Stop processing, block event */
312
313 case LLE_FILTER_TRANSFORM:
314 filter->events_transformed++;
315 /* Continue to next filter with transformed event */
316 break;
317
318 case LLE_FILTER_ERROR:
319 filter->events_errored++;
320 /* Continue to next filter on error (don't block) */
321 break;
322 }
323 }
324
325 pthread_mutex_unlock(&filter_sys->filter_mutex);
326
327 return LLE_FILTER_PASS; /* All filters passed or transformed */
328}
