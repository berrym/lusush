5 * Get grapheme break property for a codepoint
6 *
7 * Implements UAX #29 grapheme break property detection.
8 */
9grapheme_break_property_t get_grapheme_break_property(uint32_t cp) {
10 /* CR/LF */
11 if (cp == 0x000D)
12 return GB_CR;
13 if (cp == 0x000A)
14 return GB_LF;
15
16 /* Control characters (C0 and C1) */
17 if (cp < 0x20)
18 return GB_CONTROL;
19 if (cp >= 0x7F && cp <= 0x9F)
20 return GB_CONTROL;
21
22 /* Zero Width Joiner */
23 if (cp == 0x200D)
24 return GB_ZWJ;
25
26 /* Combining marks (Extend) */
27 if (cp >= 0x0300 && cp <= 0x036F)
28 return GB_EXTEND; /* Combining Diacritical Marks */
29 if (cp >= 0x1AB0 && cp <= 0x1AFF)
30 return GB_EXTEND; /* Combining Diacritical Marks Extended */
31 if (cp >= 0x1DC0 && cp <= 0x1DFF)
32 return GB_EXTEND; /* Combining Diacritical Marks Supplement */
33 if (cp >= 0x20D0 && cp <= 0x20FF)
34 return GB_EXTEND; /* Combining Diacritical Marks for Symbols */
35 if (cp >= 0xFE20 && cp <= 0xFE2F)
36 return GB_EXTEND; /* Combining Half Marks */
37
38 /* Regional Indicators (flags) */
39 if (cp >= 0x1F1E6 && cp <= 0x1F1FF)
40 return GB_REGIONAL_INDICATOR;
41
42 /* Hangul Jamo */
43 if (cp >= 0x1100 && cp <= 0x115F)
44 return GB_L; /* Hangul Choseong */
45 if (cp >= 0x1160 && cp <= 0x11A7)
46 return GB_V; /* Hangul Jungseong */
47 if (cp >= 0x11A8 && cp <= 0x11FF)
48 return GB_T; /* Hangul Jongseong */
49
50 /* Hangul Syllables */
51 if (cp >= 0xAC00 && cp <= 0xD7A3) {
52 int s_index = cp - 0xAC00;
53 int t_index = s_index % 28;
54 if (t_index == 0)
55 return GB_LV;
56 else
57 return GB_LVT;
58 }
59
60 /* Extended Pictographic (Emoji) */
61 if (cp >= 0x1F300 && cp <= 0x1F9FF)
62 return GB_EXTENDED_PICTOGRAPHIC; /* Emoji blocks */
63 if (cp >= 0x2600 && cp <= 0x27BF)
64 return GB_EXTENDED_PICTOGRAPHIC; /* Miscellaneous Symbols */
65
66 /* SpacingMark */
67 if (cp >= 0x0903 && cp <= 0x0903)
68 return GB_SPACING_MARK; /* Devanagari */
69
70 return GB_OTHER;
71}

73/**
74 * Check if there's a boundary between two codepoints per UAX #29
75 */
76bool is_grapheme_cluster_boundary(uint32_t cp1, uint32_t cp2, bool prev_was_zwj,
77 int ri_sequence_count) {
78 grapheme_break_property_t prop1 = get_grapheme_break_property(cp1);
79 grapheme_break_property_t prop2 = get_grapheme_break_property(cp2);
80
81 /* GB3: CR × LF */
82 if (prop1 == GB_CR && prop2 == GB_LF) {
83 return false; /* No break */
84 }
85
86 /* GB4: (Control | CR | LF) ÷ */
87 if (prop1 == GB_CONTROL || prop1 == GB_CR || prop1 == GB_LF) {
88 return true; /* Break */
89 }
90
91 /* GB5: ÷ (Control | CR | LF) */
92 if (prop2 == GB_CONTROL || prop2 == GB_CR || prop2 == GB_LF) {
93 return true; /* Break */
94 }
95
96 /* GB6: L × (L | V | LV | LVT) */
97 if (prop1 == GB_L &&
98 (prop2 == GB_L || prop2 == GB_V || prop2 == GB_LV || prop2 == GB_LVT)) {
99 return false; /* No break */
100 }
101
102 /* GB7: (LV | V) × (V | T) */
103 if ((prop1 == GB_LV || prop1 == GB_V) && (prop2 == GB_V || prop2 == GB_T)) {
104 return false; /* No break */
105 }
106
107 /* GB8: (LVT | T) × T */
108 if ((prop1 == GB_LVT || prop1 == GB_T) && prop2 == GB_T) {
109 return false; /* No break */
110 }
111
112 /* GB9: × (Extend | ZWJ) */
113 if (prop2 == GB_EXTEND || prop2 == GB_ZWJ) {
114 return false; /* No break */
115 }
116
117 /* GB9a: × SpacingMark */
118 if (prop2 == GB_SPACING_MARK) {
119 return false; /* No break */
120 }
121
122 /* GB9b: Prepend × */
123 if (prop1 == GB_PREPEND) {
124 return false; /* No break */
125 }
126
127 /* GB11: \p{Extended_Pictographic} Extend* ZWJ × \p{Extended_Pictographic}
128 */
129 if (prev_was_zwj && prop1 == GB_EXTENDED_PICTOGRAPHIC &&
130 prop2 == GB_EXTENDED_PICTOGRAPHIC) {
131 return false; /* No break (emoji sequence) */
132 }
133
134 /* GB12/GB13: Regional Indicator × Regional Indicator (pairs) */
135 if (prop1 == GB_REGIONAL_INDICATOR && prop2 == GB_REGIONAL_INDICATOR) {
136 /* Regional indicators pair up (for flags) */
137 /* Odd position = no break, even position = break */
138 return (ri_sequence_count % 2) == 0;
139 }
140
141 /* GB999: Any ÷ Any */
142 return true; /* Default: break */
143}

145/**
146 * Check if position is at grapheme boundary
147 */
148bool is_grapheme_boundary_at_position(const char *pos, const char *text_start,
149 const char *text_end) {
150 /* Start of text is always a boundary */
151 if (pos == text_start) {
152 return true;
153 }
154
155 /* End of text is always a boundary */
156 if (pos >= text_end) {
157 return true;
158 }
159
160 /* Decode current codepoint */
161 uint32_t cp_current;
162 int len_current = lle_utf8_sequence_length(*pos);
163 if (len_current == 0 || pos + len_current > text_end) {
164 return true; /* Invalid UTF-8, treat as boundary */
165 }
166 lle_utf8_decode_codepoint(pos, len_current, &cp_current);
167
168 /* Find previous codepoint */
169 const char *prev_pos = pos - 1;
170
171 /* Scan backward to find start of previous UTF-8 sequence */
172 while (prev_pos >= text_start && ((*prev_pos & 0xC0) == 0x80)) {
173 prev_pos--; /* Skip continuation bytes */
174 }
175
176 if (prev_pos < text_start) {
177 return true; /* Start of text */
178 }
179
180 /* Decode previous codepoint */
181 uint32_t cp_prev;
182 int len_prev = lle_utf8_sequence_length(*prev_pos);
183 if (len_prev == 0) {
184 return true; /* Invalid UTF-8 */
185 }
186 lle_utf8_decode_codepoint(prev_pos, len_prev, &cp_prev);
187
188 /* Check if previous was ZWJ (for emoji sequences) */
189 bool prev_was_zwj = false;
190 if (prev_pos > text_start) {
191 const char *before_prev = prev_pos - 1;
192 while (before_prev >= text_start && ((*before_prev & 0xC0) == 0x80)) {
193 before_prev--;
194 }
195 if (before_prev >= text_start) {
196 uint32_t cp_before;
197 int len_before = lle_utf8_sequence_length(*before_prev);
198 lle_utf8_decode_codepoint(before_prev, len_before, &cp_before);
199 prev_was_zwj = (get_grapheme_break_property(cp_before) == GB_ZWJ);
200 }
201 }
202
203 /* Count regional indicator sequence length */
204 int ri_count = 0;
205 if (get_grapheme_break_property(cp_prev) == GB_REGIONAL_INDICATOR) {
206 const char *check_pos = prev_pos;
207 while (check_pos >= text_start) {
208 const char *cp_start = check_pos - 1;
209 while (cp_start >= text_start && ((*cp_start & 0xC0) == 0x80)) {
210 cp_start--;
211 }
212 if (cp_start < text_start)
213 break;
214
215 uint32_t cp;
216 int len = lle_utf8_sequence_length(*cp_start);
217 lle_utf8_decode_codepoint(cp_start, len, &cp);
218
219 if (get_grapheme_break_property(cp) != GB_REGIONAL_INDICATOR) {
220 break;
221 }
222
223 ri_count++;
224 check_pos = cp_start;
225 }
226 }
227
228 /* Apply boundary rules */
229 return is_grapheme_cluster_boundary(cp_prev, cp_current, prev_was_zwj,
230 ri_count);
231}
