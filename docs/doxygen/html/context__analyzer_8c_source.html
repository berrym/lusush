25/**
26 * Check if character is a word boundary
27 */
28static bool is_word_boundary(char c) {
29 return isspace(c) || c == '|' || c == ';' || c == '&' || c == '(' ||
30 c == ')' || c == '<' || c == '>' || c == '\0';
31}
32
33/**
34 * Find start of current word
35 */
36static size_t find_word_start(const char *buffer, size_t cursor_pos) {
37 if (cursor_pos == 0) {
38 return 0;
39 }
40
41 size_t pos = cursor_pos;
42
43 /* Move back to start of word */
44 while (pos > 0 && !is_word_boundary(buffer[pos - 1])) {
45 pos--;
46 }
47
48 return pos;
49}

51/**
52 * Extract word from buffer
53 */
54static char *extract_word(const char *buffer, size_t start, size_t end,
55 lle_memory_pool_t *pool) {
56 (void)pool; /* Reserved for pool-based allocation */
57 if (end <= start) {
58 /* Empty string - must null-terminate! */
59 char *empty = lle_pool_alloc(1);
60 if (empty) {
61 empty[0] = '\0';
62 }
63 return empty;
64 }
65
66 size_t len = end - start;
67 char *word = lle_pool_alloc(len + 1);
68 if (!word) {
69 return NULL;
70 }
71
72 memcpy(word, buffer + start, len);
73 word[len] = '\0';
74
75 return word;
76}

78/**
79 * Check if position is at start of command
80 */
81static bool is_command_position(const char *buffer, size_t pos) {
82 /* Start of buffer is always command position */
83 if (pos == 0) {
84 return true;
85 }
86
87 /* Look backwards for command separator */
88 for (size_t i = pos; i > 0; i--) {
89 char c = buffer[i - 1];
90
91 if (c == '|' || c == ';' || c == '&') {
92 return true; /* After separator = command position */
93 }
94
95 if (!isspace(c)) {
96 return false; /* Found non-separator = argument position */
97 }
98 }
99
100 return true; /* Only whitespace before = command position */
101}

103/**
104 * Check if inside quotes
105 */
106static bool is_inside_quotes(const char *buffer, size_t pos) {
107 bool in_single = false;
108 bool in_double = false;
109
110 for (size_t i = 0; i < pos; i++) {
111 char c = buffer[i];
112
113 if (c == '\'' && !in_double) {
114 in_single = !in_single;
115 } else if (c == '"' && !in_single) {
116 in_double = !in_double;
117 }
118 }
119
120 return in_single || in_double;
121}

123/**
124 * Check if after redirect operator
125 */
126static bool is_after_redirect(const char *buffer, size_t pos) {
127 if (pos == 0) {
128 return false;
129 }
130
131 /* Look backwards for redirect operator */
132 for (size_t i = pos; i > 0; i--) {
133 char c = buffer[i - 1];
134
135 if (c == '>' || c == '<') {
136 return true;
137 }
138
139 if (!isspace(c)) {
140 return false; /* Found non-whitespace before redirect */
141 }
142 }
143
144 return false;
145}
146
147/**
148 * Check if in variable assignment
149 */
150static bool is_in_assignment(const char *buffer, size_t pos) {
151 /* Look backwards for '=' without spaces before it */
152 for (size_t i = pos; i > 0; i--) {
153 char c = buffer[i - 1];
154
155 if (c == '=') {
156 /* Check if there's a valid variable name before = */
157 if (i > 1 && (isalnum(buffer[i - 2]) || buffer[i - 2] == '_')) {
158 return true;
159 }
160 }
161
162 if (isspace(c) || c == ';' || c == '|') {
163 return false; /* Hit separator */
164 }
165 }
166
167 return false;
168}

170/**
171 * Determine context type
172 */
173static lle_completion_context_type_t determine_context_type(const char *buffer,
174 size_t word_start) {
175 /* Check for variable expansion */
176 if (word_start > 0 && buffer[word_start - 1] == '$') {
177 return LLE_CONTEXT_VARIABLE;
178 }
179
180 /* Check for assignment */
181 if (is_in_assignment(buffer, word_start)) {
182 return LLE_CONTEXT_ASSIGNMENT;
183 }
184
185 /* Check for redirect */
186 if (is_after_redirect(buffer, word_start)) {
187 return LLE_CONTEXT_REDIRECT;
188 }
189
190 /* Check if at command position */
191 if (is_command_position(buffer, word_start)) {
192 return LLE_CONTEXT_COMMAND;
193 }
194
195 /* Default: argument completion */
196 return LLE_CONTEXT_ARGUMENT;
197}

199/**
200 * Extract command context (for argument position)
201 */
202static void extract_command_context(const char *buffer, size_t pos,
203 char **command, int *arg_index,
204 lle_memory_pool_t *pool) {
205 *command = NULL;
206 *arg_index = -1;
207
208 /* Find start of current command */
209 size_t cmd_start = 0;
210 for (size_t i = pos; i > 0; i--) {
211 char c = buffer[i - 1];
212 if (c == '|' || c == ';' || c == '&') {
213 cmd_start = i;
214 break;
215 }
216 }
217
218 /* Skip whitespace */
219 while (cmd_start < pos && isspace(buffer[cmd_start])) {
220 cmd_start++;
221 }
222
223 if (cmd_start >= pos) {
224 return; /* No command found */
225 }
226
227 /* Extract command name */
228 size_t cmd_end = cmd_start;
229 while (cmd_end < pos && !isspace(buffer[cmd_end])) {
230 cmd_end++;
231 }
232
233 *command = extract_word(buffer, cmd_start, cmd_end, pool);
234
235 /* Count arguments */
236 int arg_count = 0;
237 bool in_word = false;
238
239 for (size_t i = cmd_end; i < pos; i++) {
240 if (isspace(buffer[i])) {
241 in_word = false;
242 } else if (!in_word) {
243 arg_count++;
244 in_word = true;
245 }
246 }
247
248 *arg_index = arg_count;
249}

255lle_result_t lle_context_analyze(const char *buffer, size_t cursor_pos,
256 lle_memory_pool_t *pool,
257 lle_context_analyzer_t **out_context) {
258 if (!buffer || !pool || !out_context) {
259 return LLE_ERROR_INVALID_PARAMETER;
260 }
261
262 /* Find word being completed */
263 size_t word_start = find_word_start(buffer, cursor_pos);
264 size_t word_end = cursor_pos;
265
266 /* Determine context type */
267 lle_completion_context_type_t type =
268 determine_context_type(buffer, word_start);
269
270 /* Extract partial word */
271 char *partial_word = extract_word(buffer, word_start, word_end, pool);
272 if (!partial_word) {
273 return LLE_ERROR_OUT_OF_MEMORY;
274 }
275
276 /* Extract command context if in argument position */
277 char *command_name = NULL;
278 int arg_index = -1;
279 if (type == LLE_CONTEXT_ARGUMENT) {
280 extract_command_context(buffer, word_start, &command_name, &arg_index,
281 pool);
282 }
283
284 /* Create context structure */
285 lle_context_analyzer_t *context = lle_pool_alloc(sizeof(*context));
286 if (!context) {
287 return LLE_ERROR_OUT_OF_MEMORY;
288 }
289
290 context->type = type;
291 context->word_start = word_start;
292 context->word_end = word_end;
293 context->partial_word = partial_word;
294 context->command_name = command_name;
295 context->argument_index = arg_index;
296 context->in_quotes = is_inside_quotes(buffer, cursor_pos);
297 context->after_redirect = is_after_redirect(buffer, word_start);
298 context->in_assignment = is_in_assignment(buffer, word_start);
299 context->pool = pool;
300
301 *out_context = context;
302 return LLE_SUCCESS;
303}
