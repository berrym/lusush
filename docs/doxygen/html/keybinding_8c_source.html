67/**
68 * Get current time in microseconds
69 */
70static uint64_t get_time_us(void) {
71 struct timeval tv;
72 gettimeofday(&tv, NULL);
73 return (uint64_t)tv.tv_sec * 1000000 + tv.tv_usec;

76/**

79static char *keybinding_strdup(lusush_memory_pool_t *pool, const char *str) {
80 if (str == NULL) {
81 return NULL;
82 }
83
84 size_t len = strlen(str);
85 char *copy;
86
87 if (pool != NULL) {
88 copy = (char *)lusush_pool_alloc(len + 1);
89 } else {
90 copy = (char *)malloc(len + 1);
91 }
92
93 if (copy != NULL) {
94 memcpy(copy, str, len + 1);
95 }
96
97 return copy;

135/**
136 * Parse special key name to key code
137 */
138static lle_result_t parse_special_key(const char *name,
139 lle_special_key_t *key_out) {
140 if (strcmp(name, "RET") == 0 || strcmp(name, "RETURN") == 0) {
141 *key_out = LLE_KEY_ENTER;
142 } else if (strcmp(name, "TAB") == 0) {
143 *key_out = LLE_KEY_TAB;
144 } else if (strcmp(name, "DEL") == 0 || strcmp(name, "DELETE") == 0) {
145 *key_out = LLE_KEY_DELETE;
146 } else if (strcmp(name, "BACKSPACE") == 0 || strcmp(name, "BS") == 0) {
147 *key_out = LLE_KEY_BACKSPACE;
148 } else if (strcmp(name, "UP") == 0) {
149 *key_out = LLE_KEY_UP;
150 } else if (strcmp(name, "DOWN") == 0) {
151 *key_out = LLE_KEY_DOWN;
152 } else if (strcmp(name, "LEFT") == 0) {
153 *key_out = LLE_KEY_LEFT;
154 } else if (strcmp(name, "RIGHT") == 0) {
155 *key_out = LLE_KEY_RIGHT;
156 } else if (strcmp(name, "HOME") == 0) {
157 *key_out = LLE_KEY_HOME;
158 } else if (strcmp(name, "END") == 0) {
159 *key_out = LLE_KEY_END;
160 } else if (strcmp(name, "PAGEUP") == 0 || strcmp(name, "PGUP") == 0) {
161 *key_out = LLE_KEY_PAGE_UP;
162 } else if (strcmp(name, "PAGEDOWN") == 0 || strcmp(name, "PGDN") == 0) {
163 *key_out = LLE_KEY_PAGE_DOWN;
164 } else if (strcmp(name, "INSERT") == 0 || strcmp(name, "INS") == 0) {
165 *key_out = LLE_KEY_INSERT;
166 } else if (strcmp(name, "ESC") == 0 || strcmp(name, "ESCAPE") == 0) {
167 *key_out = LLE_KEY_ESCAPE;
168 } else if (name[0] == 'F' && isdigit(name[1])) {
169 /* F1-F12 */
170 int num = atoi(name + 1);
171 if (num >= 1 && num <= 12) {
172 *key_out = LLE_KEY_F1 + (num - 1);
173 } else {
174 return LLE_ERROR_INVALID_FORMAT;
175 }
176 } else {
177 return LLE_ERROR_INVALID_FORMAT;
178 }
179
180 return LLE_SUCCESS;
181}

188lle_result_t lle_keybinding_manager_create(lle_keybinding_manager_t **manager,
189 lusush_memory_pool_t *pool) {
190 if (manager == NULL) {
191 return LLE_ERROR_NULL_POINTER;
192 }
193
194 /* Allocate manager structure */
195 lle_keybinding_manager_t *new_manager;
196 if (pool != NULL) {
197 new_manager = (lle_keybinding_manager_t *)lusush_pool_alloc(
198 sizeof(lle_keybinding_manager_t));
199 } else {
200 new_manager = (lle_keybinding_manager_t *)malloc(
201 sizeof(lle_keybinding_manager_t));
202 }
203
204 if (new_manager == NULL) {
205 return LLE_ERROR_OUT_OF_MEMORY;
206 }
207
208 memset(new_manager, 0, sizeof(lle_keybinding_manager_t));
209 new_manager->pool = pool;
210 new_manager->current_mode = LLE_KEYMAP_EMACS;
211
212 /* Create hashtable for bindings */
213 lle_hashtable_config_t config;
214 lle_hashtable_config_init_default(&config);
215 config.initial_capacity = LLE_KEYBINDING_INITIAL_SIZE;
216 config.memory_pool = pool;
217 config.use_memory_pool = (pool != NULL);
218 config.hashtable_name = "keybindings";
219
220 /* Use factory to create hashtable */
221 lle_hashtable_factory_t *factory = NULL;
222 lle_result_t result = lle_hashtable_factory_init(&factory, pool);
223 if (result != LLE_SUCCESS) {
224 if (pool != NULL) {
225 lusush_pool_free(new_manager);
226 } else {
227 free(new_manager);
228 }
229 return result;
230 }
231
232 result = lle_hashtable_factory_create_strstr(factory, &config,
233 &new_manager->bindings);
234 lle_hashtable_factory_destroy(factory);
235
236 if (result != LLE_SUCCESS) {
237 if (pool != NULL) {
238 lusush_pool_free(new_manager);
239 } else {
240 free(new_manager);
241 }
242 return result;
243 }
244
245 *manager = new_manager;
246 return LLE_SUCCESS;
247}

249lle_result_t lle_keybinding_manager_destroy(lle_keybinding_manager_t *manager) {
250 if (manager == NULL) {
251 return LLE_ERROR_NULL_POINTER;
252 }
253
254 /* Free all keybinding entries before destroying hashtable */
255 if (manager->bindings != NULL) {
256 /* Use libhashtable enumeration to iterate through all entries */
257 ht_enum_t *enumerator = ht_strstr_enum_create(manager->bindings->ht);
258 if (enumerator != NULL) {
259 const char *key;
260 const char *value_str;
261
262 /* Iterate through all key-value pairs */
263 while (ht_strstr_enum_next(enumerator, &key, &value_str)) {
264 /* Value is a pointer stored as string - convert back to pointer
265 */
266 lle_keybinding_entry_t *entry;
267 sscanf(value_str, "%p", (void **)&entry);
268
269 /* Free the entry and its allocated strings */
270 free_keybinding_entry(manager->pool, entry);
271 }
272
273 ht_strstr_enum_destroy(enumerator);
274 }
275
276 /* Now destroy the hashtable itself */
277 lle_strstr_hashtable_destroy(manager->bindings);
278 }
279
280 /* Free manager structure */
281 if (manager->pool != NULL) {
282 lusush_pool_free(manager);
283 } else {
284 free(manager);
285 }
286
287 return LLE_SUCCESS;
288}

295lle_result_t lle_key_sequence_parse(const char *key_sequence,
296 lle_key_event_t *key_event_out) {
297 if (key_sequence == NULL || key_event_out == NULL) {
298 return LLE_ERROR_NULL_POINTER;
299 }
300
301 memset(key_event_out, 0, sizeof(lle_key_event_t));
302
303 const char *p = key_sequence;
304
305 /* Parse modifiers */
306 while (*p != '\0') {
307 if (p[0] == 'C' && p[1] == '-') {
308 key_event_out->ctrl = true;
309 p += 2;
310 } else if (p[0] == 'M' && p[1] == '-') {
311 key_event_out->alt = true;
312 p += 2;
313 } else if (p[0] == 'S' && p[1] == '-') {
314 key_event_out->shift = true;
315 p += 2;
316 } else {
317 break;
318 }
319 }
320
321 /* Parse key */
322 if (*p == '\0') {
323 return LLE_ERROR_INVALID_FORMAT;
324 }
325
326 /* Check for special key name */
327 if (isupper(*p) || strcmp(p, "RET") == 0) {
328 lle_special_key_t special_key;
329 lle_result_t result = parse_special_key(p, &special_key);
330 if (result == LLE_SUCCESS) {
331 key_event_out->is_special = true;
332 key_event_out->special_key = special_key;
333 return LLE_SUCCESS;
334 }
335 }
336
337 /* Regular character */
338 if (key_event_out->ctrl && islower(*p)) {
339 /* Ctrl+letter is typically uppercase in ASCII control codes */
340 key_event_out->codepoint = toupper(*p);
341 } else {
342 key_event_out->codepoint = (uint32_t)(*p);
343 }
344
345 return LLE_SUCCESS;
346}

348lle_result_t lle_key_event_to_string(const lle_key_event_t *key_event,
349 char *buffer, size_t buffer_size) {
350 if (key_event == NULL || buffer == NULL || buffer_size == 0) {
351 return LLE_ERROR_NULL_POINTER;
352 }
353
354 char *p = buffer;
355 size_t remaining = buffer_size;
356
357 /* Add modifiers */
358 if (key_event->ctrl) {
359 if (remaining < 3)
360 return LLE_ERROR_BUFFER_OVERFLOW;
361 *p++ = 'C';
362 *p++ = '-';
363 remaining -= 2;
364 }
365 if (key_event->alt) {
366 if (remaining < 3)
367 return LLE_ERROR_BUFFER_OVERFLOW;
368 *p++ = 'M';
369 *p++ = '-';
370 remaining -= 2;
371 }
372 if (key_event->shift) {
373 if (remaining < 3)
374 return LLE_ERROR_BUFFER_OVERFLOW;
375 *p++ = 'S';
376 *p++ = '-';
377 remaining -= 2;
378 }
379
380 /* Add key */
381 if (key_event->is_special) {
382 const char *name = NULL;
383 switch (key_event->special_key) {
384 case LLE_KEY_ENTER:
385 name = "RET";
386 break;
387 case LLE_KEY_TAB:
388 name = "TAB";
389 break;
390 case LLE_KEY_BACKSPACE:
391 name = "BACKSPACE";
392 break;
393 case LLE_KEY_DELETE:
394 name = "DEL";
395 break;
396 case LLE_KEY_UP:
397 name = "UP";
398 break;
399 case LLE_KEY_DOWN:
400 name = "DOWN";
401 break;
402 case LLE_KEY_LEFT:
403 name = "LEFT";
404 break;
405 case LLE_KEY_RIGHT:
406 name = "RIGHT";
407 break;
408 case LLE_KEY_HOME:
409 name = "HOME";
410 break;
411 case LLE_KEY_END:
412 name = "END";
413 break;
414 case LLE_KEY_PAGE_UP:
415 name = "PAGEUP";
416 break;
417 case LLE_KEY_PAGE_DOWN:
418 name = "PAGEDOWN";
419 break;
420 case LLE_KEY_INSERT:
421 name = "INSERT";
422 break;
423 case LLE_KEY_ESCAPE:
424 name = "ESC";
425 break;
426 default:
427 if (key_event->special_key >= LLE_KEY_F1 &&
428 key_event->special_key <= LLE_KEY_F12) {
429 int fn = key_event->special_key - LLE_KEY_F1 + 1;
430 if (remaining < 4)
431 return LLE_ERROR_BUFFER_OVERFLOW;
432 snprintf(p, remaining, "F%d", fn);
433 return LLE_SUCCESS;
434 }
435 return LLE_ERROR_INVALID_PARAMETER;
436 }
437
438 size_t len = strlen(name);
439 if (remaining <= len)
440 return LLE_ERROR_BUFFER_OVERFLOW;
441 strcpy(p, name);
442 } else {
443 if (remaining < 2)
444 return LLE_ERROR_BUFFER_OVERFLOW;
445 /* Convert uppercase back to lowercase for ctrl+letter combinations */
446 char ch = (char)key_event->codepoint;
447 if (key_event->ctrl && isupper(ch)) {
448 ch = tolower(ch);
449 }
450 *p++ = ch;
451 *p = '\0';
452 }
453
454 return LLE_SUCCESS;
455}

462lle_result_t lle_keybinding_manager_bind(lle_keybinding_manager_t *manager,
463 const char *key_sequence,
464 lle_action_simple_t action,
465 const char *function_name) {
466 if (manager == NULL || key_sequence == NULL || action == NULL) {
467 return LLE_ERROR_NULL_POINTER;
468 }
469
470 /* Create keybinding entry */
471 lle_keybinding_entry_t *entry;
472 if (manager->pool != NULL) {
473 entry = (lle_keybinding_entry_t *)lusush_pool_alloc(
474 sizeof(lle_keybinding_entry_t));
475 } else {
476 entry =
477 (lle_keybinding_entry_t *)malloc(sizeof(lle_keybinding_entry_t));
478 }
479
480 if (entry == NULL) {
481 return LLE_ERROR_OUT_OF_MEMORY;
482 }
483
484 /* Initialize action as simple type */
485 entry->action.type = LLE_ACTION_TYPE_SIMPLE;
486 entry->action.func.simple = action;
487 entry->action.name = function_name;
488 entry->mode = manager->current_mode;
489 entry->function_name =
490 function_name ? keybinding_strdup(manager->pool, function_name) : NULL;
491
492 /* Convert entry to string for storage (hackish but works with strstr
493 * hashtable) */
494 char entry_str[32];
495 snprintf(entry_str, sizeof(entry_str), "%p", (void *)entry);
496
497 /* Insert into hashtable */
498 lle_result_t result =
499 lle_strstr_hashtable_insert(manager->bindings, key_sequence, entry_str);
500
501 if (result != LLE_SUCCESS) {
502 free_keybinding_entry(manager->pool, entry);
503 return result;
504 }
505
506 return LLE_SUCCESS;
507}

556 const char *key_sequence) {
557 if (manager == NULL || key_sequence == NULL) {
558 return LLE_ERROR_NULL_POINTER;
559 }
560
561 /* Lookup entry first to free it */
562 const char *entry_str =
563 lle_strstr_hashtable_lookup(manager->bindings, key_sequence);
564 if (entry_str != NULL) {

593lle_result_t
594lle_keybinding_manager_process_key(lle_keybinding_manager_t *manager,
595 lle_editor_t *editor,
596 const lle_key_event_t *key_event) {
597 if (manager == NULL || editor == NULL || key_event == NULL) {
598 return LLE_ERROR_NULL_POINTER;
599 }
600
601 uint64_t start_time = get_time_us();
602
603 /* Convert key event to string */
604 char key_str[LLE_MAX_KEY_SEQUENCE_LENGTH];
605 lle_result_t result =
606 lle_key_event_to_string(key_event, key_str, sizeof(key_str));
607 if (result != LLE_SUCCESS) {
608 return result;
609 }
610
611 /* Lookup binding */
612 const char *entry_str =
613 lle_strstr_hashtable_lookup(manager->bindings, key_str);
614 if (entry_str == NULL) {
615 /* Update stats */
616 uint64_t elapsed = get_time_us() - start_time;
617 manager->total_lookups++;
618 manager->total_lookup_time_us += elapsed;
619 if (elapsed > manager->max_lookup_time_us) {
620 manager->max_lookup_time_us = elapsed;
621 }
622
623 return LLE_ERROR_NOT_FOUND;
624 }
625
626 /* Parse entry pointer */
627 lle_keybinding_entry_t *entry;
628 sscanf(entry_str, "%p", (void **)&entry);
629
630 /* Update stats */
631 uint64_t elapsed = get_time_us() - start_time;
632 manager->total_lookups++;
633 manager->total_lookup_time_us += elapsed;
634 if (elapsed > manager->max_lookup_time_us) {
635 manager->max_lookup_time_us = elapsed;
636 }
637
638 /* Execute action - only simple actions supported here (no readline context)
639 */
640 if (entry->action.type == LLE_ACTION_TYPE_SIMPLE) {
641 return entry->action.func.simple(editor);
642 } else {
643 /* Context-aware actions cannot be executed without readline context */
644 return LLE_ERROR_INVALID_STATE;
645 }
646}

683/* ============================================================================
684 * PRESET LOADING
685 * ============================================================================
686 */
687
688lle_result_t
689lle_keybinding_manager_load_emacs_preset(lle_keybinding_manager_t *manager) {
690 if (manager == NULL) {
691 return LLE_ERROR_NULL_POINTER;
692 }
693
694 /* Set mode to emacs */
695 manager->current_mode = LLE_KEYMAP_EMACS;
696
697 /* ========================================================================
698 * GNU Readline Emacs-style Keybindings
699 *
700 * This loads all SIMPLE actions that operate on lle_editor_t only.
701 * Context-aware actions (requiring readline_context_t) must be registered
702 * separately in lle_readline.c after this preset loads.
703 *
704 * Context-aware bindings registered in lle_readline.c will OVERRIDE
705 * simple bindings registered here for: RIGHT, END, C-e, C-f, C-g,
706 * C-RIGHT, ESC, ENTER (these need autosuggestion/completion/readline
707 * access)
708 * ========================================================================
709 */
710
711 lle_result_t result;
712
713 /* ------------------------------------------------------------------------
714 * MOVEMENT - Character Level
715 * ------------------------------------------------------------------------
716 */
717 result = lle_keybinding_manager_bind(manager, "C-f", lle_forward_char,
718 "forward-char");
719 if (result != LLE_SUCCESS)
720 return result;
721
722 result = lle_keybinding_manager_bind(manager, "C-b", lle_backward_char,
723 "backward-char");
724 if (result != LLE_SUCCESS)
725 return result;
726
727 result = lle_keybinding_manager_bind(manager, "RIGHT", lle_forward_char,
728 "forward-char");
729 if (result != LLE_SUCCESS)
730 return result;
731
732 result = lle_keybinding_manager_bind(manager, "LEFT", lle_backward_char,
733 "backward-char");
734 if (result != LLE_SUCCESS)
735 return result;
736
737 /* ------------------------------------------------------------------------
738 * MOVEMENT - Word Level
739 * ------------------------------------------------------------------------
740 */
741 result = lle_keybinding_manager_bind(manager, "M-f", lle_forward_word,
742 "forward-word");
743 if (result != LLE_SUCCESS)
744 return result;
745
746 result = lle_keybinding_manager_bind(manager, "M-b", lle_backward_word,
747 "backward-word");
748 if (result != LLE_SUCCESS)
749 return result;
750
751 /* ------------------------------------------------------------------------
752 * MOVEMENT - Line Level
753 * ------------------------------------------------------------------------
754 */
