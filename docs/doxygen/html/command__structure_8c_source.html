25lle_result_t lle_command_structure_create(lle_command_structure_t **structure,
26 lle_memory_pool_t *memory_pool) {
27 if (!structure) {
28 return LLE_ERROR_INVALID_PARAMETER;
29 }
30
31 /* memory_pool can be NULL - will use global pool */
32
33 lle_command_structure_t *cmd_struct =
34 lle_pool_alloc(sizeof(lle_command_structure_t));
35 if (!cmd_struct) {
36 return LLE_ERROR_OUT_OF_MEMORY;
37 }
38
39 memset(cmd_struct, 0, sizeof(lle_command_structure_t));
40 cmd_struct->pool = memory_pool; /* Can be NULL */
41 cmd_struct->primary_type = LLE_CONSTRUCT_SIMPLE;
42 cmd_struct->is_complete = false;
43 cmd_struct->has_syntax_error = false;
44
45 *structure = cmd_struct;
46 return LLE_SUCCESS;
47}

49lle_result_t lle_command_structure_destroy(lle_command_structure_t *structure) {
50 if (!structure) {
51 return LLE_ERROR_INVALID_PARAMETER;
52 }
53
54 /* Free keyword list */
55 lle_keyword_position_t *keyword = structure->first_keyword;
56 while (keyword) {
57 lle_keyword_position_t *next = keyword->next;
58 lle_pool_free(keyword);
59 keyword = next;
60 }
61
62 /* Free indentation info */
63 if (structure->indentation) {
64 if (structure->indentation->level_per_line) {
65 lle_pool_free(structure->indentation->level_per_line);
66 }
67 lle_pool_free(structure->indentation);
68 }
69
70 /* Free line mapping */
71 if (structure->line_mapping) {
72 if (structure->line_mapping->line_offsets) {
73 lle_pool_free(structure->line_mapping->line_offsets);
74 }
75 if (structure->line_mapping->line_lengths) {
76 lle_pool_free(structure->line_mapping->line_lengths);
77 }
78 lle_pool_free(structure->line_mapping);
79 }
80
81 /* Free nested construct tree (recursive) */
82 if (structure->root_construct) {
83 lle_free_nested_construct_tree(structure->root_construct);
84 }
85
86 /* Free structure itself */
87 lle_pool_free(structure);
88
89 return LLE_SUCCESS;
90}

97lle_result_t
98lle_command_structure_add_keyword(lle_command_structure_t *structure,
99 lle_keyword_type_t type, size_t offset,
100 size_t line_number, size_t indent_level) {
101 if (!structure) {
102 return LLE_ERROR_INVALID_PARAMETER;
103 }
104
105 /* Allocate new keyword position */
106 lle_keyword_position_t *keyword =
107 lle_pool_alloc(sizeof(lle_keyword_position_t));
108 if (!keyword) {
109 return LLE_ERROR_OUT_OF_MEMORY;
110 }
111
112 keyword->type = type;
113 keyword->offset = offset;
114 keyword->line_number = line_number;
115 keyword->indent_level = indent_level;
116 keyword->next = NULL;
117
118 /* Add to end of list */
119 if (!structure->first_keyword) {
120 structure->first_keyword = keyword;
121 } else {
122 lle_keyword_position_t *current = structure->first_keyword;
123 while (current->next) {
124 current = current->next;
125 }
126 current->next = keyword;
127 }
128
129 structure->keyword_count++;
130
131 return LLE_SUCCESS;
132}

155lle_result_t
156lle_command_structure_find_matching_keyword(lle_command_structure_t *structure,
157 lle_keyword_position_t *keyword,
158 lle_keyword_position_t **match) {
159 if (!structure || !keyword || !match) {
160 return LLE_ERROR_INVALID_PARAMETER;
161 }
162
163 *match = NULL;
164
165 /* Define matching pairs */
166 lle_keyword_type_t search_type = LLE_KEYWORD_NONE;
167
168 switch (keyword->type) {
169 case LLE_KEYWORD_DO:
170 search_type = LLE_KEYWORD_DONE;
171 break;
172 case LLE_KEYWORD_IF:
173 search_type = LLE_KEYWORD_FI;
174 break;
175 case LLE_KEYWORD_THEN:
176 search_type = LLE_KEYWORD_FI;
177 break;
178 case LLE_KEYWORD_CASE:
179 search_type = LLE_KEYWORD_ESAC;
180 break;
181 case LLE_KEYWORD_OPEN_BRACE:
182 search_type = LLE_KEYWORD_CLOSE_BRACE;
183 break;
184 case LLE_KEYWORD_OPEN_PAREN:
185 search_type = LLE_KEYWORD_CLOSE_PAREN;
186 break;
187 default:
188 return LLE_ERROR_NOT_FOUND;
189 }
190
191 /* Search forward for matching keyword at same or lower indent */
192 lle_keyword_position_t *current = keyword->next;
193 int depth = 0;
194
195 while (current) {
196 if (current->type == keyword->type) {
197 depth++;
198 } else if (current->type == search_type) {
199 if (depth == 0) {
200 *match = current;
201 return LLE_SUCCESS;
202 }
203 depth--;
204 }
205 current = current->next;
206 }
207
208 return LLE_ERROR_NOT_FOUND;
209}

216/**

218 */
219static void lle_free_nested_construct_tree(lle_nested_construct_t *construct) {
220 if (!construct) {
221 return;
222 }
223
224 /* Free children first */
225 lle_nested_construct_t *child = construct->first_child;
226 while (child) {
227 lle_nested_construct_t *next_sibling = child->next_sibling;
228 lle_free_nested_construct_tree(child);
229 child = next_sibling;
230 }
231
232 /* Free this construct */
233 lle_pool_free(construct);
234}
