78// Utility function to get current timestamp in microseconds
79static uint64_t get_timestamp_us(void) {
80 struct timeval tv;
81 gettimeofday(&tv, NULL);
82 return (uint64_t)tv.tv_sec * 1000000 + (uint64_t)tv.tv_usec;
83}

85// Utility function to get current timestamp in nanoseconds
86static uint64_t get_timestamp_ns(void) {
87 struct timespec ts;
88 clock_gettime(CLOCK_MONOTONIC, &ts);
89 return (uint64_t)ts.tv_sec * 1000000000 + (uint64_t)ts.tv_nsec;
90}

100// Initialize a single memory pool
101static lusush_pool_error_t init_single_pool(lusush_pool_t *pool,
102 size_t block_size,
103 size_t initial_blocks) {
104 if (!pool || block_size == 0 || initial_blocks == 0) {
105 return LUSUSH_POOL_ERROR_INVALID_SIZE;
106 }
107
108 // Initialize pool structure
109 memset(pool, 0, sizeof(lusush_pool_t));
110 pool->block_size = block_size;
111 pool->initial_blocks = initial_blocks;
112 pool->max_blocks = initial_blocks * 4; // Allow growth up to 4x initial
113
114 // Allocate block management array
115 pool->all_blocks = calloc(pool->max_blocks, sizeof(lusush_pool_block_t));
116 if (!pool->all_blocks) {
117 POOL_ERROR(
118 "Failed to allocate block management array for pool (size=%zu)",
119 block_size);
120 return LUSUSH_POOL_ERROR_MALLOC_FAILED;
121 }
122
123 // Pre-allocate initial blocks
124 for (size_t i = 0; i < initial_blocks; i++) {
125 lusush_pool_block_t *block = &pool->all_blocks[i];
126
127 // Allocate memory for this block
128 block->memory = malloc(block_size);
129 if (!block->memory) {
130 POOL_ERROR("Failed to allocate memory block %zu (size=%zu)", i,
131 block_size);
132 // Cleanup previously allocated blocks
133 for (size_t j = 0; j < i; j++) {
134 free(pool->all_blocks[j].memory);
135 }
136 free(pool->all_blocks);
137 return LUSUSH_POOL_ERROR_MALLOC_FAILED;
138 }
139
140 // Initialize block metadata
141 block->size = block_size;
142 block->in_use = false;
143 block->allocation_time_us = 0;
144 block->allocation_id = 0;
145
146 // Add to free list
147 block->next = pool->free_list;
148 block->prev = NULL;
149 if (pool->free_list) {
150 pool->free_list->prev = block;
151 }
152 pool->free_list = block;
153 }
154
155 pool->current_blocks = initial_blocks;
156 pool->free_blocks = initial_blocks;
157
158 POOL_DEBUG("Initialized pool: block_size=%zu, initial_blocks=%zu",
159 block_size, initial_blocks);
160 return LUSUSH_POOL_SUCCESS;
161}

163// Cleanup a single memory pool
164static void cleanup_single_pool(lusush_pool_t *pool) {
165 if (!pool || !pool->all_blocks) {
166 return;
167 }
168
169 // Free all allocated memory blocks
170 for (size_t i = 0; i < pool->current_blocks; i++) {
171 if (pool->all_blocks[i].memory) {
172 free(pool->all_blocks[i].memory);
173 }
174 }
175
176 // Free block management array
177 free(pool->all_blocks);
178
179 // Clear pool structure
180 memset(pool, 0, sizeof(lusush_pool_t));
181
182 POOL_DEBUG("Cleaned up pool");
183}

185// Find the appropriate pool for a given size
186static lusush_pool_size_t find_pool_for_size(size_t size) {
187 for (int i = 0; i < LUSUSH_POOL_COUNT; i++) {
188 if (size <= POOL_SIZES[i]) {
189 return (lusush_pool_size_t)i;
190 }
191 }
192 // Size is larger than any pool - will use malloc fallback
193 return LUSUSH_POOL_COUNT; // Invalid pool index indicates malloc fallback
194}

196// Allocate a block from specific pool
197static void *allocate_from_pool(lusush_pool_t *pool) {
198 if (!pool || !pool->free_list) {
199 return NULL;
200 }
201
202 // Take first block from free list
203 lusush_pool_block_t *block = pool->free_list;
204
205 // Remove from free list
206 pool->free_list = block->next;
207 if (pool->free_list) {
208 pool->free_list->prev = NULL;
209 }
210
211 // Mark as in use
212 block->in_use = true;
213 block->allocation_time_us = get_timestamp_us();
214 block->allocation_id = ++global_memory_pool->next_allocation_id;
215
216 // Update pool statistics
217 pool->free_blocks--;
218 pool->pool_allocations++;
219
220 POOL_DEBUG("Allocated block: size=%zu, id=%u, ptr=%p", block->size,
221 block->allocation_id, block->memory);
222
223 return block->memory;
224}

226// Return a block to specific pool
227static bool return_to_pool(lusush_pool_t *pool, void *ptr) {
228 if (!pool || !ptr) {
229 return false;
230 }
231
232 // Find the block that contains this pointer
233 lusush_pool_block_t *block = NULL;
234 for (size_t i = 0; i < pool->current_blocks; i++) {
235 if (pool->all_blocks[i].memory == ptr) {
236 block = &pool->all_blocks[i];
237 break;
238 }
239 }
240
241 if (!block || !block->in_use) {
242 return false; // Not found or double free
243 }
244
245 // Mark as free
246 block->in_use = false;
247 block->allocation_time_us = 0;
248 block->allocation_id = 0;
249
250 // Add back to free list
251 block->next = pool->free_list;
252 block->prev = NULL;
253 if (pool->free_list) {
254 pool->free_list->prev = block;
255 }
256 pool->free_list = block;
257
258 // Update statistics
259 pool->free_blocks++;
260 pool->pool_deallocations++;
261
262 POOL_DEBUG("Returned block to pool: size=%zu, ptr=%p", block->size, ptr);
263 return true;
264}

266// Update global statistics
267static void update_stats(bool pool_hit, size_t size,
268 uint64_t allocation_time_ns) {
269 if (!global_memory_pool->enable_statistics) {
270 return;
271 }
272
273 lusush_pool_stats_t *stats = &global_memory_pool->stats;
274
275 stats->total_allocations++;
276 stats->total_bytes_allocated += size;
277 stats->active_allocations++;
278
279 if (pool_hit) {
280 stats->pool_hits++;
281 stats->current_pool_usage += size;
282 if (stats->current_pool_usage > stats->peak_pool_usage) {
283 stats->peak_pool_usage = stats->current_pool_usage;
284 }
285 } else {
286 stats->pool_misses++;
287 stats->malloc_fallbacks++;
288 }
289
290 // Update hit rate
291 stats->pool_hit_rate =
292 (double)stats->pool_hits / stats->total_allocations * 100.0;
293
294 // Update average allocation time
295 if (stats->total_allocations == 1) {
296 stats->avg_allocation_time_ns = allocation_time_ns;
297 } else {
298 stats->avg_allocation_time_ns =
299 (stats->avg_allocation_time_ns * (stats->total_allocations - 1) +
300 allocation_time_ns) /
301 stats->total_allocations;
302 }
303}

309lusush_pool_error_t lusush_pool_init(const lusush_pool_config_t *config) {
310 pthread_mutex_lock(&pool_mutex);
311
312 // Check if already initialized
313 if (global_memory_pool) {
314 pthread_mutex_unlock(&pool_mutex);
315 set_last_error(LUSUSH_POOL_SUCCESS);
316 return LUSUSH_POOL_SUCCESS; // Allow multiple inits
317 }
318
319 // Allocate global pool system
320 global_memory_pool = calloc(1, sizeof(lusush_memory_pool_system_t));
321 if (!global_memory_pool) {
322 pthread_mutex_unlock(&pool_mutex);
323 set_last_error(LUSUSH_POOL_ERROR_MALLOC_FAILED);
324 return LUSUSH_POOL_ERROR_MALLOC_FAILED;
325 }
326
327 // Use provided config or defaults
328 lusush_pool_config_t default_config = lusush_pool_get_default_config();
329 if (!config) {
330 config = &default_config;
331 }
332
333 // Set system configuration
334 global_memory_pool->enable_statistics = config->enable_statistics;
335 global_memory_pool->enable_malloc_fallback = config->enable_malloc_fallback;
336 debug_mode = config->enable_debugging;
337
338 // Initialize timestamp
339 clock_gettime(CLOCK_MONOTONIC, &global_memory_pool->init_time);
340
341 // Initialize individual pools
342 size_t block_counts[LUSUSH_POOL_COUNT] = {
343 config->small_pool_blocks, config->medium_pool_blocks,
344 config->large_pool_blocks, config->xlarge_pool_blocks};
345
346 for (int i = 0; i < LUSUSH_POOL_COUNT; i++) {
347 lusush_pool_error_t result = init_single_pool(
348 &global_memory_pool->pools[i], POOL_SIZES[i], block_counts[i]);
349
350 if (result != LUSUSH_POOL_SUCCESS) {
351 POOL_ERROR("Failed to initialize pool %d", i);
352 // Cleanup already initialized pools
353 for (int j = 0; j < i; j++) {
354 cleanup_single_pool(&global_memory_pool->pools[j]);
355 }
356 free(global_memory_pool);
357 global_memory_pool = NULL;
358 pthread_mutex_unlock(&pool_mutex);
359 set_last_error(result);
360 return result;
361 }
362 }
363
364 global_memory_pool->initialized = true;
365 pool_was_ever_initialized = true;
366
367 pthread_mutex_unlock(&pool_mutex);
368
369 POOL_DEBUG("Memory pool system initialized successfully");
370 POOL_DEBUG("Pools: Small=%zu, Medium=%zu, Large=%zu, XLarge=%zu",
371 block_counts[0], block_counts[1], block_counts[2],
372 block_counts[3]);
373
374 set_last_error(LUSUSH_POOL_SUCCESS);
375 return LUSUSH_POOL_SUCCESS;
376}

378void lusush_pool_shutdown(void) {
379 pthread_mutex_lock(&pool_mutex);
380
381 if (!global_memory_pool) {
382 pthread_mutex_unlock(&pool_mutex);
383 return;
384 }
385
386 POOL_DEBUG("Shutting down memory pool system");
387
388 // Print final statistics if enabled
389 if (global_memory_pool->enable_statistics && debug_mode) {
390 lusush_pool_print_status_report();
391 }
392
393 // Cleanup all pools
394 for (int i = 0; i < LUSUSH_POOL_COUNT; i++) {
395 cleanup_single_pool(&global_memory_pool->pools[i]);
396 }
397
398 // Free global pool system
399 free(global_memory_pool);
400 global_memory_pool = NULL;
401
402 pthread_mutex_unlock(&pool_mutex);
403
404 POOL_DEBUG("Memory pool system shutdown complete");
405}

407void *lusush_pool_alloc(size_t size) {
408 if (size == 0) {
409 set_last_error(LUSUSH_POOL_ERROR_INVALID_SIZE);
410 return NULL;
411 }
412
413 uint64_t start_time = get_timestamp_ns();
414 void *result = NULL;
415 bool pool_hit = false;
416
417 pthread_mutex_lock(&pool_mutex);
418
419 if (!global_memory_pool || !global_memory_pool->initialized) {
420 // Fallback to malloc if pool not initialized
421 pthread_mutex_unlock(&pool_mutex);
422 result = malloc(size);
423 if (result && global_memory_pool) {
424 update_stats(false, size, get_timestamp_ns() - start_time);
425 }
426 set_last_error(result ? LUSUSH_POOL_SUCCESS
427 : LUSUSH_POOL_ERROR_MALLOC_FAILED);
428 return result;
429 }
430
431 // Find appropriate pool
432 lusush_pool_size_t pool_type = find_pool_for_size(size);
433
434 if (pool_type < LUSUSH_POOL_COUNT) {
435 // Try to allocate from pool
436 result = allocate_from_pool(&global_memory_pool->pools[pool_type]);
437 if (result) {
438 pool_hit = true;
439 POOL_DEBUG("Pool allocation: size=%zu, pool=%d", size, pool_type);
440 }
441 }
442
443 // Fallback to malloc if pool allocation failed or size too large
444 if (!result && global_memory_pool->enable_malloc_fallback) {
445 result = malloc(size);
446
447 // Track fallback sizes for optimization analysis
448 if (fallback_count < 100) {
449 fallback_sizes[fallback_count++] = size;
450 }
451
452 POOL_DEBUG("Malloc fallback: size=%zu (total fallbacks: %d)", size,
453 fallback_count);
454 }
455
456 // Update statistics
457 if (result) {
458 update_stats(pool_hit, size, get_timestamp_ns() - start_time);
459 }
460
461 pthread_mutex_unlock(&pool_mutex);
462
463 set_last_error(result ? LUSUSH_POOL_SUCCESS
464 : LUSUSH_POOL_ERROR_MALLOC_FAILED);
465 return result;
466}

468void lusush_pool_free(void *ptr) {
469 if (!ptr) {
470 return;
471 }
472
473 pthread_mutex_lock(&pool_mutex);
474
475 // Handle case where pool is NULL
476 if (!global_memory_pool) {
477 pthread_mutex_unlock(&pool_mutex);
478 if (pool_was_ever_initialized) {
479 // Pool was shut down - memory was already freed during shutdown.
480 // Do NOT call free() here as it would cause double-free.
481 return;
482 } else {
483 // Pool was never initialized - this memory came from malloc
484 // fallback. We must free it to avoid leaking.
485 free(ptr);
486 return;
487 }
488 }
489
490 bool returned_to_pool = false;
491
492 if (global_memory_pool->initialized) {
493 // Try to return to appropriate pool
494 for (int i = 0; i < LUSUSH_POOL_COUNT; i++) {
495 if (return_to_pool(&global_memory_pool->pools[i], ptr)) {
496 returned_to_pool = true;
497
498 // Update statistics
499 if (global_memory_pool->enable_statistics) {
500 global_memory_pool->stats.active_allocations--;
501 global_memory_pool->stats.current_pool_usage -=
502 POOL_SIZES[i];
503 }
504
505 POOL_DEBUG("Freed to pool: size=%zu, pool=%d", POOL_SIZES[i],
506 i);
507 break;
508 }
509 }
510 }
511
512 pthread_mutex_unlock(&pool_mutex);
513
514 // If not returned to pool, use standard free
515 if (!returned_to_pool) {
516 uintptr_t ptr_addr =
517 (uintptr_t)ptr; // Save address as integer before free
518 free(ptr);
519 POOL_DEBUG("Standard free: ptr=%#" PRIxPTR, ptr_addr);
520
521 // Update statistics for malloc fallback free
522 if (global_memory_pool && global_memory_pool->enable_statistics) {
523 pthread_mutex_lock(&pool_mutex);
524 global_memory_pool->stats.active_allocations--;
525 pthread_mutex_unlock(&pool_mutex);
526 }
527 }
528
529 set_last_error(LUSUSH_POOL_SUCCESS);
530}

532void *lusush_pool_realloc(void *ptr, size_t new_size) {
533 if (new_size == 0) {
534 lusush_pool_free(ptr);
535 return NULL;
536 }
537
538 if (!ptr) {
539 return lusush_pool_alloc(new_size);
540 }
541
542 // For simplicity, we use malloc/free for realloc operations
543 // This could be optimized in the future to use pools when possible
544 void *new_ptr = lusush_pool_alloc(new_size);
545 if (new_ptr && ptr) {
546 // Copy data (we don't know the original size, so copy conservatively)
547 // This is a limitation - in practice, we might need to track allocation
548 // sizes
549 memcpy(new_ptr, ptr,
550 new_size); // Assumes new_size <= old_size or undefined behavior
551 lusush_pool_free(ptr);
552 }
553
554 return new_ptr;
555}

607bool lusush_pool_is_healthy(void) {
608 if (!global_memory_pool || !global_memory_pool->initialized) {
609 return false;
610 }
611
612 pthread_mutex_lock(&pool_mutex);
613
614 // Check each pool has some free blocks available
615 bool healthy = true;
616 for (int i = 0; i < LUSUSH_POOL_COUNT; i++) {
617 if (global_memory_pool->pools[i].free_blocks == 0 &&
618 global_memory_pool->pools[i].current_blocks >=
619 global_memory_pool->pools[i].max_blocks) {
620 healthy = false;
621 break;
622 }
623 }
624
625 pthread_mutex_unlock(&pool_mutex);
626 return healthy;
627}

633void lusush_pool_analyze_fallback_patterns(void) {
634 if (fallback_count == 0) {
635 printf("No malloc fallbacks recorded\n");
636 return;
637 }
638
639 printf("=== Memory Pool Fallback Analysis ===\n");
640 printf("Total fallbacks: %d\n", fallback_count);
641
642 // Count fallbacks by size ranges
643 int small_misses = 0, medium_misses = 0, large_misses = 0,
644 xlarge_misses = 0, oversized = 0;
645
646 for (int i = 0; i < fallback_count; i++) {
647 size_t size = fallback_sizes[i];
648 if (size <= 128)
649 small_misses++;
650 else if (size <= 512)
651 medium_misses++;
652 else if (size <= 4096)
653 large_misses++;
654 else if (size <= 16384)
655 xlarge_misses++;
656 else
657 oversized++;
658 }
659
660 printf("Fallback breakdown:\n");
661 printf(" 128B (SMALL pool): %d fallbacks\n", small_misses);
662 printf(" 512B (MEDIUM pool): %d fallbacks\n", medium_misses);
663 printf(" 4KB (LARGE pool): %d fallbacks\n", large_misses);
664 printf(" 16KB (XLARGE pool): %d fallbacks\n", xlarge_misses);
665 printf(" > 16KB (oversized): %d fallbacks\n", oversized);
666
667 // Show pool status
668 if (global_memory_pool && global_memory_pool->initialized) {
669 printf("Pool Status:\n");
670 for (int i = 0; i < LUSUSH_POOL_COUNT; i++) {
671 lusush_pool_t *pool = &global_memory_pool->pools[i];
672 printf(" Pool %d (%zuB): %zu/%zu blocks free\n", i, POOL_SIZES[i],
673 pool->free_blocks, pool->current_blocks);
674 }
675 }
676
677 // Show actual sizes for first 20 fallbacks
678 printf("First %d fallback sizes: ",
679 (fallback_count < 20) ? fallback_count : 20);
680 for (int i = 0; i < fallback_count && i < 20; i++) {
681 printf("%zu ", fallback_sizes[i]);
682 }
683 printf("\n=====================================\n");
684}

686bool lusush_pool_is_pool_pointer(const void *ptr) {
687 if (!ptr || !global_memory_pool) {
688 return false;
689 }
690
691 pthread_mutex_lock(&pool_mutex);
692
693 bool is_pool_ptr = false;
694 for (int i = 0; i < LUSUSH_POOL_COUNT; i++) {
695 lusush_pool_t *pool = &global_memory_pool->pools[i];
696 for (size_t j = 0; j < pool->current_blocks; j++) {
697 if (pool->all_blocks[j].memory == ptr) {
698 is_pool_ptr = true;
699 goto cleanup;
700 }
701 }
702 }
703
704cleanup:
705 pthread_mutex_unlock(&pool_mutex);
706 return is_pool_ptr;
707}

804void lusush_pool_get_memory_usage(uint64_t *pool_bytes, uint64_t *malloc_bytes,
805 double *pool_efficiency) {
806 if (!global_memory_pool) {
807 if (pool_bytes)
808 *pool_bytes = 0;
809 if (malloc_bytes)
810 *malloc_bytes = 0;
811 if (pool_efficiency)
812 *pool_efficiency = 0.0;
813 return;
814 }
815
816 pthread_mutex_lock(&pool_mutex);
817
818 if (pool_bytes) {
819 *pool_bytes = global_memory_pool->stats.current_pool_usage;
820 }
821
822 if (malloc_bytes) {
823 *malloc_bytes = global_memory_pool->stats.total_bytes_allocated -
824 global_memory_pool->stats.current_pool_usage;
825 }
826
827 if (pool_efficiency) {
828 *pool_efficiency = global_memory_pool->stats.pool_hit_rate;
829 }
830
831 pthread_mutex_unlock(&pool_mutex);
832}

834bool lusush_pool_meets_performance_targets(void) {
835 if (!global_memory_pool) {
836 return false;
837 }
838
839 pthread_mutex_lock(&pool_mutex);
840
841 // Performance targets:
842 // - Pool hit rate > 80%
843 // - Average allocation time < 1000 ns
844 // - System healthy (pools not exhausted)
845
846 bool meets_targets =
847 (global_memory_pool->stats.pool_hit_rate > 80.0) &&
848 (global_memory_pool->stats.avg_allocation_time_ns < 1000) &&
849 lusush_pool_is_healthy();
850
851 pthread_mutex_unlock(&pool_mutex);
852 return meets_targets;
