20 * LLE COMPLETION MENU STATE
21 *
22 * Menu state management for completion system.
23 * This module contains ONLY state structures and lifecycle - NO rendering.
24 * All rendering is handled by the display layer.
25 *
26 * State includes:
27 * - Selected item index
28 * - Visible window (for scrolling)
29 * - Category positions
30 * - Configuration
31 */

51/**
52 * Menu configuration
53 */
54typedef struct {
55 size_t max_visible_items; // Maximum items visible at once
56 bool show_category_headers; // Show category headers
57 bool show_type_indicators; // Show type indicators
58 bool show_descriptions; // Show item descriptions
59 bool enable_scrolling; // Enable scrolling for long lists
60 size_t min_items_for_menu; // Minimum items before showing menu
61} lle_completion_menu_config_t;

63// ============================================================================
64// MENU STATE
65// ============================================================================
66
67/**
68 * Completion menu state
69 */
70typedef struct {
71 lle_completion_result_t *result; // Completion result (not owned)
72
73 // Navigation state
74 size_t selected_index; // Currently selected item (global index)
75 size_t first_visible; // First visible item index (for scrolling)
76 size_t visible_count; // Number of visible items
77 size_t target_column; // Sticky column for UP/DOWN navigation
78
79 // Layout state (for multi-column navigation)
80 size_t terminal_width; // Current terminal width
81 size_t column_width; // Width of each column
82 size_t num_columns; // Number of columns in layout
83
84 // Category tracking
85 size_t *category_positions; // Start position of each category
86 size_t category_count; // Number of categories present
87
88 // State flags
89 bool menu_active; // Whether menu is currently active
90
91 // Configuration
92 lle_completion_menu_config_t config; // Menu configuration
93
94 // Memory pool
95 lle_memory_pool_t *memory_pool; // Memory pool for allocations
96} lle_completion_menu_state_t;
